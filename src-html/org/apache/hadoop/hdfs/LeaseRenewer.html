040/**
041 * <p>
042 * Used by {@link DFSClient} for renewing file-being-written leases
043 * on the namenode.
044 * When a file is opened for write (create or append),
045 * namenode stores a file lease for recording the identity of the writer.
046 * The writer (i.e. the DFSClient) is required to renew the lease periodically.
047 * When the lease is not renewed before it expires,
048 * the namenode considers the writer as failed and then it may either let
049 * another writer to obtain the lease or close the file.
050 * </p>
051 * <p>
052 * This class also provides the following functionality:
053 * <ul>
054 * <li>
055 * It maintains a map from (namenode, user) pairs to lease renewers. 
056 * The same {@link LeaseRenewer} instance is used for renewing lease
057 * for all the {@link DFSClient} to the same namenode and the same user.
058 * </li>
059 * <li>
060 * Each renewer maintains a list of {@link DFSClient}.
061 * Periodically the leases for all the clients are renewed.
062 * A client is removed from the list when the client is closed.
063 * </li>
064 * <li>
065 * A thread per namenode per user is used by the {@link LeaseRenewer}
066 * to renew the leases.
067 * </li>
068 * </ul>
069 * </p>
070 */
071class LeaseRenewer {
072  static final Log LOG = LogFactory.getLog(LeaseRenewer.class);
073
074  static final long LEASE_RENEWER_GRACE_DEFAULT = 60*1000L;
075  static final long LEASE_RENEWER_SLEEP_DEFAULT = 1000L;
076
077  /** Get a {@link LeaseRenewer} instance */
078  static LeaseRenewer getInstance(final String authority,
079      final UserGroupInformation ugi, final DFSClient dfsc) throws IOException {
080    final LeaseRenewer r = Factory.INSTANCE.get(authority, ugi);
081    r.addClient(dfsc);
082    return r;
083  }
084
085  /** 
086   * A factory for sharing {@link LeaseRenewer} objects
087   * among {@link DFSClient} instances
088   * so that there is only one renewer per authority per user.
089   */
090  private static class Factory {
091    private static final Factory INSTANCE = new Factory();
092
093    private static class Key {
094      /** Namenode info */
095      final String authority;
096      /** User info */
097      final UserGroupInformation ugi;
098
099      private Key(final String authority, final UserGroupInformation ugi) {
100        if (authority == null) {
101          throw new HadoopIllegalArgumentException("authority == null");
102        } else if (ugi == null) {
103          throw new HadoopIllegalArgumentException("ugi == null");
104        }
105
106        this.authority = authority;
107        this.ugi = ugi;
108      }
109
110      @Override
111      public int hashCode() {
112        return authority.hashCode() ^ ugi.hashCode();
113      }
114
115      @Override
116      public boolean equals(Object obj) {
117        if (obj == this) {
118          return true;
119        }
120        if (obj != null && obj instanceof Key) {
121          final Key that = (Key)obj;
122          return this.authority.equals(that.authority)
123                 && this.ugi.equals(that.ugi);
124        }
125        return false;        
126      }
127
128      @Override
129      public String toString() {
130        return ugi.getShortUserName() + "@" + authority;
131      }
132    }
133
134    /** A map for per user per namenode renewers. */
135    private final Map<Key, LeaseRenewer> renewers = new HashMap<Key, LeaseRenewer>();
136
137    /** Get a renewer. */
138    private synchronized LeaseRenewer get(final String authority,
139        final UserGroupInformation ugi) {
140      final Key k = new Key(authority, ugi);
141      LeaseRenewer r = renewers.get(k);
142      if (r == null) {
143        r = new LeaseRenewer(k);
144        renewers.put(k, r);
145      }
146      return r;
147    }
148
149    /** Remove the given renewer. */
150    private synchronized void remove(final LeaseRenewer r) {
151      final LeaseRenewer stored = renewers.get(r.factorykey);
152      //Since a renewer may expire, the stored renewer can be different.
153      if (r == stored) {
154        if (!r.clientsRunning()) {
155          renewers.remove(r.factorykey);
156        }
157      }
158    }
159  }
160
161  /** The time in milliseconds that the map became empty. */
162  private long emptyTime = Long.MAX_VALUE;
163  /** A fixed lease renewal time period in milliseconds */
164  private long renewal = HdfsConstants.LEASE_SOFTLIMIT_PERIOD/2;
165
166  /** A daemon for renewing lease */
167  private Daemon daemon = null;
168  /** Only the daemon with currentId should run. */
169  private int currentId = 0;
170
171  /** 
172   * A period in milliseconds that the lease renewer thread should run
173   * after the map became empty.
174   * In other words,
175   * if the map is empty for a time period longer than the grace period,
176   * the renewer should terminate.  
177   */
178  private long gracePeriod;
179  /**
180   * The time period in milliseconds
181   * that the renewer sleeps for each iteration. 
182   */
183  private long sleepPeriod;
184
185  private final Factory.Key factorykey;
186
187  /** A list of clients corresponding to this renewer. */
188  private final List<DFSClient> dfsclients = new ArrayList<DFSClient>();
189
190  /**
191   * A stringified stack trace of the call stack when the Lease Renewer
192   * was instantiated. This is only generated if trace-level logging is
193   * enabled on this class.
194   */
195  private final String instantiationTrace;
196
197  private LeaseRenewer(Factory.Key factorykey) {
198    this.factorykey = factorykey;
199    unsyncSetGraceSleepPeriod(LEASE_RENEWER_GRACE_DEFAULT);
200    
201    if (LOG.isTraceEnabled()) {
202      instantiationTrace = StringUtils.stringifyException(
203        new Throwable("TRACE"));
204    } else {
205      instantiationTrace = null;
206    }
207  }
208
209  /** @return the renewal time in milliseconds. */
210  private synchronized long getRenewalTime() {
211    return renewal;
212  }
213
214  /** Add a client. */
215  private synchronized void addClient(final DFSClient dfsc) {
216    for(DFSClient c : dfsclients) {
217      if (c == dfsc) {
218        //client already exists, nothing to do.
219        return;
220      }
221    }
222    //client not found, add it
223    dfsclients.add(dfsc);
224
225    //update renewal time
226    if (dfsc.getHdfsTimeout() > 0) {
227      final long half = dfsc.getHdfsTimeout()/2;
228      if (half < renewal) {
229        this.renewal = half;
230      }
231    }
232  }
233
234  private synchronized boolean clientsRunning() {
235    for(Iterator<DFSClient> i = dfsclients.iterator(); i.hasNext(); ) {
236      if (!i.next().isClientRunning()) {
237        i.remove();
238      }
239    }
240    return !dfsclients.isEmpty();
241  }
242
243  private synchronized long getSleepPeriod() {
244    return sleepPeriod;    
245  }
246
247  /** Set the grace period and adjust the sleep period accordingly. */
248  synchronized void setGraceSleepPeriod(final long gracePeriod) {
249    unsyncSetGraceSleepPeriod(gracePeriod);
250  }
251
252  private void unsyncSetGraceSleepPeriod(final long gracePeriod) {
253    if (gracePeriod < 100L) {
254      throw new HadoopIllegalArgumentException(gracePeriod
255          + " = gracePeriod < 100ms is too small.");
256    }
257    this.gracePeriod = gracePeriod;
258    final long half = gracePeriod/2;
259    this.sleepPeriod = half < LEASE_RENEWER_SLEEP_DEFAULT?
260        half: LEASE_RENEWER_SLEEP_DEFAULT;
261  }
262
263  /** Is the daemon running? */
264  synchronized boolean isRunning() {
265    return daemon != null && daemon.isAlive();
266  }
267
268  /** Does this renewer have nothing to renew? */
269  public boolean isEmpty() {
270    return dfsclients.isEmpty();
271  }
272  
273  /** Used only by tests */
274  synchronized String getDaemonName() {
275    return daemon.getName();
276  }
277
278  /** Is the empty period longer than the grace period? */  
279  private synchronized boolean isRenewerExpired() {
280    return emptyTime != Long.MAX_VALUE
281        && Time.monotonicNow() - emptyTime > gracePeriod;
282  }
283
284  synchronized void put(final long inodeId, final DFSOutputStream out,
285      final DFSClient dfsc) {
286    if (dfsc.isClientRunning()) {
287      if (!isRunning() || isRenewerExpired()) {
288        //start a new deamon with a new id.
289        final int id = ++currentId;
290        daemon = new Daemon(new Runnable() {
291          @Override
292          public void run() {
293            try {
294              if (LOG.isDebugEnabled()) {
295                LOG.debug("Lease renewer daemon for " + clientsString()
296                    + " with renew id " + id + " started");
297              }
298              LeaseRenewer.this.run(id);
299            } catch(InterruptedException e) {
300              if (LOG.isDebugEnabled()) {
301                LOG.debug(LeaseRenewer.this.getClass().getSimpleName()
302                    + " is interrupted.", e);
303              }
304            } finally {
305              synchronized(LeaseRenewer.this) {
306                Factory.INSTANCE.remove(LeaseRenewer.this);
307              }
308              if (LOG.isDebugEnabled()) {
309                LOG.debug("Lease renewer daemon for " + clientsString()
310                    + " with renew id " + id + " exited");
311              }
312            }
313          }
314          
315          @Override
316          public String toString() {
317            return String.valueOf(LeaseRenewer.this);
318          }
319        });
320        daemon.start();
321      }
322      dfsc.putFileBeingWritten(inodeId, out);
323      emptyTime = Long.MAX_VALUE;
324    }
325  }
326
327  @VisibleForTesting
328  synchronized void setEmptyTime(long time) {
329    emptyTime = time;
330  }
331
332  /** Close a file. */
333  void closeFile(final long inodeId, final DFSClient dfsc) {
334    dfsc.removeFileBeingWritten(inodeId);
335
336    synchronized(this) {
337      if (dfsc.isFilesBeingWrittenEmpty()) {
338        dfsclients.remove(dfsc);
339      }
340      //update emptyTime if necessary
341      if (emptyTime == Long.MAX_VALUE) {
342        for(DFSClient c : dfsclients) {
343          if (!c.isFilesBeingWrittenEmpty()) {
344            //found a non-empty file-being-written map
345            return;
346          }
347        }
348        //discover the first time that all file-being-written maps are empty.
349        emptyTime = Time.monotonicNow();
350      }
351    }
352  }
353
354  /** Close the given client. */
355  synchronized void closeClient(final DFSClient dfsc) {
356    dfsclients.remove(dfsc);
357    if (dfsclients.isEmpty()) {
358      if (!isRunning() || isRenewerExpired()) {
359        Factory.INSTANCE.remove(LeaseRenewer.this);
360        return;
361      }
362      if (emptyTime == Long.MAX_VALUE) {
363        //discover the first time that the client list is empty.
364        emptyTime = Time.monotonicNow();
365      }
366    }
367
368    //update renewal time
369    if (renewal == dfsc.getHdfsTimeout()/2) {
370      long min = HdfsConstants.LEASE_SOFTLIMIT_PERIOD;
371      for(DFSClient c : dfsclients) {
372        if (c.getHdfsTimeout() > 0) {
373          final long timeout = c.getHdfsTimeout();
374          if (timeout < min) {
375            min = timeout;
376          }
377        }
378      }
379      renewal = min/2;
380    }
381  }
382
383  void interruptAndJoin() throws InterruptedException {
384    Daemon daemonCopy = null;
385    synchronized (this) {
386      if (isRunning()) {
387        daemon.interrupt();
388        daemonCopy = daemon;
389      }
390    }
391   
392    if (daemonCopy != null) {
393      if(LOG.isDebugEnabled()) {
394        LOG.debug("Wait for lease checker to terminate");
395      }
396      daemonCopy.join();
397    }
398  }
399
400  private void renew() throws IOException {
401    final List<DFSClient> copies;
402    synchronized(this) {
403      copies = new ArrayList<DFSClient>(dfsclients);
404    }
405    //sort the client names for finding out repeated names.
406    Collections.sort(copies, new Comparator<DFSClient>() {
407      @Override
408      public int compare(final DFSClient left, final DFSClient right) {
409        return left.getClientName().compareTo(right.getClientName());
410      }
411    });
412    String previousName = "";
413    for(int i = 0; i < copies.size(); i++) {
414      final DFSClient c = copies.get(i);
415      //skip if current client name is the same as the previous name.
416      if (!c.getClientName().equals(previousName)) {
417        if (!c.renewLease()) {
418          if (LOG.isDebugEnabled()) {
419            LOG.debug("Did not renew lease for client " +
420                c);
421          }
422          continue;
423        }
424        previousName = c.getClientName();
425        if (LOG.isDebugEnabled()) {
426          LOG.debug("Lease renewed for client " + previousName);
427        }
428      }
429    }
430  }
431
432  /**
433   * Periodically check in with the namenode and renew all the leases
434   * when the lease period is half over.
435   */
436  private void run(final int id) throws InterruptedException {
437    for(long lastRenewed = Time.monotonicNow(); !Thread.interrupted();
438        Thread.sleep(getSleepPeriod())) {
439      final long elapsed = Time.monotonicNow() - lastRenewed;
440      if (elapsed >= getRenewalTime()) {
441        try {
442          renew();
443          if (LOG.isDebugEnabled()) {
444            LOG.debug("Lease renewer daemon for " + clientsString()
445                + " with renew id " + id + " executed");
446          }
447          lastRenewed = Time.monotonicNow();
448        } catch (SocketTimeoutException ie) {
449          LOG.warn("Failed to renew lease for " + clientsString() + " for "
450              + (elapsed/1000) + " seconds.  Aborting ...", ie);
451          synchronized (this) {
452            while (!dfsclients.isEmpty()) {
453              dfsclients.get(0).abort();
454            }
455          }
456          break;
457        } catch (IOException ie) {
458          LOG.warn("Failed to renew lease for " + clientsString() + " for "
459              + (elapsed/1000) + " seconds.  Will retry shortly ...", ie);
460        }
461      }
462
463      synchronized(this) {
464        if (id != currentId || isRenewerExpired()) {
465          if (LOG.isDebugEnabled()) {
466            if (id != currentId) {
467              LOG.debug("Lease renewer daemon for " + clientsString()
468                  + " with renew id " + id + " is not current");
469            } else {
470               LOG.debug("Lease renewer daemon for " + clientsString()
471                  + " with renew id " + id + " expired");
472            }
473          }
474          //no longer the current daemon or expired
475          return;
476        }
477
478        // if no clients are in running state or there is no more clients
479        // registered with this renewer, stop the daemon after the grace
480        // period.
481        if (!clientsRunning() && emptyTime == Long.MAX_VALUE) {
482          emptyTime = Time.monotonicNow();
483        }
484      }
485    }
486  }
487
488  @Override
489  public String toString() {
490    String s = getClass().getSimpleName() + ":" + factorykey;
491    if (LOG.isTraceEnabled()) {
492      return s + ", clients=" +  clientsString()
493        + ", created at " + instantiationTrace;
494    }
495    return s;
496  }
497
498  /** Get the names of all clients */
499  private synchronized String clientsString() {
500    if (dfsclients.isEmpty()) {
501      return "[]";
502    } else {
503      final StringBuilder b = new StringBuilder("[").append(
504          dfsclients.get(0).getClientName());
505      for(int i = 1; i < dfsclients.size(); i++) {
506        b.append(", ").append(dfsclients.get(i).getClientName());
507      }
508      return b.append("]").toString();
509    }
510  }
