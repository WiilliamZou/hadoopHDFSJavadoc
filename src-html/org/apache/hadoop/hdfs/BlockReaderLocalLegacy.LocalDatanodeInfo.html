054/**
055 * BlockReaderLocalLegacy enables local short circuited reads. If the DFS client is on
056 * the same machine as the datanode, then the client can read files directly
057 * from the local file system rather than going through the datanode for better
058 * performance. <br>
059 *
060 * This is the legacy implementation based on HDFS-2246, which requires
061 * permissions on the datanode to be set so that clients can directly access the
062 * blocks. The new implementation based on HDFS-347 should be preferred on UNIX
063 * systems where the required native code has been implemented.<br>
064 *
065 * {@link BlockReaderLocalLegacy} works as follows:
066 * <ul>
067 * <li>The client performing short circuit reads must be configured at the
068 * datanode.</li>
069 * <li>The client gets the path to the file where block is stored using
070 * {@link org.apache.hadoop.hdfs.protocol.ClientDatanodeProtocol#getBlockLocalPathInfo(ExtendedBlock, Token)}
071 * RPC call</li>
072 * <li>Client uses kerberos authentication to connect to the datanode over RPC,
073 * if security is enabled.</li>
074 * </ul>
075 */
076class BlockReaderLocalLegacy implements BlockReader {
077  private static final Log LOG = LogFactory.getLog(BlockReaderLocalLegacy.class);

079  //Stores the cache and proxy for a local datanode.
080  private static class LocalDatanodeInfo {
081    private ClientDatanodeProtocol proxy = null;
082    private final Map<ExtendedBlock, BlockLocalPathInfo> cache;
083
084    LocalDatanodeInfo() {
085      final int cacheSize = 10000;
086      final float hashTableLoadFactor = 0.75f;
087      int hashTableCapacity = (int) Math.ceil(cacheSize / hashTableLoadFactor) + 1;
088      cache = Collections
089          .synchronizedMap(new LinkedHashMap<ExtendedBlock, BlockLocalPathInfo>(
090              hashTableCapacity, hashTableLoadFactor, true) {
091            private static final long serialVersionUID = 1;
092
093            @Override
094            protected boolean removeEldestEntry(
095                Map.Entry<ExtendedBlock, BlockLocalPathInfo> eldest) {
096              return size() > cacheSize;
097            }
098          });
099    }

101    private synchronized ClientDatanodeProtocol getDatanodeProxy(
102        UserGroupInformation ugi, final DatanodeInfo node,
103        final Configuration conf, final int socketTimeout,
104        final boolean connectToDnViaHostname) throws IOException {
105      if (proxy == null) {
106        try {
107          proxy = ugi.doAs(new PrivilegedExceptionAction<ClientDatanodeProtocol>() {
108            @Override
109            public ClientDatanodeProtocol run() throws Exception {
110              return DFSUtil.createClientDatanodeProtocolProxy(node, conf,
111                  socketTimeout, connectToDnViaHostname);
112            }
113          });
114        } catch (InterruptedException e) {
115          LOG.warn("encountered exception ", e);
116        }
117      }
118      return proxy;
119    }

122      if (null != proxy) {
123        RPC.stopProxy(proxy);
124        proxy = null;
125      }
126    }

181  static BlockReaderLocalLegacy newBlockReader(DFSClient.Conf conf,
182      UserGroupInformation userGroupInformation,
183      Configuration configuration, String file, ExtendedBlock blk,
184      Token<BlockTokenIdentifier> token, DatanodeInfo node, 
185      long startOffset, long length, StorageType storageType)
186      throws IOException {
187    LocalDatanodeInfo localDatanodeInfo = getLocalDatanodeInfo(node
188        .getIpcPort());
189    // check the cache first
190    BlockLocalPathInfo pathinfo = localDatanodeInfo.getBlockLocalPathInfo(blk);
191    if (pathinfo == null) {
192      if (userGroupInformation == null) {
193        userGroupInformation = UserGroupInformation.getCurrentUser();
194      }
195      pathinfo = getBlockPathInfo(userGroupInformation, blk, node,
196          configuration, conf.socketTimeout, token,
197          conf.connectToDnViaHostname, storageType);
198    }
199
200    // check to see if the file exists. It may so happen that the
201    // HDFS file has been deleted and this block-lookup is occurring
202    // on behalf of a new HDFS file. This time, the block file could
203    // be residing in a different portion of the fs.data.dir directory.
204    // In this case, we remove this entry from the cache. The next
205    // call to this method will re-populate the cache.
206    FileInputStream dataIn = null;
207    FileInputStream checksumIn = null;
208    BlockReaderLocalLegacy localBlockReader = null;
209    boolean skipChecksumCheck = conf.skipShortCircuitChecksums ||
210        storageType.isTransient();
211    try {
212      // get a local file system
213      File blkfile = new File(pathinfo.getBlockPath());
214      dataIn = new FileInputStream(blkfile);
215
216      if (LOG.isDebugEnabled()) {
217        LOG.debug("New BlockReaderLocalLegacy for file " + blkfile + " of size "
218            + blkfile.length() + " startOffset " + startOffset + " length "
219            + length + " short circuit checksum " + !skipChecksumCheck);
220      }
221
222      if (!skipChecksumCheck) {
223        // get the metadata file
224        File metafile = new File(pathinfo.getMetaPath());
225        checksumIn = new FileInputStream(metafile);
226
227        final DataChecksum checksum = BlockMetadataHeader.readDataChecksum(
228            new DataInputStream(checksumIn), blk);
229        long firstChunkOffset = startOffset
230            - (startOffset % checksum.getBytesPerChecksum());
231        localBlockReader = new BlockReaderLocalLegacy(conf, file, blk, token,
232            startOffset, length, pathinfo, checksum, true, dataIn,
233            firstChunkOffset, checksumIn);
234      } else {
235        localBlockReader = new BlockReaderLocalLegacy(conf, file, blk, token,
236            startOffset, length, pathinfo, dataIn);
237      }
238    } catch (IOException e) {
239      // remove from cache
240      localDatanodeInfo.removeBlockLocalPathInfo(blk);
241      DFSClient.LOG.warn("BlockReaderLocalLegacy: Removing " + blk
242          + " from cache because local file " + pathinfo.getBlockPath()
243          + " could not be opened.");
244      throw e;
245    } finally {
246      if (localBlockReader == null) {
247        if (dataIn != null) {
248          dataIn.close();
249        }
250        if (checksumIn != null) {
251          checksumIn.close();
252        }
253      }
254    }
255    return localBlockReader;
256  }

260    if (ldInfo == null) {

262      localDatanodeInfoMap.put(port, ldInfo);
263    }

267  private static BlockLocalPathInfo getBlockPathInfo(UserGroupInformation ugi,
268      ExtendedBlock blk, DatanodeInfo node, Configuration conf, int timeout,
269      Token<BlockTokenIdentifier> token, boolean connectToDnViaHostname,
270      StorageType storageType) throws IOException {
271    LocalDatanodeInfo localDatanodeInfo = getLocalDatanodeInfo(node.getIpcPort());
272    BlockLocalPathInfo pathinfo = null;
273    ClientDatanodeProtocol proxy = localDatanodeInfo.getDatanodeProxy(ugi, node,
274        conf, timeout, connectToDnViaHostname);
275    try {
276      // make RPC to local datanode to find local pathnames of blocks
277      pathinfo = proxy.getBlockLocalPathInfo(blk, token);
278      // We cannot cache the path information for a replica on transient storage.
279      // If the replica gets evicted, then it moves to a different path.  Then,
280      // our next attempt to read from the cached path would fail to find the
281      // file.  Additionally, the failure would cause us to disable legacy
282      // short-circuit read for all subsequent use in the ClientContext.  Unlike
283      // the newer short-circuit read implementation, we have no communication
284      // channel for the DataNode to notify the client that the path has been
285      // invalidated.  Therefore, our only option is to skip caching.
286      if (pathinfo != null && !storageType.isTransient()) {
287        if (LOG.isDebugEnabled()) {
288          LOG.debug("Cached location of block " + blk + " as " + pathinfo);
289        }
290        localDatanodeInfo.setBlockLocalPathInfo(blk, pathinfo);
291      }
292    } catch (IOException e) {
293      localDatanodeInfo.resetDatanodeProxy(); // Reset proxy on error
294      throw e;
295    }
296    return pathinfo;
297  }

299  private static int getSlowReadBufferNumChunks(int bufferSizeBytes,
300      int bytesPerChecksum) {
301    if (bufferSizeBytes < bytesPerChecksum) {
302      throw new IllegalArgumentException("Configured BlockReaderLocalLegacy " +
303          "buffer size (" + bufferSizeBytes + ") is not large enough to hold " +
304          "a single chunk (" + bytesPerChecksum +  "). Please configure " +
305          DFSConfigKeys.DFS_CLIENT_READ_SHORTCIRCUIT_BUFFER_SIZE_KEY +
306          " appropriately");
307    }
308
309    // Round down to nearest chunk size
310    return bufferSizeBytes / bytesPerChecksum;
311  }

322  private BlockReaderLocalLegacy(DFSClient.Conf conf, String hdfsfile,
323      ExtendedBlock block, Token<BlockTokenIdentifier> token, long startOffset,
324      long length, BlockLocalPathInfo pathinfo, DataChecksum checksum,
325      boolean verifyChecksum, FileInputStream dataIn, long firstChunkOffset,
326      FileInputStream checksumIn) throws IOException {
327    this.filename = hdfsfile;
328    this.checksum = checksum;
329    this.verifyChecksum = verifyChecksum;
330    this.startOffset = Math.max(startOffset, 0);
331    this.blockId = block.getBlockId();
332
333    bytesPerChecksum = this.checksum.getBytesPerChecksum();
334    checksumSize = this.checksum.getChecksumSize();
335
336    this.dataIn = dataIn;
337    this.checksumIn = checksumIn;
338    this.offsetFromChunkBoundary = (int) (startOffset-firstChunkOffset);
339
340    int chunksPerChecksumRead = getSlowReadBufferNumChunks(
341        conf.shortCircuitBufferSize, bytesPerChecksum);
342    slowReadBuff = bufferPool.getBuffer(bytesPerChecksum * chunksPerChecksumRead);
343    checksumBuff = bufferPool.getBuffer(checksumSize * chunksPerChecksumRead);
344    // Initially the buffers have nothing to read.
345    slowReadBuff.flip();
346    checksumBuff.flip();
347    boolean success = false;
348    try {
349      // Skip both input streams to beginning of the chunk containing startOffset
350      IOUtils.skipFully(dataIn, firstChunkOffset);
351      if (checksumIn != null) {
352        long checkSumOffset = (firstChunkOffset / bytesPerChecksum) * checksumSize;
353        IOUtils.skipFully(checksumIn, checkSumOffset);
354      }
355      success = true;
356    } finally {
357      if (!success) {
358        bufferPool.returnBuffer(slowReadBuff);
359        bufferPool.returnBuffer(checksumBuff);
360      }
361    }
362  }

364  /**
365   * Reads bytes into a buffer until EOF or the buffer's limit is reached
366   */
367  private int fillBuffer(FileInputStream stream, ByteBuffer buf)
368      throws IOException {
369    TraceScope scope = Trace.startSpan("BlockReaderLocalLegacy#fillBuffer(" +
370        blockId + ")", Sampler.NEVER);
371    try {
372      int bytesRead = stream.getChannel().read(buf);
373      if (bytesRead < 0) {
374        //EOF
375        return bytesRead;
376      }
377      while (buf.remaining() > 0) {
378        int n = stream.getChannel().read(buf);
379        if (n < 0) {
380          //EOF
381          return bytesRead;
382        }
383        bytesRead += n;
384      }
385      return bytesRead;
386    } finally {
387      scope.close();
388    }
389  }

391  /**
392   * Utility method used by read(ByteBuffer) to partially copy a ByteBuffer into
393   * another.
394   */
395  private void writeSlice(ByteBuffer from, ByteBuffer to, int length) {
396    int oldLimit = from.limit();
397    from.limit(from.position() + length);
398    try {
399      to.put(from);
400    } finally {
401      from.limit(oldLimit);
402    }
403  }

405  @Override
406  public synchronized int read(ByteBuffer buf) throws IOException {
407    int nRead = 0;
408    if (verifyChecksum) {
409      // A 'direct' read actually has three phases. The first drains any
410      // remaining bytes from the slow read buffer. After this the read is
411      // guaranteed to be on a checksum chunk boundary. If there are still bytes
412      // to read, the fast direct path is used for as many remaining bytes as
413      // possible, up to a multiple of the checksum chunk size. Finally, any
414      // 'odd' bytes remaining at the end of the read cause another slow read to
415      // be issued, which involves an extra copy.
416
417      // Every 'slow' read tries to fill the slow read buffer in one go for
418      // efficiency's sake. As described above, all non-checksum-chunk-aligned
419      // reads will be served from the slower read path.
420
421      if (slowReadBuff.hasRemaining()) {
422        // There are remaining bytes from a small read available. This usually
423        // means this read is unaligned, which falls back to the slow path.
424        int fromSlowReadBuff = Math.min(buf.remaining(), slowReadBuff.remaining());
425        writeSlice(slowReadBuff, buf, fromSlowReadBuff);
426        nRead += fromSlowReadBuff;
427      }
428
429      if (buf.remaining() >= bytesPerChecksum && offsetFromChunkBoundary == 0) {
430        // Since we have drained the 'small read' buffer, we are guaranteed to
431        // be chunk-aligned
432        int len = buf.remaining() - (buf.remaining() % bytesPerChecksum);
433
434        // There's only enough checksum buffer space available to checksum one
435        // entire slow read buffer. This saves keeping the number of checksum
436        // chunks around.
437        len = Math.min(len, slowReadBuff.capacity());
438        int oldlimit = buf.limit();
439        buf.limit(buf.position() + len);
440        int readResult = 0;
441        try {
442          readResult = doByteBufferRead(buf);
443        } finally {
444          buf.limit(oldlimit);
445        }
446        if (readResult == -1) {
447          return nRead;
448        } else {
449          nRead += readResult;
450          buf.position(buf.position() + readResult);
451        }
452      }
453
454      // offsetFromChunkBoundary > 0 => unaligned read, use slow path to read
455      // until chunk boundary
456      if ((buf.remaining() > 0 && buf.remaining() < bytesPerChecksum) || offsetFromChunkBoundary > 0) {
457        int toRead = Math.min(buf.remaining(), bytesPerChecksum - offsetFromChunkBoundary);
458        int readResult = fillSlowReadBuffer(toRead);
459        if (readResult == -1) {
460          return nRead;
461        } else {
462          int fromSlowReadBuff = Math.min(readResult, buf.remaining());
463          writeSlice(slowReadBuff, buf, fromSlowReadBuff);
464          nRead += fromSlowReadBuff;
465        }
466      }
467    } else {
468      // Non-checksummed reads are much easier; we can just fill the buffer directly.
469      nRead = doByteBufferRead(buf);
470      if (nRead > 0) {
471        buf.position(buf.position() + nRead);
472      }
473    }
474    return nRead;
475  }

477  /**
478   * Tries to read as many bytes as possible into supplied buffer, checksumming
479   * each chunk if needed.
480   *
481   * <b>Preconditions:</b>
482   * <ul>
483   * <li>
484   * If checksumming is enabled, buf.remaining must be a multiple of
485   * bytesPerChecksum. Note that this is not a requirement for clients of
486   * read(ByteBuffer) - in the case of non-checksum-sized read requests,
487   * read(ByteBuffer) will substitute a suitably sized buffer to pass to this
488   * method.
489   * </li>
490   * </ul>
491   * <b>Postconditions:</b>
492   * <ul>
493   * <li>buf.limit and buf.mark are unchanged.</li>
494   * <li>buf.position += min(offsetFromChunkBoundary, totalBytesRead) - so the
495   * requested bytes can be read straight from the buffer</li>
496   * </ul>
497   *
498   * @param buf
499   *          byte buffer to write bytes to. If checksums are not required, buf
500   *          can have any number of bytes remaining, otherwise there must be a
501   *          multiple of the checksum chunk size remaining.
502   * @return <tt>max(min(totalBytesRead, len) - offsetFromChunkBoundary, 0)</tt>
503   *         that is, the the number of useful bytes (up to the amount
504   *         requested) readable from the buffer by the client.
505   */
506  private synchronized int doByteBufferRead(ByteBuffer buf) throws IOException {
507    if (verifyChecksum) {
508      assert buf.remaining() % bytesPerChecksum == 0;
509    }
510    int dataRead = -1;
511
512    int oldpos = buf.position();
513    // Read as much as we can into the buffer.
514    dataRead = fillBuffer(dataIn, buf);
515
516    if (dataRead == -1) {
517      return -1;
518    }
519
520    if (verifyChecksum) {
521      ByteBuffer toChecksum = buf.duplicate();
522      toChecksum.position(oldpos);
523      toChecksum.limit(oldpos + dataRead);
524
525      checksumBuff.clear();
526      // Equivalent to (int)Math.ceil(toChecksum.remaining() * 1.0 / bytesPerChecksum );
527      int numChunks =
528        (toChecksum.remaining() + bytesPerChecksum - 1) / bytesPerChecksum;
529      checksumBuff.limit(checksumSize * numChunks);
530
531      fillBuffer(checksumIn, checksumBuff);
532      checksumBuff.flip();
533
534      checksum.verifyChunkedSums(toChecksum, checksumBuff, filename,
535          this.startOffset);
536    }
537
538    if (dataRead >= 0) {
539      buf.position(oldpos + Math.min(offsetFromChunkBoundary, dataRead));
540    }
541
542    if (dataRead < offsetFromChunkBoundary) {
543      // yikes, didn't even get enough bytes to honour offset. This can happen
544      // even if we are verifying checksums if we are at EOF.
545      offsetFromChunkBoundary -= dataRead;
546      dataRead = 0;
547    } else {
548      dataRead -= offsetFromChunkBoundary;
549      offsetFromChunkBoundary = 0;
550    }
551
552    return dataRead;
553  }

555  /**
556   * Ensures that up to len bytes are available and checksummed in the slow read
557   * buffer. The number of bytes available to read is returned. If the buffer is
558   * not already empty, the number of remaining bytes is returned and no actual
559   * read happens.
560   *
561   * @param len
562   *          the maximum number of bytes to make available. After len bytes
563   *          are read, the underlying bytestream <b>must</b> be at a checksum
564   *          boundary, or EOF. That is, (len + currentPosition) %
565   *          bytesPerChecksum == 0.
566   * @return the number of bytes available to read, or -1 if EOF.
567   */
568  private synchronized int fillSlowReadBuffer(int len) throws IOException {
569    int nRead = -1;
570    if (slowReadBuff.hasRemaining()) {
571      // Already got data, good to go.
572      nRead = Math.min(len, slowReadBuff.remaining());
573    } else {
574      // Round a complete read of len bytes (plus any implicit offset) to the
575      // next chunk boundary, since we try and read in multiples of a chunk
576      int nextChunk = len + offsetFromChunkBoundary +
577          (bytesPerChecksum - ((len + offsetFromChunkBoundary) % bytesPerChecksum));
578      int limit = Math.min(nextChunk, slowReadBuff.capacity());
579      assert limit % bytesPerChecksum == 0;
580
581      slowReadBuff.clear();
582      slowReadBuff.limit(limit);
583
584      nRead = doByteBufferRead(slowReadBuff);
585
586      if (nRead > 0) {
587        // So that next time we call slowReadBuff.hasRemaining(), we don't get a
588        // false positive.
589        slowReadBuff.limit(nRead + slowReadBuff.position());
590      }
591    }
592    return nRead;
593  }

595  @Override
596  public synchronized int read(byte[] buf, int off, int len) throws IOException {
597    if (LOG.isTraceEnabled()) {
598      LOG.trace("read off " + off + " len " + len);
599    }
600    if (!verifyChecksum) {
601      return dataIn.read(buf, off, len);
602    }
603
604    int nRead = fillSlowReadBuffer(slowReadBuff.capacity());
605
606    if (nRead > 0) {
607      // Possible that buffer is filled with a larger read than we need, since
608      // we tried to read as much as possible at once
609      nRead = Math.min(len, nRead);
610      slowReadBuff.get(buf, off, nRead);
611    }
612
613    return nRead;
614  }

616  @Override
617  public synchronized long skip(long n) throws IOException {
618    if (LOG.isDebugEnabled()) {
619      LOG.debug("skip " + n);
620    }
621    if (n <= 0) {
622      return 0;
623    }
624    if (!verifyChecksum) {
625      return dataIn.skip(n);
626    }
627  
628    // caller made sure newPosition is not beyond EOF.
629    int remaining = slowReadBuff.remaining();
630    int position = slowReadBuff.position();
631    int newPosition = position + (int)n;
632  
633    // if the new offset is already read into dataBuff, just reposition
634    if (n <= remaining) {
635      assert offsetFromChunkBoundary == 0;
636      slowReadBuff.position(newPosition);
637      return n;
638    }
639  
640    // for small gap, read through to keep the data/checksum in sync
641    if (n - remaining <= bytesPerChecksum) {
642      slowReadBuff.position(position + remaining);
643      if (skipBuf == null) {
644        skipBuf = new byte[bytesPerChecksum];
645      }
646      int ret = read(skipBuf, 0, (int)(n - remaining));
647      return (remaining + ret);
648    }
649  
650    // optimize for big gap: discard the current buffer, skip to
651    // the beginning of the appropriate checksum chunk and then
652    // read to the middle of that chunk to be in sync with checksums.
653  
654    // We can't use this.offsetFromChunkBoundary because we need to know how
655    // many bytes of the offset were really read. Calling read(..) with a
656    // positive this.offsetFromChunkBoundary causes that many bytes to get
657    // silently skipped.
658    int myOffsetFromChunkBoundary = newPosition % bytesPerChecksum;
659    long toskip = n - remaining - myOffsetFromChunkBoundary;
660
661    slowReadBuff.position(slowReadBuff.limit());
662    checksumBuff.position(checksumBuff.limit());
663  
664    IOUtils.skipFully(dataIn, toskip);
665    long checkSumOffset = (toskip / bytesPerChecksum) * checksumSize;
666    IOUtils.skipFully(checksumIn, checkSumOffset);
667
668    // read into the middle of the chunk
669    if (skipBuf == null) {
670      skipBuf = new byte[bytesPerChecksum];
671    }
672    assert skipBuf.length == bytesPerChecksum;
673    assert myOffsetFromChunkBoundary < bytesPerChecksum;
674
675    int ret = read(skipBuf, 0, myOffsetFromChunkBoundary);
676
677    if (ret == -1) {  // EOS
678      return (toskip + remaining);
679    } else {
680      return (toskip + remaining + ret);
681    }
682  }
683
684  @Override
685  public synchronized void close() throws IOException {
686    IOUtils.cleanup(LOG, dataIn, checksumIn);
687    if (slowReadBuff != null) {
688      bufferPool.returnBuffer(slowReadBuff);
689      slowReadBuff = null;
690    }
691    if (checksumBuff != null) {
692      bufferPool.returnBuffer(checksumBuff);
693      checksumBuff = null;
694    }
695    startOffset = -1;
696    checksum = null;
697  }
