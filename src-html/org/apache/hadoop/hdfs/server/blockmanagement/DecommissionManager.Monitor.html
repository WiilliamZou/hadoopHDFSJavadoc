050/**
051 * Manages datanode decommissioning. A background monitor thread 
052 * periodically checks the status of datanodes that are in-progress of 
053 * decommissioning.
054 * <p/>
055 * A datanode can be decommissioned in a few situations:
056 * <ul>
057 * <li>If a DN is dead, it is decommissioned immediately.</li>
058 * <li>If a DN is alive, it is decommissioned after all of its blocks 
059 * are sufficiently replicated. Merely under-replicated blocks do not 
060 * block decommissioning as long as they are above a replication 
061 * threshold.</li>
062 * </ul>
063 * In the second case, the datanode transitions to a 
064 * decommission-in-progress state and is tracked by the monitor thread. The 
065 * monitor periodically scans through the list of insufficiently replicated
066 * blocks on these datanodes to 
067 * determine if they can be decommissioned. The monitor also prunes this list 
068 * as blocks become replicated, so monitor scans will become more efficient 
069 * over time.
070 * <p/>
071 * Decommission-in-progress nodes that become dead do not progress to 
072 * decommissioned until they become live again. This prevents potential 
073 * durability loss for singly-replicated blocks (see HDFS-6791).
074 * <p/>
075 * This class depends on the FSNamesystem lock for synchronization.
076 */
077@InterfaceAudience.Private
078public class DecommissionManager {
079  private static final Logger LOG = LoggerFactory.getLogger(DecommissionManager
080      .class);
081
082  private final Namesystem namesystem;
083  private final BlockManager blockManager;
084  private final HeartbeatManager hbManager;
085  private final ScheduledExecutorService executor;
086
087  /**
088   * Map containing the decommission-in-progress datanodes that are being
089   * tracked so they can be be marked as decommissioned.
090   * <p/>
091   * This holds a set of references to the under-replicated blocks on the DN at
092   * the time the DN is added to the map, i.e. the blocks that are preventing
093   * the node from being marked as decommissioned. During a monitor tick, this
094   * list is pruned as blocks becomes replicated.
095   * <p/>
096   * Note also that the reference to the list of under-replicated blocks 
097   * will be null on initial add
098   * <p/>
099   * However, this map can become out-of-date since it is not updated by block
100   * reports or other events. Before being finally marking as decommissioned,
101   * another check is done with the actual block map.
102   */
103  private final TreeMap<DatanodeDescriptor, AbstractList<BlockInfoContiguous>>
104      decomNodeBlocks;
105
106  /**
107   * Tracking a node in decomNodeBlocks consumes additional memory. To limit
108   * the impact on NN memory consumption, we limit the number of nodes in 
109   * decomNodeBlocks. Additional nodes wait in pendingNodes.
110   */
111  private final Queue<DatanodeDescriptor> pendingNodes;
112
113  private Monitor monitor = null;
114
115  DecommissionManager(final Namesystem namesystem,
116      final BlockManager blockManager, final HeartbeatManager hbManager) {
117    this.namesystem = namesystem;
118    this.blockManager = blockManager;
119    this.hbManager = hbManager;
120
121    executor = Executors.newScheduledThreadPool(1,
122        new ThreadFactoryBuilder().setNameFormat("DecommissionMonitor-%d")
123            .setDaemon(true).build());
124    decomNodeBlocks = new TreeMap<>();
125    pendingNodes = new LinkedList<>();
126  }

128  /**
129   * Start the decommission monitor thread.
130   * @param conf
131   */
132  void activate(Configuration conf) {
133    final int intervalSecs =
134        conf.getInt(DFSConfigKeys.DFS_NAMENODE_DECOMMISSION_INTERVAL_KEY,
135            DFSConfigKeys.DFS_NAMENODE_DECOMMISSION_INTERVAL_DEFAULT);
136    checkArgument(intervalSecs >= 0, "Cannot set a negative " +
137        "value for " + DFSConfigKeys.DFS_NAMENODE_DECOMMISSION_INTERVAL_KEY);
138
139    // By default, the new configuration key overrides the deprecated one.
140    // No # node limit is set.
141    int blocksPerInterval = conf.getInt(
142        DFSConfigKeys.DFS_NAMENODE_DECOMMISSION_BLOCKS_PER_INTERVAL_KEY,
143        DFSConfigKeys.DFS_NAMENODE_DECOMMISSION_BLOCKS_PER_INTERVAL_DEFAULT);
144    int nodesPerInterval = Integer.MAX_VALUE;
145
146    // If the expected key isn't present and the deprecated one is, 
147    // use the deprecated one into the new one. This overrides the 
148    // default.
149    //
150    // Also print a deprecation warning.
151    final String deprecatedKey =
152        "dfs.namenode.decommission.nodes.per.interval";
153    final String strNodes = conf.get(deprecatedKey);
154    if (strNodes != null) {
155      nodesPerInterval = Integer.parseInt(strNodes);
156      blocksPerInterval = Integer.MAX_VALUE;
157      LOG.warn("Using deprecated configuration key {} value of {}.",
158          deprecatedKey, nodesPerInterval); 
159      LOG.warn("Please update your configuration to use {} instead.", 
160          DFSConfigKeys.DFS_NAMENODE_DECOMMISSION_BLOCKS_PER_INTERVAL_KEY);
161    }
162    checkArgument(blocksPerInterval > 0,
163        "Must set a positive value for "
164        + DFSConfigKeys.DFS_NAMENODE_DECOMMISSION_BLOCKS_PER_INTERVAL_KEY);
165
166    final int maxConcurrentTrackedNodes = conf.getInt(
167        DFSConfigKeys.DFS_NAMENODE_DECOMMISSION_MAX_CONCURRENT_TRACKED_NODES,
168        DFSConfigKeys.DFS_NAMENODE_DECOMMISSION_MAX_CONCURRENT_TRACKED_NODES_DEFAULT);
169    checkArgument(maxConcurrentTrackedNodes >= 0, "Cannot set a negative " +
170        "value for "
171        + DFSConfigKeys.DFS_NAMENODE_DECOMMISSION_MAX_CONCURRENT_TRACKED_NODES);
172
173    monitor = new Monitor(blocksPerInterval, 
174        nodesPerInterval, maxConcurrentTrackedNodes);
175    executor.scheduleAtFixedRate(monitor, intervalSecs, intervalSecs,
176        TimeUnit.SECONDS);
177
178    LOG.debug("Activating DecommissionManager with interval {} seconds, " +
179            "{} max blocks per interval, {} max nodes per interval, " +
180            "{} max concurrently tracked nodes.", intervalSecs,
181        blocksPerInterval, nodesPerInterval, maxConcurrentTrackedNodes);
182  }
183
184  /**
185   * Stop the decommission monitor thread, waiting briefly for it to terminate.
186   */
187  void close() {
188    executor.shutdownNow();
189    try {
190      executor.awaitTermination(3000, TimeUnit.MILLISECONDS);
191    } catch (InterruptedException e) {}
192  }

194  /**
195   * Start decommissioning the specified datanode. 
196   * @param node
197   */
198  @VisibleForTesting
199  public void startDecommission(DatanodeDescriptor node) {
200    if (!node.isDecommissionInProgress()) {
201      if (!node.isAlive) {
202        LOG.info("Dead node {} is decommissioned immediately.", node);
203        node.setDecommissioned();
204      } else if (!node.isDecommissioned()) {
205        for (DatanodeStorageInfo storage : node.getStorageInfos()) {
206          LOG.info("Starting decommission of {} {} with {} blocks", 
207              node, storage, storage.numBlocks());
208        }
209        // Update DN stats maintained by HeartbeatManager
210        hbManager.startDecommission(node);
211        node.decommissioningStatus.setStartTime(monotonicNow());
212        pendingNodes.add(node);
213      }
214    } else {
215      LOG.trace("startDecommission: Node {} is already decommission in "
216              + "progress, nothing to do.", node);
217    }
218  }
219
220  /**
221   * Stop decommissioning the specified datanode. 
222   * @param node
223   */
224  void stopDecommission(DatanodeDescriptor node) {
225    if (node.isDecommissionInProgress() || node.isDecommissioned()) {
226      LOG.info("Stopping decommissioning of node {}", node);
227      // Update DN stats maintained by HeartbeatManager
228      hbManager.stopDecommission(node);
229      // Over-replicated blocks will be detected and processed when 
230      // the dead node comes back and send in its full block report.
231      if (node.isAlive) {
232        blockManager.processOverReplicatedBlocksOnReCommission(node);
233      }
234      // Remove from tracking in DecommissionManager
235      pendingNodes.remove(node);
236      decomNodeBlocks.remove(node);
237    } else {
238      LOG.trace("stopDecommission: Node {} is not decommission in progress " +
239          "or decommissioned, nothing to do.", node);
240    }
241  }
242
243  private void setDecommissioned(DatanodeDescriptor dn) {
244    dn.setDecommissioned();
245    LOG.info("Decommissioning complete for node {}", dn);
246  }

248  /**
249   * Checks whether a block is sufficiently replicated for decommissioning.
250   * Full-strength replication is not always necessary, hence "sufficient".
251   * @return true if sufficient, else false.
252   */
253  private boolean isSufficientlyReplicated(BlockInfoContiguous block, 
254      BlockCollection bc,
255      NumberReplicas numberReplicas) {
256    final int numExpected = bc.getBlockReplication();
257    final int numLive = numberReplicas.liveReplicas();
258    if (!blockManager.isNeededReplication(block, numExpected, numLive)) {
259      // Block doesn't need replication. Skip.
260      LOG.trace("Block {} does not need replication.", block);
261      return true;
262    }
263
264    // Block is under-replicated
265    LOG.trace("Block {} numExpected={}, numLive={}", block, numExpected, 
266        numLive);
267    if (numExpected > numLive) {
268      if (bc.isUnderConstruction() && block.equals(bc.getLastBlock())) {
269        // Can decom a UC block as long as there will still be minReplicas
270        if (numLive >= blockManager.minReplication) {
271          LOG.trace("UC block {} sufficiently-replicated since numLive ({}) "
272              + ">= minR ({})", block, numLive, blockManager.minReplication);
273          return true;
274        } else {
275          LOG.trace("UC block {} insufficiently-replicated since numLive "
276              + "({}) < minR ({})", block, numLive,
277              blockManager.minReplication);
278        }
279      } else {
280        // Can decom a non-UC as long as the default replication is met
281        if (numLive >= blockManager.defaultReplication) {
282          return true;
283        }
284      }
285    }
286    return false;
287  }

289  private static void logBlockReplicationInfo(Block block, BlockCollection bc,
290      DatanodeDescriptor srcNode, NumberReplicas num,
291      Iterable<DatanodeStorageInfo> storages) {
292    int curReplicas = num.liveReplicas();
293    int curExpectedReplicas = bc.getBlockReplication();
294    StringBuilder nodeList = new StringBuilder();
295    for (DatanodeStorageInfo storage : storages) {
296      final DatanodeDescriptor node = storage.getDatanodeDescriptor();
297      nodeList.append(node);
298      nodeList.append(" ");
299    }
300    LOG.info("Block: " + block + ", Expected Replicas: "
301        + curExpectedReplicas + ", live replicas: " + curReplicas
302        + ", corrupt replicas: " + num.corruptReplicas()
303        + ", decommissioned replicas: " + num.decommissionedReplicas()
304        + ", excess replicas: " + num.excessReplicas()
305        + ", Is Open File: " + bc.isUnderConstruction()
306        + ", Datanodes having this block: " + nodeList + ", Current Datanode: "
307        + srcNode + ", Is current datanode decommissioning: "
308        + srcNode.isDecommissionInProgress());
309  }

326  /**
327   * Checks to see if DNs have finished decommissioning.
328   * <p/>
329   * Since this is done while holding the namesystem lock, 
330   * the amount of work per monitor tick is limited.
331   */
332  private class Monitor implements Runnable {
333    /**
334     * The maximum number of blocks to check per tick.
335     */
336    private final int numBlocksPerCheck;
337    /**
338     * The maximum number of nodes to check per tick.
339     */
340    private final int numNodesPerCheck;
341    /**
342     * The maximum number of nodes to track in decomNodeBlocks. A value of 0
343     * means no limit.
344     */
345    private final int maxConcurrentTrackedNodes;
346    /**
347     * The number of blocks that have been checked on this tick.
348     */
349    private int numBlocksChecked = 0;
350    /**
351     * The number of nodes that have been checked on this tick. Used for 
352     * testing.
353     */
354    private int numNodesChecked = 0;
355    /**
356     * The last datanode in decomNodeBlocks that we've processed
357     */
358    private DatanodeDescriptor iterkey = new DatanodeDescriptor(new 
359        DatanodeID("", "", "", 0, 0, 0, 0));
360
361    Monitor(int numBlocksPerCheck, int numNodesPerCheck, int 
362        maxConcurrentTrackedNodes) {
363      this.numBlocksPerCheck = numBlocksPerCheck;
364      this.numNodesPerCheck = numNodesPerCheck;
365      this.maxConcurrentTrackedNodes = maxConcurrentTrackedNodes;
366    }
367
368    private boolean exceededNumBlocksPerCheck() {
369      LOG.trace("Processed {} blocks so far this tick", numBlocksChecked);
370      return numBlocksChecked >= numBlocksPerCheck;
371    }
372
373    @Deprecated
374    private boolean exceededNumNodesPerCheck() {
375      LOG.trace("Processed {} nodes so far this tick", numNodesChecked);
376      return numNodesChecked >= numNodesPerCheck;
377    }
378
379    @Override
380    public void run() {
381      if (!namesystem.isRunning()) {
382        LOG.info("Namesystem is not running, skipping decommissioning checks"
383            + ".");
384        return;
385      }
386      // Reset the checked count at beginning of each iteration
387      numBlocksChecked = 0;
388      numNodesChecked = 0;
389      // Check decom progress
390      namesystem.writeLock();
391      try {
392        processPendingNodes();
393        check();
394      } finally {
395        namesystem.writeUnlock();
396      }
397      if (numBlocksChecked + numNodesChecked > 0) {
398        LOG.info("Checked {} blocks and {} nodes this tick", numBlocksChecked,
399            numNodesChecked);
400      }
401    }
402
403    /**
404     * Pop datanodes off the pending list and into decomNodeBlocks, 
405     * subject to the maxConcurrentTrackedNodes limit.
406     */
407    private void processPendingNodes() {
408      while (!pendingNodes.isEmpty() &&
409          (maxConcurrentTrackedNodes == 0 ||
410           decomNodeBlocks.size() < maxConcurrentTrackedNodes)) {
411        decomNodeBlocks.put(pendingNodes.poll(), null);
412      }
413    }
414
415    private void check() {
416      final Iterator<Map.Entry<DatanodeDescriptor, AbstractList<BlockInfoContiguous>>>
417          it = new CyclicIteration<>(decomNodeBlocks, iterkey).iterator();
418      final LinkedList<DatanodeDescriptor> toRemove = new LinkedList<>();
419
420      while (it.hasNext()
421          && !exceededNumBlocksPerCheck()
422          && !exceededNumNodesPerCheck()) {
423        numNodesChecked++;
424        final Map.Entry<DatanodeDescriptor, AbstractList<BlockInfoContiguous>>
425            entry = it.next();
426        final DatanodeDescriptor dn = entry.getKey();
427        AbstractList<BlockInfoContiguous> blocks = entry.getValue();
428        boolean fullScan = false;
429        if (blocks == null) {
430          // This is a newly added datanode, run through its list to schedule 
431          // under-replicated blocks for replication and collect the blocks 
432          // that are insufficiently replicated for further tracking
433          LOG.debug("Newly-added node {}, doing full scan to find " +
434              "insufficiently-replicated blocks.", dn);
435          blocks = handleInsufficientlyReplicated(dn);
436          decomNodeBlocks.put(dn, blocks);
437          fullScan = true;
438        } else {
439          // This is a known datanode, check if its # of insufficiently 
440          // replicated blocks has dropped to zero and if it can be decommed
441          LOG.debug("Processing decommission-in-progress node {}", dn);
442          pruneSufficientlyReplicated(dn, blocks);
443        }
444        if (blocks.size() == 0) {
445          if (!fullScan) {
446            // If we didn't just do a full scan, need to re-check with the 
447            // full block map.
448            //
449            // We've replicated all the known insufficiently replicated 
450            // blocks. Re-check with the full block map before finally 
451            // marking the datanode as decommissioned 
452            LOG.debug("Node {} has finished replicating current set of "
453                + "blocks, checking with the full block map.", dn);
454            blocks = handleInsufficientlyReplicated(dn);
455            decomNodeBlocks.put(dn, blocks);
456          }
457          // If the full scan is clean AND the node liveness is okay, 
458          // we can finally mark as decommissioned.
459          final boolean isHealthy =
460              blockManager.isNodeHealthyForDecommission(dn);
461          if (blocks.size() == 0 && isHealthy) {
462            setDecommissioned(dn);
463            toRemove.add(dn);
464            LOG.debug("Node {} is sufficiently replicated and healthy, "
465                + "marked as decommissioned.", dn);
466          } else {
467            if (LOG.isDebugEnabled()) {
468              StringBuilder b = new StringBuilder("Node {} ");
469              if (isHealthy) {
470                b.append("is ");
471              } else {
472                b.append("isn't ");
473              }
474              b.append("healthy and still needs to replicate {} more blocks," +
475                  " decommissioning is still in progress.");
476              LOG.debug(b.toString(), dn, blocks.size());
477            }
478          }
479        } else {
480          LOG.debug("Node {} still has {} blocks to replicate "
481                  + "before it is a candidate to finish decommissioning.",
482              dn, blocks.size());
483        }
484        iterkey = dn;
485      }
486      // Remove the datanodes that are decommissioned
487      for (DatanodeDescriptor dn : toRemove) {
488        Preconditions.checkState(dn.isDecommissioned(),
489            "Removing a node that is not yet decommissioned!");
490        decomNodeBlocks.remove(dn);
491      }
492    }
493
494    /**
495     * Removes sufficiently replicated blocks from the block list of a 
496     * datanode.
497     */
498    private void pruneSufficientlyReplicated(final DatanodeDescriptor datanode,
499        AbstractList<BlockInfoContiguous> blocks) {
500      processBlocksForDecomInternal(datanode, blocks.iterator(), null, true);
501    }
502
503    /**
504     * Returns a list of blocks on a datanode that are insufficiently 
505     * replicated, i.e. are under-replicated enough to prevent decommission.
506     * <p/>
507     * As part of this, it also schedules replication work for 
508     * any under-replicated blocks.
509     *
510     * @param datanode
511     * @return List of insufficiently replicated blocks 
512     */
513    private AbstractList<BlockInfoContiguous> handleInsufficientlyReplicated(
514        final DatanodeDescriptor datanode) {
515      AbstractList<BlockInfoContiguous> insufficient = new ChunkedArrayList<>();
516      processBlocksForDecomInternal(datanode, datanode.getBlockIterator(),
517          insufficient, false);
518      return insufficient;
519    }
520
521    /**
522     * Used while checking if decommission-in-progress datanodes can be marked
523     * as decommissioned. Combines shared logic of 
524     * pruneSufficientlyReplicated and handleInsufficientlyReplicated.
525     *
526     * @param datanode                    Datanode
527     * @param it                          Iterator over the blocks on the
528     *                                    datanode
529     * @param insufficientlyReplicated    Return parameter. If it's not null,
530     *                                    will contain the insufficiently
531     *                                    replicated-blocks from the list.
532     * @param pruneSufficientlyReplicated whether to remove sufficiently
533     *                                    replicated blocks from the iterator
534     * @return true if there are under-replicated blocks in the provided block
535     * iterator, else false.
536     */
537    private void processBlocksForDecomInternal(
538        final DatanodeDescriptor datanode,
539        final Iterator<BlockInfoContiguous> it,
540        final List<BlockInfoContiguous> insufficientlyReplicated,
541        boolean pruneSufficientlyReplicated) {
542      boolean firstReplicationLog = true;
543      int underReplicatedBlocks = 0;
544      int decommissionOnlyReplicas = 0;
545      int underReplicatedInOpenFiles = 0;
546      while (it.hasNext()) {
547        numBlocksChecked++;
548        final BlockInfoContiguous block = it.next();
549        // Remove the block from the list if it's no longer in the block map,
550        // e.g. the containing file has been deleted
551        if (blockManager.blocksMap.getStoredBlock(block) == null) {
552          LOG.trace("Removing unknown block {}", block);
553          it.remove();
554          continue;
555        }
556        BlockCollection bc = blockManager.blocksMap.getBlockCollection(block);
557        if (bc == null) {
558          // Orphan block, will be invalidated eventually. Skip.
559          continue;
560        }
561
562        final NumberReplicas num = blockManager.countNodes(block);
563        final int liveReplicas = num.liveReplicas();
564        final int curReplicas = liveReplicas;
565
566        // Schedule under-replicated blocks for replication if not already
567        // pending
568        if (blockManager.isNeededReplication(block, bc.getBlockReplication(),
569            liveReplicas)) {
570          if (!blockManager.neededReplications.contains(block) &&
571              blockManager.pendingReplications.getNumReplicas(block) == 0 &&
572              namesystem.isPopulatingReplQueues()) {
573            // Process these blocks only when active NN is out of safe mode.
574            blockManager.neededReplications.add(block,
575                curReplicas,
576                num.decommissionedReplicas(),
577                bc.getBlockReplication());
578          }
579        }
580
581        // Even if the block is under-replicated, 
582        // it doesn't block decommission if it's sufficiently replicated 
583        if (isSufficientlyReplicated(block, bc, num)) {
584          if (pruneSufficientlyReplicated) {
585            it.remove();
586          }
587          continue;
588        }
589
590        // We've found an insufficiently replicated block.
591        if (insufficientlyReplicated != null) {
592          insufficientlyReplicated.add(block);
593        }
594        // Log if this is our first time through
595        if (firstReplicationLog) {
596          logBlockReplicationInfo(block, bc, datanode, num,
597              blockManager.blocksMap.getStorages(block));
598          firstReplicationLog = false;
599        }
600        // Update various counts
601        underReplicatedBlocks++;
602        if (bc.isUnderConstruction()) {
603          underReplicatedInOpenFiles++;
604        }
605        if ((curReplicas == 0) && (num.decommissionedReplicas() > 0)) {
606          decommissionOnlyReplicas++;
607        }
608      }
609
610      datanode.decommissioningStatus.set(underReplicatedBlocks,
611          decommissionOnlyReplicas,
612          underReplicatedInOpenFiles);
613    }
614  }
615
616  @VisibleForTesting
617  void runMonitor() throws ExecutionException, InterruptedException {
618    Future f = executor.submit(monitor);
619    f.get();
620  }
