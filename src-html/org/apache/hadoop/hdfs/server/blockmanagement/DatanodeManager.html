272  private static long getStaleIntervalFromConf(Configuration conf,
273      long heartbeatExpireInterval) {
274    long staleInterval = conf.getLong(
275        DFSConfigKeys.DFS_NAMENODE_STALE_DATANODE_INTERVAL_KEY, 
276        DFSConfigKeys.DFS_NAMENODE_STALE_DATANODE_INTERVAL_DEFAULT);
277    Preconditions.checkArgument(staleInterval > 0,
278        DFSConfigKeys.DFS_NAMENODE_STALE_DATANODE_INTERVAL_KEY +
279        " = '" + staleInterval + "' is invalid. " +
280        "It should be a positive non-zero value.");
281    
282    final long heartbeatIntervalSeconds = conf.getLong(
283        DFSConfigKeys.DFS_HEARTBEAT_INTERVAL_KEY,
284        DFSConfigKeys.DFS_HEARTBEAT_INTERVAL_DEFAULT);
285    // The stale interval value cannot be smaller than 
286    // 3 times of heartbeat interval 
287    final long minStaleInterval = conf.getInt(
288        DFSConfigKeys.DFS_NAMENODE_STALE_DATANODE_MINIMUM_INTERVAL_KEY,
289        DFSConfigKeys.DFS_NAMENODE_STALE_DATANODE_MINIMUM_INTERVAL_DEFAULT)
290        * heartbeatIntervalSeconds * 1000;
291    if (staleInterval < minStaleInterval) {
292      LOG.warn("The given interval for marking stale datanode = "
293          + staleInterval + ", which is less than "
294          + DFSConfigKeys.DFS_NAMENODE_STALE_DATANODE_MINIMUM_INTERVAL_DEFAULT
295          + " heartbeat intervals. This may cause too frequent changes of " 
296          + "stale states of DataNodes since a heartbeat msg may be missing " 
297          + "due to temporary short-term failures. Reset stale interval to " 
298          + minStaleInterval + ".");
299      staleInterval = minStaleInterval;
300    }
301    if (staleInterval > heartbeatExpireInterval) {
302      LOG.warn("The given interval for marking stale datanode = "
303          + staleInterval + ", which is larger than heartbeat expire interval "
304          + heartbeatExpireInterval + ".");
305    }
306    return staleInterval;
307  }

353  private boolean isInactive(DatanodeInfo datanode) {
354    if (datanode.isDecommissioned()) {
355      return true;
356    }
357
358    if (avoidStaleDataNodesForRead) {
359      return datanode.isStale(staleInterval);
360    }
361      
362    return false;
363  }

365  /** Sort the located blocks by the distance to the target host. */
366  public void sortLocatedBlocks(final String targethost,
367      final List<LocatedBlock> locatedblocks) {
368    //sort the blocks
369    // As it is possible for the separation of node manager and datanode, 
370    // here we should get node but not datanode only .
371    Node client = getDatanodeByHost(targethost);
372    if (client == null) {
373      List<String> hosts = new ArrayList<String> (1);
374      hosts.add(targethost);
375      String rName = dnsToSwitchMapping.resolve(hosts).get(0);
376      if (rName != null)
377        client = new NodeBase(rName + NodeBase.PATH_SEPARATOR_STR + targethost);
378    }
379    
380    Comparator<DatanodeInfo> comparator = avoidStaleDataNodesForRead ?
381        new DFSUtil.DecomStaleComparator(staleInterval) : 
382        DFSUtil.DECOM_COMPARATOR;
383        
384    for (LocatedBlock b : locatedblocks) {
385      DatanodeInfo[] di = b.getLocations();
386      // Move decommissioned/stale datanodes to the bottom
387      Arrays.sort(di, comparator);
388      
389      int lastActiveIndex = di.length - 1;
390      while (lastActiveIndex > 0 && isInactive(di[lastActiveIndex])) {
391          --lastActiveIndex;
392      }
393      int activeLen = lastActiveIndex + 1;      
394      networktopology.sortByDistance(client, b.getLocations(), activeLen);
395      // must update cache since we modified locations array
396      b.updateCachedStorageInfo();
397    }
398  }

421  /**
422   * Given datanode address or host name, returns the DatanodeDescriptor for the
423   * same, or if it doesn't find the datanode, it looks for a machine local and
424   * then rack local datanode, if a rack local datanode is not possible either,
425   * it returns the DatanodeDescriptor of any random node in the cluster.
426   *
427   * @param address hostaddress:transfer address
428   * @return the best match for the given datanode
429   */
430  DatanodeDescriptor getDatanodeDescriptor(String address) {
431    DatanodeID dnId = parseDNFromHostsEntry(address);
432    String host = dnId.getIpAddr();
433    int xferPort = dnId.getXferPort();
434    DatanodeDescriptor node = getDatanodeByXferAddr(host, xferPort);
435    if (node == null) {
436      node = getDatanodeByHost(host);
437    }
438    if (node == null) {
439      String networkLocation = 
440          resolveNetworkLocationWithFallBackToDefaultLocation(dnId);
441
442      // If the current cluster doesn't contain the node, fallback to
443      // something machine local and then rack local.
444      List<Node> rackNodes = getNetworkTopology()
445                                   .getDatanodesInRack(networkLocation);
446      if (rackNodes != null) {
447        // Try something machine local.
448        for (Node rackNode : rackNodes) {
449          if (((DatanodeDescriptor) rackNode).getIpAddr().equals(host)) {
450            node = (DatanodeDescriptor) rackNode;
451            break;
452          }
453        }
454
455        // Try something rack local.
456        if (node == null && !rackNodes.isEmpty()) {
457          node = (DatanodeDescriptor) (rackNodes
458              .get(DFSUtil.getRandom().nextInt(rackNodes.size())));
459        }
460      }
461
462      // If we can't even choose rack local, just choose any node in the
463      // cluster.
464      if (node == null) {
465        node = (DatanodeDescriptor)getNetworkTopology()
466                                   .chooseRandom(NodeBase.ROOT);
467      }
468    }
469    return node;
470  }

534    assert namesystem.hasWriteLock();

550  public void removeDatanode(final DatanodeID node
551      ) throws UnregisteredNodeException {
552    namesystem.writeLock();
553    try {
554      final DatanodeDescriptor descriptor = getDatanode(node);
555      if (descriptor != null) {

557      } else {

560      }
561    } finally {
562      namesystem.writeUnlock();
563    }

566  /** Remove a dead datanode. */
567  void removeDeadDatanode(final DatanodeID nodeID) {
568      synchronized(datanodeMap) {
569        DatanodeDescriptor d;
570        try {
571          d = getDatanode(nodeID);
572        } catch(IOException e) {
573          d = null;
574        }
575        if (d != null && isDatanodeDead(d)) {
576          NameNode.stateChangeLog.info(
577              "BLOCK* removeDeadDatanode: lost heartbeat from " + d);
578          removeDatanode(d);
579        }
580      }
581  }

589  /** Add a datanode. */
590  void addDatanode(final DatanodeDescriptor node) {
591    // To keep host2DatanodeMap consistent with datanodeMap,
592    // remove  from host2DatanodeMap the datanodeDescriptor removed
593    // from datanodeMap before adding node to host2DatanodeMap.
594    synchronized(datanodeMap) {
595      host2DatanodeMap.remove(datanodeMap.put(node.getDatanodeUuid(), node));
596    }
597
598    networktopology.add(node); // may throw InvalidTopologyException
599    host2DatanodeMap.add(node);
600    checkIfClusterIsNowMultiRack(node);
601
602    if (LOG.isDebugEnabled()) {
603      LOG.debug(getClass().getSimpleName() + ".addDatanode: "
604          + "node " + node + " is added to datanodeMap.");
605    }
606  }

636      return;
637    }

646      }
647    }

784  /**
785   * Remove an already decommissioned data node who is neither in include nor
786   * exclude hosts lists from the the list of live or dead nodes.  This is used
787   * to not display an already decommssioned data node to the operators.
788   * The operation procedure of making a already decommissioned data node not
789   * to be displayed is as following:
790   * <ol>
791   *   <li> 
792   *   Host must have been in the include hosts list and the include hosts list
793   *   must not be empty.
794   *   </li>
795   *   <li>
796   *   Host is decommissioned by remaining in the include hosts list and added
797   *   into the exclude hosts list. Name node is updated with the new 
798   *   information by issuing dfsadmin -refreshNodes command.
799   *   </li>
800   *   <li>
801   *   Host is removed from both include hosts and exclude hosts lists.  Name 
802   *   node is updated with the new informationby issuing dfsamin -refreshNodes 
803   *   command.
804   *   <li>
805   * </ol>
806   * 
807   * @param nodeList
808   *          , array list of live or dead nodes.
809   */
810  private void removeDecomNodeFromList(final List<DatanodeDescriptor> nodeList) {
811    // If the include list is empty, any nodes are welcomed and it does not
812    // make sense to exclude any nodes from the cluster. Therefore, no remove.
813    if (!hostFileManager.hasIncludes()) {
814      return;
815    }
816
817    for (Iterator<DatanodeDescriptor> it = nodeList.iterator(); it.hasNext();) {
818      DatanodeDescriptor node = it.next();
819      if ((!hostFileManager.isIncluded(node)) && (!hostFileManager.isExcluded(node))
820          && node.isDecommissioned()) {
821        // Include list is not empty, an existing datanode does not appear
822        // in both include or exclude lists and it has been decommissioned.
823        it.remove();
824      }
825    }
826  }

840  /**
841   * Register the given datanode with the namenode. NB: the given
842   * registration is mutated and given back to the datanode.
843   *
844   * @param nodeReg the datanode registration
845   * @throws DisallowedDatanodeException if the registration request is
846   *    denied because the datanode does not match includes/excludes
847   * @throws UnresolvedTopologyException if the registration request is 
848   *    denied because resolving datanode network location fails.
849   */
850  public void registerDatanode(DatanodeRegistration nodeReg)
851      throws DisallowedDatanodeException, UnresolvedTopologyException {
852    InetAddress dnAddress = Server.getRemoteIp();
853    if (dnAddress != null) {
854      // Mostly called inside an RPC, update ip and peer hostname
855      String hostname = dnAddress.getHostName();
856      String ip = dnAddress.getHostAddress();
857      if (checkIpHostnameInRegistration && !isNameResolved(dnAddress)) {
858        // Reject registration of unresolved datanode to prevent performance
859        // impact of repetitive DNS lookups later.
860        final String message = "hostname cannot be resolved (ip="
861            + ip + ", hostname=" + hostname + ")";
862        LOG.warn("Unresolved datanode registration: " + message);
863        throw new DisallowedDatanodeException(nodeReg, message);
864      }
865      // update node registration with the ip and hostname from rpc request
866      nodeReg.setIpAddr(ip);
867      nodeReg.setPeerHostName(hostname);
868    }
869    
870    try {
871      nodeReg.setExportedKeys(blockManager.getBlockKeys());
872  
873      // Checks if the node is not on the hosts list.  If it is not, then
874      // it will be disallowed from registering. 
875      if (!hostFileManager.isIncluded(nodeReg)) {
876        throw new DisallowedDatanodeException(nodeReg);
877      }
878        
879      NameNode.stateChangeLog.info("BLOCK* registerDatanode: from "
880          + nodeReg + " storage " + nodeReg.getDatanodeUuid());
881  
882      DatanodeDescriptor nodeS = getDatanode(nodeReg.getDatanodeUuid());
883      DatanodeDescriptor nodeN = host2DatanodeMap.getDatanodeByXferAddr(
884          nodeReg.getIpAddr(), nodeReg.getXferPort());
885        
886      if (nodeN != null && nodeN != nodeS) {
887        NameNode.LOG.info("BLOCK* registerDatanode: " + nodeN);
888        // nodeN previously served a different data storage, 
889        // which is not served by anybody anymore.
890        removeDatanode(nodeN);
891        // physically remove node from datanodeMap
892        wipeDatanode(nodeN);
893        nodeN = null;
894      }
895  
896      if (nodeS != null) {
897        if (nodeN == nodeS) {
898          // The same datanode has been just restarted to serve the same data 
899          // storage. We do not need to remove old data blocks, the delta will
900          // be calculated on the next block report from the datanode
901          if(NameNode.stateChangeLog.isDebugEnabled()) {
902            NameNode.stateChangeLog.debug("BLOCK* registerDatanode: "
903                + "node restarted.");
904          }
905        } else {
906          // nodeS is found
907          /* The registering datanode is a replacement node for the existing 
908            data storage, which from now on will be served by a new node.
909            If this message repeats, both nodes might have same storageID 
910            by (insanely rare) random chance. User needs to restart one of the
911            nodes with its data cleared (or user can just remove the StorageID
912            value in "VERSION" file under the data directory of the datanode,
913            but this is might not work if VERSION file format has changed 
914         */        
915          NameNode.stateChangeLog.info("BLOCK* registerDatanode: " + nodeS
916              + " is replaced by " + nodeReg + " with the same storageID "
917              + nodeReg.getDatanodeUuid());
918        }
919        
920        boolean success = false;
921        try {
922          // update cluster map
923          getNetworkTopology().remove(nodeS);
924          if(shouldCountVersion(nodeS)) {
925            decrementVersionCount(nodeS.getSoftwareVersion());
926          }
927          nodeS.updateRegInfo(nodeReg);
928
929          nodeS.setSoftwareVersion(nodeReg.getSoftwareVersion());
930          nodeS.setDisallowed(false); // Node is in the include list
931
932          // resolve network location
933          if(this.rejectUnresolvedTopologyDN) {
934            nodeS.setNetworkLocation(resolveNetworkLocation(nodeS));
935            nodeS.setDependentHostNames(getNetworkDependencies(nodeS));
936          } else {
937            nodeS.setNetworkLocation(
938                resolveNetworkLocationWithFallBackToDefaultLocation(nodeS));
939            nodeS.setDependentHostNames(
940                getNetworkDependenciesWithDefault(nodeS));
941          }
942          getNetworkTopology().add(nodeS);
943            
944          // also treat the registration message as a heartbeat
945          heartbeatManager.register(nodeS);
946          incrementVersionCount(nodeS.getSoftwareVersion());
947          startDecommissioningIfExcluded(nodeS);
948          success = true;
949        } finally {
950          if (!success) {
951            removeDatanode(nodeS);
952            wipeDatanode(nodeS);
953            countSoftwareVersions();
954          }
955        }
956        return;
957      }
958
959      DatanodeDescriptor nodeDescr 
960        = new DatanodeDescriptor(nodeReg, NetworkTopology.DEFAULT_RACK);
961      boolean success = false;
962      try {
963        // resolve network location
964        if(this.rejectUnresolvedTopologyDN) {
965          nodeDescr.setNetworkLocation(resolveNetworkLocation(nodeDescr));
966          nodeDescr.setDependentHostNames(getNetworkDependencies(nodeDescr));
967        } else {
968          nodeDescr.setNetworkLocation(
969              resolveNetworkLocationWithFallBackToDefaultLocation(nodeDescr));
970          nodeDescr.setDependentHostNames(
971              getNetworkDependenciesWithDefault(nodeDescr));
972        }
973        networktopology.add(nodeDescr);
974        nodeDescr.setSoftwareVersion(nodeReg.getSoftwareVersion());
975  
976        // register new datanode
977        addDatanode(nodeDescr);
978        // also treat the registration message as a heartbeat
979        // no need to update its timestamp
980        // because its is done when the descriptor is created
981        heartbeatManager.addDatanode(nodeDescr);
982        incrementVersionCount(nodeReg.getSoftwareVersion());
983        startDecommissioningIfExcluded(nodeDescr);
984        success = true;
985      } finally {
986        if (!success) {
987          removeDatanode(nodeDescr);
988          wipeDatanode(nodeDescr);
989          countSoftwareVersions();
990        }
991      }
992    } catch (InvalidTopologyException e) {
993      // If the network location is invalid, clear the cached mappings
994      // so that we have a chance to re-add this DataNode with the
995      // correct network location later.
996      List<String> invalidNodeNames = new ArrayList<String>(3);
997      // clear cache for nodes in IP or Hostname
998      invalidNodeNames.add(nodeReg.getIpAddr());
999      invalidNodeNames.add(nodeReg.getHostName());
1000      invalidNodeNames.add(nodeReg.getPeerHostName());
1001      dnsToSwitchMapping.reloadCachedMappings(invalidNodeNames);
1002      throw e;
1003    }
1004  }

1215   * @return DatanodeID constructed from the given string
1216   */
1217  private DatanodeID parseDNFromHostsEntry(String hostLine) {
1218    DatanodeID dnId;
1219    String hostStr;
1220    int port;
1221    int idx = hostLine.indexOf(':');
1222
1223    if (-1 == idx) {
1224      hostStr = hostLine;
1225      port = DFSConfigKeys.DFS_DATANODE_DEFAULT_PORT;
1226    } else {
1227      hostStr = hostLine.substring(0, idx);
1228      port = Integer.parseInt(hostLine.substring(idx+1));
1229    }
1230
1231    if (InetAddresses.isInetAddress(hostStr)) {
1232      // The IP:port is sufficient for listing in a report
1233      dnId = new DatanodeID(hostStr, "", "", port,

1236          DFSConfigKeys.DFS_DATANODE_IPC_DEFAULT_PORT);
1237    } else {
1238      String ipAddr = "";
1239      try {
1240        ipAddr = InetAddress.getByName(hostStr).getHostAddress();
1241      } catch (UnknownHostException e) {
1242        LOG.warn("Invalid hostname " + hostStr + " in hosts file");
1243      }
1244      dnId = new DatanodeID(ipAddr, hostStr, "", port,
1245          DFSConfigKeys.DFS_DATANODE_HTTP_DEFAULT_PORT,
1246          DFSConfigKeys.DFS_DATANODE_HTTPS_DEFAULT_PORT,
1247          DFSConfigKeys.DFS_DATANODE_IPC_DEFAULT_PORT);

1256        type == DatanodeReportType.ALL ||
1257        type == DatanodeReportType.LIVE;
1258    final boolean listDeadNodes =
1259        type == DatanodeReportType.ALL ||
1260        type == DatanodeReportType.DEAD;
1261    final boolean listDecommissioningNodes =
1262        type == DatanodeReportType.ALL ||
1263        type == DatanodeReportType.DECOMMISSIONING;
1264
1265    ArrayList<DatanodeDescriptor> nodes;
1266    final HostFileManager.HostSet foundNodes = new HostFileManager.HostSet();
1267    final HostFileManager.HostSet includedNodes = hostFileManager.getIncludes();
1268    final HostFileManager.HostSet excludedNodes = hostFileManager.getExcludes();
1269
1270    synchronized(datanodeMap) {
1271      nodes = new ArrayList<DatanodeDescriptor>(datanodeMap.size());
1272      for (DatanodeDescriptor dn : datanodeMap.values()) {
1273        final boolean isDead = isDatanodeDead(dn);
1274        final boolean isDecommissioning = dn.isDecommissionInProgress();
1275        if ((listLiveNodes && !isDead) ||
1276            (listDeadNodes && isDead) ||
1277            (listDecommissioningNodes && isDecommissioning)) {
1278            nodes.add(dn);
1279        }
1280        foundNodes.add(HostFileManager.resolvedAddressFromDatanodeID(dn));
1281      }
1282    }
1283
1284    if (listDeadNodes) {
1285      for (InetSocketAddress addr : includedNodes) {
1286        if (foundNodes.matchedBy(addr) || excludedNodes.match(addr)) {
1287          continue;
1288        }
1289        // The remaining nodes are ones that are referenced by the hosts
1290        // files but that we do not know about, ie that we have never
1291        // head from. Eg. an entry that is no longer part of the cluster
1292        // or a bogus entry was given in the hosts files
1293        //
1294        // If the host file entry specified the xferPort, we use that.
1295        // Otherwise, we guess that it is the default xfer port.
1296        // We can't ask the DataNode what it had configured, because it's
1297        // dead.
1298        DatanodeDescriptor dn = new DatanodeDescriptor(new DatanodeID(addr
1299                .getAddress().getHostAddress(), addr.getHostName(), "",
1300                addr.getPort() == 0 ? defaultXferPort : addr.getPort(),
1301                defaultInfoPort, defaultInfoSecurePort, defaultIpcPort));
1302        setDatanodeDead(dn);
1303        nodes.add(dn);
1304      }
1305    }
1306
1307    if (LOG.isDebugEnabled()) {
1308      LOG.debug("getDatanodeListForReport with " +
1309          "includedNodes = " + hostFileManager.getIncludes() +
1310          ", excludedNodes = " + hostFileManager.getExcludes() +
1311          ", foundNodes = " + foundNodes +
1312          ", nodes = " + nodes);
1313    }
1314    return nodes;
1315  }

1338  /** Handle heartbeat from datanodes. */
1339  public DatanodeCommand[] handleHeartbeat(DatanodeRegistration nodeReg,
1340      StorageReport[] reports, final String blockPoolId,
1341      long cacheCapacity, long cacheUsed, int xceiverCount, 
1342      int maxTransfers, int failedVolumes,
1343      VolumeFailureSummary volumeFailureSummary) throws IOException {
1344    synchronized (heartbeatManager) {
1345      synchronized (datanodeMap) {
1346        DatanodeDescriptor nodeinfo = null;
1347        try {
1348          nodeinfo = getDatanode(nodeReg);
1349        } catch(UnregisteredNodeException e) {
1350          return new DatanodeCommand[]{RegisterCommand.REGISTER};
1351        }
1352        
1353        // Check if this datanode should actually be shutdown instead. 
1354        if (nodeinfo != null && nodeinfo.isDisallowed()) {
1355          setDatanodeDead(nodeinfo);
1356          throw new DisallowedDatanodeException(nodeinfo);
1357        }
1358
1359        if (nodeinfo == null || !nodeinfo.isAlive) {
1360          return new DatanodeCommand[]{RegisterCommand.REGISTER};
1361        }
1362
1363        heartbeatManager.updateHeartbeat(nodeinfo, reports,
1364                                         cacheCapacity, cacheUsed,
1365                                         xceiverCount, failedVolumes,
1366                                         volumeFailureSummary);
1367
1368        // If we are in safemode, do not send back any recovery / replication
1369        // requests. Don't even drain the existing queue of work.
1370        if(namesystem.isInSafeMode()) {
1371          return new DatanodeCommand[0];
1372        }
1373
1374        //check lease recovery
1375        BlockInfoContiguousUnderConstruction[] blocks = nodeinfo
1376            .getLeaseRecoveryCommand(Integer.MAX_VALUE);
1377        if (blocks != null) {
1378          BlockRecoveryCommand brCommand = new BlockRecoveryCommand(
1379              blocks.length);
1380          for (BlockInfoContiguousUnderConstruction b : blocks) {
1381            final DatanodeStorageInfo[] storages = b.getExpectedStorageLocations();
1382            // Skip stale nodes during recovery - not heart beated for some time (30s by default).
1383            final List<DatanodeStorageInfo> recoveryLocations =
1384                new ArrayList<DatanodeStorageInfo>(storages.length);
1385            for (int i = 0; i < storages.length; i++) {
1386              if (!storages[i].getDatanodeDescriptor().isStale(staleInterval)) {
1387                recoveryLocations.add(storages[i]);
1388              }
1389            }
1390            // If we are performing a truncate recovery than set recovery fields
1391            // to old block.
1392            boolean truncateRecovery = b.getTruncateBlock() != null;
1393            boolean copyOnTruncateRecovery = truncateRecovery &&
1394                b.getTruncateBlock().getBlockId() != b.getBlockId();
1395            ExtendedBlock primaryBlock = (copyOnTruncateRecovery) ?
1396                new ExtendedBlock(blockPoolId, b.getTruncateBlock()) :
1397                new ExtendedBlock(blockPoolId, b);
1398            // If we only get 1 replica after eliminating stale nodes, then choose all
1399            // replicas for recovery and let the primary data node handle failures.
1400            DatanodeInfo[] recoveryInfos;
1401            if (recoveryLocations.size() > 1) {
1402              if (recoveryLocations.size() != storages.length) {
1403                LOG.info("Skipped stale nodes for recovery : " +
1404                    (storages.length - recoveryLocations.size()));
1405              }
1406              recoveryInfos =
1407                  DatanodeStorageInfo.toDatanodeInfos(recoveryLocations);
1408            } else {
1409              // If too many replicas are stale, then choose all replicas to participate
1410              // in block recovery.
1411              recoveryInfos = DatanodeStorageInfo.toDatanodeInfos(storages);
1412            }
1413            if(truncateRecovery) {
1414              Block recoveryBlock = (copyOnTruncateRecovery) ? b :
1415                  b.getTruncateBlock();
1416              brCommand.add(new RecoveringBlock(primaryBlock, recoveryInfos,
1417                                                recoveryBlock));
1418            } else {
1419              brCommand.add(new RecoveringBlock(primaryBlock, recoveryInfos,
1420                                                b.getBlockRecoveryId()));
1421            }
1422          }
1423          return new DatanodeCommand[] { brCommand };
1424        }
1425
1426        final List<DatanodeCommand> cmds = new ArrayList<DatanodeCommand>();
1427        //check pending replication
1428        List<BlockTargetPair> pendingList = nodeinfo.getReplicationCommand(
1429              maxTransfers);
1430        if (pendingList != null) {
1431          cmds.add(new BlockCommand(DatanodeProtocol.DNA_TRANSFER, blockPoolId,
1432              pendingList));
1433        }
1434        //check block invalidation
1435        Block[] blks = nodeinfo.getInvalidateBlocks(blockInvalidateLimit);
1436        if (blks != null) {
1437          cmds.add(new BlockCommand(DatanodeProtocol.DNA_INVALIDATE,
1438              blockPoolId, blks));
1439        }
1440        boolean sendingCachingCommands = false;
1441        long nowMs = monotonicNow();
1442        if (shouldSendCachingCommands && 
1443            ((nowMs - nodeinfo.getLastCachingDirectiveSentTimeMs()) >=
1444                timeBetweenResendingCachingDirectivesMs)) {
1445          DatanodeCommand pendingCacheCommand =
1446              getCacheCommand(nodeinfo.getPendingCached(), nodeinfo,
1447                DatanodeProtocol.DNA_CACHE, blockPoolId);
1448          if (pendingCacheCommand != null) {
1449            cmds.add(pendingCacheCommand);
1450            sendingCachingCommands = true;
1451          }
1452          DatanodeCommand pendingUncacheCommand =
1453              getCacheCommand(nodeinfo.getPendingUncached(), nodeinfo,
1454                DatanodeProtocol.DNA_UNCACHE, blockPoolId);
1455          if (pendingUncacheCommand != null) {
1456            cmds.add(pendingUncacheCommand);
1457            sendingCachingCommands = true;
1458          }
1459          if (sendingCachingCommands) {
1460            nodeinfo.setLastCachingDirectiveSentTimeMs(nowMs);
1461          }
1462        }
1463
1464        blockManager.addKeyUpdateCommand(cmds, nodeinfo);
1465
1466        // check for balancer bandwidth update
1467        if (nodeinfo.getBalancerBandwidth() > 0) {
1468          cmds.add(new BalancerBandwidthCommand(nodeinfo.getBalancerBandwidth()));
1469          // set back to 0 to indicate that datanode has been sent the new value
1470          nodeinfo.setBalancerBandwidth(0);
1471        }
1472
1473        if (!cmds.isEmpty()) {
1474          return cmds.toArray(new DatanodeCommand[cmds.size()]);
1475        }
1476      }
1477    }
1478
1479    return new DatanodeCommand[0];
1480  }

1482  /**
1483   * Convert a CachedBlockList into a DatanodeCommand with a list of blocks.
1484   *
1485   * @param list       The {@link CachedBlocksList}.  This function 
1486   *                   clears the list.
1487   * @param datanode   The datanode.
1488   * @param action     The action to perform in the command.
1489   * @param poolId     The block pool id.
1490   * @return           A DatanodeCommand to be sent back to the DN, or null if
1491   *                   there is nothing to be done.
1492   */
1493  private DatanodeCommand getCacheCommand(CachedBlocksList list,
1494      DatanodeDescriptor datanode, int action, String poolId) {
1495    int length = list.size();
1496    if (length == 0) {
1497      return null;
1498    }
1499    // Read the existing cache commands.
1500    long[] blockIds = new long[length];
1501    int i = 0;
1502    for (Iterator<CachedBlock> iter = list.iterator();
1503            iter.hasNext(); ) {
1504      CachedBlock cachedBlock = iter.next();
1505      blockIds[i++] = cachedBlock.getBlockId();
1506    }
1507    return new BlockIdCommand(action, poolId, blockIds);
1508  }
