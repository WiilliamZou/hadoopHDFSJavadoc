035/**
036 * Manage the heartbeats received from datanodes.
037 * The datanode list and statistics are synchronized
038 * by the heartbeat manager lock.
039 */
040class HeartbeatManager implements DatanodeStatistics {
041  static final Log LOG = LogFactory.getLog(HeartbeatManager.class);
042
043  /**
044   * Stores a subset of the datanodeMap in DatanodeManager,
045   * containing nodes that are considered alive.
046   * The HeartbeatMonitor periodically checks for out-dated entries,
047   * and removes them from the list.
048   * It is synchronized by the heartbeat manager lock.
049   */
050  private final List<DatanodeDescriptor> datanodes = new ArrayList<DatanodeDescriptor>();
051
052  /** Statistics, which are synchronized by the heartbeat manager lock. */
053  private final Stats stats = new Stats();
054
055  /** The time period to check for expired datanodes */
056  private final long heartbeatRecheckInterval;
057  /** Heartbeat monitor thread */
058  private final Daemon heartbeatThread = new Daemon(new Monitor());
059
060    
061  final Namesystem namesystem;
062  final BlockManager blockManager;
063
064  HeartbeatManager(final Namesystem namesystem,
065      final BlockManager blockManager, final Configuration conf) {
066    this.namesystem = namesystem;
067    this.blockManager = blockManager;
068    boolean avoidStaleDataNodesForWrite = conf.getBoolean(
069        DFSConfigKeys.DFS_NAMENODE_AVOID_STALE_DATANODE_FOR_WRITE_KEY,
070        DFSConfigKeys.DFS_NAMENODE_AVOID_STALE_DATANODE_FOR_WRITE_DEFAULT);
071    long recheckInterval = conf.getInt(
072        DFSConfigKeys.DFS_NAMENODE_HEARTBEAT_RECHECK_INTERVAL_KEY,
073        DFSConfigKeys.DFS_NAMENODE_HEARTBEAT_RECHECK_INTERVAL_DEFAULT); // 5 min
074    long staleInterval = conf.getLong(
075        DFSConfigKeys.DFS_NAMENODE_STALE_DATANODE_INTERVAL_KEY,
076        DFSConfigKeys.DFS_NAMENODE_STALE_DATANODE_INTERVAL_DEFAULT);// 30s
077
078    if (avoidStaleDataNodesForWrite && staleInterval < recheckInterval) {
079      this.heartbeatRecheckInterval = staleInterval;
080      LOG.info("Setting heartbeat recheck interval to " + staleInterval
081          + " since " + DFSConfigKeys.DFS_NAMENODE_STALE_DATANODE_INTERVAL_KEY
082          + " is less than "
083          + DFSConfigKeys.DFS_NAMENODE_HEARTBEAT_RECHECK_INTERVAL_KEY);
084    } else {
085      this.heartbeatRecheckInterval = recheckInterval;
086    }
087  }
088
089  void activate(Configuration conf) {
090    heartbeatThread.start();
091  }
092
093  void close() {
094    heartbeatThread.interrupt();
095    try {
096      // This will no effect if the thread hasn't yet been started.
097      heartbeatThread.join(3000);
098    } catch (InterruptedException e) {
099    }
100  }

191  synchronized void register(final DatanodeDescriptor d) {
192    if (!d.isAlive) {
193      addDatanode(d);
194
195      //update its timestamp
196      d.updateHeartbeatState(StorageReport.EMPTY_ARRAY, 0L, 0L, 0, 0, null);
197    }
198  }
199
200  synchronized DatanodeDescriptor[] getDatanodes() {
201    return datanodes.toArray(new DatanodeDescriptor[datanodes.size()]);
202  }
203
204  synchronized void addDatanode(final DatanodeDescriptor d) {
205    // update in-service node count
206    stats.add(d);
207    datanodes.add(d);
208    d.isAlive = true;
209  }
210
211  synchronized void removeDatanode(DatanodeDescriptor node) {
212    if (node.isAlive) {
213      stats.subtract(node);
214      datanodes.remove(node);
215      node.isAlive = false;
216    }
217  }
218
219  synchronized void updateHeartbeat(final DatanodeDescriptor node,
220      StorageReport[] reports, long cacheCapacity, long cacheUsed,
221      int xceiverCount, int failedVolumes,
222      VolumeFailureSummary volumeFailureSummary) {
223    stats.subtract(node);
224    node.updateHeartbeat(reports, cacheCapacity, cacheUsed,
225      xceiverCount, failedVolumes, volumeFailureSummary);
226    stats.add(node);
227  }
228
229  synchronized void startDecommission(final DatanodeDescriptor node) {
230    stats.subtract(node);
231    node.startDecommission();
232    stats.add(node);
233  }
234
235  synchronized void stopDecommission(final DatanodeDescriptor node) {
236    stats.subtract(node);
237    node.stopDecommission();
238    stats.add(node);
239  }

241  /**
242   * Check if there are any expired heartbeats, and if so,
243   * whether any blocks have to be re-replicated.
244   * While removing dead datanodes, make sure that only one datanode is marked
245   * dead at a time within the synchronized section. Otherwise, a cascading
246   * effect causes more datanodes to be declared dead.
247   * Check if there are any failed storage and if so,
248   * Remove all the blocks on the storage. It also covers the following less
249   * common scenarios. After DatanodeStorage is marked FAILED, it is still
250   * possible to receive IBR for this storage.
251   * 1) DN could deliver IBR for failed storage due to its implementation.
252   *    a) DN queues a pending IBR request.
253   *    b) The storage of the block fails.
254   *    c) DN first sends HB, NN will mark the storage FAILED.
255   *    d) DN then sends the pending IBR request.
256   * 2) SBN processes block request from pendingDNMessages.
257   *    It is possible to have messages in pendingDNMessages that refer
258   *    to some failed storage.
259   *    a) SBN receives a IBR and put it in pendingDNMessages.
260   *    b) The storage of the block fails.
261   *    c) Edit log replay get the IBR from pendingDNMessages.
262   * Alternatively, we can resolve these scenarios with the following approaches.
263   * A. Make sure DN don't deliver IBR for failed storage.
264   * B. Remove all blocks in PendingDataNodeMessages for the failed storage
265   *    when we remove all blocks from BlocksMap for that storage.
266   */
267  void heartbeatCheck() {
268    final DatanodeManager dm = blockManager.getDatanodeManager();
269    // It's OK to check safe mode w/o taking the lock here, we re-check
270    // for safe mode after taking the lock before removing a datanode.
271    if (namesystem.isInStartupSafeMode()) {
272      return;
273    }
274    boolean allAlive = false;
275    while (!allAlive) {
276      // locate the first dead node.
277      DatanodeID dead = null;
278
279      // locate the first failed storage that isn't on a dead node.
280      DatanodeStorageInfo failedStorage = null;
281
282      // check the number of stale nodes
283      int numOfStaleNodes = 0;
284      int numOfStaleStorages = 0;
285      synchronized(this) {
286        for (DatanodeDescriptor d : datanodes) {
287          if (dead == null && dm.isDatanodeDead(d)) {
288            stats.incrExpiredHeartbeats();
289            dead = d;
290          }
291          if (d.isStale(dm.getStaleInterval())) {
292            numOfStaleNodes++;
293          }
294          DatanodeStorageInfo[] storageInfos = d.getStorageInfos();
295          for(DatanodeStorageInfo storageInfo : storageInfos) {
296            if (storageInfo.areBlockContentsStale()) {
297              numOfStaleStorages++;
298            }
299
300            if (failedStorage == null &&
301                storageInfo.areBlocksOnFailedStorage() &&
302                d != dead) {
303              failedStorage = storageInfo;
304            }
305          }
306
307        }
308        
309        // Set the number of stale nodes in the DatanodeManager
310        dm.setNumStaleNodes(numOfStaleNodes);
311        dm.setNumStaleStorages(numOfStaleStorages);
312      }
313
314      allAlive = dead == null && failedStorage == null;
315      if (dead != null) {
316        // acquire the fsnamesystem lock, and then remove the dead node.
317        namesystem.writeLock();
318        try {
319          if (namesystem.isInStartupSafeMode()) {
320            return;
321          }
322          synchronized(this) {
323            dm.removeDeadDatanode(dead);
324          }
325        } finally {
326          namesystem.writeUnlock();
327        }
328      }
329      if (failedStorage != null) {
330        // acquire the fsnamesystem lock, and remove blocks on the storage.
331        namesystem.writeLock();
332        try {
333          if (namesystem.isInStartupSafeMode()) {
334            return;
335          }
336          synchronized(this) {
337            blockManager.removeBlocksAssociatedTo(failedStorage);
338          }
339        } finally {
340          namesystem.writeUnlock();
341        }
342      }
343    }
344  }

347  /** Periodically check heartbeat and update block key */
348  private class Monitor implements Runnable {
349    private long lastHeartbeatCheck;
350    private long lastBlockKeyUpdate;
351
352    @Override
353    public void run() {
354      while(namesystem.isRunning()) {
355        try {
356          final long now = Time.monotonicNow();
357          if (lastHeartbeatCheck + heartbeatRecheckInterval < now) {
358            heartbeatCheck();
359            lastHeartbeatCheck = now;
360          }
361          if (blockManager.shouldUpdateBlockKey(now - lastBlockKeyUpdate)) {
362            synchronized(HeartbeatManager.this) {
363              for(DatanodeDescriptor d : datanodes) {
364                d.needKeyUpdate = true;
365              }
366            }
367            lastBlockKeyUpdate = now;
368          }
369        } catch (Exception e) {
370          LOG.error("Exception while checking heartbeat", e);
371        }
372        try {
373          Thread.sleep(5000);  // 5 seconds
374        } catch (InterruptedException ie) {
375        }
376      }
377    }
378  }

380  /** Datanode statistics.
381   * For decommissioning/decommissioned nodes, only used capacity is counted.
382   */
383  private static class Stats {
384    private long capacityTotal = 0L;
385    private long capacityUsed = 0L;
386    private long capacityRemaining = 0L;
387    private long blockPoolUsed = 0L;
388    private int xceiverCount = 0;
389    private long cacheCapacity = 0L;
390    private long cacheUsed = 0L;
391
392    private int nodesInService = 0;
393    private int nodesInServiceXceiverCount = 0;
394
395    private int expiredHeartbeats = 0;
396
397    private void add(final DatanodeDescriptor node) {
398      capacityUsed += node.getDfsUsed();
399      blockPoolUsed += node.getBlockPoolUsed();
400      xceiverCount += node.getXceiverCount();
401      if (!(node.isDecommissionInProgress() || node.isDecommissioned())) {
402        nodesInService++;
403        nodesInServiceXceiverCount += node.getXceiverCount();
404        capacityTotal += node.getCapacity();
405        capacityRemaining += node.getRemaining();
406      } else {
407        capacityTotal += node.getDfsUsed();
408      }
409      cacheCapacity += node.getCacheCapacity();
410      cacheUsed += node.getCacheUsed();
411    }
412
413    private void subtract(final DatanodeDescriptor node) {
414      capacityUsed -= node.getDfsUsed();
415      blockPoolUsed -= node.getBlockPoolUsed();
416      xceiverCount -= node.getXceiverCount();
417      if (!(node.isDecommissionInProgress() || node.isDecommissioned())) {
418        nodesInService--;
419        nodesInServiceXceiverCount -= node.getXceiverCount();
420        capacityTotal -= node.getCapacity();
421        capacityRemaining -= node.getRemaining();
422      } else {
423        capacityTotal -= node.getDfsUsed();
424      }
425      cacheCapacity -= node.getCacheCapacity();
426      cacheUsed -= node.getCacheUsed();
427    }
428    
429    /** Increment expired heartbeat counter. */
430    private void incrExpiredHeartbeats() {
431      expiredHeartbeats++;
432    }
433  }
