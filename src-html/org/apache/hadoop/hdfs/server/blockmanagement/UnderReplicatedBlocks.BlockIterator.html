031/**
032 * Keep prioritized queues of under replicated blocks.
033 * Blocks have replication priority, with priority {@link #QUEUE_HIGHEST_PRIORITY}
034 * indicating the highest priority.
035 * </p>
036 * Having a prioritised queue allows the {@link BlockManager} to select
037 * which blocks to replicate first -it tries to give priority to data
038 * that is most at risk or considered most valuable.
039 *
040 * <p/>
041 * The policy for choosing which priority to give added blocks
042 * is implemented in {@link #getPriority(Block, int, int, int)}.
043 * </p>
044 * <p>The queue order is as follows:</p>
045 * <ol>
046 *   <li>{@link #QUEUE_HIGHEST_PRIORITY}: the blocks that must be replicated
047 *   first. That is blocks with only one copy, or blocks with zero live
048 *   copies but a copy in a node being decommissioned. These blocks
049 *   are at risk of loss if the disk or server on which they
050 *   remain fails.</li>
051 *   <li>{@link #QUEUE_VERY_UNDER_REPLICATED}: blocks that are very
052 *   under-replicated compared to their expected values. Currently
053 *   that means the ratio of the ratio of actual:expected means that
054 *   there is <i>less than</i> 1:3.</li>. These blocks may not be at risk,
055 *   but they are clearly considered "important".
056 *   <li>{@link #QUEUE_UNDER_REPLICATED}: blocks that are also under
057 *   replicated, and the ratio of actual:expected is good enough that
058 *   they do not need to go into the {@link #QUEUE_VERY_UNDER_REPLICATED}
059 *   queue.</li>
060 *   <li>{@link #QUEUE_REPLICAS_BADLY_DISTRIBUTED}: there are as least as
061 *   many copies of a block as required, but the blocks are not adequately
062 *   distributed. Loss of a rack/switch could take all copies off-line.</li>
063 *   <li>{@link #QUEUE_WITH_CORRUPT_BLOCKS} This is for blocks that are corrupt
064 *   and for which there are no-non-corrupt copies (currently) available.
065 *   The policy here is to keep those corrupt blocks replicated, but give
066 *   blocks that are not corrupt higher priority.</li>
067 * </ol>
068 */
069class UnderReplicatedBlocks implements Iterable<Block> {
070  /** The total number of queues : {@value} */
071  static final int LEVEL = 5;
072  /** The queue with the highest priority: {@value} */
073  static final int QUEUE_HIGHEST_PRIORITY = 0;
074  /** The queue for blocks that are way below their expected value : {@value} */
075  static final int QUEUE_VERY_UNDER_REPLICATED = 1;
076  /** The queue for "normally" under-replicated blocks: {@value} */
077  static final int QUEUE_UNDER_REPLICATED = 2;
078  /** The queue for blocks that have the right number of replicas,
079   * but which the block manager felt were badly distributed: {@value}
080   */
081  static final int QUEUE_REPLICAS_BADLY_DISTRIBUTED = 3;
082  /** The queue for corrupt blocks: {@value} */
083  static final int QUEUE_WITH_CORRUPT_BLOCKS = 4;
084  /** the queues themselves */
085  private final List<LightWeightLinkedSet<Block>> priorityQueues
086      = new ArrayList<LightWeightLinkedSet<Block>>();
087
088  /** Stores the replication index for each priority */
089  private Map<Integer, Integer> priorityToReplIdx = new HashMap<Integer, Integer>(LEVEL);
090  /** The number of corrupt blocks with replication factor 1 */
091  private int corruptReplOneBlocks = 0;
092
093  /** Create an object. */
094  UnderReplicatedBlocks() {
095    for (int i = 0; i < LEVEL; i++) {
096      priorityQueues.add(new LightWeightLinkedSet<Block>());
097      priorityToReplIdx.put(i, 0);
098    }
099  }
100
101  /**
102   * Empty the queues.
103   */
104  void clear() {
105    for (int i = 0; i < LEVEL; i++) {
106      priorityQueues.get(i).clear();
107    }
108  }
109
110  /** Return the total number of under replication blocks */
111  synchronized int size() {
112    int size = 0;
113    for (int i = 0; i < LEVEL; i++) {
114      size += priorityQueues.get(i).size();
115    }
116    return size;
117  }
118
119  /** Return the number of under replication blocks excluding corrupt blocks */
120  synchronized int getUnderReplicatedBlockCount() {
121    int size = 0;
122    for (int i = 0; i < LEVEL; i++) {
123      if (i != QUEUE_WITH_CORRUPT_BLOCKS) {
124        size += priorityQueues.get(i).size();
125      }
126    }
127    return size;
128  }
129
130  /** Return the number of corrupt blocks */
131  synchronized int getCorruptBlockSize() {
132    return priorityQueues.get(QUEUE_WITH_CORRUPT_BLOCKS).size();
133  }
134
135  /** Return the number of corrupt blocks with replication factor 1 */
136  synchronized int getCorruptReplOneBlockSize() {
137    return corruptReplOneBlocks;
138  }
139
140  /** Check if a block is in the neededReplication queue */
141  synchronized boolean contains(Block block) {
142    for(LightWeightLinkedSet<Block> set : priorityQueues) {
143      if (set.contains(block)) {
144        return true;
145      }
146    }
147    return false;
148  }
149
150  /** Return the priority of a block
151   * @param block a under replicated block
152   * @param curReplicas current number of replicas of the block
153   * @param expectedReplicas expected number of replicas of the block
154   * @return the priority for the blocks, between 0 and ({@link #LEVEL}-1)
155   */
156  private int getPriority(Block block,
157                          int curReplicas, 
158                          int decommissionedReplicas,
159                          int expectedReplicas) {
160    assert curReplicas >= 0 : "Negative replicas!";
161    if (curReplicas >= expectedReplicas) {
162      // Block has enough copies, but not enough racks
163      return QUEUE_REPLICAS_BADLY_DISTRIBUTED;
164    } else if (curReplicas == 0) {
165      // If there are zero non-decommissioned replicas but there are
166      // some decommissioned replicas, then assign them highest priority
167      if (decommissionedReplicas > 0) {
168        return QUEUE_HIGHEST_PRIORITY;
169      }
170      //all we have are corrupt blocks
171      return QUEUE_WITH_CORRUPT_BLOCKS;
172    } else if (curReplicas == 1) {
173      //only on replica -risk of loss
174      // highest priority
175      return QUEUE_HIGHEST_PRIORITY;
176    } else if ((curReplicas * 3) < expectedReplicas) {
177      //there is less than a third as many blocks as requested;
178      //this is considered very under-replicated
179      return QUEUE_VERY_UNDER_REPLICATED;
180    } else {
181      //add to the normal queue for under replicated blocks
182      return QUEUE_UNDER_REPLICATED;
183    }
184  }
185
186  /** add a block to a under replication queue according to its priority
187   * @param block a under replication block
188   * @param curReplicas current number of replicas of the block
189   * @param decomissionedReplicas the number of decommissioned replicas
190   * @param expectedReplicas expected number of replicas of the block
191   * @return true if the block was added to a queue.
192   */
193  synchronized boolean add(Block block,
194                           int curReplicas, 
195                           int decomissionedReplicas,
196                           int expectedReplicas) {
197    assert curReplicas >= 0 : "Negative replicas!";
198    int priLevel = getPriority(block, curReplicas, decomissionedReplicas,
199                               expectedReplicas);
200    if(priorityQueues.get(priLevel).add(block)) {
201      if (priLevel == QUEUE_WITH_CORRUPT_BLOCKS &&
202          expectedReplicas == 1) {
203        corruptReplOneBlocks++;
204      }
205      NameNode.blockStateChangeLog.debug(
206          "BLOCK* NameSystem.UnderReplicationBlock.add: {}"
207              + " has only {} replicas and need {} replicas so is added to" +
208              " neededReplications at priority level {}", block, curReplicas,
209          expectedReplicas, priLevel);
210
211      return true;
212    }
213    return false;
214  }
215
216  /** remove a block from a under replication queue */
217  synchronized boolean remove(Block block, 
218                              int oldReplicas, 
219                              int decommissionedReplicas,
220                              int oldExpectedReplicas) {
221    int priLevel = getPriority(block, oldReplicas, 
222                               decommissionedReplicas,
223                               oldExpectedReplicas);
224    boolean removedBlock = remove(block, priLevel);
225    if (priLevel == QUEUE_WITH_CORRUPT_BLOCKS &&
226        oldExpectedReplicas == 1 &&
227        removedBlock) {
228      corruptReplOneBlocks--;
229      assert corruptReplOneBlocks >= 0 :
230          "Number of corrupt blocks with replication factor 1 " +
231              "should be non-negative";
232    }
233    return removedBlock;
234  }
235
236  /**
237   * Remove a block from the under replication queues.
238   *
239   * The priLevel parameter is a hint of which queue to query
240   * first: if negative or &gt;= {@link #LEVEL} this shortcutting
241   * is not attmpted.
242   *
243   * If the block is not found in the nominated queue, an attempt is made to
244   * remove it from all queues.
245   *
246   * <i>Warning:</i> This is not a synchronized method.
247   * @param block block to remove
248   * @param priLevel expected privilege level
249   * @return true if the block was found and removed from one of the priority queues
250   */
251  boolean remove(Block block, int priLevel) {
252    if(priLevel >= 0 && priLevel < LEVEL 
253        && priorityQueues.get(priLevel).remove(block)) {
254      NameNode.blockStateChangeLog.debug(
255        "BLOCK* NameSystem.UnderReplicationBlock.remove: Removing block {}" +
256            " from priority queue {}", block, priLevel);
257      return true;
258    } else {
259      // Try to remove the block from all queues if the block was
260      // not found in the queue for the given priority level.
261      for (int i = 0; i < LEVEL; i++) {
262        if (priorityQueues.get(i).remove(block)) {
263          NameNode.blockStateChangeLog.debug(
264              "BLOCK* NameSystem.UnderReplicationBlock.remove: Removing block" +
265                  " {} from priority queue {}", block, priLevel);
266          return true;
267        }
268      }
269    }
270    return false;
271  }
272
273  /**
274   * Recalculate and potentially update the priority level of a block.
275   *
276   * If the block priority has changed from before an attempt is made to
277   * remove it from the block queue. Regardless of whether or not the block
278   * is in the block queue of (recalculate) priority, an attempt is made
279   * to add it to that queue. This ensures that the block will be
280   * in its expected priority queue (and only that queue) by the end of the
281   * method call.
282   * @param block a under replicated block
283   * @param curReplicas current number of replicas of the block
284   * @param decommissionedReplicas  the number of decommissioned replicas
285   * @param curExpectedReplicas expected number of replicas of the block
286   * @param curReplicasDelta the change in the replicate count from before
287   * @param expectedReplicasDelta the change in the expected replica count from before
288   */
289  synchronized void update(Block block, int curReplicas,
290                           int decommissionedReplicas,
291                           int curExpectedReplicas,
292                           int curReplicasDelta, int expectedReplicasDelta) {
293    int oldReplicas = curReplicas-curReplicasDelta;
294    int oldExpectedReplicas = curExpectedReplicas-expectedReplicasDelta;
295    int curPri = getPriority(block, curReplicas, decommissionedReplicas, curExpectedReplicas);
296    int oldPri = getPriority(block, oldReplicas, decommissionedReplicas, oldExpectedReplicas);
297    if(NameNode.stateChangeLog.isDebugEnabled()) {
298      NameNode.stateChangeLog.debug("UnderReplicationBlocks.update " + 
299        block +
300        " curReplicas " + curReplicas +
301        " curExpectedReplicas " + curExpectedReplicas +
302        " oldReplicas " + oldReplicas +
303        " oldExpectedReplicas  " + oldExpectedReplicas +
304        " curPri  " + curPri +
305        " oldPri  " + oldPri);
306    }
307    if(oldPri != curPri) {
308      remove(block, oldPri);
309    }
310    if(priorityQueues.get(curPri).add(block)) {
311      NameNode.blockStateChangeLog.debug(
312          "BLOCK* NameSystem.UnderReplicationBlock.update: {} has only {} " +
313              "replicas and needs {} replicas so is added to " +
314              "neededReplications at priority level {}", block, curReplicas,
315          curExpectedReplicas, curPri);
316
317    }
318    if (oldPri != curPri || expectedReplicasDelta != 0) {
319      // corruptReplOneBlocks could possibly change
320      if (curPri == QUEUE_WITH_CORRUPT_BLOCKS &&
321          curExpectedReplicas == 1) {
322        // add a new corrupt block with replication factor 1
323        corruptReplOneBlocks++;
324      } else if (oldPri == QUEUE_WITH_CORRUPT_BLOCKS &&
325          curExpectedReplicas - expectedReplicasDelta == 1) {
326        // remove an existing corrupt block with replication factor 1
327        corruptReplOneBlocks--;
328      }
329    }
330  }

332  /**
333   * Get a list of block lists to be replicated. The index of block lists
334   * represents its replication priority. Replication index will be tracked for
335   * each priority list separately in priorityToReplIdx map. Iterates through
336   * all priority lists and find the elements after replication index. Once the
337   * last priority lists reaches to end, all replication indexes will be set to
338   * 0 and start from 1st priority list to fulfill the blockToProces count.
339   * 
340   * @param blocksToProcess - number of blocks to fetch from underReplicated blocks.
341   * @return Return a list of block lists to be replicated. The block list index
342   *         represents its replication priority.
343   */
344  public synchronized List<List<Block>> chooseUnderReplicatedBlocks(
345      int blocksToProcess) {
346    // initialize data structure for the return value
347    List<List<Block>> blocksToReplicate = new ArrayList<List<Block>>(LEVEL);
348    for (int i = 0; i < LEVEL; i++) {
349      blocksToReplicate.add(new ArrayList<Block>());
350    }
351
352    if (size() == 0) { // There are no blocks to collect.
353      return blocksToReplicate;
354    }
355    
356    int blockCount = 0;
357    for (int priority = 0; priority < LEVEL; priority++) { 
358      // Go through all blocks that need replications with current priority.
359      BlockIterator neededReplicationsIterator = iterator(priority);
360      Integer replIndex = priorityToReplIdx.get(priority);
361      
362      // skip to the first unprocessed block, which is at replIndex
363      for (int i = 0; i < replIndex && neededReplicationsIterator.hasNext(); i++) {
364        neededReplicationsIterator.next();
365      }
366
367      blocksToProcess = Math.min(blocksToProcess, size());
368      
369      if (blockCount == blocksToProcess) {
370        break;  // break if already expected blocks are obtained
371      }
372      
373      // Loop through all remaining blocks in the list.
374      while (blockCount < blocksToProcess
375          && neededReplicationsIterator.hasNext()) {
376        Block block = neededReplicationsIterator.next();
377        blocksToReplicate.get(priority).add(block);
378        replIndex++;
379        blockCount++;
380      }
381      
382      if (!neededReplicationsIterator.hasNext()
383          && neededReplicationsIterator.getPriority() == LEVEL - 1) {
384        // reset all priorities replication index to 0 because there is no
385        // recently added blocks in any list.
386        for (int i = 0; i < LEVEL; i++) {
387          priorityToReplIdx.put(i, 0);
388        }
389        break;
390      }
391      priorityToReplIdx.put(priority, replIndex); 
392    }
393    return blocksToReplicate;
394  }
395
396  /** returns an iterator of all blocks in a given priority queue */
397  synchronized BlockIterator iterator(int level) {
398    return new BlockIterator(level);
399  }
400
401  /** return an iterator of all the under replication blocks */
402  @Override
403  public synchronized BlockIterator iterator() {
404    return new BlockIterator();
405  }

407  /**
408   * An iterator over blocks.
409   */
410  class BlockIterator implements Iterator<Block> {
411    private int level;
412    private boolean isIteratorForLevel = false;
413    private final List<Iterator<Block>> iterators = new ArrayList<Iterator<Block>>();
414
415    /**
416     * Construct an iterator over all queues.
417     */
418    private BlockIterator() {
419      level=0;
420      for(int i=0; i<LEVEL; i++) {
421        iterators.add(priorityQueues.get(i).iterator());
422      }
423    }
424
425    /**
426     * Constrict an iterator for a single queue level
427     * @param l the priority level to iterate over
428     */
429    private BlockIterator(int l) {
430      level = l;
431      isIteratorForLevel = true;
432      iterators.add(priorityQueues.get(level).iterator());
433    }
434
435    private void update() {
436      if (isIteratorForLevel) {
437        return;
438      }
439      while(level< LEVEL-1 && !iterators.get(level).hasNext()) {
440        level++;
441      }
442    }
443
444    @Override
445    public Block next() {
446      if (isIteratorForLevel) {
447        return iterators.get(0).next();
448      }
449      update();
450      return iterators.get(level).next();
451    }
452
453    @Override
454    public boolean hasNext() {
455      if (isIteratorForLevel) {
456        return iterators.get(0).hasNext();
457      }
458      update();
459      return iterators.get(level).hasNext();
460    }
461
462    @Override
463    public void remove() {
464      if (isIteratorForLevel) {
465        iterators.get(0).remove();
466      } else {
467        iterators.get(level).remove();
468      }
469    }
470
471    int getPriority() {
472      return level;
473    }
474  }
475
476  /**
477   * This method is to decrement the replication index for the given priority
478   * 
479   * @param priority  - int priority level
480   */
481  public void decrementReplicationIndex(int priority) {
482    Integer replIdx = priorityToReplIdx.get(priority);
483    priorityToReplIdx.put(priority, --replIdx); 
484  }
