019package org.apache.hadoop.hdfs.server.datanode.fsdataset.impl;
020
021import com.google.common.base.Preconditions;
022
023import org.apache.commons.logging.Log;
024import org.apache.commons.logging.LogFactory;
025import org.apache.hadoop.classification.InterfaceAudience;
026import org.apache.hadoop.classification.InterfaceStability;
027import org.apache.hadoop.conf.Configuration;
028import org.apache.hadoop.hdfs.DFSConfigKeys;
029import org.apache.hadoop.hdfs.server.datanode.fsdataset.FsVolumeSpi;
030import org.apache.hadoop.util.ReflectionUtils;
031import org.apache.hadoop.util.Time;
032
033import java.io.File;
034import java.util.concurrent.atomic.AtomicLong;
035
036@InterfaceAudience.Private
037@InterfaceStability.Unstable
038public abstract class RamDiskReplicaTracker {
039  static final Log LOG = LogFactory.getLog(RamDiskReplicaTracker.class);
040
041  FsDatasetImpl fsDataset;
042
043  static class RamDiskReplica implements Comparable<RamDiskReplica>  {
044    private final String bpid;
045    private final long blockId;
046    private File savedBlockFile;
047    private File savedMetaFile;
048
049    private long creationTime;
050    protected AtomicLong numReads = new AtomicLong(0);
051    protected boolean isPersisted;
052
053    /**
054     * RAM_DISK volume that holds the original replica.
055     */
056    final FsVolumeSpi ramDiskVolume;
057
058    /**
059     * Persistent volume that holds or will hold the saved replica.
060     */
061    FsVolumeImpl lazyPersistVolume;
062
063    RamDiskReplica(final String bpid, final long blockId,
064                   final FsVolumeImpl ramDiskVolume) {
065      this.bpid = bpid;
066      this.blockId = blockId;
067      this.ramDiskVolume = ramDiskVolume;
068      lazyPersistVolume = null;
069      savedMetaFile = null;
070      savedBlockFile = null;
071      creationTime = Time.monotonicNow();
072      isPersisted = false;
073    }
074
075    long getBlockId() {
076      return blockId;
077    }
078
079    String getBlockPoolId() {
080      return bpid;
081    }
082
083    FsVolumeImpl getLazyPersistVolume() {
084      return lazyPersistVolume;
085    }
086
087    void setLazyPersistVolume(FsVolumeImpl volume) {
088      Preconditions.checkState(!volume.isTransientStorage());
089      this.lazyPersistVolume = volume;
090    }
091
092    File getSavedBlockFile() {
093      return savedBlockFile;
094    }
095
096    File getSavedMetaFile() {
097      return savedMetaFile;
098    }
099
100    long getNumReads() { return numReads.get(); }
101
102    long getCreationTime() { return creationTime; }
103
104    boolean getIsPersisted() {return isPersisted; }
105
106    /**
107     * Record the saved meta and block files on the given volume.
108     *
109     * @param files Meta and block files, in that order.
110     */
111    void recordSavedBlockFiles(File[] files) {
112      this.savedMetaFile = files[0];
113      this.savedBlockFile = files[1];
114    }
115
116    @Override
117    public int hashCode() {
118      return bpid.hashCode() ^ (int) blockId;
119    }
120
121    @Override
122    public boolean equals(Object other) {
123      if (this == other) {
124        return true;
125      }
126
127      if (other == null || getClass() != other.getClass()) {
128        return false;
129      }
130
131      RamDiskReplica otherState = (RamDiskReplica) other;
132      return (otherState.bpid.equals(bpid) && otherState.blockId == blockId);
133    }
134
135    // Delete the saved meta and block files. Failure to delete can be
136    // ignored, the directory scanner will retry the deletion later.
137    void deleteSavedFiles() {
138      if (savedBlockFile != null) {
139        if (!savedBlockFile.delete()) {
140          LOG.warn("Failed to delete block file " + savedBlockFile);
141        }
142        savedBlockFile = null;
143      }
144
145      if (savedMetaFile != null) {
146        if (!savedMetaFile.delete()) {
147          LOG.warn("Failed to delete meta file " + savedMetaFile);
148        }
149        savedMetaFile = null;
150      }
151    }
152
153    @Override
154    public int compareTo(RamDiskReplica other) {
155      int bpidResult = bpid.compareTo(other.bpid);
156      if (bpidResult == 0)
157        if (blockId == other.blockId) {
158          return 0;
159        } else if (blockId < other.blockId) {
160          return -1;
161        } else {
162          return 1;
163        }
164      return bpidResult;
165    }
166
167    @Override
168    public String toString() {
169      return "[BlockPoolID=" + bpid + "; BlockId=" + blockId + "]";
170    }
171  }
172
173  /**
174   * Get an instance of the configured RamDiskReplicaTracker based on the
175   * the configuration property
176   * {@link org.apache.hadoop.hdfs.DFSConfigKeys#DFS_DATANODE_RAM_DISK_REPLICA_TRACKER_KEY}.
177   *
178   * @param conf the configuration to be used
179   * @param dataset the FsDataset object.
180   * @return an instance of RamDiskReplicaTracker
181   */
182  static RamDiskReplicaTracker getInstance(final Configuration conf,
183                                           final FsDatasetImpl fsDataset) {
184    final Class<? extends RamDiskReplicaTracker> trackerClass = conf.getClass(
185        DFSConfigKeys.DFS_DATANODE_RAM_DISK_REPLICA_TRACKER_KEY,
186        DFSConfigKeys.DFS_DATANODE_RAM_DISK_REPLICA_TRACKER_DEFAULT,
187        RamDiskReplicaTracker.class);
188    final RamDiskReplicaTracker tracker = ReflectionUtils.newInstance(
189        trackerClass, conf);
190    tracker.initialize(fsDataset);
191    return tracker;
192  }
193
194  void initialize(final FsDatasetImpl fsDataset) {
195    this.fsDataset = fsDataset;
196  }
197
198  /**
199   * Start tracking a new finalized replica on RAM disk.
200   *
201   * @param transientVolume RAM disk volume that stores the replica.
202   */
203  abstract void addReplica(final String bpid, final long blockId,
204                           final FsVolumeImpl transientVolume);
205
206  /**
207   * Invoked when a replica is opened by a client. This may be used as
208   * a heuristic by the eviction scheme.
209   */
210  abstract void touch(final String bpid, final long blockId);
211
212  /**
213   * Get the next replica to write to persistent storage.
214   */
215  abstract RamDiskReplica dequeueNextReplicaToPersist();
216
217  /**
218   * Invoked if a replica that was previously dequeued for persistence
219   * could not be successfully persisted. Add it back so it can be retried
220   * later.
221   */
222  abstract void reenqueueReplicaNotPersisted(
223      final RamDiskReplica ramDiskReplica);
224
225  /**
226   * Invoked when the Lazy persist operation is started by the DataNode.
227   * @param checkpointVolume
228   */
229  abstract void recordStartLazyPersist(
230      final String bpid, final long blockId, FsVolumeImpl checkpointVolume);
231
232  /**
233   * Invoked when the Lazy persist operation is complete.
234   *
235   * @param savedFiles The saved meta and block files, in that order.
236   */
237  abstract void recordEndLazyPersist(
238      final String bpid, final long blockId, final File[] savedFiles);
239
240  /**
241   * Return a candidate replica to remove from RAM Disk. The exact replica
242   * to be returned may depend on the eviction scheme utilized.
243   *
244   * @return
245   */
246  abstract RamDiskReplica getNextCandidateForEviction();
247
248  /**
249   * Return the number of replicas pending persistence to disk.
250   */
251  abstract int numReplicasNotPersisted();
252
253  /**
254   * Discard all state we are tracking for the given replica.
255   */
256  abstract void discardReplica(
257      final String bpid, final long blockId,
258      boolean deleteSavedCopies);
259
260  /**
261   * Return RamDiskReplica info given block pool id and block id
262   * Return null if it does not exist in RamDisk
263   */
264  abstract RamDiskReplica getReplica(
265    final String bpid, final long blockId);
266}
