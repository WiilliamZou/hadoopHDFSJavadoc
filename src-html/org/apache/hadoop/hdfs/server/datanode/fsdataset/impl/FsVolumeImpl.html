109  FsVolumeImpl(FsDatasetImpl dataset, String storageID, File currentDir,
110      Configuration conf, StorageType storageType) throws IOException {
111    this.dataset = dataset;
112    this.storageID = storageID;
113    this.reserved = conf.getLong(
114        DFSConfigKeys.DFS_DATANODE_DU_RESERVED_KEY,
115        DFSConfigKeys.DFS_DATANODE_DU_RESERVED_DEFAULT);
116    this.reservedForRbw = new AtomicLong(0L);
117    this.currentDir = currentDir; 
118    File parent = currentDir.getParentFile();
119    this.usage = new DF(parent, conf);
120    this.storageType = storageType;
121    this.configuredCapacity = -1;
122    cacheExecutor = initializeCacheExecutor(parent);
123  }
124
125  protected ThreadPoolExecutor initializeCacheExecutor(File parent) {
126    if (storageType.isTransient()) {
127      return null;
128    }
129    if (dataset.datanode == null) {
130      // FsVolumeImpl is used in test.
131      return null;
132    }
133
134    final int maxNumThreads = dataset.datanode.getConf().getInt(
135        DFSConfigKeys.DFS_DATANODE_FSDATASETCACHE_MAX_THREADS_PER_VOLUME_KEY,
136        DFSConfigKeys.DFS_DATANODE_FSDATASETCACHE_MAX_THREADS_PER_VOLUME_DEFAULT);
137
138    ThreadFactory workerFactory = new ThreadFactoryBuilder()
139        .setDaemon(true)
140        .setNameFormat("FsVolumeImplWorker-" + parent.toString() + "-%d")
141        .build();
142    ThreadPoolExecutor executor = new ThreadPoolExecutor(
143        1, maxNumThreads,
144        60, TimeUnit.SECONDS,
145        new LinkedBlockingQueue<Runnable>(),
146        workerFactory);
147    executor.allowCoreThreadTimeOut(true);
148    return executor;
149  }

151  private void printReferenceTraceInfo(String op) {
152    StackTraceElement[] stack = Thread.currentThread().getStackTrace();
153    for (StackTraceElement ste : stack) {
154      switch (ste.getMethodName()) {
155      case "getDfsUsed":
156      case "getBlockPoolUsed":
157      case "getAvailable":
158      case "getVolumeMap":
159        return;
160      default:
161        break;
162      }
163    }
164    FsDatasetImpl.LOG.trace("Reference count: " + op + " " + this + ": " +
165        this.reference.getReferenceCount());
166    FsDatasetImpl.LOG.trace(
167        Joiner.on("\n").join(Thread.currentThread().getStackTrace()));
168  }

170  /**
171   * Increase the reference count. The caller must increase the reference count
172   * before issuing IOs.
173   *
174   * @throws IOException if the volume is already closed.
175   */
176  private void reference() throws ClosedChannelException {
177    this.reference.reference();
178    if (FsDatasetImpl.LOG.isTraceEnabled()) {
179      printReferenceTraceInfo("incr");
180    }
181  }
182
183  /**
184   * Decrease the reference count.
185   */
186  private void unreference() {
187    if (FsDatasetImpl.LOG.isTraceEnabled()) {
188      printReferenceTraceInfo("desc");
189    }
190    if (FsDatasetImpl.LOG.isDebugEnabled()) {
191      if (reference.getReferenceCount() <= 0) {
192        FsDatasetImpl.LOG.debug("Decrease reference count <= 0 on " + this +
193          Joiner.on("\n").join(Thread.currentThread().getStackTrace()));
194      }
195    }
196    checkReference();
197    this.reference.unreference();
198  }

232  /**
233   * Close this volume and wait all other threads to release the reference count
234   * on this volume.
235   * @throws IOException if the volume is closed or the waiting is interrupted.
236   */
237  void closeAndWait() throws IOException {
238    try {
239      this.reference.setClosed();
240    } catch (ClosedChannelException e) {
241      throw new IOException("The volume has already closed.", e);
242    }
243    final int SLEEP_MILLIS = 500;
244    while (this.reference.getReferenceCount() > 0) {
245      if (FsDatasetImpl.LOG.isDebugEnabled()) {
246        FsDatasetImpl.LOG.debug(String.format(
247            "The reference count for %s is %d, wait to be 0.",
248            this, reference.getReferenceCount()));
249      }
250      try {
251        Thread.sleep(SLEEP_MILLIS);
252      } catch (InterruptedException e) {
253        throw new IOException(e);
254      }
255    }
256  }

403  @Override
404  public void releaseReservedSpace(long bytesToRelease) {
405    if (bytesToRelease != 0) {
406
407      long oldReservation, newReservation;
408      do {
409        oldReservation = reservedForRbw.get();
410        newReservation = oldReservation - bytesToRelease;
411        if (newReservation < 0) {
412          // Failsafe, this should never occur in practice, but if it does we don't
413          // want to start advertising more space than we have available.
414          newReservation = 0;
415        }
416      } while (!reservedForRbw.compareAndSet(oldReservation, newReservation));
417    }

439  @VisibleForTesting
440  public static String nextSorted(List<String> arr, String prev) {
441    int res = 0;
442    if (prev != null) {
443      res = Collections.binarySearch(arr, prev);
444      if (res < 0) {
445        res = -1 - res;
446      } else {
447        res++;
448      }
449    }
450    if (res >= arr.size()) {
451      return null;
452    }
453    return arr.get(res);
454  }

486  /**
487   * A BlockIterator implementation for FsVolumeImpl.
488   */
489  private class BlockIteratorImpl implements FsVolumeSpi.BlockIterator {
490    private final File bpidDir;
491    private final String name;
492    private final String bpid;
493    private long maxStalenessMs = 0;
494
495    private List<String> cache;
496    private long cacheMs;
497
498    private BlockIteratorState state;
499
500    BlockIteratorImpl(String bpid, String name) {
501      this.bpidDir = new File(currentDir, bpid);
502      this.name = name;
503      this.bpid = bpid;
504      rewind();
505    }

507    /**
508     * Get the next subdirectory within the block pool slice.
509     *
510     * @return         The next subdirectory within the block pool slice, or
511     *                   null if there are no more.
512     */
513    private String getNextSubDir(String prev, File dir)
514          throws IOException {
515      List<String> children =
516          IOUtils.listDirectory(dir, SubdirFilter.INSTANCE);
517      cache = null;
518      cacheMs = 0;
519      if (children.size() == 0) {
520        LOG.trace("getNextSubDir({}, {}): no subdirectories found in {}",
521            storageID, bpid, dir.getAbsolutePath());
522        return null;
523      }
524      Collections.sort(children);
525      String nextSubDir = nextSorted(children, prev);
526      if (nextSubDir == null) {
527        LOG.trace("getNextSubDir({}, {}): no more subdirectories found in {}",
528            storageID, bpid, dir.getAbsolutePath());
529      } else {
530        LOG.trace("getNextSubDir({}, {}): picking next subdirectory {} " +
531            "within {}", storageID, bpid, nextSubDir, dir.getAbsolutePath());
532      }
533      return nextSubDir;
534    }
535
536    private String getNextFinalizedDir() throws IOException {
537      File dir = Paths.get(
538          bpidDir.getAbsolutePath(), "current", "finalized").toFile();
539      return getNextSubDir(state.curFinalizedDir, dir);
540    }
541
542    private String getNextFinalizedSubDir() throws IOException {
543      if (state.curFinalizedDir == null) {
544        return null;
545      }
546      File dir = Paths.get(
547          bpidDir.getAbsolutePath(), "current", "finalized",
548              state.curFinalizedDir).toFile();
549      return getNextSubDir(state.curFinalizedSubDir, dir);
550    }
551
552    private List<String> getSubdirEntries() throws IOException {
553      if (state.curFinalizedSubDir == null) {
554        return null; // There are no entries in the null subdir.
555      }
556      long now = Time.monotonicNow();
557      if (cache != null) {
558        long delta = now - cacheMs;
559        if (delta < maxStalenessMs) {
560          return cache;
561        } else {
562          LOG.trace("getSubdirEntries({}, {}): purging entries cache for {} " +
563            "after {} ms.", storageID, bpid, state.curFinalizedSubDir, delta);
564          cache = null;
565        }
566      }
567      File dir = Paths.get(bpidDir.getAbsolutePath(), "current", "finalized",
568                    state.curFinalizedDir, state.curFinalizedSubDir).toFile();
569      List<String> entries =
570          IOUtils.listDirectory(dir, BlockFileFilter.INSTANCE);
571      if (entries.size() == 0) {
572        entries = null;
573      } else {
574        Collections.sort(entries);
575      }
576      if (entries == null) {
577        LOG.trace("getSubdirEntries({}, {}): no entries found in {}",
578            storageID, bpid, dir.getAbsolutePath());
579      } else {
580        LOG.trace("getSubdirEntries({}, {}): listed {} entries in {}", 
581            storageID, bpid, entries.size(), dir.getAbsolutePath());
582      }
583      cache = entries;
584      cacheMs = now;
585      return cache;
586    }
587
588    /**
589     * Get the next block.<p/>
590     *
591     * Each volume has a hierarchical structure.<p/>
592     *
593     * <code>
594     * BPID B0
595     *   finalized/
596     *     subdir0
597     *       subdir0
598     *         blk_000
599     *         blk_001
600     *       ...
601     *     subdir1
602     *       subdir0
603     *         ...
604     *   rbw/
605     * </code>
606     *
607     * When we run out of entries at one level of the structure, we search
608     * progressively higher levels.  For example, when we run out of blk_
609     * entries in a subdirectory, we search for the next subdirectory.
610     * And so on.
611     */
612    @Override
613    public ExtendedBlock nextBlock() throws IOException {
614      if (state.atEnd) {
615        return null;
616      }
617      try {
618        while (true) {
619          List<String> entries = getSubdirEntries();
620          if (entries != null) {
621            state.curEntry = nextSorted(entries, state.curEntry);
622            if (state.curEntry == null) {
623              LOG.trace("nextBlock({}, {}): advancing from {} to next " +
624                  "subdirectory.", storageID, bpid, state.curFinalizedSubDir);
625            } else {
626              ExtendedBlock block =
627                  new ExtendedBlock(bpid, Block.filename2id(state.curEntry));
628              LOG.trace("nextBlock({}, {}): advancing to {}",
629                  storageID, bpid, block);
630              return block;
631            }
632          }
633          state.curFinalizedSubDir = getNextFinalizedSubDir();
634          if (state.curFinalizedSubDir == null) {
635            state.curFinalizedDir = getNextFinalizedDir();
636            if (state.curFinalizedDir == null) {
637              state.atEnd = true;
638              return null;
639            }
640          }
641        }
642      } catch (IOException e) {
643        state.atEnd = true;
644        LOG.error("nextBlock({}, {}): I/O error", storageID, bpid, e);
645        throw e;
646      }
647    }

661    @Override
662    public void save() throws IOException {
663      state.lastSavedMs = Time.now();
664      boolean success = false;
665      ObjectMapper mapper = new ObjectMapper();
666      try (BufferedWriter writer = new BufferedWriter(new OutputStreamWriter(
667                new FileOutputStream(getTempSaveFile(), false), "UTF-8"))) {
668        mapper.writerWithDefaultPrettyPrinter().writeValue(writer, state);
669        success = true;
670      } finally {
671        if (!success) {
672          if (getTempSaveFile().delete()) {
673            LOG.debug("save({}, {}): error deleting temporary file.",
674                storageID, bpid);
675          }
676        }
677      }
678      Files.move(getTempSaveFile().toPath(), getSaveFile().toPath(),
679          StandardCopyOption.ATOMIC_MOVE);
680      if (LOG.isTraceEnabled()) {
681        LOG.trace("save({}, {}): saved {}", storageID, bpid,
682            mapper.writerWithDefaultPrettyPrinter().writeValueAsString(state));
683      }
684    }

687      ObjectMapper mapper = new ObjectMapper();
688      File file = getSaveFile();
689      this.state = mapper.reader(BlockIteratorState.class).readValue(file);
690      LOG.trace("load({}, {}): loaded iterator {} from {}: {}", storageID,
691          bpid, name, file.getAbsoluteFile(),
692          mapper.writerWithDefaultPrettyPrinter().writeValueAsString(state));
693    }

764   */

766                         File f, long bytesReservedForRbw)
767      throws IOException {
768    releaseReservedSpace(bytesReservedForRbw);
769    return getBlockPoolSlice(bpid).addBlock(b, f);
770  }

826  boolean isBPDirEmpty(String bpid) throws IOException {
827    File volumeCurrentDir = this.getCurrentDir();
828    File bpDir = new File(volumeCurrentDir, bpid);
829    File bpCurrentDir = new File(bpDir, DataStorage.STORAGE_DIR_CURRENT);
830    File finalizedDir = new File(bpCurrentDir,
831        DataStorage.STORAGE_DIR_FINALIZED);
832    File rbwDir = new File(bpCurrentDir, DataStorage.STORAGE_DIR_RBW);
833    if (finalizedDir.exists() && !DatanodeUtil.dirNoFilesRecursive(
834        finalizedDir)) {
835      return false;
836    }
837    if (rbwDir.exists() && FileUtil.list(rbwDir).length != 0) {
838      return false;
839    }
840    return true;
841  }

843  void deleteBPDirectories(String bpid, boolean force) throws IOException {
844    File volumeCurrentDir = this.getCurrentDir();
845    File bpDir = new File(volumeCurrentDir, bpid);
846    if (!bpDir.isDirectory()) {
847      // nothing to be deleted
848      return;
849    }
850    File tmpDir = new File(bpDir, DataStorage.STORAGE_DIR_TMP); 
851    File bpCurrentDir = new File(bpDir, DataStorage.STORAGE_DIR_CURRENT);
852    File finalizedDir = new File(bpCurrentDir,
853        DataStorage.STORAGE_DIR_FINALIZED);
854    File lazypersistDir = new File(bpCurrentDir,
855        DataStorage.STORAGE_DIR_LAZY_PERSIST);
856    File rbwDir = new File(bpCurrentDir, DataStorage.STORAGE_DIR_RBW);
857    if (force) {
858      FileUtil.fullyDelete(bpDir);
859    } else {
860      if (!rbwDir.delete()) {
861        throw new IOException("Failed to delete " + rbwDir);
862      }
863      if (!DatanodeUtil.dirNoFilesRecursive(finalizedDir) ||
864          !FileUtil.fullyDelete(finalizedDir)) {
865        throw new IOException("Failed to delete " + finalizedDir);
866      }
867      if (lazypersistDir.exists() &&
868        ((!DatanodeUtil.dirNoFilesRecursive(lazypersistDir) ||
869          !FileUtil.fullyDelete(lazypersistDir)))) {
870        throw new IOException("Failed to delete " + lazypersistDir);
871      }
872      FileUtil.fullyDelete(tmpDir);
873      for (File f : FileUtil.listFiles(bpCurrentDir)) {
874        if (!f.delete()) {
875          throw new IOException("Failed to delete " + f);
876        }
877      }
878      if (!bpCurrentDir.delete()) {
879        throw new IOException("Failed to delete " + bpCurrentDir);
880      }
881      for (File f : FileUtil.listFiles(bpDir)) {
882        if (!f.delete()) {
883          throw new IOException("Failed to delete " + f);
884        }
885      }
886      if (!bpDir.delete()) {
887        throw new IOException("Failed to delete " + bpDir);
888      }
889    }
890  }
