151    List<StorageReport> reports;
152    synchronized (statsLock) {
153      List<FsVolumeImpl> curVolumes = getVolumes();
154      reports = new ArrayList<>(curVolumes.size());
155      for (FsVolumeImpl volume : curVolumes) {
156        try (FsVolumeReference ref = volume.obtainReference()) {
157          StorageReport sr = new StorageReport(volume.toDatanodeStorage(),
158              false,
159              volume.getCapacity(),
160              volume.getDfsUsed(),
161              volume.getAvailable(),
162              volume.getBlockPoolUsed(bpid));
163          reports.add(sr);
164        } catch (ClosedChannelException e) {
165          continue;
166        }
167      }
168    }
169
170    return reports.toArray(new StorageReport[reports.size()]);
171  }

256  /**
257   * An FSDataset has a directory where it loads its data files.
258   */
259  FsDatasetImpl(DataNode datanode, DataStorage storage, Configuration conf
260      ) throws IOException {
261    this.fsRunning = true;
262    this.datanode = datanode;
263    this.dataStorage = storage;
264    this.conf = conf;
265    // The number of volumes required for operation is the total number 
266    // of volumes minus the number of failed volumes we can tolerate.
267    final int volFailuresTolerated =
268      conf.getInt(DFSConfigKeys.DFS_DATANODE_FAILED_VOLUMES_TOLERATED_KEY,
269                  DFSConfigKeys.DFS_DATANODE_FAILED_VOLUMES_TOLERATED_DEFAULT);
270
271    String[] dataDirs = conf.getTrimmedStrings(DFSConfigKeys.DFS_DATANODE_DATA_DIR_KEY);
272    Collection<StorageLocation> dataLocations = DataNode.getStorageLocations(conf);
273    List<VolumeFailureInfo> volumeFailureInfos = getInitialVolumeFailureInfos(
274        dataLocations, storage);
275
276    int volsConfigured = (dataDirs == null) ? 0 : dataDirs.length;
277    int volsFailed = volumeFailureInfos.size();
278    this.validVolsRequired = volsConfigured - volFailuresTolerated;
279
280    if (volFailuresTolerated < 0 || volFailuresTolerated >= volsConfigured) {
281      throw new DiskErrorException("Invalid volume failure "
282          + " config value: " + volFailuresTolerated);
283    }
284    if (volsFailed > volFailuresTolerated) {
285      throw new DiskErrorException("Too many failed volumes - "
286          + "current valid volumes: " + storage.getNumStorageDirs() 
287          + ", volumes configured: " + volsConfigured 
288          + ", volumes failed: " + volsFailed
289          + ", volume failures tolerated: " + volFailuresTolerated);
290    }
291
292    storageMap = new ConcurrentHashMap<String, DatanodeStorage>();
293    volumeMap = new ReplicaMap(this);
294    ramDiskReplicaTracker = RamDiskReplicaTracker.getInstance(conf, this);
295
296    @SuppressWarnings("unchecked")
297    final VolumeChoosingPolicy<FsVolumeImpl> blockChooserImpl =
298        ReflectionUtils.newInstance(conf.getClass(
299            DFSConfigKeys.DFS_DATANODE_FSDATASET_VOLUME_CHOOSING_POLICY_KEY,
300            RoundRobinVolumeChoosingPolicy.class,
301            VolumeChoosingPolicy.class), conf);
302    volumes = new FsVolumeList(volumeFailureInfos, datanode.getBlockScanner(),
303        blockChooserImpl);
304    asyncDiskService = new FsDatasetAsyncDiskService(datanode, this);
305    asyncLazyPersistService = new RamDiskAsyncLazyPersistService(datanode);
306    deletingBlock = new HashMap<String, Set<Long>>();
307
308    for (int idx = 0; idx < storage.getNumStorageDirs(); idx++) {
309      addVolume(dataLocations, storage.getStorageDir(idx));
310    }
311    setupAsyncLazyPersistThreads();
312
313    cacheManager = new FsDatasetCache(this);
314
315    // Start the lazy writer once we have built the replica maps.
316    lazyWriter = new Daemon(new LazyWriter(conf));
317    lazyWriter.start();
318    registerMBean(datanode.getDatanodeUuid());
319    localFS = FileSystem.getLocal(conf);
320    blockPinningEnabled = conf.getBoolean(
321      DFSConfigKeys.DFS_DATANODE_BLOCK_PINNING_ENABLED,
322      DFSConfigKeys.DFS_DATANODE_BLOCK_PINNING_ENABLED_DEFAULT);
323  }

325  /**
326   * Gets initial volume failure information for all volumes that failed
327   * immediately at startup.  The method works by determining the set difference
328   * between all configured storage locations and the actual storage locations in
329   * use after attempting to put all of them into service.
330   *
331   * @return each storage location that has failed
332   */
333  private static List<VolumeFailureInfo> getInitialVolumeFailureInfos(
334      Collection<StorageLocation> dataLocations, DataStorage storage) {
335    Set<String> failedLocationSet = Sets.newHashSetWithExpectedSize(
336        dataLocations.size());
337    for (StorageLocation sl: dataLocations) {
338      failedLocationSet.add(sl.getFile().getAbsolutePath());
339    }
340    for (Iterator<Storage.StorageDirectory> it = storage.dirIterator();
341         it.hasNext(); ) {
342      Storage.StorageDirectory sd = it.next();
343      failedLocationSet.remove(sd.getRoot().getAbsolutePath());
344    }
345    List<VolumeFailureInfo> volumeFailureInfos = Lists.newArrayListWithCapacity(
346        failedLocationSet.size());
347    long failureDate = Time.now();
348    for (String failedStorageLocation: failedLocationSet) {
349      volumeFailureInfos.add(new VolumeFailureInfo(failedStorageLocation,
350          failureDate));
351    }
352    return volumeFailureInfos;
353  }

355  private void addVolume(Collection<StorageLocation> dataLocations,
356      Storage.StorageDirectory sd) throws IOException {
357    final File dir = sd.getCurrentDir();
358    final StorageType storageType =
359        getStorageTypeFromLocations(dataLocations, sd.getRoot());
360
361    // If IOException raises from FsVolumeImpl() or getVolumeMap(), there is
362    // nothing needed to be rolled back to make various data structures, e.g.,
363    // storageMap and asyncDiskService, consistent.
364    FsVolumeImpl fsVolume = new FsVolumeImpl(
365        this, sd.getStorageUuid(), dir, this.conf, storageType);
366    FsVolumeReference ref = fsVolume.obtainReference();
367    ReplicaMap tempVolumeMap = new ReplicaMap(this);
368    fsVolume.getVolumeMap(tempVolumeMap, ramDiskReplicaTracker);
369
370    synchronized (this) {
371      volumeMap.addAll(tempVolumeMap);
372      storageMap.put(sd.getStorageUuid(),
373          new DatanodeStorage(sd.getStorageUuid(),
374              DatanodeStorage.State.NORMAL,
375              storageType));
376      asyncDiskService.addVolume(sd.getCurrentDir());
377      volumes.addVolume(ref);
378    }
379
380    LOG.info("Added volume - " + dir + ", StorageType: " + storageType);
381  }

389  @Override
390  public void addVolume(final StorageLocation location,
391      final List<NamespaceInfo> nsInfos)
392      throws IOException {
393    final File dir = location.getFile();
394
395    // Prepare volume in DataStorage
396    final DataStorage.VolumeBuilder builder;
397    try {
398      builder = dataStorage.prepareVolume(datanode, location.getFile(), nsInfos);
399    } catch (IOException e) {
400      volumes.addVolumeFailureInfo(new VolumeFailureInfo(
401          location.getFile().getAbsolutePath(), Time.now()));
402      throw e;
403    }
404
405    final Storage.StorageDirectory sd = builder.getStorageDirectory();
406
407    StorageType storageType = location.getStorageType();
408    final FsVolumeImpl fsVolume =
409        createFsVolume(sd.getStorageUuid(), sd.getCurrentDir(), storageType);
410    final ReplicaMap tempVolumeMap = new ReplicaMap(fsVolume);
411    ArrayList<IOException> exceptions = Lists.newArrayList();
412
413    for (final NamespaceInfo nsInfo : nsInfos) {
414      String bpid = nsInfo.getBlockPoolID();
415      try {
416        fsVolume.addBlockPool(bpid, this.conf);
417        fsVolume.getVolumeMap(bpid, tempVolumeMap, ramDiskReplicaTracker);
418      } catch (IOException e) {
419        LOG.warn("Caught exception when adding " + fsVolume +
420            ". Will throw later.", e);
421        exceptions.add(e);
422      }
423    }
424    if (!exceptions.isEmpty()) {
425      try {
426        sd.unlock();
427      } catch (IOException e) {
428        exceptions.add(e);
429      }
430      throw MultipleIOException.createIOException(exceptions);
431    }
432
433    final FsVolumeReference ref = fsVolume.obtainReference();
434    setupAsyncLazyPersistThread(fsVolume);
435
436    builder.build();
437    synchronized (this) {
438      volumeMap.addAll(tempVolumeMap);
439      storageMap.put(sd.getStorageUuid(),
440          new DatanodeStorage(sd.getStorageUuid(),
441              DatanodeStorage.State.NORMAL,
442              storageType));
443      asyncDiskService.addVolume(sd.getCurrentDir());
444      volumes.addVolume(ref);
445    }
446    LOG.info("Added volume - " + dir + ", StorageType: " + storageType);
447  }

449  /**
450   * Removes a set of volumes from FsDataset.
451   * @param volumesToRemove a set of absolute root path of each volume.
452   * @param clearFailure set true to clear failure information.
453   *
454   * DataNode should call this function before calling
455   * {@link DataStorage#removeVolumes(java.util.Collection)}.
456   */
457  @Override
458  public synchronized void removeVolumes(
459      Set<File> volumesToRemove, boolean clearFailure) {
460    // Make sure that all volumes are absolute path.
461    for (File vol : volumesToRemove) {
462      Preconditions.checkArgument(vol.isAbsolute(),
463          String.format("%s is not absolute path.", vol.getPath()));
464    }
465    for (int idx = 0; idx < dataStorage.getNumStorageDirs(); idx++) {
466      Storage.StorageDirectory sd = dataStorage.getStorageDir(idx);
467      final File absRoot = sd.getRoot().getAbsoluteFile();
468      if (volumesToRemove.contains(absRoot)) {
469        LOG.info("Removing " + absRoot + " from FsDataset.");
470
471        // Disable the volume from the service.
472        asyncDiskService.removeVolume(sd.getCurrentDir());
473        volumes.removeVolume(absRoot, clearFailure);
474
475        // Removed all replica information for the blocks on the volume. Unlike
476        // updating the volumeMap in addVolume(), this operation does not scan
477        // disks.
478        for (String bpid : volumeMap.getBlockPoolList()) {
479          for (Iterator<ReplicaInfo> it = volumeMap.replicas(bpid).iterator();
480               it.hasNext(); ) {
481            ReplicaInfo block = it.next();
482            final File absBasePath =
483                new File(block.getVolume().getBasePath()).getAbsoluteFile();
484            if (absBasePath.equals(absRoot)) {
485              invalidate(bpid, block);
486              it.remove();
487            }
488          }
489        }
490
491        storageMap.remove(sd.getStorageUuid());
492      }
493    }
494    setupAsyncLazyPersistThreads();
495  }

598    VolumeFailureInfo[] infos = volumes.getVolumeFailureInfos();
599    if (infos.length == 0) {
600      return null;
601    }
602    List<String> failedStorageLocations = Lists.newArrayListWithCapacity(
603        infos.length);
604    long lastVolumeFailureDate = 0;
605    long estimatedCapacityLostTotal = 0;
606    for (VolumeFailureInfo info: infos) {
607      failedStorageLocations.add(info.getFailedStorageLocation());
608      long failureDate = info.getFailureDate();
609      if (failureDate > lastVolumeFailureDate) {
610        lastVolumeFailureDate = failureDate;
611      }
612      estimatedCapacityLostTotal += info.getEstimatedCapacityLost();
613    }
614    return new VolumeFailureSummary(
615        failedStorageLocations.toArray(new String[failedStorageLocations.size()]),
616        lastVolumeFailureDate, estimatedCapacityLostTotal);
617  }

851   * Move block files from one storage to another storage.
852   * @return Returns the Old replicaInfo
853   * @throws IOException
854   */
855  @Override
856  public ReplicaInfo moveBlockAcrossStorage(ExtendedBlock block,
857      StorageType targetStorageType) throws IOException {
858    ReplicaInfo replicaInfo = getReplicaInfo(block);
859    if (replicaInfo.getState() != ReplicaState.FINALIZED) {
860      throw new ReplicaNotFoundException(
861          ReplicaNotFoundException.UNFINALIZED_REPLICA + block);
862    }
863    if (replicaInfo.getNumBytes() != block.getNumBytes()) {
864      throw new IOException("Corrupted replica " + replicaInfo
865          + " with a length of " + replicaInfo.getNumBytes()
866          + " expected length is " + block.getNumBytes());
867    }
868    if (replicaInfo.getVolume().getStorageType() == targetStorageType) {
869      throw new ReplicaAlreadyExistsException("Replica " + replicaInfo
870          + " already exists on storage " + targetStorageType);
871    }
872
873    if (replicaInfo.isOnTransientStorage()) {
874      // Block movement from RAM_DISK will be done by LazyPersist mechanism
875      throw new IOException("Replica " + replicaInfo
876          + " cannot be moved from storageType : "
877          + replicaInfo.getVolume().getStorageType());
878    }
879
880    try (FsVolumeReference volumeRef = volumes.getNextVolume(
881        targetStorageType, block.getNumBytes())) {
882      File oldBlockFile = replicaInfo.getBlockFile();
883      File oldMetaFile = replicaInfo.getMetaFile();
884      FsVolumeImpl targetVolume = (FsVolumeImpl) volumeRef.getVolume();
885      // Copy files to temp dir first
886      File[] blockFiles = copyBlockFiles(block.getBlockId(),
887          block.getGenerationStamp(), oldMetaFile, oldBlockFile,
888          targetVolume.getTmpDir(block.getBlockPoolId()),
889          replicaInfo.isOnTransientStorage());
890
891      ReplicaInfo newReplicaInfo = new ReplicaInPipeline(
892          replicaInfo.getBlockId(), replicaInfo.getGenerationStamp(),
893          targetVolume, blockFiles[0].getParentFile(), 0);
894      newReplicaInfo.setNumBytes(blockFiles[1].length());
895      // Finalize the copied files
896      newReplicaInfo = finalizeReplica(block.getBlockPoolId(), newReplicaInfo);
897
898      removeOldReplica(replicaInfo, newReplicaInfo, oldBlockFile, oldMetaFile,
899          oldBlockFile.length(), oldMetaFile.length(), block.getBlockPoolId());
900    }
901
902    // Replace the old block if any to reschedule the scanning.
903    return replicaInfo;
904  }

906  /**
907   * Compute and store the checksum for a block file that does not already have
908   * its checksum computed.
909   *
910   * @param srcMeta source meta file, containing only the checksum header, not a
911   *     calculated checksum
912   * @param dstMeta destination meta file, into which this method will write a
913   *     full computed checksum
914   * @param blockFile block file for which the checksum will be computed
915   * @throws IOException
916   */
917  private static void computeChecksum(File srcMeta, File dstMeta, File blockFile)
918      throws IOException {
919    final DataChecksum checksum = BlockMetadataHeader.readDataChecksum(srcMeta);
920    final byte[] data = new byte[1 << 16];
921    final byte[] crcs = new byte[checksum.getChecksumSize(data.length)];
922
923    DataOutputStream metaOut = null;
924    try {
925      File parentFile = dstMeta.getParentFile();
926      if (parentFile != null) {
927        if (!parentFile.mkdirs() && !parentFile.isDirectory()) {
928          throw new IOException("Destination '" + parentFile
929              + "' directory cannot be created");
930        }
931      }
932      metaOut = new DataOutputStream(new BufferedOutputStream(
933          new FileOutputStream(dstMeta), HdfsConstants.SMALL_BUFFER_SIZE));
934      BlockMetadataHeader.writeHeader(metaOut, checksum);
935
936      int offset = 0;
937      try (InputStream dataIn = isNativeIOAvailable ?
938          NativeIO.getShareDeleteFileInputStream(blockFile) :
939          new FileInputStream(blockFile)) {
940
941        for (int n; (n = dataIn.read(data, offset, data.length - offset)) != -1; ) {
942          if (n > 0) {
943            n += offset;
944            offset = n % checksum.getBytesPerChecksum();
945            final int length = n - offset;
946
947            if (length > 0) {
948              checksum.calculateChunkedSums(data, 0, length, crcs, 0);
949              metaOut.write(crcs, 0, checksum.getChecksumSize(length));
950
951              System.arraycopy(data, length, data, 0, offset);
952            }
953          }
954        }
955      }
956
957      // calculate and write the last crc
958      checksum.calculateChunkedSums(data, 0, offset, crcs, 0);
959      metaOut.write(crcs, 0, 4);
960    } finally {
961      IOUtils.cleanup(LOG, metaOut);
962    }
963  }

965  static private void truncateBlock(File blockFile, File metaFile,
966      long oldlen, long newlen) throws IOException {
967    LOG.info("truncateBlock: blockFile=" + blockFile
968        + ", metaFile=" + metaFile
969        + ", oldlen=" + oldlen
970        + ", newlen=" + newlen);
971
972    if (newlen == oldlen) {
973      return;
974    }
975    if (newlen > oldlen) {
976      throw new IOException("Cannot truncate block to from oldlen (=" + oldlen
977          + ") to newlen (=" + newlen + ")");
978    }
979
980    DataChecksum dcs = BlockMetadataHeader.readHeader(metaFile).getChecksum(); 
981    int checksumsize = dcs.getChecksumSize();
982    int bpc = dcs.getBytesPerChecksum();
983    long n = (newlen - 1)/bpc + 1;
984    long newmetalen = BlockMetadataHeader.getHeaderSize() + n*checksumsize;
985    long lastchunkoffset = (n - 1)*bpc;
986    int lastchunksize = (int)(newlen - lastchunkoffset); 
987    byte[] b = new byte[Math.max(lastchunksize, checksumsize)]; 
988
989    RandomAccessFile blockRAF = new RandomAccessFile(blockFile, "rw");
990    try {
991      //truncate blockFile 
992      blockRAF.setLength(newlen);
993 
994      //read last chunk
995      blockRAF.seek(lastchunkoffset);
996      blockRAF.readFully(b, 0, lastchunksize);
997    } finally {
998      blockRAF.close();
999    }
1000
1001    //compute checksum
1002    dcs.update(b, 0, lastchunksize);
1003    dcs.writeValue(b, 0, false);
1004
1005    //update metaFile 
1006    RandomAccessFile metaRAF = new RandomAccessFile(metaFile, "rw");
1007    try {
1008      metaRAF.setLength(newmetalen);
1009      metaRAF.seek(newmetalen - checksumsize);
1010      metaRAF.write(b, 0, checksumsize);
1011    } finally {
1012      metaRAF.close();
1013    }
1014  }

1019      long newGS, long expectedBlockLen) throws IOException {
1020    // If the block was successfully finalized because all packets
1021    // were successfully processed at the Datanode but the ack for
1022    // some of the packets were not received by the client. The client 
1023    // re-opens the connection and retries sending those packets.
1024    // The other reason is that an "append" is occurring to this block.
1025    
1026    // check the validity of the parameter
1027    if (newGS < b.getGenerationStamp()) {
1028      throw new IOException("The new generation stamp " + newGS + 
1029          " should be greater than the replica " + b + "'s generation stamp");
1030    }
1031    ReplicaInfo replicaInfo = getReplicaInfo(b);
1032    LOG.info("Appending to " + replicaInfo);
1033    if (replicaInfo.getState() != ReplicaState.FINALIZED) {
1034      throw new ReplicaNotFoundException(
1035          ReplicaNotFoundException.UNFINALIZED_REPLICA + b);
1036    }
1037    if (replicaInfo.getNumBytes() != expectedBlockLen) {
1038      throw new IOException("Corrupted replica " + replicaInfo + 
1039          " with a length of " + replicaInfo.getNumBytes() + 
1040          " expected length is " + expectedBlockLen);
1041    }
1042
1043    FsVolumeReference ref = replicaInfo.getVolume().obtainReference();
1044    ReplicaBeingWritten replica = null;
1045    try {
1046      replica = append(b.getBlockPoolId(), (FinalizedReplica)replicaInfo, newGS,
1047          b.getNumBytes());
1048    } catch (IOException e) {
1049      IOUtils.cleanup(null, ref);
1050      throw e;
1051    }
1052    return new ReplicaHandler(replica, ref);
1053  }

1055  /** Append to a finalized replica
1056   * Change a finalized replica to be a RBW replica and 
1057   * bump its generation stamp to be the newGS
1058   * 
1059   * @param bpid block pool Id
1060   * @param replicaInfo a finalized replica
1061   * @param newGS new generation stamp
1062   * @param estimateBlockLen estimate generation stamp
1063   * @return a RBW replica
1064   * @throws IOException if moving the replica from finalized directory 
1065   *         to rbw directory fails
1066   */
1067  private synchronized ReplicaBeingWritten append(String bpid,
1068      FinalizedReplica replicaInfo, long newGS, long estimateBlockLen)
1069      throws IOException {
1070    // If the block is cached, start uncaching it.
1071    cacheManager.uncacheBlock(bpid, replicaInfo.getBlockId());
1072    // unlink the finalized replica
1073    replicaInfo.unlinkBlock(1);
1074    
1075    // construct a RBW replica with the new GS
1076    File blkfile = replicaInfo.getBlockFile();
1077    FsVolumeImpl v = (FsVolumeImpl)replicaInfo.getVolume();
1078    if (v.getAvailable() < estimateBlockLen - replicaInfo.getNumBytes()) {
1079      throw new DiskOutOfSpaceException("Insufficient space for appending to "
1080          + replicaInfo);
1081    }
1082    File newBlkFile = new File(v.getRbwDir(bpid), replicaInfo.getBlockName());
1083    File oldmeta = replicaInfo.getMetaFile();
1084    ReplicaBeingWritten newReplicaInfo = new ReplicaBeingWritten(
1085        replicaInfo.getBlockId(), replicaInfo.getNumBytes(), newGS,
1086        v, newBlkFile.getParentFile(), Thread.currentThread(), estimateBlockLen);
1087    File newmeta = newReplicaInfo.getMetaFile();
1088
1089    // rename meta file to rbw directory
1090    if (LOG.isDebugEnabled()) {
1091      LOG.debug("Renaming " + oldmeta + " to " + newmeta);
1092    }
1093    try {
1094      NativeIO.renameTo(oldmeta, newmeta);
1095    } catch (IOException e) {
1096      throw new IOException("Block " + replicaInfo + " reopen failed. " +
1097                            " Unable to move meta file  " + oldmeta +
1098                            " to rbw dir " + newmeta, e);
1099    }
1100
1101    // rename block file to rbw directory
1102    if (LOG.isDebugEnabled()) {
1103      LOG.debug("Renaming " + blkfile + " to " + newBlkFile
1104          + ", file length=" + blkfile.length());
1105    }
1106    try {
1107      NativeIO.renameTo(blkfile, newBlkFile);
1108    } catch (IOException e) {
1109      try {
1110        NativeIO.renameTo(newmeta, oldmeta);
1111      } catch (IOException ex) {
1112        LOG.warn("Cannot move meta file " + newmeta + 
1113            "back to the finalized directory " + oldmeta, ex);
1114      }
1115      throw new IOException("Block " + replicaInfo + " reopen failed. " +
1116                              " Unable to move block file " + blkfile +
1117                              " to rbw dir " + newBlkFile, e);
1118    }
1119    
1120    // Replace finalized replica by a RBW replica in replicas map
1121    volumeMap.add(bpid, newReplicaInfo);
1122    v.reserveSpaceForRbw(estimateBlockLen - replicaInfo.getNumBytes());
1123    return newReplicaInfo;
1124  }

1126  private ReplicaInfo recoverCheck(ExtendedBlock b, long newGS, 
1127      long expectedBlockLen) throws IOException {
1128    ReplicaInfo replicaInfo = getReplicaInfo(b.getBlockPoolId(), b.getBlockId());
1129    
1130    // check state
1131    if (replicaInfo.getState() != ReplicaState.FINALIZED &&
1132        replicaInfo.getState() != ReplicaState.RBW) {
1133      throw new ReplicaNotFoundException(
1134          ReplicaNotFoundException.UNFINALIZED_AND_NONRBW_REPLICA + replicaInfo);
1135    }
1136
1137    // check generation stamp
1138    long replicaGenerationStamp = replicaInfo.getGenerationStamp();
1139    if (replicaGenerationStamp < b.getGenerationStamp() ||
1140        replicaGenerationStamp > newGS) {
1141      throw new ReplicaNotFoundException(
1142          ReplicaNotFoundException.UNEXPECTED_GS_REPLICA + replicaGenerationStamp
1143          + ". Expected GS range is [" + b.getGenerationStamp() + ", " + 
1144          newGS + "].");
1145    }
1146    
1147    // stop the previous writer before check a replica's length
1148    long replicaLen = replicaInfo.getNumBytes();
1149    if (replicaInfo.getState() == ReplicaState.RBW) {
1150      ReplicaBeingWritten rbw = (ReplicaBeingWritten)replicaInfo;
1151      // kill the previous writer
1152      rbw.stopWriter(datanode.getDnConf().getXceiverStopTimeout());
1153      rbw.setWriter(Thread.currentThread());
1154      // check length: bytesRcvd, bytesOnDisk, and bytesAcked should be the same
1155      if (replicaLen != rbw.getBytesOnDisk() 
1156          || replicaLen != rbw.getBytesAcked()) {
1157        throw new ReplicaAlreadyExistsException("RBW replica " + replicaInfo + 
1158            "bytesRcvd(" + rbw.getNumBytes() + "), bytesOnDisk(" + 
1159            rbw.getBytesOnDisk() + "), and bytesAcked(" + rbw.getBytesAcked() +
1160            ") are not the same.");
1161      }
1162    }
1163    
1164    // check block length
1165    if (replicaLen != expectedBlockLen) {
1166      throw new IOException("Corrupted replica " + replicaInfo + 
1167          " with a length of " + replicaLen + 
1168          " expected length is " + expectedBlockLen);
1169    }
1170    
1171    return replicaInfo;
1172  }

1329          minBytesRcvd + ", " + maxBytesRcvd + "].");
1330    }

1334      // Truncate the potentially corrupt portion.

1839      // descriptors (and the client is using ShortCircuitShm), invalidate it.

1858    }

1862        .append(") replica(s):");

1865      }

1867    }

2067  /**
2068   * Reconcile the difference between blocks on the disk and blocks in
2069   * volumeMap
2070   *
2071   * Check the given block for inconsistencies. Look at the
2072   * current state of the block and reconcile the differences as follows:
2073   * <ul>
2074   * <li>If the block file is missing, delete the block from volumeMap</li>
2075   * <li>If the block file exists and the block is missing in volumeMap,
2076   * add the block to volumeMap <li>
2077   * <li>If generation stamp does not match, then update the block with right
2078   * generation stamp</li>
2079   * <li>If the block length in memory does not match the actual block file length
2080   * then mark the block as corrupt and update the block length in memory</li>
2081   * <li>If the file in {@link ReplicaInfo} does not match the file on
2082   * the disk, update {@link ReplicaInfo} with the correct file</li>
2083   * </ul>
2084   *
2085   * @param blockId Block that differs
2086   * @param diskFile Block file on the disk
2087   * @param diskMetaFile Metadata file from on the disk
2088   * @param vol Volume of the block file
2089   */
2090  @Override
2091  public void checkAndUpdate(String bpid, long blockId, File diskFile,
2092      File diskMetaFile, FsVolumeSpi vol) throws IOException {
2093    Block corruptBlock = null;
2094    ReplicaInfo memBlockInfo;
2095    synchronized (this) {
2096      memBlockInfo = volumeMap.get(bpid, blockId);
2097      if (memBlockInfo != null && memBlockInfo.getState() != ReplicaState.FINALIZED) {
2098        // Block is not finalized - ignore the difference
2099        return;
2100      }
2101
2102      final long diskGS = diskMetaFile != null && diskMetaFile.exists() ?
2103          Block.getGenerationStamp(diskMetaFile.getName()) :
2104            GenerationStamp.GRANDFATHER_GENERATION_STAMP;
2105
2106      if (diskFile == null || !diskFile.exists()) {
2107        if (memBlockInfo == null) {
2108          // Block file does not exist and block does not exist in memory
2109          // If metadata file exists then delete it
2110          if (diskMetaFile != null && diskMetaFile.exists()
2111              && diskMetaFile.delete()) {
2112            LOG.warn("Deleted a metadata file without a block "
2113                + diskMetaFile.getAbsolutePath());
2114          }
2115          return;
2116        }
2117        if (!memBlockInfo.getBlockFile().exists()) {
2118          // Block is in memory and not on the disk
2119          // Remove the block from volumeMap
2120          volumeMap.remove(bpid, blockId);
2121          if (vol.isTransientStorage()) {
2122            ramDiskReplicaTracker.discardReplica(bpid, blockId, true);
2123          }
2124          LOG.warn("Removed block " + blockId
2125              + " from memory with missing block file on the disk");
2126          // Finally remove the metadata file
2127          if (diskMetaFile != null && diskMetaFile.exists()
2128              && diskMetaFile.delete()) {
2129            LOG.warn("Deleted a metadata file for the deleted block "
2130                + diskMetaFile.getAbsolutePath());
2131          }
2132        }
2133        return;
2134      }
2135      /*
2136       * Block file exists on the disk
2137       */
2138      if (memBlockInfo == null) {
2139        // Block is missing in memory - add the block to volumeMap
2140        ReplicaInfo diskBlockInfo = new FinalizedReplica(blockId, 
2141            diskFile.length(), diskGS, vol, diskFile.getParentFile());
2142        volumeMap.add(bpid, diskBlockInfo);
2143        if (vol.isTransientStorage()) {
2144          ramDiskReplicaTracker.addReplica(bpid, blockId, (FsVolumeImpl) vol);
2145        }
2146        LOG.warn("Added missing block to memory " + diskBlockInfo);
2147        return;
2148      }
2149      /*
2150       * Block exists in volumeMap and the block file exists on the disk
2151       */
2152      // Compare block files
2153      File memFile = memBlockInfo.getBlockFile();
2154      if (memFile.exists()) {
2155        if (memFile.compareTo(diskFile) != 0) {
2156          if (diskMetaFile.exists()) {
2157            if (memBlockInfo.getMetaFile().exists()) {
2158              // We have two sets of block+meta files. Decide which one to
2159              // keep.
2160              ReplicaInfo diskBlockInfo = new FinalizedReplica(
2161                  blockId, diskFile.length(), diskGS, vol, diskFile.getParentFile());
2162              ((FsVolumeImpl) vol).getBlockPoolSlice(bpid).resolveDuplicateReplicas(
2163                  memBlockInfo, diskBlockInfo, volumeMap);
2164            }
2165          } else {
2166            if (!diskFile.delete()) {
2167              LOG.warn("Failed to delete " + diskFile + ". Will retry on next scan");
2168            }
2169          }
2170        }
2171      } else {
2172        // Block refers to a block file that does not exist.
2173        // Update the block with the file found on the disk. Since the block
2174        // file and metadata file are found as a pair on the disk, update
2175        // the block based on the metadata file found on the disk
2176        LOG.warn("Block file in volumeMap "
2177            + memFile.getAbsolutePath()
2178            + " does not exist. Updating it to the file found during scan "
2179            + diskFile.getAbsolutePath());
2180        memBlockInfo.setDir(diskFile.getParentFile());
2181        memFile = diskFile;
2182
2183        LOG.warn("Updating generation stamp for block " + blockId
2184            + " from " + memBlockInfo.getGenerationStamp() + " to " + diskGS);
2185        memBlockInfo.setGenerationStamp(diskGS);
2186      }
2187
2188      // Compare generation stamp
2189      if (memBlockInfo.getGenerationStamp() != diskGS) {
2190        File memMetaFile = FsDatasetUtil.getMetaFile(diskFile, 
2191            memBlockInfo.getGenerationStamp());
2192        if (memMetaFile.exists()) {
2193          if (memMetaFile.compareTo(diskMetaFile) != 0) {
2194            LOG.warn("Metadata file in memory "
2195                + memMetaFile.getAbsolutePath()
2196                + " does not match file found by scan "
2197                + (diskMetaFile == null? null: diskMetaFile.getAbsolutePath()));
2198          }
2199        } else {
2200          // Metadata file corresponding to block in memory is missing
2201          // If metadata file found during the scan is on the same directory
2202          // as the block file, then use the generation stamp from it
2203          long gs = diskMetaFile != null && diskMetaFile.exists()
2204              && diskMetaFile.getParent().equals(memFile.getParent()) ? diskGS
2205              : GenerationStamp.GRANDFATHER_GENERATION_STAMP;
2206
2207          LOG.warn("Updating generation stamp for block " + blockId
2208              + " from " + memBlockInfo.getGenerationStamp() + " to " + gs);
2209
2210          memBlockInfo.setGenerationStamp(gs);
2211        }
2212      }
2213
2214      // Compare block size
2215      if (memBlockInfo.getNumBytes() != memFile.length()) {
2216        // Update the length based on the block file
2217        corruptBlock = new Block(memBlockInfo);
2218        LOG.warn("Updating size of block " + blockId + " from "
2219            + memBlockInfo.getNumBytes() + " to " + memFile.length());
2220        memBlockInfo.setNumBytes(memFile.length());
2221      }
2222    }
2223
2224    // Send corrupt block report outside the lock
2225    if (corruptBlock != null) {
2226      LOG.warn("Reporting the block " + corruptBlock
2227          + " as corrupt due to length mismatch");
2228      try {
2229        datanode.reportBadBlocks(new ExtendedBlock(bpid, corruptBlock));  
2230      } catch (IOException e) {
2231        LOG.warn("Failed to repot bad block " + corruptBlock, e);
2232      }
2233    }
2234  }

2768  class LazyWriter implements Runnable {
2769    private volatile boolean shouldRun = true;
2770    final int checkpointerInterval;
2771    final float lowWatermarkFreeSpacePercentage;
2772    final long lowWatermarkFreeSpaceBytes;
2773
2774
2775    public LazyWriter(Configuration conf) {
2776      this.checkpointerInterval = conf.getInt(
2777          DFSConfigKeys.DFS_DATANODE_LAZY_WRITER_INTERVAL_SEC,
2778          DFSConfigKeys.DFS_DATANODE_LAZY_WRITER_INTERVAL_DEFAULT_SEC);
2779      this.lowWatermarkFreeSpacePercentage = conf.getFloat(
2780          DFSConfigKeys.DFS_DATANODE_RAM_DISK_LOW_WATERMARK_PERCENT,
2781          DFSConfigKeys.DFS_DATANODE_RAM_DISK_LOW_WATERMARK_PERCENT_DEFAULT);
2782      this.lowWatermarkFreeSpaceBytes = conf.getLong(
2783          DFSConfigKeys.DFS_DATANODE_RAM_DISK_LOW_WATERMARK_BYTES,
2784          DFSConfigKeys.DFS_DATANODE_RAM_DISK_LOW_WATERMARK_BYTES_DEFAULT);
2785    }
2786
2787    /**
2788     * Checkpoint a pending replica to persistent storage now.
2789     * If we fail then move the replica to the end of the queue.
2790     * @return true if there is more work to be done, false otherwise.
2791     */
2792    private boolean saveNextReplica() {
2793      RamDiskReplica block = null;
2794      FsVolumeReference targetReference;
2795      FsVolumeImpl targetVolume;
2796      ReplicaInfo replicaInfo;
2797      boolean succeeded = false;
2798
2799      try {
2800        block = ramDiskReplicaTracker.dequeueNextReplicaToPersist();
2801        if (block != null) {
2802          synchronized (FsDatasetImpl.this) {
2803            replicaInfo = volumeMap.get(block.getBlockPoolId(), block.getBlockId());
2804
2805            // If replicaInfo is null, the block was either deleted before
2806            // it could be checkpointed or it is already on persistent storage.
2807            // This can occur if a second replica on persistent storage was found
2808            // after the lazy write was scheduled.
2809            if (replicaInfo != null &&
2810                replicaInfo.getVolume().isTransientStorage()) {
2811              // Pick a target volume to persist the block.
2812              targetReference = volumes.getNextVolume(
2813                  StorageType.DEFAULT, replicaInfo.getNumBytes());
2814              targetVolume = (FsVolumeImpl) targetReference.getVolume();
2815
2816              ramDiskReplicaTracker.recordStartLazyPersist(
2817                  block.getBlockPoolId(), block.getBlockId(), targetVolume);
2818
2819              if (LOG.isDebugEnabled()) {
2820                LOG.debug("LazyWriter: Start persisting RamDisk block:"
2821                    + " block pool Id: " + block.getBlockPoolId()
2822                    + " block id: " + block.getBlockId()
2823                    + " on target volume " + targetVolume);
2824              }
2825
2826              asyncLazyPersistService.submitLazyPersistTask(
2827                  block.getBlockPoolId(), block.getBlockId(),
2828                  replicaInfo.getGenerationStamp(), block.getCreationTime(),
2829                  replicaInfo.getMetaFile(), replicaInfo.getBlockFile(),
2830                  targetReference);
2831            }
2832          }
2833        }
2834        succeeded = true;
2835      } catch(IOException ioe) {
2836        LOG.warn("Exception saving replica " + block, ioe);
2837      } finally {
2838        if (!succeeded && block != null) {
2839          LOG.warn("Failed to save replica " + block + ". re-enqueueing it.");
2840          onFailLazyPersist(block.getBlockPoolId(), block.getBlockId());
2841        }
2842      }
2843      return succeeded;
2844    }

2846    private boolean transientFreeSpaceBelowThreshold() throws IOException {
2847      long free = 0;
2848      long capacity = 0;
2849      float percentFree = 0.0f;
2850
2851      // Don't worry about fragmentation for now. We don't expect more than one
2852      // transient volume per DN.
2853      for (FsVolumeImpl v : getVolumes()) {
2854        try (FsVolumeReference ref = v.obtainReference()) {
2855          if (v.isTransientStorage()) {
2856            capacity += v.getCapacity();
2857            free += v.getAvailable();
2858          }
2859        } catch (ClosedChannelException e) {
2860          // ignore.
2861        }
2862      }
2863
2864      if (capacity == 0) {
2865        return false;
2866      }
2867
2868      percentFree = (float) ((double)free * 100 / capacity);
2869      return (percentFree < lowWatermarkFreeSpacePercentage) ||
2870          (free < lowWatermarkFreeSpaceBytes);
2871    }

2873    /**
2874     * Attempt to evict one or more transient block replicas we have at least
2875     * spaceNeeded bytes free.
2876     */
2877    private void evictBlocks() throws IOException {
2878      int iterations = 0;
2879
2880      while (iterations++ < MAX_BLOCK_EVICTIONS_PER_ITERATION &&
2881             transientFreeSpaceBelowThreshold()) {
2882        RamDiskReplica replicaState = ramDiskReplicaTracker.getNextCandidateForEviction();
2883
2884        if (replicaState == null) {
2885          break;
2886        }
2887
2888        if (LOG.isDebugEnabled()) {
2889          LOG.debug("Evicting block " + replicaState);
2890        }
2891
2892        ReplicaInfo replicaInfo, newReplicaInfo;
2893        File blockFile, metaFile;
2894        long blockFileUsed, metaFileUsed;
2895        final String bpid = replicaState.getBlockPoolId();
2896
2897        synchronized (FsDatasetImpl.this) {
2898          replicaInfo = getReplicaInfo(replicaState.getBlockPoolId(), replicaState.getBlockId());
2899          Preconditions.checkState(replicaInfo.getVolume().isTransientStorage());
2900          blockFile = replicaInfo.getBlockFile();
2901          metaFile = replicaInfo.getMetaFile();
2902          blockFileUsed = blockFile.length();
2903          metaFileUsed = metaFile.length();
2904          ramDiskReplicaTracker.discardReplica(replicaState.getBlockPoolId(),
2905              replicaState.getBlockId(), false);
2906
2907          // Move the replica from lazyPersist/ to finalized/ on target volume
2908          BlockPoolSlice bpSlice =
2909              replicaState.getLazyPersistVolume().getBlockPoolSlice(bpid);
2910          File newBlockFile = bpSlice.activateSavedReplica(
2911              replicaInfo, replicaState.getSavedMetaFile(),
2912              replicaState.getSavedBlockFile());
2913
2914          newReplicaInfo =
2915              new FinalizedReplica(replicaInfo.getBlockId(),
2916                                   replicaInfo.getBytesOnDisk(),
2917                                   replicaInfo.getGenerationStamp(),
2918                                   replicaState.getLazyPersistVolume(),
2919                                   newBlockFile.getParentFile());
2920
2921          // Update the volumeMap entry.
2922          volumeMap.add(bpid, newReplicaInfo);
2923
2924          // Update metrics
2925          datanode.getMetrics().incrRamDiskBlocksEvicted();
2926          datanode.getMetrics().addRamDiskBlocksEvictionWindowMs(
2927              Time.monotonicNow() - replicaState.getCreationTime());
2928          if (replicaState.getNumReads() == 0) {
2929            datanode.getMetrics().incrRamDiskBlocksEvictedWithoutRead();
2930          }
2931        }
2932
2933        removeOldReplica(replicaInfo, newReplicaInfo, blockFile, metaFile,
2934            blockFileUsed, metaFileUsed, bpid);
2935      }
2936    }

2938    @Override
2939    public void run() {
2940      int numSuccessiveFailures = 0;
2941
2942      while (fsRunning && shouldRun) {
2943        try {
2944          numSuccessiveFailures = saveNextReplica() ? 0 : (numSuccessiveFailures + 1);
2945          evictBlocks();
2946
2947          // Sleep if we have no more work to do or if it looks like we are not
2948          // making any forward progress. This is to ensure that if all persist
2949          // operations are failing we don't keep retrying them in a tight loop.
2950          if (numSuccessiveFailures >= ramDiskReplicaTracker.numReplicasNotPersisted()) {
2951            Thread.sleep(checkpointerInterval * 1000);
2952            numSuccessiveFailures = 0;
2953          }
2954        } catch (InterruptedException e) {
2955          LOG.info("LazyWriter was interrupted, exiting");
2956          break;
2957        } catch (Exception e) {
2958          LOG.warn("Ignoring exception in LazyWriter:", e);
2959        }
2960      }
2961    }
2962
2963    public void stop() {
2964      shouldRun = false;
2965    }
2966  }
