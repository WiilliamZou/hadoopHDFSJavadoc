058/**
059 * A block pool slice represents a portion of a block pool stored on a volume.  
060 * Taken together, all BlockPoolSlices sharing a block pool ID across a 
061 * cluster represent a single block pool.
062 * 
063 * This class is synchronized by {@link FsVolumeImpl}.
064 */
065class BlockPoolSlice {
066  static final Log LOG = LogFactory.getLog(BlockPoolSlice.class);

068  private final String bpid;
069  private final FsVolumeImpl volume; // volume to which this BlockPool belongs to
070  private final File currentDir; // StorageDirectory/current/bpid/current

073  private final File lazypersistDir;
074  private final File rbwDir; // directory store RBW replica
075  private final File tmpDir; // directory store Temporary replica
076  private static final String DU_CACHE_FILE = "dfsUsed";
077  private volatile boolean dfsUsedSaved = false;
078  private static final int SHUTDOWN_HOOK_PRIORITY = 30;
079  private final boolean deleteDuplicateReplicas;

084  /**
085   * Create a blook pool slice 
086   * @param bpid Block pool Id
087   * @param volume {@link FsVolumeImpl} to which this BlockPool belongs to
088   * @param bpDir directory corresponding to the BlockPool
089   * @param conf configuration
090   * @throws IOException
091   */
092  BlockPoolSlice(String bpid, FsVolumeImpl volume, File bpDir,
093      Configuration conf) throws IOException {
094    this.bpid = bpid;
095    this.volume = volume;
096    this.currentDir = new File(bpDir, DataStorage.STORAGE_DIR_CURRENT); 
097    this.finalizedDir = new File(
098        currentDir, DataStorage.STORAGE_DIR_FINALIZED);
099    this.lazypersistDir = new File(currentDir, DataStorage.STORAGE_DIR_LAZY_PERSIST);
100    if (!this.finalizedDir.exists()) {
101      if (!this.finalizedDir.mkdirs()) {
102        throw new IOException("Failed to mkdirs " + this.finalizedDir);
103      }
104    }
105
106    this.deleteDuplicateReplicas = conf.getBoolean(
107        DFSConfigKeys.DFS_DATANODE_DUPLICATE_REPLICA_DELETION,
108        DFSConfigKeys.DFS_DATANODE_DUPLICATE_REPLICA_DELETION_DEFAULT);
109
110    // Files that were being written when the datanode was last shutdown
111    // are now moved back to the data directory. It is possible that
112    // in the future, we might want to do some sort of datanode-local
113    // recovery for these blocks. For example, crc validation.
114    //
115    this.tmpDir = new File(bpDir, DataStorage.STORAGE_DIR_TMP);
116    if (tmpDir.exists()) {
117      FileUtil.fullyDelete(tmpDir);
118    }
119    this.rbwDir = new File(currentDir, DataStorage.STORAGE_DIR_RBW);
120    final boolean supportAppends = conf.getBoolean(
121        DFSConfigKeys.DFS_SUPPORT_APPEND_KEY,
122        DFSConfigKeys.DFS_SUPPORT_APPEND_DEFAULT);
123    if (rbwDir.exists() && !supportAppends) {
124      FileUtil.fullyDelete(rbwDir);
125    }
126    if (!rbwDir.mkdirs()) {  // create rbw directory if not exist
127      if (!rbwDir.isDirectory()) {
128        throw new IOException("Mkdirs failed to create " + rbwDir.toString());
129      }
130    }
131    if (!tmpDir.mkdirs()) {
132      if (!tmpDir.isDirectory()) {
133        throw new IOException("Mkdirs failed to create " + tmpDir.toString());
134      }
135    }
136    // Use cached value initially if available. Or the following call will
137    // block until the initial du command completes.
138    this.dfsUsage = new DU(bpDir, conf, loadDfsUsed());
139    this.dfsUsage.start();
140
141    // Make the dfs usage to be saved during shutdown.
142    ShutdownHookManager.get().addShutdownHook(
143      new Runnable() {
144        @Override
145        public void run() {
146          if (!dfsUsedSaved) {
147            saveDfsUsed();
148          }
149        }
150      }, SHUTDOWN_HOOK_PRIORITY);
151  }

186   /**
187   * Read in the cached DU value and return it if it is less than 600 seconds
188   * old (DU update interval). Slight imprecision of dfsUsed is not critical
189   * and skipping DU can significantly shorten the startup time.
190   * If the cached value is not available or too old, -1 is returned.
191   */
192  long loadDfsUsed() {
193    long cachedDfsUsed;
194    long mtime;
195    Scanner sc;
196
197    try {
198      sc = new Scanner(new File(currentDir, DU_CACHE_FILE), "UTF-8");
199    } catch (FileNotFoundException fnfe) {
200      return -1;
201    }
202
203    try {
204      // Get the recorded dfsUsed from the file.
205      if (sc.hasNextLong()) {
206        cachedDfsUsed = sc.nextLong();
207      } else {
208        return -1;
209      }
210      // Get the recorded mtime from the file.
211      if (sc.hasNextLong()) {
212        mtime = sc.nextLong();
213      } else {
214        return -1;
215      }
216
217      // Return the cached value if mtime is okay.
218      if (mtime > 0 && (Time.now() - mtime < 600000L)) {
219        FsDatasetImpl.LOG.info("Cached dfsUsed found for " + currentDir + ": " +
220            cachedDfsUsed);
221        return cachedDfsUsed;
222      }
223      return -1;
224    } finally {
225      sc.close();
226    }
227  }
228
229  /**
230   * Write the current dfsUsed to the cache file.
231   */
232  void saveDfsUsed() {
233    File outFile = new File(currentDir, DU_CACHE_FILE);
234    if (outFile.exists() && !outFile.delete()) {
235      FsDatasetImpl.LOG.warn("Failed to delete old dfsUsed file in " +
236        outFile.getParent());
237    }
238
239    try {
240      long used = getDfsUsed();
241      try (Writer out = new OutputStreamWriter(
242          new FileOutputStream(outFile), "UTF-8")) {
243        // mtime is written last, so that truncated writes won't be valid.
244        out.write(Long.toString(used) + " " + Long.toString(Time.now()));
245        out.flush();
246      }
247    } catch (IOException ioe) {
248      // If write failed, the volume might be bad. Since the cache file is
249      // not critical, log the error and continue.
250      FsDatasetImpl.LOG.warn("Failed to write dfsUsed to " + outFile, ioe);
251    }
252  }

272  File addBlock(Block b, File f) throws IOException {
273    File blockDir = DatanodeUtil.idToBlockDir(finalizedDir, b.getBlockId());
274    if (!blockDir.exists()) {
275      if (!blockDir.mkdirs()) {
276        throw new IOException("Failed to mkdirs " + blockDir);
277      }
278    }
279    File blockFile = FsDatasetImpl.moveBlockFiles(b, f, blockDir);
280    File metaFile = FsDatasetUtil.getMetaFile(blockFile, b.getGenerationStamp());
281    dfsUsage.incDfsUsed(b.getNumBytes()+metaFile.length());
282    return blockFile;
283  }

309  void getVolumeMap(ReplicaMap volumeMap,
310                    final RamDiskReplicaTracker lazyWriteReplicaMap)
311      throws IOException {
312    // Recover lazy persist replicas, they will be added to the volumeMap
313    // when we scan the finalized directory.
314    if (lazypersistDir.exists()) {
315      int numRecovered = moveLazyPersistReplicasToFinalized(lazypersistDir);
316      FsDatasetImpl.LOG.info(
317          "Recovered " + numRecovered + " replicas from " + lazypersistDir);
318    }
319
320    // add finalized replicas
321    addToReplicasMap(volumeMap, finalizedDir, lazyWriteReplicaMap, true);
322    // add rbw replicas
323    addToReplicasMap(volumeMap, rbwDir, lazyWriteReplicaMap, false);
324  }

326  /**
327   * Recover an unlinked tmp file on datanode restart. If the original block
328   * does not exist, then the tmp file is renamed to be the
329   * original file name and the original name is returned; otherwise the tmp
330   * file is deleted and null is returned.
331   */
332  File recoverTempUnlinkedBlock(File unlinkedTmp) throws IOException {
333    File blockFile = FsDatasetUtil.getOrigFile(unlinkedTmp);
334    if (blockFile.exists()) {
335      // If the original block file still exists, then no recovery is needed.
336      if (!unlinkedTmp.delete()) {
337        throw new IOException("Unable to cleanup unlinked tmp file " +
338            unlinkedTmp);
339      }
340      return null;
341    } else {
342      if (!unlinkedTmp.renameTo(blockFile)) {
343        throw new IOException("Unable to rename unlinked tmp file " +
344            unlinkedTmp);
345      }
346      return blockFile;
347    }
348  }

351  /**
352   * Move replicas in the lazy persist directory to their corresponding locations
353   * in the finalized directory.
354   * @return number of replicas recovered.
355   */
356  private int moveLazyPersistReplicasToFinalized(File source)
357      throws IOException {
358    File files[] = FileUtil.listFiles(source);
359    int numRecovered = 0;
360    for (File file : files) {
361      if (file.isDirectory()) {
362        numRecovered += moveLazyPersistReplicasToFinalized(file);
363      }
364
365      if (Block.isMetaFilename(file.getName())) {
366        File metaFile = file;
367        File blockFile = Block.metaToBlockFile(metaFile);
368        long blockId = Block.filename2id(blockFile.getName());
369        File targetDir = DatanodeUtil.idToBlockDir(finalizedDir, blockId);
370
371        if (blockFile.exists()) {
372
373          if (!targetDir.exists() && !targetDir.mkdirs()) {
374            LOG.warn("Failed to mkdirs " + targetDir);
375            continue;
376          }
377
378          final File targetMetaFile = new File(targetDir, metaFile.getName());
379          try {
380            NativeIO.renameTo(metaFile, targetMetaFile);
381          } catch (IOException e) {
382            LOG.warn("Failed to move meta file from "
383                + metaFile + " to " + targetMetaFile, e);
384            continue;
385
386          }
387
388          final File targetBlockFile = new File(targetDir, blockFile.getName());
389          try {
390            NativeIO.renameTo(blockFile, targetBlockFile);
391          } catch (IOException e) {
392            LOG.warn("Failed to move block file from "
393                + blockFile + " to " + targetBlockFile, e);
394            continue;
395          }
396
397          if (targetBlockFile.exists() && targetMetaFile.exists()) {
398            ++numRecovered;
399          } else {
400            // Failure should be rare.
401            LOG.warn("Failed to move " + blockFile + " to " + targetDir);
402          }
403        }
404      }
405    }
406
407    FileUtil.fullyDelete(source);
408    return numRecovered;
409  }

411  /**
412   * Add replicas under the given directory to the volume map
413   * @param volumeMap the replicas map
414   * @param dir an input directory
415   * @param lazyWriteReplicaMap Map of replicas on transient
416   *                                storage.
417   * @param isFinalized true if the directory has finalized replicas;
418   *                    false if the directory has rbw replicas
419   */
420  void addToReplicasMap(ReplicaMap volumeMap, File dir,
421                        final RamDiskReplicaTracker lazyWriteReplicaMap,
422                        boolean isFinalized)
423      throws IOException {
424    File files[] = FileUtil.listFiles(dir);
425    for (File file : files) {
426      if (file.isDirectory()) {
427        addToReplicasMap(volumeMap, file, lazyWriteReplicaMap, isFinalized);
428      }
429
430      if (isFinalized && FsDatasetUtil.isUnlinkTmpFile(file)) {
431        file = recoverTempUnlinkedBlock(file);
432        if (file == null) { // the original block still exists, so we cover it
433          // in another iteration and can continue here
434          continue;
435        }
436      }
437      if (!Block.isBlockFilename(file))
438        continue;
439      
440      long genStamp = FsDatasetUtil.getGenerationStampFromFile(
441          files, file);
442      long blockId = Block.filename2id(file.getName());
443      ReplicaInfo newReplica = null;
444      if (isFinalized) {
445        newReplica = new FinalizedReplica(blockId, 
446            file.length(), genStamp, volume, file.getParentFile());
447      } else {
448
449        boolean loadRwr = true;
450        File restartMeta = new File(file.getParent()  +
451            File.pathSeparator + "." + file.getName() + ".restart");
452        Scanner sc = null;
453        try {
454          sc = new Scanner(restartMeta, "UTF-8");
455          // The restart meta file exists
456          if (sc.hasNextLong() && (sc.nextLong() > Time.now())) {
457            // It didn't expire. Load the replica as a RBW.
458            // We don't know the expected block length, so just use 0
459            // and don't reserve any more space for writes.
460            newReplica = new ReplicaBeingWritten(blockId,
461                validateIntegrityAndSetLength(file, genStamp), 
462                genStamp, volume, file.getParentFile(), null, 0);
463            loadRwr = false;
464          }
465          sc.close();
466          if (!restartMeta.delete()) {
467            FsDatasetImpl.LOG.warn("Failed to delete restart meta file: " +
468              restartMeta.getPath());
469          }
470        } catch (FileNotFoundException fnfe) {
471          // nothing to do hereFile dir =
472        } finally {
473          if (sc != null) {
474            sc.close();
475          }
476        }
477        // Restart meta doesn't exist or expired.
478        if (loadRwr) {
479          newReplica = new ReplicaWaitingToBeRecovered(blockId,
480              validateIntegrityAndSetLength(file, genStamp),
481              genStamp, volume, file.getParentFile());
482        }
483      }
484
485      ReplicaInfo oldReplica = volumeMap.get(bpid, newReplica.getBlockId());
486      if (oldReplica == null) {
487        volumeMap.add(bpid, newReplica);
488      } else {
489        // We have multiple replicas of the same block so decide which one
490        // to keep.
491        newReplica = resolveDuplicateReplicas(newReplica, oldReplica, volumeMap);
492      }
493
494      // If we are retaining a replica on transient storage make sure
495      // it is in the lazyWriteReplicaMap so it can be persisted
496      // eventually.
497      if (newReplica.getVolume().isTransientStorage()) {
498        lazyWriteReplicaMap.addReplica(bpid, blockId,
499                                       (FsVolumeImpl) newReplica.getVolume());
500      } else {
501        lazyWriteReplicaMap.discardReplica(bpid, blockId, false);
502      }
503    }
504  }

506  /**
507   * This method is invoked during DN startup when volumes are scanned to
508   * build up the volumeMap.
509   *
510   * Given two replicas, decide which one to keep. The preference is as
511   * follows:
512   *   1. Prefer the replica with the higher generation stamp.
513   *   2. If generation stamps are equal, prefer the replica with the
514   *      larger on-disk length.
515   *   3. If on-disk length is the same, prefer the replica on persistent
516   *      storage volume.
517   *   4. All other factors being equal, keep replica1.
518   *
519   * The other replica is removed from the volumeMap and is deleted from
520   * its storage volume.
521   *
522   * @param replica1
523   * @param replica2
524   * @param volumeMap
525   * @return the replica that is retained.
526   * @throws IOException
527   */
528  ReplicaInfo resolveDuplicateReplicas(
529      final ReplicaInfo replica1, final ReplicaInfo replica2,
530      final ReplicaMap volumeMap) throws IOException {
531
532    if (!deleteDuplicateReplicas) {
533      // Leave both block replicas in place.
534      return replica1;
535    }
536
537    ReplicaInfo replicaToKeep;
538    ReplicaInfo replicaToDelete;
539
540    if (replica1.getGenerationStamp() != replica2.getGenerationStamp()) {
541      replicaToKeep = replica1.getGenerationStamp() > replica2.getGenerationStamp()
542          ? replica1 : replica2;
543    } else if (replica1.getNumBytes() != replica2.getNumBytes()) {
544      replicaToKeep = replica1.getNumBytes() > replica2.getNumBytes() ?
545          replica1 : replica2;
546    } else if (replica1.getVolume().isTransientStorage() &&
547               !replica2.getVolume().isTransientStorage()) {
548      replicaToKeep = replica2;
549    } else {
550      replicaToKeep = replica1;
551    }
552
553    replicaToDelete = (replicaToKeep == replica1) ? replica2 : replica1;
554
555    if (LOG.isDebugEnabled()) {
556      LOG.debug("resolveDuplicateReplicas decide to keep " + replicaToKeep
557          + ".  Will try to delete " + replicaToDelete);
558    }
559
560    // Update volumeMap.
561    volumeMap.add(bpid, replicaToKeep);
562
563    // Delete the files on disk. Failure here is okay.
564    final File blockFile = replicaToDelete.getBlockFile();
565    if (!blockFile.delete()) {
566      LOG.warn("Failed to delete block file " + blockFile);
567    }
568    final File metaFile = replicaToDelete.getMetaFile();
569    if (!metaFile.delete()) {
570      LOG.warn("Failed to delete meta file " + metaFile);
571    }
572
573    return replicaToKeep;
574  }

576  /**
577   * Find out the number of bytes in the block that match its crc.
578   * 
579   * This algorithm assumes that data corruption caused by unexpected 
580   * datanode shutdown occurs only in the last crc chunk. So it checks
581   * only the last chunk.
582   * 
583   * @param blockFile the block file
584   * @param genStamp generation stamp of the block
585   * @return the number of valid bytes
586   */
587  private long validateIntegrityAndSetLength(File blockFile, long genStamp) {
588    DataInputStream checksumIn = null;
589    InputStream blockIn = null;
590    try {
591      final File metaFile = FsDatasetUtil.getMetaFile(blockFile, genStamp);
592      long blockFileLen = blockFile.length();
593      long metaFileLen = metaFile.length();
594      int crcHeaderLen = DataChecksum.getChecksumHeaderSize();
595      if (!blockFile.exists() || blockFileLen == 0 ||
596          !metaFile.exists() || metaFileLen < crcHeaderLen) {
597        return 0;
598      }
599      checksumIn = new DataInputStream(
600          new BufferedInputStream(new FileInputStream(metaFile),
601              HdfsConstants.IO_FILE_BUFFER_SIZE));
602
603      // read and handle the common header here. For now just a version
604      final DataChecksum checksum = BlockMetadataHeader.readDataChecksum(
605          checksumIn, metaFile);
606      int bytesPerChecksum = checksum.getBytesPerChecksum();
607      int checksumSize = checksum.getChecksumSize();
608      long numChunks = Math.min(
609          (blockFileLen + bytesPerChecksum - 1)/bytesPerChecksum, 
610          (metaFileLen - crcHeaderLen)/checksumSize);
611      if (numChunks == 0) {
612        return 0;
613      }
614      IOUtils.skipFully(checksumIn, (numChunks-1)*checksumSize);
615      blockIn = new FileInputStream(blockFile);
616      long lastChunkStartPos = (numChunks-1)*bytesPerChecksum;
617      IOUtils.skipFully(blockIn, lastChunkStartPos);
618      int lastChunkSize = (int)Math.min(
619          bytesPerChecksum, blockFileLen-lastChunkStartPos);
620      byte[] buf = new byte[lastChunkSize+checksumSize];
621      checksumIn.readFully(buf, lastChunkSize, checksumSize);
622      IOUtils.readFully(blockIn, buf, 0, lastChunkSize);
623
624      checksum.update(buf, 0, lastChunkSize);
625      long validFileLength;
626      if (checksum.compare(buf, lastChunkSize)) { // last chunk matches crc
627        validFileLength = lastChunkStartPos + lastChunkSize;
628      } else { // last chunck is corrupt
629        validFileLength = lastChunkStartPos;
630      }
631
632      // truncate if extra bytes are present without CRC
633      if (blockFile.length() > validFileLength) {
634        RandomAccessFile blockRAF = new RandomAccessFile(blockFile, "rw");
635        try {
636          // truncate blockFile
637          blockRAF.setLength(validFileLength);
638        } finally {
639          blockRAF.close();
640        }
641      }
642
643      return validFileLength;
644    } catch (IOException e) {
645      FsDatasetImpl.LOG.warn(e);
646      return 0;
647    } finally {
648      IOUtils.closeStream(checksumIn);
649      IOUtils.closeStream(blockIn);
650    }
651  }

653  @Override
