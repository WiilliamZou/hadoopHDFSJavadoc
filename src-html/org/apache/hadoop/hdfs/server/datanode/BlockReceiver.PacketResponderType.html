138  BlockReceiver(final ExtendedBlock block, final StorageType storageType,
139      final DataInputStream in,
140      final String inAddr, final String myAddr,
141      final BlockConstructionStage stage, 
142      final long newGs, final long minBytesRcvd, final long maxBytesRcvd, 
143      final String clientname, final DatanodeInfo srcDataNode,
144      final DataNode datanode, DataChecksum requestedChecksum,
145      CachingStrategy cachingStrategy,
146      final boolean allowLazyPersist,
147      final boolean pinning) throws IOException {
148    try{
149      this.block = block;
150      this.in = in;
151      this.inAddr = inAddr;
152      this.myAddr = myAddr;
153      this.srcDataNode = srcDataNode;
154      this.datanode = datanode;
155
156      this.clientname = clientname;
157      this.isDatanode = clientname.length() == 0;
158      this.isClient = !this.isDatanode;
159      this.restartBudget = datanode.getDnConf().restartReplicaExpiry;
160      this.datanodeSlowLogThresholdMs = datanode.getDnConf().datanodeSlowIoWarningThresholdMs;
161      // For replaceBlock() calls response should be sent to avoid socketTimeout
162      // at clients. So sending with the interval of 0.5 * socketTimeout
163      this.responseInterval = (long) (datanode.getDnConf().socketTimeout * 0.5);
164      //for datanode, we have
165      //1: clientName.length() == 0, and
166      //2: stage == null or PIPELINE_SETUP_CREATE
167      this.stage = stage;
168      this.isTransfer = stage == BlockConstructionStage.TRANSFER_RBW
169          || stage == BlockConstructionStage.TRANSFER_FINALIZED;
170
171      this.pinning = pinning;
172      if (LOG.isDebugEnabled()) {
173        LOG.debug(getClass().getSimpleName() + ": " + block
174            + "\n  isClient  =" + isClient + ", clientname=" + clientname
175            + "\n  isDatanode=" + isDatanode + ", srcDataNode=" + srcDataNode
176            + "\n  inAddr=" + inAddr + ", myAddr=" + myAddr
177            + "\n  cachingStrategy = " + cachingStrategy
178            + "\n  pinning=" + pinning
179            );
180      }
181
182      //
183      // Open local disk out
184      //
185      if (isDatanode) { //replication or move
186        replicaHandler = datanode.data.createTemporary(storageType, block);
187      } else {
188        switch (stage) {
189        case PIPELINE_SETUP_CREATE:
190          replicaHandler = datanode.data.createRbw(storageType, block, allowLazyPersist);
191          datanode.notifyNamenodeReceivingBlock(
192              block, replicaHandler.getReplica().getStorageUuid());
193          break;
194        case PIPELINE_SETUP_STREAMING_RECOVERY:
195          replicaHandler = datanode.data.recoverRbw(
196              block, newGs, minBytesRcvd, maxBytesRcvd);
197          block.setGenerationStamp(newGs);
198          break;
199        case PIPELINE_SETUP_APPEND:
200          replicaHandler = datanode.data.append(block, newGs, minBytesRcvd);
201          block.setGenerationStamp(newGs);
202          datanode.notifyNamenodeReceivingBlock(
203              block, replicaHandler.getReplica().getStorageUuid());
204          break;
205        case PIPELINE_SETUP_APPEND_RECOVERY:
206          replicaHandler = datanode.data.recoverAppend(block, newGs, minBytesRcvd);
207          block.setGenerationStamp(newGs);
208          datanode.notifyNamenodeReceivingBlock(
209              block, replicaHandler.getReplica().getStorageUuid());
210          break;
211        case TRANSFER_RBW:
212        case TRANSFER_FINALIZED:
213          // this is a transfer destination
214          replicaHandler =
215              datanode.data.createTemporary(storageType, block);
216          break;
217        default: throw new IOException("Unsupported stage " + stage + 
218              " while receiving block " + block + " from " + inAddr);
219        }
220      }
221      replicaInfo = replicaHandler.getReplica();
222      this.dropCacheBehindWrites = (cachingStrategy.getDropBehind() == null) ?
223        datanode.getDnConf().dropCacheBehindWrites :
224          cachingStrategy.getDropBehind();
225      this.syncBehindWrites = datanode.getDnConf().syncBehindWrites;
226      this.syncBehindWritesInBackground = datanode.getDnConf().
227          syncBehindWritesInBackground;
228      
229      final boolean isCreate = isDatanode || isTransfer 
230          || stage == BlockConstructionStage.PIPELINE_SETUP_CREATE;
231      streams = replicaInfo.createStreams(isCreate, requestedChecksum);
232      assert streams != null : "null streams!";
233
234      // read checksum meta information
235      this.clientChecksum = requestedChecksum;
236      this.diskChecksum = streams.getChecksum();
237      this.needsChecksumTranslation = !clientChecksum.equals(diskChecksum);
238      this.bytesPerChecksum = diskChecksum.getBytesPerChecksum();
239      this.checksumSize = diskChecksum.getChecksumSize();
240
241      this.out = streams.getDataOut();
242      if (out instanceof FileOutputStream) {
243        this.outFd = ((FileOutputStream)out).getFD();
244      } else {
245        LOG.warn("Could not get file descriptor for outputstream of class " +
246            out.getClass());
247      }
248      this.checksumOut = new DataOutputStream(new BufferedOutputStream(
249          streams.getChecksumOut(), HdfsConstants.SMALL_BUFFER_SIZE));
250      // write data chunk header if creating a new replica
251      if (isCreate) {
252        BlockMetadataHeader.writeHeader(checksumOut, diskChecksum);
253      } 
254    } catch (ReplicaAlreadyExistsException bae) {
255      throw bae;
256    } catch (ReplicaNotFoundException bne) {
257      throw bne;
258    } catch(IOException ioe) {
259      IOUtils.closeStream(this);
260      cleanupBlock();
261      
262      // check if there is a disk error
263      IOException cause = DatanodeUtil.getCauseIfDiskError(ioe);
264      DataNode.LOG.warn("IOException in BlockReceiver constructor. Cause is ",
265          cause);
266      
267      if (cause != null) { // possible disk error
268        ioe = cause;
269        datanode.checkDiskErrorAsync();
270      }
271      
272      throw ioe;
273    }
274  }

283  /**
284   * close files and release volume reference.
285   */
286  @Override
287  public void close() throws IOException {
288    packetReceiver.close();
289
290    IOException ioe = null;
291    if (syncOnClose && (out != null || checksumOut != null)) {
292      datanode.metrics.incrFsyncCount();      
293    }
294    long flushTotalNanos = 0;
295    boolean measuredFlushTime = false;
296    // close checksum file
297    try {
298      if (checksumOut != null) {
299        long flushStartNanos = System.nanoTime();
300        checksumOut.flush();
301        long flushEndNanos = System.nanoTime();
302        if (syncOnClose) {
303          long fsyncStartNanos = flushEndNanos;
304          streams.syncChecksumOut();
305          datanode.metrics.addFsyncNanos(System.nanoTime() - fsyncStartNanos);
306        }
307        flushTotalNanos += flushEndNanos - flushStartNanos;
308        measuredFlushTime = true;
309        checksumOut.close();
310        checksumOut = null;
311      }
312    } catch(IOException e) {
313      ioe = e;
314    }
315    finally {
316      IOUtils.closeStream(checksumOut);
317    }
318    // close block file
319    try {
320      if (out != null) {
321        long flushStartNanos = System.nanoTime();
322        out.flush();
323        long flushEndNanos = System.nanoTime();
324        if (syncOnClose) {
325          long fsyncStartNanos = flushEndNanos;
326          streams.syncDataOut();
327          datanode.metrics.addFsyncNanos(System.nanoTime() - fsyncStartNanos);
328        }
329        flushTotalNanos += flushEndNanos - flushStartNanos;
330        measuredFlushTime = true;
331        out.close();
332        out = null;
333      }
334    } catch (IOException e) {
335      ioe = e;
336    }
337    finally{
338      IOUtils.closeStream(out);
339    }
340    if (replicaHandler != null) {
341      IOUtils.cleanup(null, replicaHandler);
342      replicaHandler = null;
343    }
344    if (measuredFlushTime) {
345      datanode.metrics.addFlushNanos(flushTotalNanos);
346    }
347    // disk check
348    if(ioe != null) {
349      datanode.checkDiskErrorAsync();
350      throw ioe;
351    }
352  }

354  /**
355   * Flush block data and metadata files to disk.
356   * @throws IOException
357   */
358  void flushOrSync(boolean isSync) throws IOException {
359    long flushTotalNanos = 0;
360    long begin = Time.monotonicNow();
361    if (checksumOut != null) {
362      long flushStartNanos = System.nanoTime();
363      checksumOut.flush();
364      long flushEndNanos = System.nanoTime();
365      if (isSync) {
366        long fsyncStartNanos = flushEndNanos;
367        streams.syncChecksumOut();
368        datanode.metrics.addFsyncNanos(System.nanoTime() - fsyncStartNanos);
369      }
370      flushTotalNanos += flushEndNanos - flushStartNanos;
371    }
372    if (out != null) {
373      long flushStartNanos = System.nanoTime();
374      out.flush();
375      long flushEndNanos = System.nanoTime();
376      if (isSync) {
377        long fsyncStartNanos = flushEndNanos;
378        streams.syncDataOut();
379        datanode.metrics.addFsyncNanos(System.nanoTime() - fsyncStartNanos);
380      }
381      flushTotalNanos += flushEndNanos - flushStartNanos;
382    }
383    if (checksumOut != null || out != null) {
384      datanode.metrics.addFlushNanos(flushTotalNanos);
385      if (isSync) {
386          datanode.metrics.incrFsyncCount();      
387      }
388    }
389    long duration = Time.monotonicNow() - begin;
390    if (duration > datanodeSlowLogThresholdMs) {
391      LOG.warn("Slow flushOrSync took " + duration + "ms (threshold="
392          + datanodeSlowLogThresholdMs + "ms), isSync:" + isSync + ", flushTotalNanos="
393          + flushTotalNanos + "ns");
394    }
395  }

397  /**
398   * While writing to mirrorOut, failure to write to mirror should not
399   * affect this datanode unless it is caused by interruption.
400   */
401  private void handleMirrorOutError(IOException ioe) throws IOException {
402    String bpid = block.getBlockPoolId();
403    LOG.info(datanode.getDNRegistrationForBP(bpid)
404        + ":Exception writing " + block + " to mirror " + mirrorAddr, ioe);
405    if (Thread.interrupted()) { // shut down if the thread is interrupted
406      throw ioe;
407    } else { // encounter an error while writing to mirror
408      // continue to run even if can not write to mirror
409      // notify client of the error
410      // and wait for the client to shut down the pipeline
411      mirrorError = true;
412    }
413  }

415  /**
416   * Verify multiple CRC chunks. 
417   */
418  private void verifyChunks(ByteBuffer dataBuf, ByteBuffer checksumBuf)
419      throws IOException {
420    try {
421      clientChecksum.verifyChunkedSums(dataBuf, checksumBuf, clientname, 0);
422    } catch (ChecksumException ce) {
423      LOG.warn("Checksum error in block " + block + " from " + inAddr, ce);
424      // No need to report to namenode when client is writing.
425      if (srcDataNode != null && isDatanode) {
426        try {
427          LOG.info("report corrupt " + block + " from datanode " +
428                    srcDataNode + " to namenode");
429          datanode.reportRemoteBadBlock(srcDataNode, block);
430        } catch (IOException e) {
431          LOG.warn("Failed to report bad " + block + 
432                    " from datanode " + srcDataNode + " to namenode");
433        }
434      }
435      throw new IOException("Unexpected checksum mismatch while writing "
436          + block + " from " + inAddr);
437    }
438  }

445   * This does not verify the original checksums, under the assumption
446   * that they have already been validated.

448  private void translateChunks(ByteBuffer dataBuf, ByteBuffer checksumBuf) {

452  /** 
453   * Check whether checksum needs to be verified.
454   * Skip verifying checksum iff this is not the last one in the 
455   * pipeline and clientName is non-null. i.e. Checksum is verified
456   * on all the datanodes when the data is being written by a 
457   * datanode rather than a client. Whe client is writing the data, 
458   * protocol includes acks and only the last datanode needs to verify 
459   * checksum.
460   * @return true if checksum verification is needed, otherwise false.
461   */
462  private boolean shouldVerifyChecksum() {
463    return (mirrorOut == null || isDatanode || needsChecksumTranslation);
464  }

466  /** 
467   * Receives and processes a packet. It can contain many chunks.
468   * returns the number of data bytes that the packet has.
469   */
470  private int receivePacket() throws IOException {
471    // read the next packet
472    packetReceiver.receiveNextPacket(in);
473
474    PacketHeader header = packetReceiver.getHeader();
475    if (LOG.isDebugEnabled()){
476      LOG.debug("Receiving one packet for block " + block +
477                ": " + header);
478    }
479
480    // Sanity check the header
481    if (header.getOffsetInBlock() > replicaInfo.getNumBytes()) {
482      throw new IOException("Received an out-of-sequence packet for " + block + 
483          "from " + inAddr + " at offset " + header.getOffsetInBlock() +
484          ". Expecting packet starting at " + replicaInfo.getNumBytes());
485    }
486    if (header.getDataLen() < 0) {
487      throw new IOException("Got wrong length during writeBlock(" + block + 
488                            ") from " + inAddr + " at offset " + 
489                            header.getOffsetInBlock() + ": " +
490                            header.getDataLen()); 
491    }
492
493    long offsetInBlock = header.getOffsetInBlock();
494    long seqno = header.getSeqno();
495    boolean lastPacketInBlock = header.isLastPacketInBlock();
496    final int len = header.getDataLen();
497    boolean syncBlock = header.getSyncBlock();
498
499    // avoid double sync'ing on close
500    if (syncBlock && lastPacketInBlock) {
501      this.syncOnClose = false;
502    }
503
504    // update received bytes
505    final long firstByteInBlock = offsetInBlock;
506    offsetInBlock += len;
507    if (replicaInfo.getNumBytes() < offsetInBlock) {
508      replicaInfo.setNumBytes(offsetInBlock);
509    }
510    
511    // put in queue for pending acks, unless sync was requested
512    if (responder != null && !syncBlock && !shouldVerifyChecksum()) {
513      ((PacketResponder) responder.getRunnable()).enqueue(seqno,
514          lastPacketInBlock, offsetInBlock, Status.SUCCESS);
515    }
516
517    //First write the packet to the mirror:
518    if (mirrorOut != null && !mirrorError) {
519      try {
520        long begin = Time.monotonicNow();
521        packetReceiver.mirrorPacketTo(mirrorOut);
522        mirrorOut.flush();
523        long duration = Time.monotonicNow() - begin;
524        if (duration > datanodeSlowLogThresholdMs) {
525          LOG.warn("Slow BlockReceiver write packet to mirror took " + duration
526              + "ms (threshold=" + datanodeSlowLogThresholdMs + "ms)");
527        }
528      } catch (IOException e) {
529        handleMirrorOutError(e);
530      }
531    }
532    
533    ByteBuffer dataBuf = packetReceiver.getDataSlice();
534    ByteBuffer checksumBuf = packetReceiver.getChecksumSlice();
535    
536    if (lastPacketInBlock || len == 0) {
537      if(LOG.isDebugEnabled()) {
538        LOG.debug("Receiving an empty packet or the end of the block " + block);
539      }
540      // sync block if requested
541      if (syncBlock) {
542        flushOrSync(true);
543      }
544    } else {
545      final int checksumLen = diskChecksum.getChecksumSize(len);
546      final int checksumReceivedLen = checksumBuf.capacity();
547
548      if (checksumReceivedLen > 0 && checksumReceivedLen != checksumLen) {
549        throw new IOException("Invalid checksum length: received length is "
550            + checksumReceivedLen + " but expected length is " + checksumLen);
551      }
552
553      if (checksumReceivedLen > 0 && shouldVerifyChecksum()) {
554        try {
555          verifyChunks(dataBuf, checksumBuf);
556        } catch (IOException ioe) {
557          // checksum error detected locally. there is no reason to continue.
558          if (responder != null) {
559            try {
560              ((PacketResponder) responder.getRunnable()).enqueue(seqno,
561                  lastPacketInBlock, offsetInBlock,
562                  Status.ERROR_CHECKSUM);
563              // Wait until the responder sends back the response
564              // and interrupt this thread.
565              Thread.sleep(3000);
566            } catch (InterruptedException e) { }
567          }
568          throw new IOException("Terminating due to a checksum error." + ioe);
569        }
570 
571        if (needsChecksumTranslation) {
572          // overwrite the checksums in the packet buffer with the
573          // appropriate polynomial for the disk storage.
574          translateChunks(dataBuf, checksumBuf);
575        }
576      }
577
578      if (checksumReceivedLen == 0 && !streams.isTransientStorage()) {
579        // checksum is missing, need to calculate it
580        checksumBuf = ByteBuffer.allocate(checksumLen);
581        diskChecksum.calculateChunkedSums(dataBuf, checksumBuf);
582      }
583      
584      // by this point, the data in the buffer uses the disk checksum
585
586      final boolean shouldNotWriteChecksum = checksumReceivedLen == 0
587          && streams.isTransientStorage();
588      try {
589        long onDiskLen = replicaInfo.getBytesOnDisk();
590        if (onDiskLen<offsetInBlock) {
591          //finally write to the disk :
592          
593          if (onDiskLen % bytesPerChecksum != 0) { 
594            // prepare to overwrite last checksum
595            adjustCrcFilePosition();
596          }
597          
598          // If this is a partial chunk, then read in pre-existing checksum
599          Checksum partialCrc = null;
600          if (!shouldNotWriteChecksum && firstByteInBlock % bytesPerChecksum != 0) {
601            if (LOG.isDebugEnabled()) {
602              LOG.debug("receivePacket for " + block 
603                  + ": bytesPerChecksum=" + bytesPerChecksum                  
604                  + " does not divide firstByteInBlock=" + firstByteInBlock);
605            }
606            long offsetInChecksum = BlockMetadataHeader.getHeaderSize() +
607                onDiskLen / bytesPerChecksum * checksumSize;
608            partialCrc = computePartialChunkCrc(onDiskLen, offsetInChecksum);
609          }
610
611          int startByteToDisk = (int)(onDiskLen-firstByteInBlock) 
612              + dataBuf.arrayOffset() + dataBuf.position();
613
614          int numBytesToDisk = (int)(offsetInBlock-onDiskLen);
615          
616          // Write data to disk.
617          long begin = Time.monotonicNow();
618          out.write(dataBuf.array(), startByteToDisk, numBytesToDisk);
619          long duration = Time.monotonicNow() - begin;
620          if (duration > datanodeSlowLogThresholdMs) {
621            LOG.warn("Slow BlockReceiver write data to disk cost:" + duration
622                + "ms (threshold=" + datanodeSlowLogThresholdMs + "ms)");
623          }
624
625          final byte[] lastCrc;
626          if (shouldNotWriteChecksum) {
627            lastCrc = null;
628          } else if (partialCrc != null) {
629            // If this is a partial chunk, then verify that this is the only
630            // chunk in the packet. Calculate new crc for this chunk.
631            if (len > bytesPerChecksum) {
632              throw new IOException("Unexpected packet data length for "
633                  +  block + " from " + inAddr + ": a partial chunk must be "
634                  + " sent in an individual packet (data length = " + len
635                  +  " > bytesPerChecksum = " + bytesPerChecksum + ")");
636            }
637            partialCrc.update(dataBuf.array(), startByteToDisk, numBytesToDisk);
638            byte[] buf = FSOutputSummer.convertToByteStream(partialCrc, checksumSize);
639            lastCrc = copyLastChunkChecksum(buf, checksumSize, buf.length);
640            checksumOut.write(buf);
641            if(LOG.isDebugEnabled()) {
642              LOG.debug("Writing out partial crc for data len " + len);
643            }
644            partialCrc = null;
645          } else {
646            // write checksum
647            final int offset = checksumBuf.arrayOffset() +
648                checksumBuf.position();
649            final int end = offset + checksumLen;
650            lastCrc = copyLastChunkChecksum(checksumBuf.array(), checksumSize,
651                end);
652            checksumOut.write(checksumBuf.array(), offset, checksumLen);
653          }
654
655          /// flush entire packet, sync if requested
656          flushOrSync(syncBlock);
657          
658          replicaInfo.setLastChecksumAndDataLen(offsetInBlock, lastCrc);
659
660          datanode.metrics.incrBytesWritten(len);
661          datanode.metrics.incrTotalWriteTime(duration);
662
663          manageWriterOsCache(offsetInBlock);
664        }
665      } catch (IOException iex) {
666        datanode.checkDiskErrorAsync();
667        throw iex;
668      }
669    }
670
671    // if sync was requested, put in queue for pending acks here
672    // (after the fsync finished)
673    if (responder != null && (syncBlock || shouldVerifyChecksum())) {
674      ((PacketResponder) responder.getRunnable()).enqueue(seqno,
675          lastPacketInBlock, offsetInBlock, Status.SUCCESS);
676    }
677
678    /*
679     * Send in-progress responses for the replaceBlock() calls back to caller to
680     * avoid timeouts due to balancer throttling. HDFS-6247
681     */
682    if (isReplaceBlock
683        && (Time.monotonicNow() - lastResponseTime > responseInterval)) {
684      BlockOpResponseProto.Builder response = BlockOpResponseProto.newBuilder()
685          .setStatus(Status.IN_PROGRESS);
686      response.build().writeDelimitedTo(replyOut);
687      replyOut.flush();
688
689      lastResponseTime = Time.monotonicNow();
690    }
691
692    if (throttler != null) { // throttle I/O
693      throttler.throttle(len);
694    }
695    
696    return lastPacketInBlock?-1:len;
697  }
698
699  private static byte[] copyLastChunkChecksum(byte[] array, int size, int end) {
700    return Arrays.copyOfRange(array, end - size, end);
701  }

703  private void manageWriterOsCache(long offsetInBlock) {
704    try {
705      if (outFd != null &&
706          offsetInBlock > lastCacheManagementOffset + CACHE_DROP_LAG_BYTES) {
707        long begin = Time.monotonicNow();
708        //
709        // For SYNC_FILE_RANGE_WRITE, we want to sync from
710        // lastCacheManagementOffset to a position "two windows ago"
711        //
712        //                         <========= sync ===========>
713        // +-----------------------O--------------------------X
714        // start                  last                      curPos
715        // of file                 
716        //
717        if (syncBehindWrites) {
718          if (syncBehindWritesInBackground) {
719            this.datanode.getFSDataset().submitBackgroundSyncFileRangeRequest(
720                block, outFd, lastCacheManagementOffset,
721                offsetInBlock - lastCacheManagementOffset,
722                NativeIO.POSIX.SYNC_FILE_RANGE_WRITE);
723          } else {
724            NativeIO.POSIX.syncFileRangeIfPossible(outFd,
725                lastCacheManagementOffset, offsetInBlock
726                    - lastCacheManagementOffset,
727                NativeIO.POSIX.SYNC_FILE_RANGE_WRITE);
728          }
729        }
730        //
731        // For POSIX_FADV_DONTNEED, we want to drop from the beginning 
732        // of the file to a position prior to the current position.
733        //
734        // <=== drop =====> 
735        //                 <---W--->
736        // +--------------+--------O--------------------------X
737        // start        dropPos   last                      curPos
738        // of file             
739        //                     
740        long dropPos = lastCacheManagementOffset - CACHE_DROP_LAG_BYTES;
741        if (dropPos > 0 && dropCacheBehindWrites) {
742          NativeIO.POSIX.getCacheManipulator().posixFadviseIfPossible(
743              block.getBlockName(), outFd, 0, dropPos,
744              NativeIO.POSIX.POSIX_FADV_DONTNEED);
745        }
746        lastCacheManagementOffset = offsetInBlock;
747        long duration = Time.monotonicNow() - begin;
748        if (duration > datanodeSlowLogThresholdMs) {
749          LOG.warn("Slow manageWriterOsCache took " + duration
750              + "ms (threshold=" + datanodeSlowLogThresholdMs + "ms)");
751        }
752      }
753    } catch (Throwable t) {
754      LOG.warn("Error managing cache for writer of block " + block, t);
755    }
756  }

763  void receiveBlock(
764      DataOutputStream mirrOut, // output to next datanode
765      DataInputStream mirrIn,   // input from next datanode
766      DataOutputStream replyOut,  // output to previous datanode
767      String mirrAddr, DataTransferThrottler throttlerArg,
768      DatanodeInfo[] downstreams,
769      boolean isReplaceBlock) throws IOException {
770
771      syncOnClose = datanode.getDnConf().syncOnClose;
772      boolean responderClosed = false;
773      mirrorOut = mirrOut;
774      mirrorAddr = mirrAddr;
775      throttler = throttlerArg;
776
777      this.replyOut = replyOut;
778      this.isReplaceBlock = isReplaceBlock;
779
780    try {
781      if (isClient && !isTransfer) {
782        responder = new Daemon(datanode.threadGroup, 
783            new PacketResponder(replyOut, mirrIn, downstreams));
784        responder.start(); // start thread to processes responses
785      }
786
787      while (receivePacket() >= 0) { /* Receive until the last packet */ }
788
789      // wait for all outstanding packet responses. And then
790      // indicate responder to gracefully shutdown.
791      // Mark that responder has been closed for future processing
792      if (responder != null) {
793        ((PacketResponder)responder.getRunnable()).close();
794        responderClosed = true;
795      }
796
797      // If this write is for a replication or transfer-RBW/Finalized,
798      // then finalize block or convert temporary to RBW.
799      // For client-writes, the block is finalized in the PacketResponder.
800      if (isDatanode || isTransfer) {
801        // Hold a volume reference to finalize block.
802        try (ReplicaHandler handler = claimReplicaHandler()) {
803          // close the block/crc files
804          close();
805          block.setNumBytes(replicaInfo.getNumBytes());
806
807          if (stage == BlockConstructionStage.TRANSFER_RBW) {
808            // for TRANSFER_RBW, convert temporary to RBW
809            datanode.data.convertTemporaryToRbw(block);
810          } else {
811            // for isDatnode or TRANSFER_FINALIZED
812            // Finalize the block.
813            datanode.data.finalizeBlock(block);
814          }
815        }
816        datanode.metrics.incrBlocksWritten();
817      }
818
819    } catch (IOException ioe) {
820      replicaInfo.releaseAllBytesReserved();
821      if (datanode.isRestarting()) {
822        // Do not throw if shutting down for restart. Otherwise, it will cause
823        // premature termination of responder.
824        LOG.info("Shutting down for restart (" + block + ").");
825      } else {
826        LOG.info("Exception for " + block, ioe);
827        throw ioe;
828      }
829    } finally {
830      // Clear the previous interrupt state of this thread.
831      Thread.interrupted();
832
833      // If a shutdown for restart was initiated, upstream needs to be notified.
834      // There is no need to do anything special if the responder was closed
835      // normally.
836      if (!responderClosed) { // Data transfer was not complete.
837        if (responder != null) {
838          // In case this datanode is shutting down for quick restart,
839          // send a special ack upstream.
840          if (datanode.isRestarting() && isClient && !isTransfer) {
841            File blockFile = ((ReplicaInPipeline)replicaInfo).getBlockFile();
842            File restartMeta = new File(blockFile.getParent()  + 
843                File.pathSeparator + "." + blockFile.getName() + ".restart");
844            if (restartMeta.exists() && !restartMeta.delete()) {
845              LOG.warn("Failed to delete restart meta file: " +
846                  restartMeta.getPath());
847            }
848            try (Writer out = new OutputStreamWriter(
849                new FileOutputStream(restartMeta), "UTF-8")) {
850              // write out the current time.
851              out.write(Long.toString(Time.now() + restartBudget));
852              out.flush();
853            } catch (IOException ioe) {
854              // The worst case is not recovering this RBW replica. 
855              // Client will fall back to regular pipeline recovery.
856            } finally {
857              IOUtils.cleanup(LOG, out);
858            }
859            try {              
860              // Even if the connection is closed after the ack packet is
861              // flushed, the client can react to the connection closure 
862              // first. Insert a delay to lower the chance of client 
863              // missing the OOB ack.
864              Thread.sleep(1000);
865            } catch (InterruptedException ie) {
866              // It is already going down. Ignore this.
867            }
868          }
869          responder.interrupt();
870        }
871        IOUtils.closeStream(this);
872        cleanupBlock();
873      }
874      if (responder != null) {
875        try {
876          responder.interrupt();
877          // join() on the responder should timeout a bit earlier than the
878          // configured deadline. Otherwise, the join() on this thread will
879          // likely timeout as well.
880          long joinTimeout = datanode.getDnConf().getXceiverStopTimeout();
881          joinTimeout = joinTimeout > 1  ? joinTimeout*8/10 : joinTimeout;
882          responder.join(joinTimeout);
883          if (responder.isAlive()) {
884            String msg = "Join on responder thread " + responder
885                + " timed out";
886            LOG.warn(msg + "\n" + StringUtils.getStackTrace(responder));
887            throw new IOException(msg);
888          }
889        } catch (InterruptedException e) {
890          responder.interrupt();
891          // do not throw if shutting down for restart.
892          if (!datanode.isRestarting()) {
893            throw new IOException("Interrupted receiveBlock");
894          }
895        }
896        responder = null;
897      }
898    }
899  }

926  /**
927   * Convert a checksum byte array to a long
928   */
929  static private long checksum2long(byte[] checksum) {
930    long crc = 0L;
931    for(int i=0; i<checksum.length; i++) {
932      crc |= (0xffL&checksum[i])<<((checksum.length-i-1)*8);
933    }
934    return crc;
935  }

937  /**
938   * reads in the partial crc chunk and computes checksum
939   * of pre-existing data in partial chunk.
940   */
941  private Checksum computePartialChunkCrc(long blkoff, long ckoff)
942      throws IOException {
943
944    // find offset of the beginning of partial chunk.
945    //
946    int sizePartialChunk = (int) (blkoff % bytesPerChecksum);
947    blkoff = blkoff - sizePartialChunk;
948    if (LOG.isDebugEnabled()) {
949      LOG.debug("computePartialChunkCrc for " + block
950          + ": sizePartialChunk=" + sizePartialChunk
951          + ", block offset=" + blkoff
952          + ", metafile offset=" + ckoff);
953    }
954
955    // create an input stream from the block file
956    // and read in partial crc chunk into temporary buffer
957    //
958    byte[] buf = new byte[sizePartialChunk];
959    byte[] crcbuf = new byte[checksumSize];
960    try (ReplicaInputStreams instr =
961        datanode.data.getTmpInputStreams(block, blkoff, ckoff)) {
962      IOUtils.readFully(instr.getDataIn(), buf, 0, sizePartialChunk);
963
964      // open meta file and read in crc value computer earlier
965      IOUtils.readFully(instr.getChecksumIn(), crcbuf, 0, crcbuf.length);
966    }
967
968    // compute crc of partial chunk from data read in the block file.
969    final Checksum partialCrc = DataChecksum.newDataChecksum(
970        diskChecksum.getChecksumType(), diskChecksum.getBytesPerChecksum());
971    partialCrc.update(buf, 0, sizePartialChunk);
972    if (LOG.isDebugEnabled()) {
973      LOG.debug("Read in partial CRC chunk from disk for " + block);
974    }
975
976    // paranoia! verify that the pre-computed crc matches what we
977    // recalculated just now
978    if (partialCrc.getValue() != checksum2long(crcbuf)) {
979      String msg = "Partial CRC " + partialCrc.getValue() +
980                   " does not match value computed the " +
981                   " last time file was closed " +
982                   checksum2long(crcbuf);
983      throw new IOException(msg);
984    }
985    return partialCrc;
986  }

995  private static enum PacketResponderType {
996    NON_PIPELINE, LAST_IN_PIPELINE, HAS_DOWNSTREAM_IN_PIPELINE
997  }
998
999  /**
1000   * Processes responses from downstream datanodes in the pipeline
1001   * and sends back replies to the originator.
1002   */
1003  class PacketResponder implements Runnable, Closeable {   
1004    /** queue for packets waiting for ack - synchronization using monitor lock */
1005    private final LinkedList<Packet> ackQueue = new LinkedList<Packet>(); 
1006    /** the thread that spawns this responder */
1007    private final Thread receiverThread = Thread.currentThread();
1008    /** is this responder running? - synchronization using monitor lock */
1009    private volatile boolean running = true;
1010    /** input from the next downstream datanode */
1011    private final DataInputStream downstreamIn;
1012    /** output to upstream datanode/client */
1013    private final DataOutputStream upstreamOut;
1014    /** The type of this responder */
1015    private final PacketResponderType type;
1016    /** for log and error messages */
1017    private final String myString; 
1018    private boolean sending = false;
1019
1020    @Override
1021    public String toString() {
1022      return myString;
1023    }
1024
1025    PacketResponder(final DataOutputStream upstreamOut,
1026        final DataInputStream downstreamIn, final DatanodeInfo[] downstreams) {
1027      this.downstreamIn = downstreamIn;
1028      this.upstreamOut = upstreamOut;
1029
1030      this.type = downstreams == null? PacketResponderType.NON_PIPELINE
1031          : downstreams.length == 0? PacketResponderType.LAST_IN_PIPELINE
1032              : PacketResponderType.HAS_DOWNSTREAM_IN_PIPELINE;
1033
1034      final StringBuilder b = new StringBuilder(getClass().getSimpleName())
1035          .append(": ").append(block).append(", type=").append(type);
1036      if (type != PacketResponderType.HAS_DOWNSTREAM_IN_PIPELINE) {
1037        b.append(", downstreams=").append(downstreams.length)
1038            .append(":").append(Arrays.asList(downstreams));
1039      }
1040      this.myString = b.toString();
1041    }
1042
1043    private boolean isRunning() {
1044      // When preparing for a restart, it should continue to run until
1045      // interrupted by the receiver thread.
1046      return running && (datanode.shouldRun || datanode.isRestarting());
1047    }
1048    
1049    /**
1050     * enqueue the seqno that is still be to acked by the downstream datanode.
1051     * @param seqno sequence number of the packet
1052     * @param lastPacketInBlock if true, this is the last packet in block
1053     * @param offsetInBlock offset of this packet in block
1054     */
1055    void enqueue(final long seqno, final boolean lastPacketInBlock,
1056        final long offsetInBlock, final Status ackStatus) {
1057      final Packet p = new Packet(seqno, lastPacketInBlock, offsetInBlock,
1058          System.nanoTime(), ackStatus);
1059      if(LOG.isDebugEnabled()) {
1060        LOG.debug(myString + ": enqueue " + p);
1061      }
1062      synchronized(ackQueue) {
1063        if (running) {
1064          ackQueue.addLast(p);
1065          ackQueue.notifyAll();
1066        }
1067      }
1068    }
1069
1070    /**
1071     * Send an OOB response. If all acks have been sent already for the block
1072     * and the responder is about to close, the delivery is not guaranteed.
1073     * This is because the other end can close the connection independently.
1074     * An OOB coming from downstream will be automatically relayed upstream
1075     * by the responder. This method is used only by originating datanode.
1076     *
1077     * @param ackStatus the type of ack to be sent
1078     */
1079    void sendOOBResponse(final Status ackStatus) throws IOException,
1080        InterruptedException {
1081      if (!running) {
1082        LOG.info("Cannot send OOB response " + ackStatus + 
1083            ". Responder not running.");
1084        return;
1085      }
1086
1087      synchronized(this) {
1088        if (sending) {
1089          wait(PipelineAck.getOOBTimeout(ackStatus));
1090          // Didn't get my turn in time. Give up.
1091          if (sending) {
1092            throw new IOException("Could not send OOB reponse in time: "
1093                + ackStatus);
1094          }
1095        }
1096        sending = true;
1097      }
1098
1099      LOG.info("Sending an out of band ack of type " + ackStatus);
1100      try {
1101        sendAckUpstreamUnprotected(null, PipelineAck.UNKOWN_SEQNO, 0L, 0L,
1102            PipelineAck.combineHeader(datanode.getECN(), ackStatus));
1103      } finally {
1104        // Let others send ack. Unless there are miltiple OOB send
1105        // calls, there can be only one waiter, the responder thread.
1106        // In any case, only one needs to be notified.
1107        synchronized(this) {
1108          sending = false;
1109          notify();
1110        }
1111      }
1112    }
1113    
1114    /** Wait for a packet with given {@code seqno} to be enqueued to ackQueue */
1115    Packet waitForAckHead(long seqno) throws InterruptedException {
1116      synchronized(ackQueue) {
1117        while (isRunning() && ackQueue.size() == 0) {
1118          if (LOG.isDebugEnabled()) {
1119            LOG.debug(myString + ": seqno=" + seqno +
1120                      " waiting for local datanode to finish write.");
1121          }
1122          ackQueue.wait();
1123        }
1124        return isRunning() ? ackQueue.getFirst() : null;
1125      }
1126    }
1127
1128    /**
1129     * wait for all pending packets to be acked. Then shutdown thread.
1130     */
1131    @Override
1132    public void close() {
1133      synchronized(ackQueue) {
1134        while (isRunning() && ackQueue.size() != 0) {
1135          try {
1136            ackQueue.wait();
1137          } catch (InterruptedException e) {
1138            running = false;
1139            Thread.currentThread().interrupt();
1140          }
1141        }
1142        if(LOG.isDebugEnabled()) {
1143          LOG.debug(myString + ": closing");
1144        }
1145        running = false;
1146        ackQueue.notifyAll();
1147      }
1148
1149      synchronized(this) {
1150        running = false;
1151        notifyAll();
1152      }
1153    }
1154
1155    /**
1156     * Thread to process incoming acks.
1157     * @see java.lang.Runnable#run()
1158     */
1159    @Override
1160    public void run() {
1161      boolean lastPacketInBlock = false;
1162      final long startTime = ClientTraceLog.isInfoEnabled() ? System.nanoTime() : 0;
1163      while (isRunning() && !lastPacketInBlock) {
1164        long totalAckTimeNanos = 0;
1165        boolean isInterrupted = false;
1166        try {
1167          Packet pkt = null;
1168          long expected = -2;
1169          PipelineAck ack = new PipelineAck();
1170          long seqno = PipelineAck.UNKOWN_SEQNO;
1171          long ackRecvNanoTime = 0;
1172          try {
1173            if (type != PacketResponderType.LAST_IN_PIPELINE && !mirrorError) {
1174              // read an ack from downstream datanode
1175              ack.readFields(downstreamIn);
1176              ackRecvNanoTime = System.nanoTime();
1177              if (LOG.isDebugEnabled()) {
1178                LOG.debug(myString + " got " + ack);
1179              }
1180              // Process an OOB ACK.
1181              Status oobStatus = ack.getOOBStatus();
1182              if (oobStatus != null) {
1183                LOG.info("Relaying an out of band ack of type " + oobStatus);
1184                sendAckUpstream(ack, PipelineAck.UNKOWN_SEQNO, 0L, 0L,
1185                    PipelineAck.combineHeader(datanode.getECN(),
1186                      Status.SUCCESS));
1187                continue;
1188              }
1189              seqno = ack.getSeqno();
1190            }
1191            if (seqno != PipelineAck.UNKOWN_SEQNO
1192                || type == PacketResponderType.LAST_IN_PIPELINE) {
1193              pkt = waitForAckHead(seqno);
1194              if (!isRunning()) {
1195                break;
1196              }
1197              expected = pkt.seqno;
1198              if (type == PacketResponderType.HAS_DOWNSTREAM_IN_PIPELINE
1199                  && seqno != expected) {
1200                throw new IOException(myString + "seqno: expected=" + expected
1201                    + ", received=" + seqno);
1202              }
1203              if (type == PacketResponderType.HAS_DOWNSTREAM_IN_PIPELINE) {
1204                // The total ack time includes the ack times of downstream
1205                // nodes.
1206                // The value is 0 if this responder doesn't have a downstream
1207                // DN in the pipeline.
1208                totalAckTimeNanos = ackRecvNanoTime - pkt.ackEnqueueNanoTime;
1209                // Report the elapsed time from ack send to ack receive minus
1210                // the downstream ack time.
1211                long ackTimeNanos = totalAckTimeNanos
1212                    - ack.getDownstreamAckTimeNanos();
1213                if (ackTimeNanos < 0) {
1214                  if (LOG.isDebugEnabled()) {
1215                    LOG.debug("Calculated invalid ack time: " + ackTimeNanos
1216                        + "ns.");
1217                  }
1218                } else {
1219                  datanode.metrics.addPacketAckRoundTripTimeNanos(ackTimeNanos);
1220                }
1221              }
1222              lastPacketInBlock = pkt.lastPacketInBlock;
1223            }
1224          } catch (InterruptedException ine) {
1225            isInterrupted = true;
1226          } catch (IOException ioe) {
1227            if (Thread.interrupted()) {
1228              isInterrupted = true;
1229            } else {
1230              // continue to run even if can not read from mirror
1231              // notify client of the error
1232              // and wait for the client to shut down the pipeline
1233              mirrorError = true;
1234              LOG.info(myString, ioe);
1235            }
1236          }
1237
1238          if (Thread.interrupted() || isInterrupted) {
1239            /*
1240             * The receiver thread cancelled this thread. We could also check
1241             * any other status updates from the receiver thread (e.g. if it is
1242             * ok to write to replyOut). It is prudent to not send any more
1243             * status back to the client because this datanode has a problem.
1244             * The upstream datanode will detect that this datanode is bad, and
1245             * rightly so.
1246             *
1247             * The receiver thread can also interrupt this thread for sending
1248             * an out-of-band response upstream.
1249             */
1250            LOG.info(myString + ": Thread is interrupted.");
1251            running = false;
1252            continue;
1253          }
1254
1255          if (lastPacketInBlock) {
1256            // Finalize the block and close the block file
1257            finalizeBlock(startTime);
1258          }
1259
1260          Status myStatus = pkt != null ? pkt.ackStatus : Status.SUCCESS;
1261          sendAckUpstream(ack, expected, totalAckTimeNanos,
1262            (pkt != null ? pkt.offsetInBlock : 0),
1263            PipelineAck.combineHeader(datanode.getECN(), myStatus));
1264          if (pkt != null) {
1265            // remove the packet from the ack queue
1266            removeAckHead();
1267          }
1268        } catch (IOException e) {
1269          LOG.warn("IOException in BlockReceiver.run(): ", e);
1270          if (running) {
1271            datanode.checkDiskErrorAsync();
1272            LOG.info(myString, e);
1273            running = false;
1274            if (!Thread.interrupted()) { // failure not caused by interruption
1275              receiverThread.interrupt();
1276            }
1277          }
1278        } catch (Throwable e) {
1279          if (running) {
1280            LOG.info(myString, e);
1281            running = false;
1282            receiverThread.interrupt();
1283          }
1284        }
1285      }
1286      LOG.info(myString + " terminating");
1287    }

1289    /**
1290     * Finalize the block and close the block file
1291     * @param startTime time when BlockReceiver started receiving the block
1292     */
1293    private void finalizeBlock(long startTime) throws IOException {
1294      long endTime = 0;
1295      // Hold a volume reference to finalize block.
1296      try (ReplicaHandler handler = BlockReceiver.this.claimReplicaHandler()) {
1297        BlockReceiver.this.close();
1298        endTime = ClientTraceLog.isInfoEnabled() ? System.nanoTime() : 0;
1299        block.setNumBytes(replicaInfo.getNumBytes());
1300        datanode.data.finalizeBlock(block);
1301      }
1302
1303      if (pinning) {
1304        datanode.data.setPinning(block);
1305      }
1306      
1307      datanode.closeBlock(
1308          block, DataNode.EMPTY_DEL_HINT, replicaInfo.getStorageUuid());
1309      if (ClientTraceLog.isInfoEnabled() && isClient) {
1310        long offset = 0;
1311        DatanodeRegistration dnR = datanode.getDNRegistrationForBP(block
1312            .getBlockPoolId());
1313        ClientTraceLog.info(String.format(DN_CLIENTTRACE_FORMAT, inAddr,
1314            myAddr, block.getNumBytes(), "HDFS_WRITE", clientname, offset,
1315            dnR.getDatanodeUuid(), block, endTime - startTime));
1316      } else {
1317        LOG.info("Received " + block + " size " + block.getNumBytes()
1318            + " from " + inAddr);
1319      }
1320    }

1322    /**
1323     * The wrapper for the unprotected version. This is only called by
1324     * the responder's run() method.
1325     *
1326     * @param ack Ack received from downstream
1327     * @param seqno sequence number of ack to be sent upstream
1328     * @param totalAckTimeNanos total ack time including all the downstream
1329     *          nodes
1330     * @param offsetInBlock offset in block for the data in packet
1331     * @param myHeader the local ack header
1332     */
1333    private void sendAckUpstream(PipelineAck ack, long seqno,
1334        long totalAckTimeNanos, long offsetInBlock,
1335        int myHeader) throws IOException {
1336      try {
1337        // Wait for other sender to finish. Unless there is an OOB being sent,
1338        // the responder won't have to wait.
1339        synchronized(this) {
1340          while(sending) {
1341            wait();
1342          }
1343          sending = true;
1344        }
1345
1346        try {
1347          if (!running) return;
1348          sendAckUpstreamUnprotected(ack, seqno, totalAckTimeNanos,
1349              offsetInBlock, myHeader);
1350        } finally {
1351          synchronized(this) {
1352            sending = false;
1353            notify();
1354          }
1355        }
1356      } catch (InterruptedException ie) {
1357        // The responder was interrupted. Make it go down without
1358        // interrupting the receiver(writer) thread.  
1359        running = false;
1360      }
1361    }

1363    /**
1364     * @param ack Ack received from downstream
1365     * @param seqno sequence number of ack to be sent upstream
1366     * @param totalAckTimeNanos total ack time including all the downstream
1367     *          nodes
1368     * @param offsetInBlock offset in block for the data in packet
1369     * @param myHeader the local ack header
1370     */
1371    private void sendAckUpstreamUnprotected(PipelineAck ack, long seqno,
1372        long totalAckTimeNanos, long offsetInBlock, int myHeader)
1373        throws IOException {
1374      final int[] replies;
1375      if (ack == null) {
1376        // A new OOB response is being sent from this node. Regardless of
1377        // downstream nodes, reply should contain one reply.
1378        replies = new int[] { myHeader };
1379      } else if (mirrorError) { // ack read error
1380        int h = PipelineAck.combineHeader(datanode.getECN(), Status.SUCCESS);
1381        int h1 = PipelineAck.combineHeader(datanode.getECN(), Status.ERROR);
1382        replies = new int[] {h, h1};
1383      } else {
1384        short ackLen = type == PacketResponderType.LAST_IN_PIPELINE ? 0 : ack
1385            .getNumOfReplies();
1386        replies = new int[ackLen + 1];
1387        replies[0] = myHeader;
1388        for (int i = 0; i < ackLen; ++i) {
1389          replies[i + 1] = ack.getHeaderFlag(i);
1390        }
1391        // If the mirror has reported that it received a corrupt packet,
1392        // do self-destruct to mark myself bad, instead of making the
1393        // mirror node bad. The mirror is guaranteed to be good without
1394        // corrupt data on disk.
1395        if (ackLen > 0 && PipelineAck.getStatusFromHeader(replies[1]) ==
1396          Status.ERROR_CHECKSUM) {
1397          throw new IOException("Shutting down writer and responder "
1398              + "since the down streams reported the data sent by this "
1399              + "thread is corrupt");
1400        }
1401      }
1402      PipelineAck replyAck = new PipelineAck(seqno, replies,
1403          totalAckTimeNanos);
1404      if (replyAck.isSuccess()
1405          && offsetInBlock > replicaInfo.getBytesAcked()) {
1406        replicaInfo.setBytesAcked(offsetInBlock);
1407      }
1408      // send my ack back to upstream datanode
1409      long begin = Time.monotonicNow();
1410      replyAck.write(upstreamOut);
1411      upstreamOut.flush();
1412      long duration = Time.monotonicNow() - begin;
1413      if (duration > datanodeSlowLogThresholdMs) {
1414        LOG.warn("Slow PacketResponder send ack to upstream took " + duration
1415            + "ms (threshold=" + datanodeSlowLogThresholdMs + "ms), " + myString
1416            + ", replyAck=" + replyAck);
1417      } else if (LOG.isDebugEnabled()) {
1418        LOG.debug(myString + ", replyAck=" + replyAck);
1419      }
1420
1421      // If a corruption was detected in the received data, terminate after
1422      // sending ERROR_CHECKSUM back.
1423      Status myStatus = PipelineAck.getStatusFromHeader(myHeader);
1424      if (myStatus == Status.ERROR_CHECKSUM) {
1425        throw new IOException("Shutting down writer and responder "
1426            + "due to a checksum error in received data. The error "
1427            + "response has been sent upstream.");
1428      }
1429    }

1432     * Remove a packet from the head of the ack queue
1433     * 
1434     * This should be called only when the ack queue is not empty
1435     */
1436    private void removeAckHead() {
1437      synchronized(ackQueue) {
1438        ackQueue.removeFirst();
1439        ackQueue.notifyAll();
1440      }
