049/**
050 * Manages storage for the set of BlockPoolSlices which share a particular 
051 * block pool id, on this DataNode.
052 * 
053 * This class supports the following functionality:
054 * <ol>
055 * <li> Formatting a new block pool storage</li>
056 * <li> Recovering a storage state to a consistent state (if possible></li>
057 * <li> Taking a snapshot of the block pool during upgrade</li>
058 * <li> Rolling back a block pool to a previous snapshot</li>
059 * <li> Finalizing block storage by deletion of a snapshot</li>
060 * </ul>
061 * 
062 * @see Storage
063 */
064@InterfaceAudience.Private
065public class BlockPoolSliceStorage extends Storage {
066  static final String TRASH_ROOT_DIR = "trash";

073   *   1. If the marker file is present, then a rolling upgrade just completed.

134  /**
135   * Load one storage directory. Recover from previous transitions if required.
136   *
137   * @param datanode datanode instance
138   * @param nsInfo namespace information
139   * @param dataDir the root path of the storage directory
140   * @param startOpt startup option
141   * @return the StorageDirectory successfully loaded.
142   * @throws IOException
143   */
144  private StorageDirectory loadStorageDirectory(DataNode datanode,
145      NamespaceInfo nsInfo, File dataDir, StartupOption startOpt) throws IOException {
146    StorageDirectory sd = new StorageDirectory(dataDir, null, true);
147    try {
148      StorageState curState = sd.analyzeStorage(startOpt, this);
149      // sd is locked but not opened
150      switch (curState) {
151      case NORMAL:
152        break;
153      case NON_EXISTENT:
154        LOG.info("Block pool storage directory " + dataDir + " does not exist");
155        throw new IOException("Storage directory " + dataDir
156            + " does not exist");
157      case NOT_FORMATTED: // format
158        LOG.info("Block pool storage directory " + dataDir
159            + " is not formatted for " + nsInfo.getBlockPoolID());
160        LOG.info("Formatting ...");
161        format(sd, nsInfo);
162        break;
163      default:  // recovery part is common
164        sd.doRecover(curState);
165      }
166
167      // 2. Do transitions
168      // Each storage directory is treated individually.
169      // During startup some of them can upgrade or roll back
170      // while others could be up-to-date for the regular startup.
171      doTransition(datanode, sd, nsInfo, startOpt);
172      if (getCTime() != nsInfo.getCTime()) {
173        throw new IOException(
174            "Data-node and name-node CTimes must be the same.");
175      }
176
177      // 3. Update successfully loaded storage.
178      setServiceLayoutVersion(getServiceLayoutVersion());
179      writeProperties(sd);
180
181      return sd;
182    } catch (IOException ioe) {
183      sd.unlock();
184      throw ioe;
185    }
186  }

188  /**
189   * Analyze and load storage directories. Recover from previous transitions if
190   * required.
191   *
192   * The block pool storages are either all analyzed or none of them is loaded.
193   * Therefore, a failure on loading any block pool storage results a faulty
194   * data volume.
195   *
196   * @param datanode Datanode to which this storage belongs to
197   * @param nsInfo namespace information
198   * @param dataDirs storage directories of block pool
199   * @param startOpt startup option
200   * @return an array of loaded block pool directories.
201   * @throws IOException on error
202   */
203  List<StorageDirectory> loadBpStorageDirectories(
204      DataNode datanode, NamespaceInfo nsInfo,
205      Collection<File> dataDirs, StartupOption startOpt) throws IOException {
206    List<StorageDirectory> succeedDirs = Lists.newArrayList();
207    try {
208      for (File dataDir : dataDirs) {
209        if (containsStorageDir(dataDir)) {
210          throw new IOException(
211              "BlockPoolSliceStorage.recoverTransitionRead: " +
212                  "attempt to load an used block storage: " + dataDir);
213        }
214        StorageDirectory sd =
215            loadStorageDirectory(datanode, nsInfo, dataDir, startOpt);
216        succeedDirs.add(sd);
217      }
218    } catch (IOException e) {
219      LOG.warn("Failed to analyze storage directories for block pool "
220          + nsInfo.getBlockPoolID(), e);
221      throw e;
222    }
223    return succeedDirs;
224  }

226  /**
227   * Analyze storage directories. Recover from previous transitions if required.
228   *
229   * The block pool storages are either all analyzed or none of them is loaded.
230   * Therefore, a failure on loading any block pool storage results a faulty
231   * data volume.
232   *
233   * @param datanode Datanode to which this storage belongs to
234   * @param nsInfo namespace information
235   * @param dataDirs storage directories of block pool
236   * @param startOpt startup option
237   * @throws IOException on error
238   */
239  void recoverTransitionRead(DataNode datanode, NamespaceInfo nsInfo,
240      Collection<File> dataDirs, StartupOption startOpt) throws IOException {
241    LOG.info("Analyzing storage directories for bpid " + nsInfo.getBlockPoolID());
242    for (StorageDirectory sd : loadBpStorageDirectories(
243        datanode, nsInfo, dataDirs, startOpt)) {
244      addStorageDir(sd);
245    }
246  }

269    bpSdir.clearDirectory(); // create directory
270    this.layoutVersion = HdfsConstants.DATANODE_LAYOUT_VERSION;

273    this.blockpoolID = nsInfo.getBlockPoolID();

277  /**
278   * Remove block pool level storage directory.
279   * @param absPathToRemove the absolute path of the root for the block pool
280   *                        level storage to remove.
281   */
282  void remove(File absPathToRemove) {
283    Preconditions.checkArgument(absPathToRemove.isAbsolute());
284    LOG.info("Removing block level storage: " + absPathToRemove);
285    for (Iterator<StorageDirectory> it = this.storageDirs.iterator();
286         it.hasNext(); ) {
287      StorageDirectory sd = it.next();
288      if (sd.getRoot().getAbsoluteFile().equals(absPathToRemove)) {
289        it.remove();
290        break;
291      }
292    }
293  }

308  /** Validate and set block pool ID */
309  private void setBlockPoolID(File storage, String bpid)
310      throws InconsistentFSStateException {
311    if (bpid == null || bpid.equals("")) {
312      throw new InconsistentFSStateException(storage, "file "
313          + STORAGE_FILE_VERSION + " is invalid.");
314    }
315    
316    if (!blockpoolID.equals("") && !blockpoolID.equals(bpid)) {
317      throw new InconsistentFSStateException(storage,
318          "Unexpected blockpoolID " + bpid + ". Expected " + blockpoolID);
319    }
320    blockpoolID = bpid;
321  }

334  /**
335   * Analyze whether a transition of the BP state is required and
336   * perform it if necessary.
337   * <br>
338   * Rollback if previousLV >= LAYOUT_VERSION && prevCTime <= namenode.cTime.
339   * Upgrade if this.LV > LAYOUT_VERSION || this.cTime < namenode.cTime Regular
340   * startup if this.LV = LAYOUT_VERSION && this.cTime = namenode.cTime
341   * 
342   * @param sd storage directory <SD>/current/<bpid>
343   * @param nsInfo namespace info
344   * @param startOpt startup option
345   * @throws IOException
346   */
347  private void doTransition(DataNode datanode, StorageDirectory sd,
348      NamespaceInfo nsInfo, StartupOption startOpt) throws IOException {
349    if (startOpt == StartupOption.ROLLBACK && sd.getPreviousDir().exists()) {
350      Preconditions.checkState(!getTrashRootDir(sd).exists(),
351          sd.getPreviousDir() + " and " + getTrashRootDir(sd) + " should not " +
352          " both be present.");
353      doRollback(sd, nsInfo); // rollback if applicable
354    } else {
355      // Restore all the files in the trash. The restored files are retained
356      // during rolling upgrade rollback. They are deleted during rolling
357      // upgrade downgrade.
358      int restored = restoreBlockFilesFromTrash(getTrashRootDir(sd));
359      LOG.info("Restored " + restored + " block files from trash.");
360    }
361    readProperties(sd);
362    checkVersionUpgradable(this.layoutVersion);
363    assert this.layoutVersion >= HdfsConstants.DATANODE_LAYOUT_VERSION 
364       : "Future version is not allowed";
365    if (getNamespaceID() != nsInfo.getNamespaceID()) {
366      throw new IOException("Incompatible namespaceIDs in "
367          + sd.getRoot().getCanonicalPath() + ": namenode namespaceID = "
368          + nsInfo.getNamespaceID() + "; datanode namespaceID = "
369          + getNamespaceID());
370    }
371    if (!blockpoolID.equals(nsInfo.getBlockPoolID())) {
372      throw new IOException("Incompatible blockpoolIDs in "
373          + sd.getRoot().getCanonicalPath() + ": namenode blockpoolID = "
374          + nsInfo.getBlockPoolID() + "; datanode blockpoolID = "
375          + blockpoolID);
376    }
377    if (this.layoutVersion == HdfsConstants.DATANODE_LAYOUT_VERSION
378        && this.cTime == nsInfo.getCTime()) {
379      return; // regular startup
380    }
381    if (this.layoutVersion > HdfsConstants.DATANODE_LAYOUT_VERSION
382        || this.cTime < nsInfo.getCTime()) {
383      doUpgrade(datanode, sd, nsInfo); // upgrade
384      return;
385    }
386    // layoutVersion == LAYOUT_VERSION && this.cTime > nsInfo.cTime
387    // must shutdown
388    throw new IOException("Datanode state: LV = " + this.getLayoutVersion()
389        + " CTime = " + this.getCTime()
390        + " is newer than the namespace state: LV = "
391        + nsInfo.getLayoutVersion() + " CTime = " + nsInfo.getCTime());
392  }

394  /**
395   * Upgrade to any release after 0.22 (0.22 included) release e.g. 0.22 => 0.23
396   * Upgrade procedure is as follows:
397   * <ol>
398   * <li>If <SD>/current/<bpid>/previous exists then delete it</li>
399   * <li>Rename <SD>/current/<bpid>/current to
400   * <SD>/current/bpid/current/previous.tmp</li>
401   * <li>Create new <SD>current/<bpid>/current directory</li>
402   * <ol>
403   * <li>Hard links for block files are created from previous.tmp to current</li>
404   * <li>Save new version file in current directory</li>
405   * </ol>
406   * <li>Rename previous.tmp to previous</li> </ol>
407   * 
408   * @param bpSd storage directory <SD>/current/<bpid>
409   * @param nsInfo Namespace Info from the namenode
410   * @throws IOException on error
411   */
412  void doUpgrade(DataNode datanode, StorageDirectory bpSd, NamespaceInfo nsInfo)
413      throws IOException {
414    // Upgrading is applicable only to release with federation or after
415    if (!DataNodeLayoutVersion.supports(
416        LayoutVersion.Feature.FEDERATION, layoutVersion)) {
417      return;
418    }
419    LOG.info("Upgrading block pool storage directory " + bpSd.getRoot()
420        + ".\n   old LV = " + this.getLayoutVersion() + "; old CTime = "
421        + this.getCTime() + ".\n   new LV = " + HdfsConstants.DATANODE_LAYOUT_VERSION
422        + "; new CTime = " + nsInfo.getCTime());
423    // get <SD>/previous directory
424    String dnRoot = getDataNodeStorageRoot(bpSd.getRoot().getCanonicalPath());
425    StorageDirectory dnSdStorage = new StorageDirectory(new File(dnRoot));
426    File dnPrevDir = dnSdStorage.getPreviousDir();
427    
428    // If <SD>/previous directory exists delete it
429    if (dnPrevDir.exists()) {
430      deleteDir(dnPrevDir);
431    }
432    File bpCurDir = bpSd.getCurrentDir();
433    File bpPrevDir = bpSd.getPreviousDir();
434    assert bpCurDir.exists() : "BP level current directory must exist.";
435    cleanupDetachDir(new File(bpCurDir, DataStorage.STORAGE_DIR_DETACHED));
436    
437    // 1. Delete <SD>/current/<bpid>/previous dir before upgrading
438    if (bpPrevDir.exists()) {
439      deleteDir(bpPrevDir);
440    }
441    File bpTmpDir = bpSd.getPreviousTmp();
442    assert !bpTmpDir.exists() : "previous.tmp directory must not exist.";
443    
444    // 2. Rename <SD>/current/<bpid>/current to
445    //    <SD>/current/<bpid>/previous.tmp
446    rename(bpCurDir, bpTmpDir);
447    
448    // 3. Create new <SD>/current with block files hardlinks and VERSION
449    linkAllBlocks(datanode, bpTmpDir, bpCurDir);
450    this.layoutVersion = HdfsConstants.DATANODE_LAYOUT_VERSION;
451    assert this.namespaceID == nsInfo.getNamespaceID() 
452        : "Data-node and name-node layout versions must be the same.";
453    this.cTime = nsInfo.getCTime();
454    writeProperties(bpSd);
455    
456    // 4.rename <SD>/current/<bpid>/previous.tmp to
457    // <SD>/current/<bpid>/previous
458    rename(bpTmpDir, bpPrevDir);
459    LOG.info("Upgrade of block pool " + blockpoolID + " at " + bpSd.getRoot()
460        + " is complete");
461  }

463  /**
464   * Cleanup the detachDir.
465   * 
466   * If the directory is not empty report an error; Otherwise remove the
467   * directory.
468   * 
469   * @param detachDir detach directory
470   * @throws IOException if the directory is not empty or it can not be removed
471   */
472  private void cleanupDetachDir(File detachDir) throws IOException {
473    if (!DataNodeLayoutVersion.supports(
474        LayoutVersion.Feature.APPEND_RBW_DIR, layoutVersion)
475        && detachDir.exists() && detachDir.isDirectory()) {
476
477      if (FileUtil.list(detachDir).length != 0) {
478        throw new IOException("Detached directory " + detachDir
479            + " is not empty. Please manually move each file under this "
480            + "directory to the finalized directory if the finalized "
481            + "directory tree does not have the file.");
482      } else if (!detachDir.delete()) {
483        throw new IOException("Cannot remove directory " + detachDir);
484      }
485    }
486  }

488  /**
489   * Restore all files from the trash directory to their corresponding
490   * locations under current/
491   */
492  private int restoreBlockFilesFromTrash(File trashRoot)
493      throws  IOException {
494    int filesRestored = 0;
495    File[] children = trashRoot.exists() ? trashRoot.listFiles() : null;
496    if (children == null) {
497      return 0;
498    }
499
500    File restoreDirectory = null;
501    for (File child : children) {
502      if (child.isDirectory()) {
503        // Recurse to process subdirectories.
504        filesRestored += restoreBlockFilesFromTrash(child);
505        continue;
506      }
507
508      if (restoreDirectory == null) {
509        restoreDirectory = new File(getRestoreDirectory(child));
510        if (!restoreDirectory.exists() && !restoreDirectory.mkdirs()) {
511          throw new IOException("Failed to create directory " + restoreDirectory);
512        }
513      }
514
515      final File newChild = new File(restoreDirectory, child.getName());
516
517      if (newChild.exists() && newChild.length() >= child.length()) {
518        // Failsafe - we should not hit this case but let's make sure
519        // we never overwrite a newer version of a block file with an
520        // older version.
521        LOG.info("Not overwriting " + newChild + " with smaller file from " +
522                     "trash directory. This message can be safely ignored.");
523      } else if (!child.renameTo(newChild)) {
524        throw new IOException("Failed to rename " + child + " to " + newChild);
525      } else {
526        ++filesRestored;
527      }
528    }
529    FileUtil.fullyDelete(trashRoot);
530    return filesRestored;
531  }

533  /*
534   * Roll back to old snapshot at the block pool level
535   * If previous directory exists: 
536   * <ol>
537   * <li>Rename <SD>/current/<bpid>/current to removed.tmp</li>
538   * <li>Rename * <SD>/current/<bpid>/previous to current</li>
539   * <li>Remove removed.tmp</li>
540   * </ol>
541   * 
542   * Do nothing if previous directory does not exist.
543   * @param bpSd Block pool storage directory at <SD>/current/<bpid>
544   */
545  void doRollback(StorageDirectory bpSd, NamespaceInfo nsInfo)
546      throws IOException {
547    File prevDir = bpSd.getPreviousDir();
548    // regular startup if previous dir does not exist
549    if (!prevDir.exists())
550      return;
551    // read attributes out of the VERSION file of previous directory
552    BlockPoolSliceStorage prevInfo = new BlockPoolSliceStorage();
553    prevInfo.readPreviousVersionProperties(bpSd);
554
555    // We allow rollback to a state, which is either consistent with
556    // the namespace state or can be further upgraded to it.
557    // In another word, we can only roll back when ( storedLV >= software LV)
558    // && ( DN.previousCTime <= NN.ctime)
559    if (!(prevInfo.getLayoutVersion() >= HdfsConstants.DATANODE_LAYOUT_VERSION && 
560        prevInfo.getCTime() <= nsInfo.getCTime())) { // cannot rollback
561      throw new InconsistentFSStateException(bpSd.getRoot(),
562          "Cannot rollback to a newer state.\nDatanode previous state: LV = "
563              + prevInfo.getLayoutVersion() + " CTime = " + prevInfo.getCTime()
564              + " is newer than the namespace state: LV = "
565              + HdfsConstants.DATANODE_LAYOUT_VERSION + " CTime = " + nsInfo.getCTime());
566    }
567    
568    LOG.info("Rolling back storage directory " + bpSd.getRoot()
569        + ".\n   target LV = " + nsInfo.getLayoutVersion()
570        + "; target CTime = " + nsInfo.getCTime());
571    File tmpDir = bpSd.getRemovedTmp();
572    assert !tmpDir.exists() : "removed.tmp directory must not exist.";
573    // 1. rename current to tmp
574    File curDir = bpSd.getCurrentDir();
575    assert curDir.exists() : "Current directory must exist.";
576    rename(curDir, tmpDir);
577    
578    // 2. rename previous to current
579    rename(prevDir, curDir);
580    
581    // 3. delete removed.tmp dir
582    deleteDir(tmpDir);
583    LOG.info("Rollback of " + bpSd.getRoot() + " is complete");
584  }

586  /*
587   * Finalize the block pool storage by deleting <BP>/previous directory
588   * that holds the snapshot.
589   */
590  void doFinalize(File dnCurDir) throws IOException {
591    File bpRoot = getBpRoot(blockpoolID, dnCurDir);
592    StorageDirectory bpSd = new StorageDirectory(bpRoot);
593    // block pool level previous directory
594    File prevDir = bpSd.getPreviousDir();
595    if (!prevDir.exists()) {
596      return; // already finalized
597    }
598    final String dataDirPath = bpSd.getRoot().getCanonicalPath();
599    LOG.info("Finalizing upgrade for storage directory " + dataDirPath
600        + ".\n   cur LV = " + this.getLayoutVersion() + "; cur CTime = "
601        + this.getCTime());
602    assert bpSd.getCurrentDir().exists() : "Current directory must exist.";
603    
604    // rename previous to finalized.tmp
605    final File tmpDir = bpSd.getFinalizedTmp();
606    rename(prevDir, tmpDir);
607
608    // delete finalized.tmp dir in a separate thread
609    new Daemon(new Runnable() {
610      @Override
611      public void run() {
612        try {
613          deleteDir(tmpDir);
614        } catch (IOException ex) {
615          LOG.error("Finalize upgrade for " + dataDirPath + " failed.", ex);
616        }
617        LOG.info("Finalize upgrade for " + dataDirPath + " is complete.");
618      }
619
620      @Override
621      public String toString() {
622        return "Finalize " + dataDirPath;
623      }
624    }).start();
625  }

627  /**
628   * Hardlink all finalized and RBW blocks in fromDir to toDir
629   * 
630   * @param fromDir directory where the snapshot is stored
631   * @param toDir the current data directory
632   * @throws IOException if error occurs during hardlink
633   */
634  private void linkAllBlocks(DataNode datanode, File fromDir, File toDir)
635      throws IOException {
636    // do the link
637    int diskLayoutVersion = this.getLayoutVersion();
638    // hardlink finalized blocks in tmpDir
639    HardLink hardLink = new HardLink();
640    DataStorage.linkBlocks(datanode, new File(fromDir, DataStorage.STORAGE_DIR_FINALIZED),
641      new File(toDir,DataStorage.STORAGE_DIR_FINALIZED), diskLayoutVersion, hardLink);
642    DataStorage.linkBlocks(datanode, new File(fromDir, DataStorage.STORAGE_DIR_RBW),
643        new File(toDir, DataStorage.STORAGE_DIR_RBW), diskLayoutVersion, hardLink);
644    LOG.info( hardLink.linkStats.report() );
645  }

683  /**
684   * Determine whether we can use trash for the given blockFile. Trash
685   * is disallowed if a 'previous' directory exists for the
686   * storage directory containing the block.
687   */
688  @VisibleForTesting

690    Matcher matcher = BLOCK_POOL_CURRENT_PATH_PATTERN.matcher(blockFile.getParent());

692    return !(new File(previousDir)).exists();

696   * Get a target subdirectory under trash/ for a given block file that is being
697   * deleted.
698   *
699   * The subdirectory structure under trash/ mirrors that under current/ to keep
700   * implicit memory of where the files are to be restored (if necessary).
701   *
702   * @return the trash directory for a given block file that is being deleted.
703   */
704  public String getTrashDirectory(File blockFile) {
705    if (isTrashAllowed(blockFile)) {
706      Matcher matcher = BLOCK_POOL_CURRENT_PATH_PATTERN.matcher(blockFile.getParent());
707      String trashDirectory = matcher.replaceFirst("$1$2" + TRASH_ROOT_DIR + "$4");
708      return trashDirectory;
709    }
710    return null;
711  }

730  /**
731   * Delete all files and directories in the trash directories.
732   */
733  public void restoreTrash() {
734    for (StorageDirectory sd : storageDirs) {
735      File trashRoot = getTrashRootDir(sd);
736      try {
737        Preconditions.checkState(!(trashRoot.exists() && sd.getPreviousDir().exists()));
738        restoreBlockFilesFromTrash(trashRoot);
739        FileUtil.fullyDelete(getTrashRootDir(sd));
740      } catch (IOException ioe) {
741        LOG.warn("Restoring trash failed for storage directory " + sd);
742      }
743    }
744  }

757  /**
758   * Create a rolling upgrade marker file for each BP storage root, if it
759   * does not exist already.
760   */
761  public void setRollingUpgradeMarkers(List<StorageDirectory> dnStorageDirs)
762      throws IOException {
763    for (StorageDirectory sd : dnStorageDirs) {
764      File bpRoot = getBpRoot(blockpoolID, sd.getCurrentDir());
765      File markerFile = new File(bpRoot, ROLLING_UPGRADE_MARKER_FILE);
766      if (!storagesWithRollingUpgradeMarker.contains(bpRoot.toString())) {
767        if (!markerFile.exists() && markerFile.createNewFile()) {
768          LOG.info("Created " + markerFile);
769        } else {
770          LOG.info(markerFile + " already exists.");
771        }
772        storagesWithRollingUpgradeMarker.add(bpRoot.toString());
773        storagesWithoutRollingUpgradeMarker.remove(bpRoot.toString());
774      }
775    }
776  }

778  /**
779   * Check whether the rolling upgrade marker file exists for each BP storage
780   * root. If it does exist, then the marker file is cleared and more
781   * importantly the layout upgrade is finalized.
782   */
783  public void clearRollingUpgradeMarkers(List<StorageDirectory> dnStorageDirs)
784      throws IOException {
785    for (StorageDirectory sd : dnStorageDirs) {
786      File bpRoot = getBpRoot(blockpoolID, sd.getCurrentDir());
787      File markerFile = new File(bpRoot, ROLLING_UPGRADE_MARKER_FILE);
788      if (!storagesWithoutRollingUpgradeMarker.contains(bpRoot.toString())) {
789        if (markerFile.exists()) {
790          LOG.info("Deleting " + markerFile);
791          doFinalize(sd.getCurrentDir());
792          if (!markerFile.delete()) {
793            LOG.warn("Failed to delete " + markerFile);
794          }
795        }
796        storagesWithoutRollingUpgradeMarker.add(bpRoot.toString());
797        storagesWithRollingUpgradeMarker.remove(bpRoot.toString());
798      }
799    }
800  }
