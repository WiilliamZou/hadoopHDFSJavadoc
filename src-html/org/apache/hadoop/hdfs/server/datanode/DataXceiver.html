125  private DataXceiver(Peer peer, DataNode datanode,
126      DataXceiverServer dataXceiverServer) throws IOException {
127
128    this.peer = peer;
129    this.dnConf = datanode.getDnConf();
130    this.socketIn = peer.getInputStream();
131    this.socketOut = peer.getOutputStream();
132    this.datanode = datanode;
133    this.dataXceiverServer = dataXceiverServer;
134    this.connectToDnViaHostname = datanode.getDnConf().connectToDnViaHostname;
135    remoteAddress = peer.getRemoteAddressString();
136    final int colonIdx = remoteAddress.indexOf(':');
137    remoteAddressWithoutPort =
138        (colonIdx < 0) ? remoteAddress : remoteAddress.substring(0, colonIdx);
139    localAddress = peer.getLocalAddressString();
140
141    if (LOG.isDebugEnabled()) {
142      LOG.debug("Number of active connections is: "
143          + datanode.getXceiverCount());
144    }
145  }

147  /**
148   * Update the current thread's name to contain the current status.
149   * Use this only after this receiver has started on its thread, i.e.,
150   * outside the constructor.
151   */
152  private void updateCurrentThreadName(String status) {
153    StringBuilder sb = new StringBuilder();
154    sb.append("DataXceiver for client ");
155    if (previousOpClientName != null) {
156      sb.append(previousOpClientName).append(" at ");
157    }
158    sb.append(remoteAddress);
159    if (status != null) {
160      sb.append(" [").append(status).append("]");
161    }
162    Thread.currentThread().setName(sb.toString());
163  }

178  /**
179   * Read/write data from/to the DataXceiverServer.
180   */
181  @Override
182  public void run() {
183    int opsProcessed = 0;
184    Op op = null;
185
186    try {
187      dataXceiverServer.addPeer(peer, Thread.currentThread(), this);
188      peer.setWriteTimeout(datanode.getDnConf().socketWriteTimeout);
189      InputStream input = socketIn;
190      try {
191        IOStreamPair saslStreams = datanode.saslServer.receive(peer, socketOut,
192          socketIn, datanode.getXferAddress().getPort(),
193          datanode.getDatanodeId());
194        input = new BufferedInputStream(saslStreams.in,
195          HdfsConstants.SMALL_BUFFER_SIZE);
196        socketOut = saslStreams.out;
197      } catch (InvalidMagicNumberException imne) {
198        if (imne.isHandshake4Encryption()) {
199          LOG.info("Failed to read expected encryption handshake from client " +
200              "at " + peer.getRemoteAddressString() + ". Perhaps the client " +
201              "is running an older version of Hadoop which does not support " +
202              "encryption");
203        } else {
204          LOG.info("Failed to read expected SASL data transfer protection " +
205              "handshake from client at " + peer.getRemoteAddressString() + 
206              ". Perhaps the client is running an older version of Hadoop " +
207              "which does not support SASL data transfer protection");
208        }
209        return;
210      }
211      
212      super.initialize(new DataInputStream(input));
213      
214      // We process requests in a loop, and stay around for a short timeout.
215      // This optimistic behaviour allows the other end to reuse connections.
216      // Setting keepalive timeout to 0 disable this behavior.
217      do {
218        updateCurrentThreadName("Waiting for operation #" + (opsProcessed + 1));
219
220        try {
221          if (opsProcessed != 0) {
222            assert dnConf.socketKeepaliveTimeout > 0;
223            peer.setReadTimeout(dnConf.socketKeepaliveTimeout);
224          } else {
225            peer.setReadTimeout(dnConf.socketTimeout);
226          }
227          op = readOp();
228        } catch (InterruptedIOException ignored) {
229          // Time out while we wait for client rpc
230          break;
231        } catch (IOException err) {
232          // Since we optimistically expect the next op, it's quite normal to get EOF here.
233          if (opsProcessed > 0 &&
234              (err instanceof EOFException || err instanceof ClosedChannelException)) {
235            if (LOG.isDebugEnabled()) {
236              LOG.debug("Cached " + peer + " closing after " + opsProcessed + " ops");
237            }
238          } else {
239            incrDatanodeNetworkErrors();
240            throw err;
241          }
242          break;
243        }
244
245        // restore normal timeout
246        if (opsProcessed != 0) {
247          peer.setReadTimeout(dnConf.socketTimeout);
248        }
249
250        opStartTime = monotonicNow();
251        processOp(op);
252        ++opsProcessed;
253      } while ((peer != null) &&
254          (!peer.isClosed() && dnConf.socketKeepaliveTimeout > 0));
255    } catch (Throwable t) {
256      String s = datanode.getDisplayName() + ":DataXceiver error processing "
257          + ((op == null) ? "unknown" : op.name()) + " operation "
258          + " src: " + remoteAddress + " dst: " + localAddress;
259      if (op == Op.WRITE_BLOCK && t instanceof ReplicaAlreadyExistsException) {
260        // For WRITE_BLOCK, it is okay if the replica already exists since
261        // client and replication may write the same block to the same datanode
262        // at the same time.
263        if (LOG.isTraceEnabled()) {
264          LOG.trace(s, t);
265        } else {
266          LOG.info(s + "; " + t);
267        }
268      } else if (op == Op.READ_BLOCK && t instanceof SocketTimeoutException) {
269        String s1 =
270            "Likely the client has stopped reading, disconnecting it";
271        s1 += " (" + s + ")";
272        if (LOG.isTraceEnabled()) {
273          LOG.trace(s1, t);
274        } else {
275          LOG.info(s1 + "; " + t);          
276        }
277      } else {
278        LOG.error(s, t);
279      }
280    } finally {
281      if (LOG.isDebugEnabled()) {
282        LOG.debug(datanode.getDisplayName() + ":Number of active connections is: "
283            + datanode.getXceiverCount());
284      }
285      updateCurrentThreadName("Cleaning up");
286      if (peer != null) {
287        dataXceiverServer.closePeer(peer);
288        IOUtils.closeStream(in);
289      }
290    }
291  }

293  @Override
294  public void requestShortCircuitFds(final ExtendedBlock blk,
295      final Token<BlockTokenIdentifier> token,
296      SlotId slotId, int maxVersion, boolean supportsReceiptVerification)
297        throws IOException {
298    updateCurrentThreadName("Passing file descriptors for block " + blk);
299    BlockOpResponseProto.Builder bld = BlockOpResponseProto.newBuilder();
300    FileInputStream fis[] = null;
301    SlotId registeredSlotId = null;
302    boolean success = false;
303    try {
304      try {
305        if (peer.getDomainSocket() == null) {
306          throw new IOException("You cannot pass file descriptors over " +
307              "anything but a UNIX domain socket.");
308        }
309        if (slotId != null) {
310          boolean isCached = datanode.data.
311              isCached(blk.getBlockPoolId(), blk.getBlockId());
312          datanode.shortCircuitRegistry.registerSlot(
313              ExtendedBlockId.fromExtendedBlock(blk), slotId, isCached);
314          registeredSlotId = slotId;
315        }
316        fis = datanode.requestShortCircuitFdsForRead(blk, token, maxVersion);
317        Preconditions.checkState(fis != null);
318        bld.setStatus(SUCCESS);
319        bld.setShortCircuitAccessVersion(DataNode.CURRENT_BLOCK_FORMAT_VERSION);
320      } catch (ShortCircuitFdsVersionException e) {
321        bld.setStatus(ERROR_UNSUPPORTED);
322        bld.setShortCircuitAccessVersion(DataNode.CURRENT_BLOCK_FORMAT_VERSION);
323        bld.setMessage(e.getMessage());
324      } catch (ShortCircuitFdsUnsupportedException e) {
325        bld.setStatus(ERROR_UNSUPPORTED);
326        bld.setMessage(e.getMessage());
327      } catch (InvalidToken e) {
328        bld.setStatus(ERROR_ACCESS_TOKEN);
329        bld.setMessage(e.getMessage());
330      } catch (IOException e) {
331        bld.setStatus(ERROR);
332        bld.setMessage(e.getMessage());
333      }
334      bld.build().writeDelimitedTo(socketOut);
335      if (fis != null) {
336        FileDescriptor fds[] = new FileDescriptor[fis.length];
337        for (int i = 0; i < fds.length; i++) {
338          fds[i] = fis[i].getFD();
339        }
340        byte buf[] = new byte[1];
341        if (supportsReceiptVerification) {
342          buf[0] = (byte)USE_RECEIPT_VERIFICATION.getNumber();
343        } else {
344          buf[0] = (byte)DO_NOT_USE_RECEIPT_VERIFICATION.getNumber();
345        }
346        DomainSocket sock = peer.getDomainSocket();
347        sock.sendFileDescriptors(fds, buf, 0, buf.length);
348        if (supportsReceiptVerification) {
349          LOG.trace("Reading receipt verification byte for " + slotId);
350          int val = sock.getInputStream().read();
351          if (val < 0) {
352            throw new EOFException();
353          }
354        } else {
355          LOG.trace("Receipt verification is not enabled on the DataNode.  " +
356                    "Not verifying " + slotId);
357        }
358        success = true;
359      }
360    } finally {
361      if ((!success) && (registeredSlotId != null)) {
362        LOG.info("Unregistering " + registeredSlotId + " because the " +
363            "requestShortCircuitFdsForRead operation failed.");
364        datanode.shortCircuitRegistry.unregisterSlot(registeredSlotId);
365      }
366      if (ClientTraceLog.isInfoEnabled()) {
367        DatanodeRegistration dnR = datanode.getDNRegistrationForBP(blk
368            .getBlockPoolId());
369        BlockSender.ClientTraceLog.info(String.format(
370            "src: 127.0.0.1, dest: 127.0.0.1, op: REQUEST_SHORT_CIRCUIT_FDS," +
371            " blockid: %s, srvID: %s, success: %b",
372            blk.getBlockId(), dnR.getDatanodeUuid(), success));
373      }
374      if (fis != null) {
375        IOUtils.cleanup(LOG, fis);
376      }
377    }
378  }

380  @Override
381  public void releaseShortCircuitFds(SlotId slotId) throws IOException {
382    boolean success = false;
383    try {
384      String error;
385      Status status;
386      try {
387        datanode.shortCircuitRegistry.unregisterSlot(slotId);
388        error = null;
389        status = Status.SUCCESS;
390      } catch (UnsupportedOperationException e) {
391        error = "unsupported operation";
392        status = Status.ERROR_UNSUPPORTED;
393      } catch (Throwable e) {
394        error = e.getMessage();
395        status = Status.ERROR_INVALID;
396      }
397      ReleaseShortCircuitAccessResponseProto.Builder bld =
398          ReleaseShortCircuitAccessResponseProto.newBuilder();
399      bld.setStatus(status);
400      if (error != null) {
401        bld.setError(error);
402      }
403      bld.build().writeDelimitedTo(socketOut);
404      success = true;
405    } finally {
406      if (ClientTraceLog.isInfoEnabled()) {
407        BlockSender.ClientTraceLog.info(String.format(
408            "src: 127.0.0.1, dest: 127.0.0.1, op: RELEASE_SHORT_CIRCUIT_FDS," +
409            " shmId: %016x%016x, slotIdx: %d, srvID: %s, success: %b",
410            slotId.getShmId().getHi(), slotId.getShmId().getLo(),
411            slotId.getSlotIdx(), datanode.getDatanodeUuid(), success));
412      }
413    }
414  }

422  private void sendShmSuccessResponse(DomainSocket sock, NewShmInfo shmInfo)
423      throws IOException {
424    ShortCircuitShmResponseProto.newBuilder().setStatus(SUCCESS).
425        setId(PBHelper.convert(shmInfo.shmId)).build().
426        writeDelimitedTo(socketOut);
427    // Send the file descriptor for the shared memory segment.
428    byte buf[] = new byte[] { (byte)0 };
429    FileDescriptor shmFdArray[] =
430        new FileDescriptor[] { shmInfo.stream.getFD() };
431    sock.sendFileDescriptors(shmFdArray, buf, 0, buf.length);
432  }

434  @Override
435  public void requestShortCircuitShm(String clientName) throws IOException {
436    NewShmInfo shmInfo = null;
437    boolean success = false;
438    DomainSocket sock = peer.getDomainSocket();
439    try {
440      if (sock == null) {
441        sendShmErrorResponse(ERROR_INVALID, "Bad request from " +
442            peer + ": must request a shared " +
443            "memory segment over a UNIX domain socket.");
444        return;
445      }
446      try {
447        shmInfo = datanode.shortCircuitRegistry.
448            createNewMemorySegment(clientName, sock);
449        // After calling #{ShortCircuitRegistry#createNewMemorySegment}, the
450        // socket is managed by the DomainSocketWatcher, not the DataXceiver.
451        releaseSocket();
452      } catch (UnsupportedOperationException e) {
453        sendShmErrorResponse(ERROR_UNSUPPORTED, 
454            "This datanode has not been configured to support " +
455            "short-circuit shared memory segments.");
456        return;
457      } catch (IOException e) {
458        sendShmErrorResponse(ERROR,
459            "Failed to create shared file descriptor: " + e.getMessage());
460        return;
461      }
462      sendShmSuccessResponse(sock, shmInfo);
463      success = true;
464    } finally {
465      if (ClientTraceLog.isInfoEnabled()) {
466        if (success) {
467          BlockSender.ClientTraceLog.info(String.format(
468              "cliID: %s, src: 127.0.0.1, dest: 127.0.0.1, " +
469              "op: REQUEST_SHORT_CIRCUIT_SHM," +
470              " shmId: %016x%016x, srvID: %s, success: true",
471              clientName, shmInfo.shmId.getHi(), shmInfo.shmId.getLo(),
472              datanode.getDatanodeUuid()));
473        } else {
474          BlockSender.ClientTraceLog.info(String.format(
475              "cliID: %s, src: 127.0.0.1, dest: 127.0.0.1, " +
476              "op: REQUEST_SHORT_CIRCUIT_SHM, " +
477              "shmId: n/a, srvID: %s, success: false",
478              clientName, datanode.getDatanodeUuid()));
479        }
480      }
481      if ((!success) && (peer == null)) {
482        // If we failed to pass the shared memory segment to the client,
483        // close the UNIX domain socket now.  This will trigger the 
484        // DomainSocketWatcher callback, cleaning up the segment.
485        IOUtils.cleanup(null, sock);
486      }
487      IOUtils.cleanup(null, shmInfo);
488    }
489  }

496  @Override
497  public void readBlock(final ExtendedBlock block,
498      final Token<BlockTokenIdentifier> blockToken,
499      final String clientName,
500      final long blockOffset,
501      final long length,
502      final boolean sendChecksum,
503      final CachingStrategy cachingStrategy) throws IOException {
504    previousOpClientName = clientName;
505    long read = 0;
506    OutputStream baseStream = getOutputStream();
507    DataOutputStream out = new DataOutputStream(new BufferedOutputStream(
508        baseStream, HdfsConstants.SMALL_BUFFER_SIZE));
509    checkAccess(out, true, block, blockToken,
510        Op.READ_BLOCK, BlockTokenSecretManager.AccessMode.READ);
511  
512    // send the block
513    BlockSender blockSender = null;
514    DatanodeRegistration dnR = 
515      datanode.getDNRegistrationForBP(block.getBlockPoolId());
516    final String clientTraceFmt =
517      clientName.length() > 0 && ClientTraceLog.isInfoEnabled()
518        ? String.format(DN_CLIENTTRACE_FORMAT, localAddress, remoteAddress,
519            "%d", "HDFS_READ", clientName, "%d",
520            dnR.getDatanodeUuid(), block, "%d")
521        : dnR + " Served block " + block + " to " +
522            remoteAddress;
523
524    updateCurrentThreadName("Sending block " + block);
525    try {
526      try {
527        blockSender = new BlockSender(block, blockOffset, length,
528            true, false, sendChecksum, datanode, clientTraceFmt,
529            cachingStrategy);
530      } catch(IOException e) {
531        String msg = "opReadBlock " + block + " received exception " + e; 
532        LOG.info(msg);
533        sendResponse(ERROR, msg);
534        throw e;
535      }
536      
537      // send op status
538      writeSuccessWithChecksumInfo(blockSender, new DataOutputStream(getOutputStream()));
539
540      long beginRead = Time.monotonicNow();
541      read = blockSender.sendBlock(out, baseStream, null); // send data
542      long duration = Time.monotonicNow() - beginRead;
543      if (blockSender.didSendEntireByteRange()) {
544        // If we sent the entire range, then we should expect the client
545        // to respond with a Status enum.
546        try {
547          ClientReadStatusProto stat = ClientReadStatusProto.parseFrom(
548              PBHelper.vintPrefixed(in));
549          if (!stat.hasStatus()) {
550            LOG.warn("Client " + peer.getRemoteAddressString() +
551                " did not send a valid status code after reading. " +
552                "Will close connection.");
553            IOUtils.closeStream(out);
554          }
555        } catch (IOException ioe) {
556          LOG.debug("Error reading client status response. Will close connection.", ioe);
557          IOUtils.closeStream(out);
558          incrDatanodeNetworkErrors();
559        }
560      } else {
561        IOUtils.closeStream(out);
562      }
563      datanode.metrics.incrBytesRead((int) read);
564      datanode.metrics.incrBlocksRead();
565      datanode.metrics.incrTotalReadTime(duration);
566    } catch ( SocketException ignored ) {
567      if (LOG.isTraceEnabled()) {
568        LOG.trace(dnR + ":Ignoring exception while serving " + block + " to " +
569            remoteAddress, ignored);
570      }
571      // Its ok for remote side to close the connection anytime.
572      datanode.metrics.incrBlocksRead();
573      IOUtils.closeStream(out);
574    } catch ( IOException ioe ) {
575      /* What exactly should we do here?
576       * Earlier version shutdown() datanode if there is disk error.
577       */
578      if (!(ioe instanceof SocketTimeoutException)) {
579        LOG.warn(dnR + ":Got exception while serving " + block + " to "
580          + remoteAddress, ioe);
581        incrDatanodeNetworkErrors();
582      }
583      throw ioe;
584    } finally {
585      IOUtils.closeStream(blockSender);
586    }
587
588    //update metrics
589    datanode.metrics.addReadBlockOp(elapsed());
590    datanode.metrics.incrReadsFromClient(peer.isLocal(), read);
591  }

601      final BlockConstructionStage stage,
602      final int pipelineSize,
603      final long minBytesRcvd,
604      final long maxBytesRcvd,
605      final long latestGenerationStamp,
606      DataChecksum requestedChecksum,
607      CachingStrategy cachingStrategy,
608      final boolean allowLazyPersist,
609      final boolean pinning,
610      final boolean[] targetPinnings) throws IOException {
611    previousOpClientName = clientname;
612    updateCurrentThreadName("Receiving block " + block);
613    final boolean isDatanode = clientname.length() == 0;
614    final boolean isClient = !isDatanode;
615    final boolean isTransfer = stage == BlockConstructionStage.TRANSFER_RBW
616        || stage == BlockConstructionStage.TRANSFER_FINALIZED;
617    long size = 0;
618    // check single target for transfer-RBW/Finalized 
619    if (isTransfer && targets.length > 0) {
620      throw new IOException(stage + " does not support multiple targets "
621          + Arrays.asList(targets));
622    }
623    
624    if (LOG.isDebugEnabled()) {
625      LOG.debug("opWriteBlock: stage=" + stage + ", clientname=" + clientname 
626                + "\n  block  =" + block + ", newGs=" + latestGenerationStamp
627                + ", bytesRcvd=[" + minBytesRcvd + ", " + maxBytesRcvd + "]"
628          + "\n  targets=" + Arrays.asList(targets)
629          + "; pipelineSize=" + pipelineSize + ", srcDataNode=" + srcDataNode
630          + ", pinning=" + pinning);
631      LOG.debug("isDatanode=" + isDatanode
632          + ", isClient=" + isClient
633          + ", isTransfer=" + isTransfer);
634      LOG.debug("writeBlock receive buf size " + peer.getReceiveBufferSize() +
635                " tcp no delay " + peer.getTcpNoDelay());
636    }
637
638    // We later mutate block's generation stamp and length, but we need to
639    // forward the original version of the block to downstream mirrors, so
640    // make a copy here.
641    final ExtendedBlock originalBlock = new ExtendedBlock(block);
642    if (block.getNumBytes() == 0) {
643      block.setNumBytes(dataXceiverServer.estimateBlockSize);
644    }
645    LOG.info("Receiving " + block + " src: " + remoteAddress + " dest: "
646        + localAddress);
647
648    // reply to upstream datanode or client 
649    final DataOutputStream replyOut = new DataOutputStream(
650        new BufferedOutputStream(
651            getOutputStream(),
652            HdfsConstants.SMALL_BUFFER_SIZE));
653    checkAccess(replyOut, isClient, block, blockToken,
654        Op.WRITE_BLOCK, BlockTokenSecretManager.AccessMode.WRITE);
655
656    DataOutputStream mirrorOut = null;  // stream to next target
657    DataInputStream mirrorIn = null;    // reply from next target
658    Socket mirrorSock = null;           // socket to next target
659    String mirrorNode = null;           // the name:port of next target
660    String firstBadLink = "";           // first datanode that failed in connection setup
661    Status mirrorInStatus = SUCCESS;
662    final String storageUuid;
663    try {
664      if (isDatanode || 
665          stage != BlockConstructionStage.PIPELINE_CLOSE_RECOVERY) {
666        // open a block receiver
667        blockReceiver = new BlockReceiver(block, storageType, in,
668            peer.getRemoteAddressString(),
669            peer.getLocalAddressString(),
670            stage, latestGenerationStamp, minBytesRcvd, maxBytesRcvd,
671            clientname, srcDataNode, datanode, requestedChecksum,
672            cachingStrategy, allowLazyPersist, pinning);
673
674        storageUuid = blockReceiver.getStorageUuid();
675      } else {
676        storageUuid = datanode.data.recoverClose(
677            block, latestGenerationStamp, minBytesRcvd);
678      }
679
680      //
681      // Connect to downstream machine, if appropriate
682      //
683      if (targets.length > 0) {
684        InetSocketAddress mirrorTarget = null;
685        // Connect to backup machine
686        mirrorNode = targets[0].getXferAddr(connectToDnViaHostname);
687        if (LOG.isDebugEnabled()) {
688          LOG.debug("Connecting to datanode " + mirrorNode);
689        }
690        mirrorTarget = NetUtils.createSocketAddr(mirrorNode);
691        mirrorSock = datanode.newSocket();
692        try {
693          int timeoutValue = dnConf.socketTimeout
694              + (HdfsServerConstants.READ_TIMEOUT_EXTENSION * targets.length);
695          int writeTimeout = dnConf.socketWriteTimeout + 
696                      (HdfsServerConstants.WRITE_TIMEOUT_EXTENSION * targets.length);
697          NetUtils.connect(mirrorSock, mirrorTarget, timeoutValue);
698          mirrorSock.setSoTimeout(timeoutValue);
699          mirrorSock.setSendBufferSize(HdfsConstants.DEFAULT_DATA_SOCKET_SIZE);
700          
701          OutputStream unbufMirrorOut = NetUtils.getOutputStream(mirrorSock,
702              writeTimeout);
703          InputStream unbufMirrorIn = NetUtils.getInputStream(mirrorSock);
704          DataEncryptionKeyFactory keyFactory =
705            datanode.getDataEncryptionKeyFactoryForBlock(block);
706          IOStreamPair saslStreams = datanode.saslClient.socketSend(mirrorSock,
707            unbufMirrorOut, unbufMirrorIn, keyFactory, blockToken, targets[0]);
708          unbufMirrorOut = saslStreams.out;
709          unbufMirrorIn = saslStreams.in;
710          mirrorOut = new DataOutputStream(new BufferedOutputStream(unbufMirrorOut,
711              HdfsConstants.SMALL_BUFFER_SIZE));
712          mirrorIn = new DataInputStream(unbufMirrorIn);
713
714          // Do not propagate allowLazyPersist to downstream DataNodes.
715          if (targetPinnings != null && targetPinnings.length > 0) {
716            new Sender(mirrorOut).writeBlock(originalBlock, targetStorageTypes[0],
717              blockToken, clientname, targets, targetStorageTypes, srcDataNode,
718              stage, pipelineSize, minBytesRcvd, maxBytesRcvd,
719              latestGenerationStamp, requestedChecksum, cachingStrategy,
720              false, targetPinnings[0], targetPinnings);
721          } else {
722            new Sender(mirrorOut).writeBlock(originalBlock, targetStorageTypes[0],
723              blockToken, clientname, targets, targetStorageTypes, srcDataNode,
724              stage, pipelineSize, minBytesRcvd, maxBytesRcvd,
725              latestGenerationStamp, requestedChecksum, cachingStrategy,
726              false, false, targetPinnings);
727          }
728
729          mirrorOut.flush();
730
731          DataNodeFaultInjector.get().writeBlockAfterFlush();
732
733          // read connect ack (only for clients, not for replication req)
734          if (isClient) {
735            BlockOpResponseProto connectAck =
736              BlockOpResponseProto.parseFrom(PBHelper.vintPrefixed(mirrorIn));
737            mirrorInStatus = connectAck.getStatus();
738            firstBadLink = connectAck.getFirstBadLink();
739            if (LOG.isDebugEnabled() || mirrorInStatus != SUCCESS) {
740              LOG.info("Datanode " + targets.length +
741                       " got response for connect ack " +
742                       " from downstream datanode with firstbadlink as " +
743                       firstBadLink);
744            }
745          }
746
747        } catch (IOException e) {
748          if (isClient) {
749            BlockOpResponseProto.newBuilder()
750              .setStatus(ERROR)
751               // NB: Unconditionally using the xfer addr w/o hostname
752              .setFirstBadLink(targets[0].getXferAddr())
753              .build()
754              .writeDelimitedTo(replyOut);
755            replyOut.flush();
756          }
757          IOUtils.closeStream(mirrorOut);
758          mirrorOut = null;
759          IOUtils.closeStream(mirrorIn);
760          mirrorIn = null;
761          IOUtils.closeSocket(mirrorSock);
762          mirrorSock = null;
763          if (isClient) {
764            LOG.error(datanode + ":Exception transfering block " +
765                      block + " to mirror " + mirrorNode + ": " + e);
766            throw e;
767          } else {
768            LOG.info(datanode + ":Exception transfering " +
769                     block + " to mirror " + mirrorNode +
770                     "- continuing without the mirror", e);
771            incrDatanodeNetworkErrors();
772          }
773        }
774      }
775
776      // send connect-ack to source for clients and not transfer-RBW/Finalized
777      if (isClient && !isTransfer) {
778        if (LOG.isDebugEnabled() || mirrorInStatus != SUCCESS) {
779          LOG.info("Datanode " + targets.length +
780                   " forwarding connect ack to upstream firstbadlink is " +
781                   firstBadLink);
782        }
783        BlockOpResponseProto.newBuilder()
784          .setStatus(mirrorInStatus)
785          .setFirstBadLink(firstBadLink)
786          .build()
787          .writeDelimitedTo(replyOut);
788        replyOut.flush();
789      }
790
791      // receive the block and mirror to the next target
792      if (blockReceiver != null) {
793        String mirrorAddr = (mirrorSock == null) ? null : mirrorNode;
794        blockReceiver.receiveBlock(mirrorOut, mirrorIn, replyOut,
795            mirrorAddr, null, targets, false);
796
797        // send close-ack for transfer-RBW/Finalized 
798        if (isTransfer) {
799          if (LOG.isTraceEnabled()) {
800            LOG.trace("TRANSFER: send close-ack");
801          }
802          writeResponse(SUCCESS, null, replyOut);
803        }
804      }
805
806      // update its generation stamp
807      if (isClient && 
808          stage == BlockConstructionStage.PIPELINE_CLOSE_RECOVERY) {
809        block.setGenerationStamp(latestGenerationStamp);
810        block.setNumBytes(minBytesRcvd);
811      }
812      
813      // if this write is for a replication request or recovering
814      // a failed close for client, then confirm block. For other client-writes,
815      // the block is finalized in the PacketResponder.
816      if (isDatanode ||
817          stage == BlockConstructionStage.PIPELINE_CLOSE_RECOVERY) {
818        datanode.closeBlock(block, DataNode.EMPTY_DEL_HINT, storageUuid);
819        LOG.info("Received " + block + " src: " + remoteAddress + " dest: "
820            + localAddress + " of size " + block.getNumBytes());
821      }
822
823      if(isClient) {
824        size = block.getNumBytes();
825      }
826    } catch (IOException ioe) {
827      LOG.info("opWriteBlock " + block + " received exception " + ioe);
828      incrDatanodeNetworkErrors();
829      throw ioe;
830    } finally {
831      // close all opened streams
832      IOUtils.closeStream(mirrorOut);
833      IOUtils.closeStream(mirrorIn);
834      IOUtils.closeStream(replyOut);
835      IOUtils.closeSocket(mirrorSock);
836      IOUtils.closeStream(blockReceiver);
837      blockReceiver = null;
838    }
839
840    //update metrics
841    datanode.metrics.addWriteBlockOp(elapsed());
842    datanode.metrics.incrWritesFromClient(peer.isLocal(), size);
843  }

871  private MD5Hash calcPartialBlockChecksum(ExtendedBlock block,
872      long requestLength, DataChecksum checksum, DataInputStream checksumIn)
873      throws IOException {
874    final int bytesPerCRC = checksum.getBytesPerChecksum();
875    final int csize = checksum.getChecksumSize();
876    final byte[] buffer = new byte[4*1024];
877    MessageDigest digester = MD5Hash.getDigester();
878
879    long remaining = requestLength / bytesPerCRC * csize;
880    for (int toDigest = 0; remaining > 0; remaining -= toDigest) {
881      toDigest = checksumIn.read(buffer, 0,
882          (int) Math.min(remaining, buffer.length));
883      if (toDigest < 0) {
884        break;
885      }
886      digester.update(buffer, 0, toDigest);
887    }
888    
889    int partialLength = (int) (requestLength % bytesPerCRC);
890    if (partialLength > 0) {
891      byte[] buf = new byte[partialLength];
892      final InputStream blockIn = datanode.data.getBlockInputStream(block,
893          requestLength - partialLength);
894      try {
895        // Get the CRC of the partialLength.
896        IOUtils.readFully(blockIn, buf, 0, partialLength);
897      } finally {
898        IOUtils.closeStream(blockIn);
899      }
900      checksum.update(buf, 0, partialLength);
901      byte[] partialCrc = new byte[csize];
902      checksum.writeValue(partialCrc, 0, true);
903      digester.update(partialCrc);
904    }
905    return new MD5Hash(digester.digest());
906  }

908  @Override
909  public void blockChecksum(final ExtendedBlock block,
910      final Token<BlockTokenIdentifier> blockToken) throws IOException {
911    final DataOutputStream out = new DataOutputStream(
912        getOutputStream());
913    checkAccess(out, true, block, blockToken,
914        Op.BLOCK_CHECKSUM, BlockTokenSecretManager.AccessMode.READ);
915    // client side now can specify a range of the block for checksum
916    long requestLength = block.getNumBytes();
917    Preconditions.checkArgument(requestLength >= 0);
918    long visibleLength = datanode.data.getReplicaVisibleLength(block);
919    boolean partialBlk = requestLength < visibleLength;
920
921    updateCurrentThreadName("Reading metadata for block " + block);
922    final LengthInputStream metadataIn = datanode.data
923        .getMetaDataInputStream(block);
924    
925    final DataInputStream checksumIn = new DataInputStream(
926        new BufferedInputStream(metadataIn, HdfsConstants.IO_FILE_BUFFER_SIZE));
927    updateCurrentThreadName("Getting checksum for block " + block);
928    try {
929      //read metadata file
930      final BlockMetadataHeader header = BlockMetadataHeader
931          .readHeader(checksumIn);
932      final DataChecksum checksum = header.getChecksum();
933      final int csize = checksum.getChecksumSize();
934      final int bytesPerCRC = checksum.getBytesPerChecksum();
935      final long crcPerBlock = csize <= 0 ? 0 : 
936        (metadataIn.getLength() - BlockMetadataHeader.getHeaderSize()) / csize;
937
938      final MD5Hash md5 = partialBlk && crcPerBlock > 0 ? 
939          calcPartialBlockChecksum(block, requestLength, checksum, checksumIn)
940            : MD5Hash.digest(checksumIn);
941      if (LOG.isDebugEnabled()) {
942        LOG.debug("block=" + block + ", bytesPerCRC=" + bytesPerCRC
943            + ", crcPerBlock=" + crcPerBlock + ", md5=" + md5);
944      }
945
946      //write reply
947      BlockOpResponseProto.newBuilder()
948        .setStatus(SUCCESS)
949        .setChecksumResponse(OpBlockChecksumResponseProto.newBuilder()             
950          .setBytesPerCrc(bytesPerCRC)
951          .setCrcPerBlock(crcPerBlock)
952          .setMd5(ByteString.copyFrom(md5.getDigest()))
953          .setCrcType(PBHelper.convert(checksum.getChecksumType())))
954        .build()
955        .writeDelimitedTo(out);
956      out.flush();
957    } catch (IOException ioe) {
958      LOG.info("blockChecksum " + block + " received exception " + ioe);
959      incrDatanodeNetworkErrors();
960      throw ioe;
961    } finally {
962      IOUtils.closeStream(out);
963      IOUtils.closeStream(checksumIn);
964      IOUtils.closeStream(metadataIn);
965    }
966
967    //update metrics
968    datanode.metrics.addBlockChecksumOp(elapsed());
969  }

971  @Override
972  public void copyBlock(final ExtendedBlock block,
973      final Token<BlockTokenIdentifier> blockToken) throws IOException {
974    updateCurrentThreadName("Copying block " + block);
975    // Read in the header
976    if (datanode.isBlockTokenEnabled) {
977      try {
978        datanode.blockPoolTokenSecretManager.checkAccess(blockToken, null, block,
979            BlockTokenSecretManager.AccessMode.COPY);
980      } catch (InvalidToken e) {
981        LOG.warn("Invalid access token in request from " + remoteAddress
982            + " for OP_COPY_BLOCK for block " + block + " : "
983            + e.getLocalizedMessage());
984        sendResponse(ERROR_ACCESS_TOKEN, "Invalid access token");
985        return;
986      }
987
988    }
989    
990    if (datanode.data.getPinning(block)) {
991      String msg = "Not able to copy block " + block.getBlockId() + " " +
992          "to " + peer.getRemoteAddressString() + " because it's pinned ";
993      LOG.info(msg);
994      sendResponse(ERROR, msg);
995    }
996    
997    if (!dataXceiverServer.balanceThrottler.acquire()) { // not able to start
998      String msg = "Not able to copy block " + block.getBlockId() + " " +
999          "to " + peer.getRemoteAddressString() + " because threads " +
1000          "quota is exceeded.";
1001      LOG.info(msg);
1002      sendResponse(ERROR, msg);
1003      return;
1004    }
1005
1006    BlockSender blockSender = null;
1007    DataOutputStream reply = null;
1008    boolean isOpSuccess = true;
1009
1010    try {
1011      // check if the block exists or not
1012      blockSender = new BlockSender(block, 0, -1, false, false, true, datanode, 
1013          null, CachingStrategy.newDropBehind());
1014
1015      // set up response stream
1016      OutputStream baseStream = getOutputStream();
1017      reply = new DataOutputStream(new BufferedOutputStream(
1018          baseStream, HdfsConstants.SMALL_BUFFER_SIZE));
1019
1020      // send status first
1021      writeSuccessWithChecksumInfo(blockSender, reply);
1022
1023      long beginRead = Time.monotonicNow();
1024      // send block content to the target
1025      long read = blockSender.sendBlock(reply, baseStream,
1026                                        dataXceiverServer.balanceThrottler);
1027      long duration = Time.monotonicNow() - beginRead;
1028      datanode.metrics.incrBytesRead((int) read);
1029      datanode.metrics.incrBlocksRead();
1030      datanode.metrics.incrTotalReadTime(duration);
1031      
1032      LOG.info("Copied " + block + " to " + peer.getRemoteAddressString());
1033    } catch (IOException ioe) {
1034      isOpSuccess = false;
1035      LOG.info("opCopyBlock " + block + " received exception " + ioe);
1036      incrDatanodeNetworkErrors();
1037      throw ioe;
1038    } finally {
1039      dataXceiverServer.balanceThrottler.release();
1040      if (isOpSuccess) {
1041        try {
1042          // send one last byte to indicate that the resource is cleaned.
1043          reply.writeChar('d');
1044        } catch (IOException ignored) {
1045        }
1046      }
1047      IOUtils.closeStream(reply);
1048      IOUtils.closeStream(blockSender);
1049    }
1050
1051    //update metrics    
1052    datanode.metrics.addCopyBlockOp(elapsed());
1053  }

1055  @Override
1056  public void replaceBlock(final ExtendedBlock block,
1057      final StorageType storageType, 
1058      final Token<BlockTokenIdentifier> blockToken,
1059      final String delHint,
1060      final DatanodeInfo proxySource) throws IOException {
1061    updateCurrentThreadName("Replacing block " + block + " from " + delHint);
1062
1063    /* read header */
1064    if (datanode.isBlockTokenEnabled) {
1065      try {
1066        datanode.blockPoolTokenSecretManager.checkAccess(blockToken, null, block,
1067            BlockTokenSecretManager.AccessMode.REPLACE);
1068      } catch (InvalidToken e) {
1069        LOG.warn("Invalid access token in request from " + remoteAddress
1070            + " for OP_REPLACE_BLOCK for block " + block + " : "
1071            + e.getLocalizedMessage());
1072        sendResponse(ERROR_ACCESS_TOKEN, "Invalid access token");
1073        return;
1074      }
1075    }
1076
1077    if (!dataXceiverServer.balanceThrottler.acquire()) { // not able to start
1078      String msg = "Not able to receive block " + block.getBlockId() +
1079          " from " + peer.getRemoteAddressString() + " because threads " +
1080          "quota is exceeded.";
1081      LOG.warn(msg);
1082      sendResponse(ERROR, msg);
1083      return;
1084    }
1085
1086    Socket proxySock = null;
1087    DataOutputStream proxyOut = null;
1088    Status opStatus = SUCCESS;
1089    String errMsg = null;
1090    BlockReceiver blockReceiver = null;
1091    DataInputStream proxyReply = null;
1092    DataOutputStream replyOut = new DataOutputStream(getOutputStream());
1093    boolean IoeDuringCopyBlockOperation = false;
1094    try {
1095      // Move the block to different storage in the same datanode
1096      if (proxySource.equals(datanode.getDatanodeId())) {
1097        ReplicaInfo oldReplica = datanode.data.moveBlockAcrossStorage(block,
1098            storageType);
1099        if (oldReplica != null) {
1100          LOG.info("Moved " + block + " from StorageType "
1101              + oldReplica.getVolume().getStorageType() + " to " + storageType);
1102        }
1103      } else {
1104        block.setNumBytes(dataXceiverServer.estimateBlockSize);
1105        // get the output stream to the proxy
1106        final String dnAddr = proxySource.getXferAddr(connectToDnViaHostname);
1107        if (LOG.isDebugEnabled()) {
1108          LOG.debug("Connecting to datanode " + dnAddr);
1109        }
1110        InetSocketAddress proxyAddr = NetUtils.createSocketAddr(dnAddr);
1111        proxySock = datanode.newSocket();
1112        NetUtils.connect(proxySock, proxyAddr, dnConf.socketTimeout);
1113        proxySock.setSoTimeout(dnConf.socketTimeout);
1114        
1115        OutputStream unbufProxyOut = NetUtils.getOutputStream(proxySock,
1116            dnConf.socketWriteTimeout);
1117        InputStream unbufProxyIn = NetUtils.getInputStream(proxySock);
1118        DataEncryptionKeyFactory keyFactory =
1119            datanode.getDataEncryptionKeyFactoryForBlock(block);
1120        IOStreamPair saslStreams = datanode.saslClient.socketSend(proxySock,
1121            unbufProxyOut, unbufProxyIn, keyFactory, blockToken, proxySource);
1122        unbufProxyOut = saslStreams.out;
1123        unbufProxyIn = saslStreams.in;
1124        
1125        proxyOut = new DataOutputStream(new BufferedOutputStream(unbufProxyOut, 
1126            HdfsConstants.SMALL_BUFFER_SIZE));
1127        proxyReply = new DataInputStream(new BufferedInputStream(unbufProxyIn,
1128            HdfsConstants.IO_FILE_BUFFER_SIZE));
1129        
1130        /* send request to the proxy */
1131        IoeDuringCopyBlockOperation = true;
1132        new Sender(proxyOut).copyBlock(block, blockToken);
1133        IoeDuringCopyBlockOperation = false;
1134        
1135        // receive the response from the proxy
1136        
1137        BlockOpResponseProto copyResponse = BlockOpResponseProto.parseFrom(
1138            PBHelper.vintPrefixed(proxyReply));
1139        
1140        String logInfo = "copy block " + block + " from "
1141            + proxySock.getRemoteSocketAddress();
1142        DataTransferProtoUtil.checkBlockOpStatus(copyResponse, logInfo);
1143
1144        // get checksum info about the block we're copying
1145        ReadOpChecksumInfoProto checksumInfo = copyResponse.getReadOpChecksumInfo();
1146        DataChecksum remoteChecksum = DataTransferProtoUtil.fromProto(
1147            checksumInfo.getChecksum());
1148        // open a block receiver and check if the block does not exist
1149        blockReceiver = new BlockReceiver(block, storageType,
1150            proxyReply, proxySock.getRemoteSocketAddress().toString(),
1151            proxySock.getLocalSocketAddress().toString(),
1152            null, 0, 0, 0, "", null, datanode, remoteChecksum,
1153            CachingStrategy.newDropBehind(), false, false);
1154        
1155        // receive a block
1156        blockReceiver.receiveBlock(null, null, replyOut, null, 
1157            dataXceiverServer.balanceThrottler, null, true);
1158        
1159        // notify name node
1160        datanode.notifyNamenodeReceivedBlock(
1161            block, delHint, blockReceiver.getStorageUuid());
1162        
1163        LOG.info("Moved " + block + " from " + peer.getRemoteAddressString()
1164            + ", delHint=" + delHint);
1165      }
1166    } catch (IOException ioe) {
1167      opStatus = ERROR;
1168      errMsg = "opReplaceBlock " + block + " received exception " + ioe; 
1169      LOG.info(errMsg);
1170      if (!IoeDuringCopyBlockOperation) {
1171        // Don't double count IO errors
1172        incrDatanodeNetworkErrors();
1173      }
1174      throw ioe;
1175    } finally {
1176      // receive the last byte that indicates the proxy released its thread resource
1177      if (opStatus == SUCCESS && proxyReply != null) {
1178        try {
1179          proxyReply.readChar();
1180        } catch (IOException ignored) {
1181        }
1182      }
1183      
1184      // now release the thread resource
1185      dataXceiverServer.balanceThrottler.release();
1186      
1187      // send response back
1188      try {
1189        sendResponse(opStatus, errMsg);
1190      } catch (IOException ioe) {
1191        LOG.warn("Error writing reply back to " + peer.getRemoteAddressString());
1192        incrDatanodeNetworkErrors();
1193      }
1194      IOUtils.closeStream(proxyOut);
1195      IOUtils.closeStream(blockReceiver);
1196      IOUtils.closeStream(proxyReply);
1197      IOUtils.closeStream(replyOut);
1198    }
1199
1200    //update metrics
1201    datanode.metrics.addReplaceBlockOp(elapsed());
1202  }

1219  private static void writeResponse(Status status, String message, OutputStream out)
1220  throws IOException {

1222      .setStatus(status);

1250  private void checkAccess(OutputStream out, final boolean reply, 
1251      final ExtendedBlock blk,
1252      final Token<BlockTokenIdentifier> t,
1253      final Op op,
1254      final BlockTokenSecretManager.AccessMode mode) throws IOException {
1255    if (datanode.isBlockTokenEnabled) {
1256      if (LOG.isDebugEnabled()) {
1257        LOG.debug("Checking block access token for block '" + blk.getBlockId()
1258            + "' with mode '" + mode + "'");
1259      }
1260      try {
1261        datanode.blockPoolTokenSecretManager.checkAccess(t, null, blk, mode);
1262      } catch(InvalidToken e) {
1263        try {
1264          if (reply) {
1265            BlockOpResponseProto.Builder resp = BlockOpResponseProto.newBuilder()
1266              .setStatus(ERROR_ACCESS_TOKEN);
1267            if (mode == BlockTokenSecretManager.AccessMode.WRITE) {
1268              DatanodeRegistration dnR = 
1269                datanode.getDNRegistrationForBP(blk.getBlockPoolId());
1270              // NB: Unconditionally using the xfer addr w/o hostname
1271              resp.setFirstBadLink(dnR.getXferAddr());
1272            }
1273            resp.build().writeDelimitedTo(out);
1274            out.flush();
1275          }
1276          LOG.warn("Block token verification failed: op=" + op
1277              + ", remoteAddress=" + remoteAddress
1278              + ", message=" + e.getLocalizedMessage());
1279          throw e;
1280        } finally {
1281          IOUtils.closeStream(out);
1282        }
1283      }
1284    }
1285  }
