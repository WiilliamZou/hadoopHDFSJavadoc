056/**
057 * Reads a block from the disk and sends it to a recipient.
058 * 
059 * Data sent from the BlockeSender in the following format:
060 * <br><b>Data format:</b> <pre>
061 *    +--------------------------------------------------+
062 *    | ChecksumHeader | Sequence of data PACKETS...     |
063 *    +--------------------------------------------------+ 
064 * </pre>   
065 * <b>ChecksumHeader format:</b> <pre>
066 *    +--------------------------------------------------+
067 *    | 1 byte CHECKSUM_TYPE | 4 byte BYTES_PER_CHECKSUM |
068 *    +--------------------------------------------------+ 
069 * </pre>   
070 * An empty packet is sent to mark the end of block and read completion.
071 * 
072 * PACKET Contains a packet header, checksum and data. Amount of data
073 * carried is set by BUFFER_SIZE.
074 * <pre>
075 *   +-----------------------------------------------------+
076 *   | Variable length header. See {@link PacketHeader}    |
077 *   +-----------------------------------------------------+
078 *   | x byte checksum data. x is defined below            |
079 *   +-----------------------------------------------------+
080 *   | actual data ......                                  |
081 *   +-----------------------------------------------------+
082 * 
083 *   Data is made of Chunks. Each chunk is of length <= BYTES_PER_CHECKSUM.
084 *   A checksum is calculated for each chunk.
085 *  
086 *   x = (length of data + BYTE_PER_CHECKSUM - 1)/BYTES_PER_CHECKSUM *
087 *       CHECKSUM_SIZE
088 *  
089 *   CHECKSUM_SIZE depends on CHECKSUM_TYPE (usually, 4 for CRC32) 
090 *  </pre>
091 *  
092 *  The client reads data until it receives a packet with 
093 *  "LastPacketInBlock" set to true or with a zero length. If there is 
094 *  no checksum error, it replies to DataNode with OP_STATUS_CHECKSUM_OK.
095 */
096class BlockSender implements java.io.Closeable {

172  /**
173   * Constructor
174   * 
175   * @param block Block that is being read
176   * @param startOffset starting offset to read from
177   * @param length length of data to read
178   * @param corruptChecksumOk if true, corrupt checksum is okay
179   * @param verifyChecksum verify checksum while reading the data
180   * @param sendChecksum send checksum to client.
181   * @param datanode datanode from which the block is being read
182   * @param clientTraceFmt format string used to print client trace logs
183   * @throws IOException
184   */
185  BlockSender(ExtendedBlock block, long startOffset, long length,
186              boolean corruptChecksumOk, boolean verifyChecksum,
187              boolean sendChecksum, DataNode datanode, String clientTraceFmt,
188              CachingStrategy cachingStrategy)
189      throws IOException {
190    try {
191      this.block = block;
192      this.corruptChecksumOk = corruptChecksumOk;
193      this.verifyChecksum = verifyChecksum;
194      this.clientTraceFmt = clientTraceFmt;
195
196      /*
197       * If the client asked for the cache to be dropped behind all reads,
198       * we honor that.  Otherwise, we use the DataNode defaults.
199       * When using DataNode defaults, we use a heuristic where we only
200       * drop the cache for large reads.
201       */
202      if (cachingStrategy.getDropBehind() == null) {
203        this.dropCacheBehindAllReads = false;
204        this.dropCacheBehindLargeReads =
205            datanode.getDnConf().dropCacheBehindReads;
206      } else {
207        this.dropCacheBehindAllReads =
208            this.dropCacheBehindLargeReads =
209                 cachingStrategy.getDropBehind().booleanValue();
210      }
211      /*
212       * Similarly, if readahead was explicitly requested, we always do it.
213       * Otherwise, we read ahead based on the DataNode settings, and only
214       * when the reads are large.
215       */
216      if (cachingStrategy.getReadahead() == null) {
217        this.alwaysReadahead = false;
218        this.readaheadLength = datanode.getDnConf().readaheadLength;
219      } else {
220        this.alwaysReadahead = true;
221        this.readaheadLength = cachingStrategy.getReadahead().longValue();
222      }
223      this.datanode = datanode;
224      
225      if (verifyChecksum) {
226        // To simplify implementation, callers may not specify verification
227        // without sending.
228        Preconditions.checkArgument(sendChecksum,
229            "If verifying checksum, currently must also send it.");
230      }
231      
232      final Replica replica;
233      final long replicaVisibleLength;
234      synchronized(datanode.data) { 
235        replica = getReplica(block, datanode);
236        replicaVisibleLength = replica.getVisibleLength();
237      }
238      // if there is a write in progress
239      ChunkChecksum chunkChecksum = null;
240      if (replica instanceof ReplicaBeingWritten) {
241        final ReplicaBeingWritten rbw = (ReplicaBeingWritten)replica;
242        waitForMinLength(rbw, startOffset + length);
243        chunkChecksum = rbw.getLastChecksumAndDataLen();
244      }
245
246      if (replica.getGenerationStamp() < block.getGenerationStamp()) {
247        throw new IOException("Replica gen stamp < block genstamp, block="
248            + block + ", replica=" + replica);
249      } else if (replica.getGenerationStamp() > block.getGenerationStamp()) {
250        if (DataNode.LOG.isDebugEnabled()) {
251          DataNode.LOG.debug("Bumping up the client provided"
252              + " block's genstamp to latest " + replica.getGenerationStamp()
253              + " for block " + block);
254        }
255        block.setGenerationStamp(replica.getGenerationStamp());
256      }
257      if (replicaVisibleLength < 0) {
258        throw new IOException("Replica is not readable, block="
259            + block + ", replica=" + replica);
260      }
261      if (DataNode.LOG.isDebugEnabled()) {
262        DataNode.LOG.debug("block=" + block + ", replica=" + replica);
263      }
264
265      // transferToFully() fails on 32 bit platforms for block sizes >= 2GB,
266      // use normal transfer in those cases
267      this.transferToAllowed = datanode.getDnConf().transferToAllowed &&
268        (!is32Bit || length <= Integer.MAX_VALUE);
269
270      // Obtain a reference before reading data
271      this.volumeRef = datanode.data.getVolume(block).obtainReference();
272
273      /* 
274       * (corruptChecksumOK, meta_file_exist): operation
275       * True,   True: will verify checksum  
276       * True,  False: No verify, e.g., need to read data from a corrupted file 
277       * False,  True: will verify checksum
278       * False, False: throws IOException file not found
279       */
280      DataChecksum csum = null;
281      if (verifyChecksum || sendChecksum) {
282        LengthInputStream metaIn = null;
283        boolean keepMetaInOpen = false;
284        try {
285          metaIn = datanode.data.getMetaDataInputStream(block);
286          if (!corruptChecksumOk || metaIn != null) {
287            if (metaIn == null) {
288              //need checksum but meta-data not found
289              throw new FileNotFoundException("Meta-data not found for " +
290                  block);
291            }
292
293            // The meta file will contain only the header if the NULL checksum
294            // type was used, or if the replica was written to transient storage.
295            // Checksum verification is not performed for replicas on transient
296            // storage.  The header is important for determining the checksum
297            // type later when lazy persistence copies the block to non-transient
298            // storage and computes the checksum.
299            if (metaIn.getLength() > BlockMetadataHeader.getHeaderSize()) {
300              checksumIn = new DataInputStream(new BufferedInputStream(
301                  metaIn, HdfsConstants.IO_FILE_BUFFER_SIZE));
302  
303              csum = BlockMetadataHeader.readDataChecksum(checksumIn, block);
304              keepMetaInOpen = true;
305            }
306          } else {
307            LOG.warn("Could not find metadata file for " + block);
308          }
309        } finally {
310          if (!keepMetaInOpen) {
311            IOUtils.closeStream(metaIn);
312          }
313        }
314      }
315      if (csum == null) {
316        // The number of bytes per checksum here determines the alignment
317        // of reads: we always start reading at a checksum chunk boundary,
318        // even if the checksum type is NULL. So, choosing too big of a value
319        // would risk sending too much unnecessary data. 512 (1 disk sector)
320        // is likely to result in minimal extra IO.
321        csum = DataChecksum.newDataChecksum(DataChecksum.Type.NULL, 512);
322      }
323
324      /*
325       * If chunkSize is very large, then the metadata file is mostly
326       * corrupted. For now just truncate bytesPerchecksum to blockLength.
327       */       
328      int size = csum.getBytesPerChecksum();
329      if (size > 10*1024*1024 && size > replicaVisibleLength) {
330        csum = DataChecksum.newDataChecksum(csum.getChecksumType(),
331            Math.max((int)replicaVisibleLength, 10*1024*1024));
332        size = csum.getBytesPerChecksum();        
333      }
334      chunkSize = size;
335      checksum = csum;
336      checksumSize = checksum.getChecksumSize();
337      length = length < 0 ? replicaVisibleLength : length;
338
339      // end is either last byte on disk or the length for which we have a 
340      // checksum
341      long end = chunkChecksum != null ? chunkChecksum.getDataLength()
342          : replica.getBytesOnDisk();
343      if (startOffset < 0 || startOffset > end
344          || (length + startOffset) > end) {
345        String msg = " Offset " + startOffset + " and length " + length
346        + " don't match block " + block + " ( blockLen " + end + " )";
347        LOG.warn(datanode.getDNRegistrationForBP(block.getBlockPoolId()) +
348            ":sendBlock() : " + msg);
349        throw new IOException(msg);
350      }
351      
352      // Ensure read offset is position at the beginning of chunk
353      offset = startOffset - (startOffset % chunkSize);
354      if (length >= 0) {
355        // Ensure endOffset points to end of chunk.
356        long tmpLen = startOffset + length;
357        if (tmpLen % chunkSize != 0) {
358          tmpLen += (chunkSize - tmpLen % chunkSize);
359        }
360        if (tmpLen < end) {
361          // will use on-disk checksum here since the end is a stable chunk
362          end = tmpLen;
363        } else if (chunkChecksum != null) {
364          // last chunk is changing. flag that we need to use in-memory checksum 
365          this.lastChunkChecksum = chunkChecksum;
366        }
367      }
368      endOffset = end;
369
370      // seek to the right offsets
371      if (offset > 0 && checksumIn != null) {
372        long checksumSkip = (offset / chunkSize) * checksumSize;
373        // note blockInStream is seeked when created below
374        if (checksumSkip > 0) {
375          // Should we use seek() for checksum file as well?
376          IOUtils.skipFully(checksumIn, checksumSkip);
377        }
378      }
379      seqno = 0;
380
381      if (DataNode.LOG.isDebugEnabled()) {
382        DataNode.LOG.debug("replica=" + replica);
383      }
384      blockIn = datanode.data.getBlockInputStream(block, offset); // seek to offset
385      if (blockIn instanceof FileInputStream) {
386        blockInFd = ((FileInputStream)blockIn).getFD();
387      } else {
388        blockInFd = null;
389      }
390    } catch (IOException ioe) {
391      IOUtils.closeStream(this);
392      IOUtils.closeStream(blockIn);
393      throw ioe;
394    }
395  }

397  /**
398   * close opened files.
399   */
400  @Override
401  public void close() throws IOException {
402    if (blockInFd != null &&
403        ((dropCacheBehindAllReads) ||
404         (dropCacheBehindLargeReads && isLongRead()))) {
405      try {
406        NativeIO.POSIX.getCacheManipulator().posixFadviseIfPossible(
407            block.getBlockName(), blockInFd, lastCacheDropOffset,
408            offset - lastCacheDropOffset,
409            NativeIO.POSIX.POSIX_FADV_DONTNEED);
410      } catch (Exception e) {
411        LOG.warn("Unable to drop cache on file close", e);
412      }
413    }
414    if (curReadahead != null) {
415      curReadahead.cancel();
416    }
417    
418    IOException ioe = null;
419    if(checksumIn!=null) {
420      try {
421        checksumIn.close(); // close checksum file
422      } catch (IOException e) {
423        ioe = e;
424      }
425      checksumIn = null;
426    }   
427    if(blockIn!=null) {
428      try {
429        blockIn.close(); // close data file
430      } catch (IOException e) {
431        ioe = e;
432      }
433      blockIn = null;
434      blockInFd = null;
435    }
436    if (volumeRef != null) {
437      IOUtils.cleanup(null, volumeRef);
438      volumeRef = null;
439    }
440    // throw IOException if there is any
441    if(ioe!= null) {
442      throw ioe;
443    }
444  }

446  private static Replica getReplica(ExtendedBlock block, DataNode datanode)
447      throws ReplicaNotFoundException {
448    Replica replica = datanode.data.getReplica(block.getBlockPoolId(),
449        block.getBlockId());
450    if (replica == null) {
451      throw new ReplicaNotFoundException(block);
452    }
453    return replica;
454  }

456  /**
457   * Wait for rbw replica to reach the length
458   * @param rbw replica that is being written to
459   * @param len minimum length to reach
460   * @throws IOException on failing to reach the len in given wait time
461   */
462  private static void waitForMinLength(ReplicaBeingWritten rbw, long len)
463      throws IOException {
464    // Wait for 3 seconds for rbw replica to reach the minimum length
465    for (int i = 0; i < 30 && rbw.getBytesOnDisk() < len; i++) {
466      try {
467        Thread.sleep(100);
468      } catch (InterruptedException ie) {
469        throw new IOException(ie);
470      }
471    }
472    long bytesOnDisk = rbw.getBytesOnDisk();
473    if (bytesOnDisk < len) {
474      throw new IOException(
475          String.format("Need %d bytes, but only %d bytes available", len,
476              bytesOnDisk));
477    }
478  }

480  /**
481   * Converts an IOExcpetion (not subclasses) to SocketException.
482   * This is typically done to indicate to upper layers that the error 
483   * was a socket error rather than often more serious exceptions like 
484   * disk errors.
485   */
486  private static IOException ioeToSocketException(IOException ioe) {
487    if (ioe.getClass().equals(IOException.class)) {
488      // "se" could be a new class in stead of SocketException.
489      IOException se = new SocketException("Original Exception : " + ioe);
490      se.initCause(ioe);
491      /* Change the stacktrace so that original trace is not truncated
492       * when printed.*/ 
493      se.setStackTrace(ioe.getStackTrace());
494      return se;
495    }
496    // otherwise just return the same exception.
497    return ioe;
498  }

500  /**
501   * @param datalen Length of data 
502   * @return number of chunks for data of given size
503   */
504  private int numberOfChunks(long datalen) {
505    return (int) ((datalen + chunkSize - 1)/chunkSize);
506  }

508  /**
509   * Sends a packet with up to maxChunks chunks of data.
510   * 
511   * @param pkt buffer used for writing packet data
512   * @param maxChunks maximum number of chunks to send
513   * @param out stream to send data to
514   * @param transferTo use transferTo to send data
515   * @param throttler used for throttling data transfer bandwidth
516   */
517  private int sendPacket(ByteBuffer pkt, int maxChunks, OutputStream out,
518      boolean transferTo, DataTransferThrottler throttler) throws IOException {
519    int dataLen = (int) Math.min(endOffset - offset,
520                             (chunkSize * (long) maxChunks));
521    
522    int numChunks = numberOfChunks(dataLen); // Number of chunks be sent in the packet
523    int checksumDataLen = numChunks * checksumSize;
524    int packetLen = dataLen + checksumDataLen + 4;
525    boolean lastDataPacket = offset + dataLen == endOffset && dataLen > 0;
526
527    // The packet buffer is organized as follows:
528    // _______HHHHCCCCD?D?D?D?
529    //        ^   ^
530    //        |   \ checksumOff
531    //        \ headerOff
532    // _ padding, since the header is variable-length
533    // H = header and length prefixes
534    // C = checksums
535    // D? = data, if transferTo is false.
536    
537    int headerLen = writePacketHeader(pkt, dataLen, packetLen);
538    
539    // Per above, the header doesn't start at the beginning of the
540    // buffer
541    int headerOff = pkt.position() - headerLen;
542    
543    int checksumOff = pkt.position();
544    byte[] buf = pkt.array();
545    
546    if (checksumSize > 0 && checksumIn != null) {
547      readChecksum(buf, checksumOff, checksumDataLen);
548
549      // write in progress that we need to use to get last checksum
550      if (lastDataPacket && lastChunkChecksum != null) {
551        int start = checksumOff + checksumDataLen - checksumSize;
552        byte[] updatedChecksum = lastChunkChecksum.getChecksum();
553        
554        if (updatedChecksum != null) {
555          System.arraycopy(updatedChecksum, 0, buf, start, checksumSize);
556        }
557      }
558    }
559    
560    int dataOff = checksumOff + checksumDataLen;
561    if (!transferTo) { // normal transfer
562      IOUtils.readFully(blockIn, buf, dataOff, dataLen);
563
564      if (verifyChecksum) {
565        verifyChecksum(buf, dataOff, dataLen, numChunks, checksumOff);
566      }
567    }
568    
569    try {
570      if (transferTo) {
571        SocketOutputStream sockOut = (SocketOutputStream)out;
572        // First write header and checksums
573        sockOut.write(buf, headerOff, dataOff - headerOff);
574        
575        // no need to flush since we know out is not a buffered stream
576        FileChannel fileCh = ((FileInputStream)blockIn).getChannel();
577        LongWritable waitTime = new LongWritable();
578        LongWritable transferTime = new LongWritable();
579        sockOut.transferToFully(fileCh, blockInPosition, dataLen, 
580            waitTime, transferTime);
581        datanode.metrics.addSendDataPacketBlockedOnNetworkNanos(waitTime.get());
582        datanode.metrics.addSendDataPacketTransferNanos(transferTime.get());
583        blockInPosition += dataLen;
584      } else {
585        // normal transfer
586        out.write(buf, headerOff, dataOff + dataLen - headerOff);
587      }
588    } catch (IOException e) {
589      if (e instanceof SocketTimeoutException) {
590        /*
591         * writing to client timed out.  This happens if the client reads
592         * part of a block and then decides not to read the rest (but leaves
593         * the socket open).
594         * 
595         * Reporting of this case is done in DataXceiver#run
596         */
597      } else {
598        /* Exception while writing to the client. Connection closure from
599         * the other end is mostly the case and we do not care much about
600         * it. But other things can go wrong, especially in transferTo(),
601         * which we do not want to ignore.
602         *
603         * The message parsing below should not be considered as a good
604         * coding example. NEVER do it to drive a program logic. NEVER.
605         * It was done here because the NIO throws an IOException for EPIPE.
606         */
607        String ioem = e.getMessage();
608        if (!ioem.startsWith("Broken pipe") && !ioem.startsWith("Connection reset")) {
609          LOG.error("BlockSender.sendChunks() exception: ", e);
610        }
611        datanode.getBlockScanner().markSuspectBlock(
612              volumeRef.getVolume().getStorageID(),
613              block);
614      }
615      throw ioeToSocketException(e);
616    }
617
618    if (throttler != null) { // rebalancing so throttle
619      throttler.throttle(packetLen);
620    }
621
622    return dataLen;
623  }

625  /**
626   * Read checksum into given buffer
627   * @param buf buffer to read the checksum into
628   * @param checksumOffset offset at which to write the checksum into buf
629   * @param checksumLen length of checksum to write
630   * @throws IOException on error
631   */
632  private void readChecksum(byte[] buf, final int checksumOffset,
633      final int checksumLen) throws IOException {
634    if (checksumSize <= 0 && checksumIn == null) {
635      return;
636    }
637    try {
638      checksumIn.readFully(buf, checksumOffset, checksumLen);
639    } catch (IOException e) {
640      LOG.warn(" Could not read or failed to veirfy checksum for data"
641          + " at offset " + offset + " for block " + block, e);
642      IOUtils.closeStream(checksumIn);
643      checksumIn = null;
644      if (corruptChecksumOk) {
645        if (checksumOffset < checksumLen) {
646          // Just fill the array with zeros.
647          Arrays.fill(buf, checksumOffset, checksumLen, (byte) 0);
648        }
649      } else {
650        throw e;
651      }
652    }
653  }

655  /**
656   * Compute checksum for chunks and verify the checksum that is read from
657   * the metadata file is correct.
658   * 
659   * @param buf buffer that has checksum and data
660   * @param dataOffset position where data is written in the buf
661   * @param datalen length of data
662   * @param numChunks number of chunks corresponding to data
663   * @param checksumOffset offset where checksum is written in the buf
664   * @throws ChecksumException on failed checksum verification
665   */
666  public void verifyChecksum(final byte[] buf, final int dataOffset,
667      final int datalen, final int numChunks, final int checksumOffset)
668      throws ChecksumException {
669    int dOff = dataOffset;
670    int cOff = checksumOffset;
671    int dLeft = datalen;
672
673    for (int i = 0; i < numChunks; i++) {
674      checksum.reset();
675      int dLen = Math.min(dLeft, chunkSize);
676      checksum.update(buf, dOff, dLen);
677      if (!checksum.compare(buf, cOff)) {
678        long failedPos = offset + datalen - dLeft;
679        throw new ChecksumException("Checksum failed at " + failedPos,
680            failedPos);
681      }
682      dLeft -= dLen;
683      dOff += dLen;
684      cOff += checksumSize;
685    }
686  }

688  /**
689   * sendBlock() is used to read block and its metadata and stream the data to
690   * either a client or to another datanode. 
691   * 
692   * @param out  stream to which the block is written to
693   * @param baseStream optional. if non-null, <code>out</code> is assumed to 
694   *        be a wrapper over this stream. This enables optimizations for
695   *        sending the data, e.g. 
696   *        {@link SocketOutputStream#transferToFully(FileChannel, 
697   *        long, int)}.
698   * @param throttler for sending data.
699   * @return total bytes read, including checksum data.
700   */
701  long sendBlock(DataOutputStream out, OutputStream baseStream, 
702                 DataTransferThrottler throttler) throws IOException {
703    TraceScope scope =
704        Trace.startSpan("sendBlock_" + block.getBlockId(), Sampler.NEVER);
705    try {
706      return doSendBlock(out, baseStream, throttler);
707    } finally {
708      scope.close();
709    }
710  }
711
712  private long doSendBlock(DataOutputStream out, OutputStream baseStream,
713        DataTransferThrottler throttler) throws IOException {
714    if (out == null) {
715      throw new IOException( "out stream is null" );
716    }
717    initialOffset = offset;
718    long totalRead = 0;
719    OutputStream streamForSendChunks = out;
720    
721    lastCacheDropOffset = initialOffset;
722
723    if (isLongRead() && blockInFd != null) {
724      // Advise that this file descriptor will be accessed sequentially.
725      NativeIO.POSIX.getCacheManipulator().posixFadviseIfPossible(
726          block.getBlockName(), blockInFd, 0, 0,
727          NativeIO.POSIX.POSIX_FADV_SEQUENTIAL);
728    }
729    
730    // Trigger readahead of beginning of file if configured.
731    manageOsCache();
732
733    final long startTime = ClientTraceLog.isDebugEnabled() ? System.nanoTime() : 0;
734    try {
735      int maxChunksPerPacket;
736      int pktBufSize = PacketHeader.PKT_MAX_HEADER_LEN;
737      boolean transferTo = transferToAllowed && !verifyChecksum
738          && baseStream instanceof SocketOutputStream
739          && blockIn instanceof FileInputStream;
740      if (transferTo) {
741        FileChannel fileChannel = ((FileInputStream)blockIn).getChannel();
742        blockInPosition = fileChannel.position();
743        streamForSendChunks = baseStream;
744        maxChunksPerPacket = numberOfChunks(TRANSFERTO_BUFFER_SIZE);
745        
746        // Smaller packet size to only hold checksum when doing transferTo
747        pktBufSize += checksumSize * maxChunksPerPacket;
748      } else {
749        maxChunksPerPacket = Math.max(1,
750            numberOfChunks(HdfsConstants.IO_FILE_BUFFER_SIZE));
751        // Packet size includes both checksum and data
752        pktBufSize += (chunkSize + checksumSize) * maxChunksPerPacket;
753      }
754
755      ByteBuffer pktBuf = ByteBuffer.allocate(pktBufSize);
756
757      while (endOffset > offset && !Thread.currentThread().isInterrupted()) {
758        manageOsCache();
759        long len = sendPacket(pktBuf, maxChunksPerPacket, streamForSendChunks,
760            transferTo, throttler);
761        offset += len;
762        totalRead += len + (numberOfChunks(len) * checksumSize);
763        seqno++;
764      }
765      // If this thread was interrupted, then it did not send the full block.
766      if (!Thread.currentThread().isInterrupted()) {
767        try {
768          // send an empty packet to mark the end of the block
769          sendPacket(pktBuf, maxChunksPerPacket, streamForSendChunks, transferTo,
770              throttler);
771          out.flush();
772        } catch (IOException e) { //socket error
773          throw ioeToSocketException(e);
774        }
775
776        sentEntireByteRange = true;
777      }
778    } finally {
779      if ((clientTraceFmt != null) && ClientTraceLog.isDebugEnabled()) {
780        final long endTime = System.nanoTime();
781        ClientTraceLog.debug(String.format(clientTraceFmt, totalRead,
782            initialOffset, endTime - startTime));
783      }
784      close();
785    }
786    return totalRead;
787  }

789  /**
790   * Manage the OS buffer cache by performing read-ahead
791   * and drop-behind.
792   */
793  private void manageOsCache() throws IOException {
794    // We can't manage the cache for this block if we don't have a file
795    // descriptor to work with.
796    if (blockInFd == null) return;
797
798    // Perform readahead if necessary
799    if ((readaheadLength > 0) && (datanode.readaheadPool != null) &&
800          (alwaysReadahead || isLongRead())) {
801      curReadahead = datanode.readaheadPool.readaheadStream(
802          clientTraceFmt, blockInFd, offset, readaheadLength, Long.MAX_VALUE,
803          curReadahead);
804    }
805
806    // Drop what we've just read from cache, since we aren't
807    // likely to need it again
808    if (dropCacheBehindAllReads ||
809        (dropCacheBehindLargeReads && isLongRead())) {
810      long nextCacheDropOffset = lastCacheDropOffset + CACHE_DROP_INTERVAL_BYTES;
811      if (offset >= nextCacheDropOffset) {
812        long dropLength = offset - lastCacheDropOffset;
813        NativeIO.POSIX.getCacheManipulator().posixFadviseIfPossible(
814            block.getBlockName(), blockInFd, lastCacheDropOffset,
815            dropLength, NativeIO.POSIX.POSIX_FADV_DONTNEED);
816        lastCacheDropOffset = offset;
817      }
818    }
819  }

821  /**
822   * Returns true if we have done a long enough read for this block to qualify
823   * for the DataNode-wide cache management defaults.  We avoid applying the
824   * cache management defaults to smaller reads because the overhead would be
825   * too high.
826   *
827   * Note that if the client explicitly asked for dropBehind, we will do it
828   * even on short reads.
829   * 
830   * This is also used to determine when to invoke
831   * posix_fadvise(POSIX_FADV_SEQUENTIAL).
832   */
833  private boolean isLongRead() {
834    return (endOffset - initialOffset) > LONG_READ_THRESHOLD_BYTES;
835  }

837  /**
838   * Write packet header into {@code pkt},
839   * return the length of the header written.
840   */
841  private int writePacketHeader(ByteBuffer pkt, int dataLen, int packetLen) {
842    pkt.clear();
843    // both syncBlock and syncPacket are false
844    PacketHeader header = new PacketHeader(packetLen, offset, seqno,
845        (dataLen == 0), dataLen, false);
846    
847    int size = header.getSerializedSize();
848    pkt.position(PacketHeader.PKT_MAX_HEADER_LEN - size);
849    header.putInBuffer(pkt);
850    return size;
851  }
