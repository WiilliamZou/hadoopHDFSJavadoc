067/**
068 * A thread per active or standby namenode to perform:
069 * <ul>
070 * <li> Pre-registration handshake with namenode</li>
071 * <li> Registration with namenode</li>
072 * <li> Send periodic heartbeats to the namenode</li>
073 * <li> Handle commands received from the namenode</li>
074 * </ul>
075 */

164  /**
165   * Perform the first part of the handshake with the NameNode.
166   * This calls <code>versionRequest</code> to determine the NN's
167   * namespace and version info. It automatically retries until
168   * the NN responds or the DN is shutting down.
169   * 
170   * @return the NamespaceInfo
171   */
172  @VisibleForTesting
173  NamespaceInfo retrieveNamespaceInfo() throws IOException {
174    NamespaceInfo nsInfo = null;
175    while (shouldRun()) {
176      try {
177        nsInfo = bpNamenode.versionRequest();
178        LOG.debug(this + " received versionRequest response: " + nsInfo);
179        break;
180      } catch(SocketTimeoutException e) {  // namenode is busy
181        LOG.warn("Problem connecting to server: " + nnAddr);
182      } catch(IOException e ) {  // namenode is not available
183        LOG.warn("Problem connecting to server: " + nnAddr);
184      }
185      
186      // try again in a second
187      sleepAndLogInterrupts(5000, "requesting version info from NN");
188    }
189    
190    if (nsInfo != null) {
191      checkNNVersion(nsInfo);
192    } else {
193      throw new IOException("DN shut down before block pool connected");
194    }
195    return nsInfo;
196  }
197
198  private void checkNNVersion(NamespaceInfo nsInfo)
199      throws IncorrectVersionException {
200    // build and layout versions should match
201    String nnVersion = nsInfo.getSoftwareVersion();
202    String minimumNameNodeVersion = dnConf.getMinimumNameNodeVersion();
203    if (VersionUtil.compareVersions(nnVersion, minimumNameNodeVersion) < 0) {
204      IncorrectVersionException ive = new IncorrectVersionException(
205          minimumNameNodeVersion, nnVersion, "NameNode", "DataNode");
206      LOG.warn(ive.getMessage());
207      throw ive;
208    }
209    String dnVersion = VersionInfo.getVersion();
210    if (!nnVersion.equals(dnVersion)) {
211      LOG.info("Reported NameNode version '" + nnVersion + "' does not match " +
212          "DataNode version '" + dnVersion + "' but is within acceptable " +
213          "limits. Note: This is normal during a rolling upgrade.");
214    }
215  }

217  private void connectToNNAndHandshake() throws IOException {
218    // get NN proxy
219    bpNamenode = dn.connectToNN(nnAddr);
220
221    // First phase of the handshake with NN - get the namespace
222    // info.
223    NamespaceInfo nsInfo = retrieveNamespaceInfo();
224    
225    // Verify that this matches the other NN in this HA pair.
226    // This also initializes our block pool in the DN if we are
227    // the first NN connection for this BP.
228    bpos.verifyAndSetNamespaceInfo(nsInfo);
229    
230    // Second phase of the handshake with the NN.
231    register(nsInfo);
232  }

234  // This is useful to make sure NN gets Heartbeat before Blockreport
235  // upon NN restart while DN keeps retrying Otherwise,
236  // 1. NN restarts.
237  // 2. Heartbeat RPC will retry and succeed. NN asks DN to reregister.
238  // 3. After reregistration completes, DN will send Blockreport first.
239  // 4. Given NN receives Blockreport after Heartbeat, it won't mark
240  //    DatanodeStorageInfo#blockContentsStale to false until the next
241  //    Blockreport.
242  void scheduleHeartbeat() {
243    lastHeartbeat = 0;
244  }

246  /**
247   * This methods  arranges for the data node to send the block report at 
248   * the next heartbeat.
249   */
250  void scheduleBlockReport(long delay) {
251    if (delay > 0) { // send BR after random delay
252      lastBlockReport = monotonicNow()
253      - ( dnConf.blockReportInterval - DFSUtil.getRandom().nextInt((int)(delay)));
254    } else { // send at next heartbeat
255      lastBlockReport = lastHeartbeat - dnConf.blockReportInterval;
256    }
257    resetBlockReportTime = true; // reset future BRs for randomness
258  }

261  /**
262   * Report received blocks and delete hints to the Namenode for each
263   * storage.
264   *
265   * @throws IOException
266   */
267  private void reportReceivedDeletedBlocks() throws IOException {
268
269    // Generate a list of the pending reports for each storage under the lock
270    ArrayList<StorageReceivedDeletedBlocks> reports =
271        new ArrayList<StorageReceivedDeletedBlocks>(pendingIncrementalBRperStorage.size());
272    synchronized (pendingIncrementalBRperStorage) {
273      for (Map.Entry<DatanodeStorage, PerStoragePendingIncrementalBR> entry :
274           pendingIncrementalBRperStorage.entrySet()) {
275        final DatanodeStorage storage = entry.getKey();
276        final PerStoragePendingIncrementalBR perStorageMap = entry.getValue();
277
278        if (perStorageMap.getBlockInfoCount() > 0) {
279          // Send newly-received and deleted blockids to namenode
280          ReceivedDeletedBlockInfo[] rdbi = perStorageMap.dequeueBlockInfos();
281          reports.add(new StorageReceivedDeletedBlocks(storage, rdbi));
282        }
283      }
284      sendImmediateIBR = false;
285    }
286
287    if (reports.size() == 0) {
288      // Nothing new to report.
289      return;
290    }
291
292    // Send incremental block reports to the Namenode outside the lock
293    boolean success = false;
294    final long startTime = monotonicNow();
295    try {
296      bpNamenode.blockReceivedAndDeleted(bpRegistration,
297          bpos.getBlockPoolId(),
298          reports.toArray(new StorageReceivedDeletedBlocks[reports.size()]));
299      success = true;
300    } finally {
301      dn.getMetrics().addIncrementalBlockReport(monotonicNow() - startTime);
302      if (!success) {
303        synchronized (pendingIncrementalBRperStorage) {
304          for (StorageReceivedDeletedBlocks report : reports) {
305            // If we didn't succeed in sending the report, put all of the
306            // blocks back onto our queue, but only in the case where we
307            // didn't put something newer in the meantime.
308            PerStoragePendingIncrementalBR perStorageMap =
309                pendingIncrementalBRperStorage.get(report.getStorage());
310            perStorageMap.putMissingBlockInfos(report.getBlocks());
311            sendImmediateIBR = true;
312          }
313        }
314      }
315    }
316  }
317
318  /**
319   * @return pending incremental block report for given {@code storage}
320   */
321  private PerStoragePendingIncrementalBR getIncrementalBRMapForStorage(
322      DatanodeStorage storage) {
323    PerStoragePendingIncrementalBR mapForStorage =
324        pendingIncrementalBRperStorage.get(storage);
325
326    if (mapForStorage == null) {
327      // This is the first time we are adding incremental BR state for
328      // this storage so create a new map. This is required once per
329      // storage, per service actor.
330      mapForStorage = new PerStoragePendingIncrementalBR();
331      pendingIncrementalBRperStorage.put(storage, mapForStorage);
332    }
333
334    return mapForStorage;
335  }

337  /**
338   * Add a blockInfo for notification to NameNode. If another entry
339   * exists for the same block it is removed.
340   *
341   * Caller must synchronize access using pendingIncrementalBRperStorage.
342   */
343  void addPendingReplicationBlockInfo(ReceivedDeletedBlockInfo bInfo,
344      DatanodeStorage storage) {
345    // Make sure another entry for the same block is first removed.
346    // There may only be one such entry.
347    for (Map.Entry<DatanodeStorage, PerStoragePendingIncrementalBR> entry :
348          pendingIncrementalBRperStorage.entrySet()) {
349      if (entry.getValue().removeBlockInfo(bInfo)) {
350        break;
351      }
352    }
353    getIncrementalBRMapForStorage(storage).putBlockInfo(bInfo);
354  }

356  /*
357   * Informing the name node could take a long long time! Should we wait
358   * till namenode is informed before responding with success to the
359   * client? For now we don't.
360   */
361  void notifyNamenodeBlock(ReceivedDeletedBlockInfo bInfo,
362      String storageUuid, boolean now) {
363    synchronized (pendingIncrementalBRperStorage) {
364      addPendingReplicationBlockInfo(
365          bInfo, dn.getFSDataset().getStorage(storageUuid));
366      sendImmediateIBR = true;
367      // If now is true, the report is sent right away.
368      // Otherwise, it will be sent out in the next heartbeat.
369      if (now) {
370        pendingIncrementalBRperStorage.notifyAll();
371      }
372    }
373  }

375  void notifyNamenodeDeletedBlock(
376      ReceivedDeletedBlockInfo bInfo, String storageUuid) {
377    synchronized (pendingIncrementalBRperStorage) {
378      addPendingReplicationBlockInfo(
379          bInfo, dn.getFSDataset().getStorage(storageUuid));
380    }
381  }

440  private long generateUniqueBlockReportId() {
441    long id = System.nanoTime();
442    if (id <= prevBlockReportId) {
443      id = prevBlockReportId + 1;
444    }
445    prevBlockReportId = id;
446    return id;
447  }

449  /**
450   * Report the list blocks to the Namenode
451   * @return DatanodeCommands returned by the NN. May be null.
452   * @throws IOException
453   */
454  List<DatanodeCommand> blockReport() throws IOException {
455    // send block report if timer has expired.
456    final long startTime = monotonicNow();
457    if (startTime - lastBlockReport <= dnConf.blockReportInterval) {
458      return null;
459    }
460
461    final ArrayList<DatanodeCommand> cmds = new ArrayList<DatanodeCommand>();
462
463    // Flush any block information that precedes the block report. Otherwise
464    // we have a chance that we will miss the delHint information
465    // or we will report an RBW replica after the BlockReport already reports
466    // a FINALIZED one.
467    reportReceivedDeletedBlocks();
468    lastDeletedReport = startTime;
469
470    long brCreateStartTime = monotonicNow();
471    Map<DatanodeStorage, BlockListAsLongs> perVolumeBlockLists =
472        dn.getFSDataset().getBlockReports(bpos.getBlockPoolId());
473
474    // Convert the reports to the format expected by the NN.
475    int i = 0;
476    int totalBlockCount = 0;
477    StorageBlockReport reports[] =
478        new StorageBlockReport[perVolumeBlockLists.size()];
479
480    for(Map.Entry<DatanodeStorage, BlockListAsLongs> kvPair : perVolumeBlockLists.entrySet()) {
481      BlockListAsLongs blockList = kvPair.getValue();
482      reports[i++] = new StorageBlockReport(kvPair.getKey(), blockList);
483      totalBlockCount += blockList.getNumberOfBlocks();
484    }
485
486    // Send the reports to the NN.
487    int numReportsSent = 0;
488    int numRPCs = 0;
489    boolean success = false;
490    long brSendStartTime = monotonicNow();
491    long reportId = generateUniqueBlockReportId();
492    try {
493      if (totalBlockCount < dnConf.blockReportSplitThreshold) {
494        // Below split threshold, send all reports in a single message.
495        DatanodeCommand cmd = bpNamenode.blockReport(
496            bpRegistration, bpos.getBlockPoolId(), reports,
497              new BlockReportContext(1, 0, reportId));
498        numRPCs = 1;
499        numReportsSent = reports.length;
500        if (cmd != null) {
501          cmds.add(cmd);
502        }
503      } else {
504        // Send one block report per message.
505        for (int r = 0; r < reports.length; r++) {
506          StorageBlockReport singleReport[] = { reports[r] };
507          DatanodeCommand cmd = bpNamenode.blockReport(
508              bpRegistration, bpos.getBlockPoolId(), singleReport,
509              new BlockReportContext(reports.length, r, reportId));
510          numReportsSent++;
511          numRPCs++;
512          if (cmd != null) {
513            cmds.add(cmd);
514          }
515        }
516      }
517      success = true;
518    } finally {
519      // Log the block report processing stats from Datanode perspective
520      long brSendCost = monotonicNow() - brSendStartTime;
521      long brCreateCost = brSendStartTime - brCreateStartTime;
522      dn.getMetrics().addBlockReport(brSendCost);
523      final int nCmds = cmds.size();
524      LOG.info((success ? "S" : "Uns") +
525          "uccessfully sent block report 0x" +
526          Long.toHexString(reportId) + ",  containing " + reports.length +
527          " storage report(s), of which we sent " + numReportsSent + "." +
528          " The reports had " + totalBlockCount +
529          " total blocks and used " + numRPCs +
530          " RPC(s). This took " + brCreateCost +
531          " msec to generate and " + brSendCost +
532          " msecs for RPC and NN processing." +
533          " Got back " +
534          ((nCmds == 0) ? "no commands" :
535              ((nCmds == 1) ? "one command: " + cmds.get(0) :
536                  (nCmds + " commands: " + Joiner.on("; ").join(cmds)))) +
537          ".");
538    }
539    scheduleNextBlockReport(startTime);
540    return cmds.size() == 0 ? null : cmds;
541  }

543  private void scheduleNextBlockReport(long previousReportStartTime) {
544    // If we have sent the first set of block reports, then wait a random
545    // time before we start the periodic block reports.
546    if (resetBlockReportTime) {
547      lastBlockReport = previousReportStartTime -
548          DFSUtil.getRandom().nextInt((int)(dnConf.blockReportInterval));
549      resetBlockReportTime = false;
550    } else {
551      /* say the last block report was at 8:20:14. The current report
552       * should have started around 9:20:14 (default 1 hour interval).
553       * If current time is :
554       *   1) normal like 9:20:18, next report should be at 10:20:14
555       *   2) unexpected like 11:35:43, next report should be at 12:20:14
556       */
557      lastBlockReport += (monotonicNow() - lastBlockReport) /
558          dnConf.blockReportInterval * dnConf.blockReportInterval;
559    }
560  }

562  DatanodeCommand cacheReport() throws IOException {
563    // If caching is disabled, do not send a cache report
564    if (dn.getFSDataset().getCacheCapacity() == 0) {
565      return null;
566    }
567    // send cache report if timer has expired.
568    DatanodeCommand cmd = null;
569    final long startTime = monotonicNow();
570    if (startTime - lastCacheReport > dnConf.cacheReportInterval) {
571      if (LOG.isDebugEnabled()) {
572        LOG.debug("Sending cacheReport from service actor: " + this);
573      }
574      lastCacheReport = startTime;
575
576      String bpid = bpos.getBlockPoolId();
577      List<Long> blockIds = dn.getFSDataset().getCacheReport(bpid);
578      long createTime = monotonicNow();
579
580      cmd = bpNamenode.cacheReport(bpRegistration, bpid, blockIds);
581      long sendTime = monotonicNow();
582      long createCost = createTime - startTime;
583      long sendCost = sendTime - createTime;
584      dn.getMetrics().addCacheReport(sendCost);
585      LOG.debug("CacheReport of " + blockIds.size()
586          + " block(s) took " + createCost + " msec to generate and "
587          + sendCost + " msecs for RPC and NN processing");
588    }
589    return cmd;
590  }

592  HeartbeatResponse sendHeartBeat() throws IOException {
593    StorageReport[] reports =
594        dn.getFSDataset().getStorageReports(bpos.getBlockPoolId());
595    if (LOG.isDebugEnabled()) {
596      LOG.debug("Sending heartbeat with " + reports.length +
597                " storage reports from service actor: " + this);
598    }
599    
600    VolumeFailureSummary volumeFailureSummary = dn.getFSDataset()
601        .getVolumeFailureSummary();
602    int numFailedVolumes = volumeFailureSummary != null ?
603        volumeFailureSummary.getFailedStorageLocations().length : 0;
604    return bpNamenode.sendHeartbeat(bpRegistration,
605        reports,
606        dn.getFSDataset().getCacheCapacity(),
607        dn.getFSDataset().getCacheUsed(),
608        dn.getXmitsInProgress(),
609        dn.getXceiverCount(),
610        numFailedVolumes,
611        volumeFailureSummary);
612  }

614  //This must be called only by BPOfferService
615  void start() {
616    if ((bpThread != null) && (bpThread.isAlive())) {
617      //Thread is started already
618      return;
619    }
620    bpThread = new Thread(this, formatThreadName());
621    bpThread.setDaemon(true); // needed for JUnit testing
622    bpThread.start();
623  }

629      " heartbeating to " + nnAddr;
630  }

657  private void handleRollingUpgradeStatus(HeartbeatResponse resp) throws IOException {
658    RollingUpgradeStatus rollingUpgradeStatus = resp.getRollingUpdateStatus();
659    if (rollingUpgradeStatus != null &&
660        rollingUpgradeStatus.getBlockPoolId().compareTo(bpos.getBlockPoolId()) != 0) {
661      // Can this ever occur?
662      LOG.error("Invalid BlockPoolId " +
663          rollingUpgradeStatus.getBlockPoolId() +
664          " in HeartbeatResponse. Expected " +
665          bpos.getBlockPoolId());
666    } else {
667      bpos.signalRollingUpgrade(rollingUpgradeStatus != null);
668    }
669  }

671  /**
672   * Main loop for each BP thread. Run until shutdown,
673   * forever calling remote NameNode functions.
674   */
675  private void offerService() throws Exception {
676    LOG.info("For namenode " + nnAddr + " using"
677        + " DELETEREPORT_INTERVAL of " + dnConf.deleteReportInterval + " msec "
678        + " BLOCKREPORT_INTERVAL of " + dnConf.blockReportInterval + "msec"
679        + " CACHEREPORT_INTERVAL of " + dnConf.cacheReportInterval + "msec"
680        + " Initial delay: " + dnConf.initialBlockReportDelay + "msec"
681        + "; heartBeatInterval=" + dnConf.heartBeatInterval);
682
683    //
684    // Now loop for a long time....
685    //
686    while (shouldRun()) {
687      try {
688        final long startTime = monotonicNow();
689
690        //
691        // Every so often, send heartbeat or block-report
692        //
693        if (startTime - lastHeartbeat >= dnConf.heartBeatInterval) {
694          //
695          // All heartbeat messages include following info:
696          // -- Datanode name
697          // -- data transfer port
698          // -- Total capacity
699          // -- Bytes remaining
700          //
701          lastHeartbeat = startTime;
702          if (!dn.areHeartbeatsDisabledForTests()) {
703            HeartbeatResponse resp = sendHeartBeat();
704            assert resp != null;
705            dn.getMetrics().addHeartbeat(monotonicNow() - startTime);
706
707            // If the state of this NN has changed (eg STANDBY->ACTIVE)
708            // then let the BPOfferService update itself.
709            //
710            // Important that this happens before processCommand below,
711            // since the first heartbeat to a new active might have commands
712            // that we should actually process.
713            bpos.updateActorStatesFromHeartbeat(
714                this, resp.getNameNodeHaState());
715            state = resp.getNameNodeHaState().getState();
716
717            if (state == HAServiceState.ACTIVE) {
718              handleRollingUpgradeStatus(resp);
719            }
720
721            long startProcessCommands = monotonicNow();
722            if (!processCommand(resp.getCommands()))
723              continue;
724            long endProcessCommands = monotonicNow();
725            if (endProcessCommands - startProcessCommands > 2000) {
726              LOG.info("Took " + (endProcessCommands - startProcessCommands)
727                  + "ms to process " + resp.getCommands().length
728                  + " commands from NN");
729            }
730          }
731        }
732        if (sendImmediateIBR ||
733            (startTime - lastDeletedReport > dnConf.deleteReportInterval)) {
734          reportReceivedDeletedBlocks();
735          lastDeletedReport = startTime;
736        }
737
738        List<DatanodeCommand> cmds = blockReport();
739        processCommand(cmds == null ? null : cmds.toArray(new DatanodeCommand[cmds.size()]));
740
741        DatanodeCommand cmd = cacheReport();
742        processCommand(new DatanodeCommand[]{ cmd });
743
744        //
745        // There is no work to do;  sleep until hearbeat timer elapses, 
746        // or work arrives, and then iterate again.
747        //
748        long waitTime = dnConf.heartBeatInterval - 
749        (monotonicNow() - lastHeartbeat);
750        synchronized(pendingIncrementalBRperStorage) {
751          if (waitTime > 0 && !sendImmediateIBR) {
752            try {
753              pendingIncrementalBRperStorage.wait(waitTime);
754            } catch (InterruptedException ie) {
755              LOG.warn("BPOfferService for " + this + " interrupted");
756            }
757          }
758        } // synchronized
759      } catch(RemoteException re) {
760        String reClass = re.getClassName();
761        if (UnregisteredNodeException.class.getName().equals(reClass) ||
762            DisallowedDatanodeException.class.getName().equals(reClass) ||
763            IncorrectVersionException.class.getName().equals(reClass)) {
764          LOG.warn(this + " is shutting down", re);
765          shouldServiceRun = false;
766          return;
767        }
768        LOG.warn("RemoteException in offerService", re);
769        try {
770          long sleepTime = Math.min(1000, dnConf.heartBeatInterval);
771          Thread.sleep(sleepTime);
772        } catch (InterruptedException ie) {
773          Thread.currentThread().interrupt();
774        }
775      } catch (IOException e) {
776        LOG.warn("IOException in offerService", e);
777      }
778      processQueueMessages();
779    } // while (shouldRun())
780  } // offerService

782  /**
783   * Register one bp with the corresponding NameNode
784   * <p>
785   * The bpDatanode needs to register with the namenode on startup in order
786   * 1) to report which storage it is serving now and 
787   * 2) to receive a registrationID
788   *  
789   * issued by the namenode to recognize registered datanodes.
790   * 
791   * @param nsInfo current NamespaceInfo
792   * @see FSNamesystem#registerDatanode(DatanodeRegistration)
793   * @throws IOException
794   */
795  void register(NamespaceInfo nsInfo) throws IOException {
796    // The handshake() phase loaded the block pool storage
797    // off disk - so update the bpRegistration object from that info
798    bpRegistration = bpos.createRegistration();
799
800    LOG.info(this + " beginning handshake with NN");
801
802    while (shouldRun()) {
803      try {
804        // Use returned registration from namenode with updated fields
805        bpRegistration = bpNamenode.registerDatanode(bpRegistration);
806        bpRegistration.setNamespaceInfo(nsInfo);
807        break;
808      } catch(EOFException e) {  // namenode might have just restarted
809        LOG.info("Problem connecting to server: " + nnAddr + " :"
810            + e.getLocalizedMessage());
811        sleepAndLogInterrupts(1000, "connecting to server");
812      } catch(SocketTimeoutException e) {  // namenode is busy
813        LOG.info("Problem connecting to server: " + nnAddr);
814        sleepAndLogInterrupts(1000, "connecting to server");
815      }
816    }
817    
818    LOG.info("Block pool " + this + " successfully registered with NN");
819    bpos.registrationSucceeded(this, bpRegistration);
820
821    // random short delay - helps scatter the BR from all DNs
822    scheduleBlockReport(dnConf.initialBlockReportDelay);
823  }

827      String stateString) {
828    try {
829      Thread.sleep(millis);
830    } catch (InterruptedException ie) {
831      LOG.info("BPOfferService " + this + " interrupted while " + stateString);
832    }

835  /**
836   * No matter what kind of exception we get, keep retrying to offerService().
837   * That's the loop that connects to the NameNode and provides basic DataNode
838   * functionality.
839   *
840   * Only stop when "shouldRun" or "shouldServiceRun" is turned off, which can
841   * happen either at shutdown or due to refreshNamenodes.
842   */
843  @Override
844  public void run() {
845    LOG.info(this + " starting to offer service");
846
847    try {
848      while (true) {
849        // init stuff
850        try {
851          // setup storage
852          connectToNNAndHandshake();
853          break;
854        } catch (IOException ioe) {
855          // Initial handshake, storage recovery or registration failed
856          runningState = RunningState.INIT_FAILED;
857          if (shouldRetryInit()) {
858            // Retry until all namenode's of BPOS failed initialization
859            LOG.error("Initialization failed for " + this + " "
860                + ioe.getLocalizedMessage());
861            sleepAndLogInterrupts(5000, "initializing");
862          } else {
863            runningState = RunningState.FAILED;
864            LOG.fatal("Initialization failed for " + this + ". Exiting. ", ioe);
865            return;
866          }
867        }
868      }
869
870      runningState = RunningState.RUNNING;
871
872      while (shouldRun()) {
873        try {
874          offerService();
875        } catch (Exception ex) {
876          LOG.error("Exception in BPOfferService for " + this, ex);
877          sleepAndLogInterrupts(5000, "offering service");
878        }
879      }
880      runningState = RunningState.EXITED;
881    } catch (Throwable ex) {
882      LOG.warn("Unexpected exception in block pool " + this, ex);
883      runningState = RunningState.FAILED;
884    } finally {
885      LOG.warn("Ending block pool service for: " + this);
886      cleanUp();
887    }
888  }
889
890  private boolean shouldRetryInit() {
891    return shouldRun() && bpos.shouldRetryInit();
892  }
893
894  private boolean shouldRun() {
895    return shouldServiceRun && dn.shouldRun();
896  }

898  /**
899   * Process an array of datanode commands
900   * 
901   * @param cmds an array of datanode commands
902   * @return true if further processing may be required or false otherwise. 
903   */
904  boolean processCommand(DatanodeCommand[] cmds) {
905    if (cmds != null) {
906      for (DatanodeCommand cmd : cmds) {
907        try {
908          if (bpos.processCommandFromActor(cmd, this) == false) {
909            return false;
910          }
911        } catch (IOException ioe) {
912          LOG.warn("Error processing datanode Command", ioe);
913        }
914      }
915    }
916    return true;
917  }

941  private static class PerStoragePendingIncrementalBR {
942    private final Map<Long, ReceivedDeletedBlockInfo> pendingIncrementalBR =
943        Maps.newHashMap();
944
945    /**
946     * Return the number of blocks on this storage that have pending
947     * incremental block reports.
948     * @return
949     */
950    int getBlockInfoCount() {
951      return pendingIncrementalBR.size();
952    }
953
954    /**
955     * Dequeue and return all pending incremental block report state.
956     * @return
957     */
958    ReceivedDeletedBlockInfo[] dequeueBlockInfos() {
959      ReceivedDeletedBlockInfo[] blockInfos =
960          pendingIncrementalBR.values().toArray(
961              new ReceivedDeletedBlockInfo[getBlockInfoCount()]);
962
963      pendingIncrementalBR.clear();
964      return blockInfos;
965    }
966
967    /**
968     * Add blocks from blockArray to pendingIncrementalBR, unless the
969     * block already exists in pendingIncrementalBR.
970     * @param blockArray list of blocks to add.
971     * @return the number of missing blocks that we added.
972     */
973    int putMissingBlockInfos(ReceivedDeletedBlockInfo[] blockArray) {
974      int blocksPut = 0;
975      for (ReceivedDeletedBlockInfo rdbi : blockArray) {
976        if (!pendingIncrementalBR.containsKey(rdbi.getBlock().getBlockId())) {
977          pendingIncrementalBR.put(rdbi.getBlock().getBlockId(), rdbi);
978          ++blocksPut;
979        }
980      }
981      return blocksPut;
982    }
983
984    /**
985     * Add pending incremental block report for a single block.
986     * @param blockInfo
987     */
988    void putBlockInfo(ReceivedDeletedBlockInfo blockInfo) {
989      pendingIncrementalBR.put(blockInfo.getBlock().getBlockId(), blockInfo);
990    }
991
992    /**
993     * Remove pending incremental block report for a single block if it
994     * exists.
995     *
996     * @param blockInfo
997     * @return true if a report was removed, false if no report existed for
998     *         the given block.
999     */
1000    boolean removeBlockInfo(ReceivedDeletedBlockInfo blockInfo) {
1001      return (pendingIncrementalBR.remove(blockInfo.getBlock().getBlockId()) != null);
1002    }
1003  }
1004
1005  void triggerBlockReport(BlockReportOptions options) throws IOException {
1006    if (options.isIncremental()) {
1007      LOG.info(bpos.toString() + ": scheduling an incremental block report.");
1008      synchronized(pendingIncrementalBRperStorage) {
1009        sendImmediateIBR = true;
1010        pendingIncrementalBRperStorage.notifyAll();
1011      }
1012    } else {
1013      LOG.info(bpos.toString() + ": scheduling a full block report.");
1014      synchronized(pendingIncrementalBRperStorage) {
1015        lastBlockReport = 0;
1016        pendingIncrementalBRperStorage.notifyAll();
1017      }
1018    }
1019  }

1021  public void bpThreadEnqueue(BPServiceActorAction action) {
1022    synchronized (bpThreadQueue) {
1023      if (!bpThreadQueue.contains(action)) {
1024        bpThreadQueue.add(action);
1025      }
1026    }
1027  }

1029  private void processQueueMessages() {
1030    LinkedList<BPServiceActorAction> duplicateQueue;
1031    synchronized (bpThreadQueue) {
1032      duplicateQueue = new LinkedList<BPServiceActorAction>(bpThreadQueue);
1033      bpThreadQueue.clear();
1034    }
1035    while (!duplicateQueue.isEmpty()) {
1036      BPServiceActorAction actionItem = duplicateQueue.remove();
1037      try {
1038        actionItem.reportTo(bpNamenode, bpRegistration);
1039      } catch (BPServiceActorActionException baae) {
1040        LOG.warn(baae.getMessage() + nnAddr , baae);
1041        // Adding it back to the queue if not present
1042        bpThreadEnqueue(actionItem);
1043      }
1044    }
1045  }
