103  static class ScanInfoPerBlockPool extends 
104                     HashMap<String, LinkedList<ScanInfo>> {
105    
106    private static final long serialVersionUID = 1L;
107
108    ScanInfoPerBlockPool() {super();}
109    
110    ScanInfoPerBlockPool(int sz) {super(sz);}
111    
112    /**
113     * Merges {@code that} ScanInfoPerBlockPool into this one
114     */
115    public void addAll(ScanInfoPerBlockPool that) {
116      if (that == null) return;
117      
118      for (Entry<String, LinkedList<ScanInfo>> entry : that.entrySet()) {
119        String bpid = entry.getKey();
120        LinkedList<ScanInfo> list = entry.getValue();
121        
122        if (this.containsKey(bpid)) {
123          //merge that per-bpid linked list with this one
124          this.get(bpid).addAll(list);
125        } else {
126          //add that new bpid and its linked list to this
127          this.put(bpid, list);
128        }
129      }
130    }
131    
132    /**
133     * Convert all the LinkedList values in this ScanInfoPerBlockPool map
134     * into sorted arrays, and return a new map of these arrays per blockpool
135     * @return a map of ScanInfo arrays per blockpool
136     */
137    public Map<String, ScanInfo[]> toSortedArrays() {
138      Map<String, ScanInfo[]> result = 
139        new HashMap<String, ScanInfo[]>(this.size());
140      
141      for (Entry<String, LinkedList<ScanInfo>> entry : this.entrySet()) {
142        String bpid = entry.getKey();
143        LinkedList<ScanInfo> list = entry.getValue();
144        
145        // convert list to array
146        ScanInfo[] record = list.toArray(new ScanInfo[list.size()]);
147        // Sort array based on blockId
148        Arrays.sort(record);
149        result.put(bpid, record);            
150      }
151      return result;
152    }
153  }

155  /**
156   * Tracks the files and other information related to a block on the disk
157   * Missing file is indicated by setting the corresponding member
158   * to null.
159   * 
160   * Because millions of these structures may be created, we try to save
161   * memory here.  So instead of storing full paths, we store path suffixes.
162   * The block file, if it exists, will have a path like this:
163   * <volume_base_path>/<block_path>
164   * So we don't need to store the volume path, since we already know what the
165   * volume is.
166   * 
167   * The metadata file, if it exists, will have a path like this:
168   * <volume_base_path>/<block_path>_<genstamp>.meta
169   * So if we have a block file, there isn't any need to store the block path
170   * again.
171   * 
172   * The accessor functions take care of these manipulations.
173   */
174  static class ScanInfo implements Comparable<ScanInfo> {
175    private final long blockId;
176    
177    /**
178     * The block file path, relative to the volume's base directory.
179     * If there was no block file found, this may be null. If 'vol'
180     * is null, then this is the full path of the block file.
181     */
182    private final String blockSuffix;
183    
184    /**
185     * The suffix of the meta file path relative to the block file.
186     * If blockSuffix is null, then this will be the entire path relative
187     * to the volume base directory, or an absolute path if vol is also
188     * null.
189     */
190    private final String metaSuffix;
191
192    private final FsVolumeSpi volume;
193
194    /**
195     * Get the file's length in async block scan
196     */
197    private final long blockFileLength;
198
199    private final static Pattern CONDENSED_PATH_REGEX =
200        Pattern.compile("(?<!^)(\\\\|/){2,}");
201    
202    private final static String QUOTED_FILE_SEPARATOR = 
203        Matcher.quoteReplacement(File.separator);
204    
205    /**
206     * Get the most condensed version of the path.
207     *
208     * For example, the condensed version of /foo//bar is /foo/bar
209     * Unlike {@link File#getCanonicalPath()}, this will never perform I/O
210     * on the filesystem.
211     */
212    private static String getCondensedPath(String path) {
213      return CONDENSED_PATH_REGEX.matcher(path).
214          replaceAll(QUOTED_FILE_SEPARATOR);
215    }
216
217    /**
218     * Get a path suffix.
219     *
220     * @param f            The file to get the suffix for.
221     * @param prefix       The prefix we're stripping off.
222     *
223     * @return             A suffix such that prefix + suffix = path to f
224     */
225    private static String getSuffix(File f, String prefix) {
226      String fullPath = getCondensedPath(f.getAbsolutePath());
227      if (fullPath.startsWith(prefix)) {
228        return fullPath.substring(prefix.length());
229      }
230      throw new RuntimeException(prefix + " is not a prefix of " + fullPath);
231    }
232
233    ScanInfo(long blockId, File blockFile, File metaFile, FsVolumeSpi vol) {
234      this.blockId = blockId;
235      String condensedVolPath = vol == null ? null :
236        getCondensedPath(vol.getBasePath());
237      this.blockSuffix = blockFile == null ? null :
238        getSuffix(blockFile, condensedVolPath);
239      this.blockFileLength = (blockFile != null) ? blockFile.length() : 0; 
240      if (metaFile == null) {
241        this.metaSuffix = null;
242      } else if (blockFile == null) {
243        this.metaSuffix = getSuffix(metaFile, condensedVolPath);
244      } else {
245        this.metaSuffix = getSuffix(metaFile,
246            condensedVolPath + blockSuffix);
247      }
248      this.volume = vol;
249    }
250
251    File getBlockFile() {
252      return (blockSuffix == null) ? null :
253        new File(volume.getBasePath(), blockSuffix);
254    }
255
256    long getBlockFileLength() {
257      return blockFileLength;
258    }
259
260    File getMetaFile() {
261      if (metaSuffix == null) {
262        return null;
263      } else if (blockSuffix == null) {
264        return new File(volume.getBasePath(), metaSuffix);
265      } else {
266        return new File(volume.getBasePath(), blockSuffix + metaSuffix);
267      }
268    }
269
270    long getBlockId() {
271      return blockId;
272    }
273
274    FsVolumeSpi getVolume() {
275      return volume;
276    }
277
278    @Override // Comparable
279    public int compareTo(ScanInfo b) {
280      if (blockId < b.blockId) {
281        return -1;
282      } else if (blockId == b.blockId) {
283        return 0;
284      } else {
285        return 1;
286      }
287    }
288
289    @Override // Object
290    public boolean equals(Object o) {
291      if (this == o) {
292        return true;
293      }
294      if (!(o instanceof ScanInfo)) {
295        return false;
296      }
297      return blockId == ((ScanInfo) o).blockId;
298    }
299
300    @Override // Object
301    public int hashCode() {
302      return (int)(blockId^(blockId>>>32));
303    }
304
305    public long getGenStamp() {
306      return metaSuffix != null ? Block.getGenerationStamp(
307          getMetaFile().getName()) : 
308            GenerationStamp.GRANDFATHER_GENERATION_STAMP;
309    }
310  }

312  DirectoryScanner(DataNode datanode, FsDatasetSpi<?> dataset, Configuration conf) {
313    this.datanode = datanode;
314    this.dataset = dataset;
315    int interval = conf.getInt(DFSConfigKeys.DFS_DATANODE_DIRECTORYSCAN_INTERVAL_KEY,
316        DFSConfigKeys.DFS_DATANODE_DIRECTORYSCAN_INTERVAL_DEFAULT);
317    scanPeriodMsecs = interval * 1000L; //msec
318    int threads = 
319        conf.getInt(DFSConfigKeys.DFS_DATANODE_DIRECTORYSCAN_THREADS_KEY,
320                    DFSConfigKeys.DFS_DATANODE_DIRECTORYSCAN_THREADS_DEFAULT);
321
322    reportCompileThreadPool = Executors.newFixedThreadPool(threads, 
323        new Daemon.DaemonFactory());
324    masterThread = new ScheduledThreadPoolExecutor(1,
325        new Daemon.DaemonFactory());
326  }
327
328  void start() {
329    shouldRun = true;
330    long offset = DFSUtil.getRandom().nextInt((int) (scanPeriodMsecs/1000L)) * 1000L; //msec
331    long firstScanTime = Time.now() + offset;
332    LOG.info("Periodic Directory Tree Verification scan starting at " 
333        + firstScanTime + " with interval " + scanPeriodMsecs);
334    masterThread.scheduleAtFixedRate(this, offset, scanPeriodMsecs, 
335                                     TimeUnit.MILLISECONDS);
336  }

340    return shouldRun;

345    stats.clear();

348  /**
349   * Main program loop for DirectoryScanner
350   * Runs "reconcile()" periodically under the masterThread.
351   */
352  @Override
353  public void run() {
354    try {
355      if (!shouldRun) {
356        //shutdown has been activated
357        LOG.warn("this cycle terminating immediately because 'shouldRun' has been deactivated");
358        return;
359      }
360
361      //We're are okay to run - do it
362      reconcile();      
363      
364    } catch (Exception e) {
365      //Log and continue - allows Executor to run again next cycle
366      LOG.error("Exception during DirectoryScanner execution - will continue next cycle", e);
367    } catch (Error er) {
368      //Non-recoverable error - re-throw after logging the problem
369      LOG.error("System Error during DirectoryScanner execution - permanently terminating periodic scanner", er);
370      throw er;
371    }
372  }
373  
374  void shutdown() {
375    if (!shouldRun) {
376      LOG.warn("DirectoryScanner: shutdown has been called, but periodic scanner not started");
377    } else {
378      LOG.warn("DirectoryScanner: shutdown has been called");      
379    }
380    shouldRun = false;
381    if (masterThread != null) masterThread.shutdown();
382    if (reportCompileThreadPool != null) reportCompileThreadPool.shutdown();
383    if (masterThread != null) {
384      try {
385        masterThread.awaitTermination(1, TimeUnit.MINUTES);
386      } catch (InterruptedException e) {
387        LOG.error("interrupted while waiting for masterThread to " +
388          "terminate", e);
389      }
390    }
391    if (reportCompileThreadPool != null) {
392      try {
393        reportCompileThreadPool.awaitTermination(1, TimeUnit.MINUTES);
394      } catch (InterruptedException e) {
395        LOG.error("interrupted while waiting for reportCompileThreadPool to " +
396          "terminate", e);
397      }
398    }
399    if (!retainDiffs) clear();
400  }

402  /**
403   * Reconcile differences between disk and in-memory blocks
404   */
405  void reconcile() throws IOException {
406    scan();
407    for (Entry<String, LinkedList<ScanInfo>> entry : diffs.entrySet()) {
408      String bpid = entry.getKey();
409      LinkedList<ScanInfo> diff = entry.getValue();
410      
411      for (ScanInfo info : diff) {
412        dataset.checkAndUpdate(bpid, info.getBlockId(), info.getBlockFile(),
413            info.getMetaFile(), info.getVolume());
414      }
415    }
416    if (!retainDiffs) clear();
417  }

419  /**
420   * Scan for the differences between disk and in-memory blocks
421   * Scan only the "finalized blocks" lists of both disk and memory.
422   */
423  void scan() {
424    clear();
425    Map<String, ScanInfo[]> diskReport = getDiskReport();
426
427    // Hold FSDataset lock to prevent further changes to the block map
428    synchronized(dataset) {
429      for (Entry<String, ScanInfo[]> entry : diskReport.entrySet()) {
430        String bpid = entry.getKey();
431        ScanInfo[] blockpoolReport = entry.getValue();
432        
433        Stats statsRecord = new Stats(bpid);
434        stats.put(bpid, statsRecord);
435        LinkedList<ScanInfo> diffRecord = new LinkedList<ScanInfo>();
436        diffs.put(bpid, diffRecord);
437        
438        statsRecord.totalBlocks = blockpoolReport.length;
439        List<FinalizedReplica> bl = dataset.getFinalizedBlocks(bpid);
440        FinalizedReplica[] memReport = bl.toArray(new FinalizedReplica[bl.size()]);
441        Arrays.sort(memReport); // Sort based on blockId
442  
443        int d = 0; // index for blockpoolReport
444        int m = 0; // index for memReprot
445        while (m < memReport.length && d < blockpoolReport.length) {
446          FinalizedReplica memBlock = memReport[m];
447          ScanInfo info = blockpoolReport[d];
448          if (info.getBlockId() < memBlock.getBlockId()) {
449            if (!dataset.isDeletingBlock(bpid, info.getBlockId())) {
450              // Block is missing in memory
451              statsRecord.missingMemoryBlocks++;
452              addDifference(diffRecord, statsRecord, info);
453            }
454            d++;
455            continue;
456          }
457          if (info.getBlockId() > memBlock.getBlockId()) {
458            // Block is missing on the disk
459            addDifference(diffRecord, statsRecord,
460                          memBlock.getBlockId(), info.getVolume());
461            m++;
462            continue;
463          }
464          // Block file and/or metadata file exists on the disk
465          // Block exists in memory
466          if (info.getBlockFile() == null) {
467            // Block metadata file exits and block file is missing
468            addDifference(diffRecord, statsRecord, info);
469          } else if (info.getGenStamp() != memBlock.getGenerationStamp()
470              || info.getBlockFileLength() != memBlock.getNumBytes()) {
471            // Block metadata file is missing or has wrong generation stamp,
472            // or block file length is different than expected
473            statsRecord.mismatchBlocks++;
474            addDifference(diffRecord, statsRecord, info);
475          } else if (info.getBlockFile().compareTo(memBlock.getBlockFile()) != 0) {
476            // volumeMap record and on-disk files don't match.
477            statsRecord.duplicateBlocks++;
478            addDifference(diffRecord, statsRecord, info);
479          }
480          d++;
481
482          if (d < blockpoolReport.length) {
483            // There may be multiple on-disk records for the same block, don't increment
484            // the memory record pointer if so.
485            ScanInfo nextInfo = blockpoolReport[Math.min(d, blockpoolReport.length - 1)];
486            if (nextInfo.getBlockId() != info.blockId) {
487              ++m;
488            }
489          } else {
490            ++m;
491          }
492        }
493        while (m < memReport.length) {
494          FinalizedReplica current = memReport[m++];
495          addDifference(diffRecord, statsRecord,
496                        current.getBlockId(), current.getVolume());
497        }
498        while (d < blockpoolReport.length) {
499          if (!dataset.isDeletingBlock(bpid, blockpoolReport[d].getBlockId())) {
500            statsRecord.missingMemoryBlocks++;
501            addDifference(diffRecord, statsRecord, blockpoolReport[d]);
502          }
503          d++;
504        }
505        LOG.info(statsRecord.toString());
506      } //end for
507    } //end synchronized
508  }

510  /**
511   * Block is found on the disk. In-memory block is missing or does not match
512   * the block on the disk
513   */
514  private void addDifference(LinkedList<ScanInfo> diffRecord, 
515                             Stats statsRecord, ScanInfo info) {
516    statsRecord.missingMetaFile += info.getMetaFile() == null ? 1 : 0;
517    statsRecord.missingBlockFile += info.getBlockFile() == null ? 1 : 0;
518    diffRecord.add(info);
519  }
520
521  /** Block is not found on the disk */
522  private void addDifference(LinkedList<ScanInfo> diffRecord,
523                             Stats statsRecord, long blockId,
524                             FsVolumeSpi vol) {
525    statsRecord.missingBlockFile++;
526    statsRecord.missingMetaFile++;
527    diffRecord.add(new ScanInfo(blockId, null, null, vol));
528  }

530  /** Is the given volume still valid in the dataset? */
531  private static boolean isValid(final FsDatasetSpi<?> dataset,
532      final FsVolumeSpi volume) {
533    for (FsVolumeSpi vol : dataset.getVolumes()) {
534      if (vol == volume) {
535        return true;
536      }
537    }
538    return false;
539  }

541  /** Get lists of blocks on the disk sorted by blockId, per blockpool */
542  private Map<String, ScanInfo[]> getDiskReport() {
543    // First get list of data directories
544    final List<? extends FsVolumeSpi> volumes = dataset.getVolumes();
545
546    // Use an array since the threads may return out of order and
547    // compilersInProgress#keySet may return out of order as well.
548    ScanInfoPerBlockPool[] dirReports = new ScanInfoPerBlockPool[volumes.size()];
549
550    Map<Integer, Future<ScanInfoPerBlockPool>> compilersInProgress =
551      new HashMap<Integer, Future<ScanInfoPerBlockPool>>();
552
553    for (int i = 0; i < volumes.size(); i++) {
554      if (isValid(dataset, volumes.get(i))) {
555        ReportCompiler reportCompiler =
556          new ReportCompiler(datanode,volumes.get(i));
557        Future<ScanInfoPerBlockPool> result = 
558          reportCompileThreadPool.submit(reportCompiler);
559        compilersInProgress.put(i, result);
560      }
561    }
562    
563    for (Entry<Integer, Future<ScanInfoPerBlockPool>> report :
564        compilersInProgress.entrySet()) {
565      try {
566        dirReports[report.getKey()] = report.getValue().get();
567      } catch (Exception ex) {
568        LOG.error("Error compiling report", ex);
569        // Propagate ex to DataBlockScanner to deal with
570        throw new RuntimeException(ex);
571      }
572    }
573
574    // Compile consolidated report for all the volumes
575    ScanInfoPerBlockPool list = new ScanInfoPerBlockPool();
576    for (int i = 0; i < volumes.size(); i++) {
577      if (isValid(dataset, volumes.get(i))) {
578        // volume is still valid
579        list.addAll(dirReports[i]);
580      }
581    }
582
583    return list.toSortedArrays();
584  }

586  private static boolean isBlockMetaFile(String blockId, String metaFile) {
587    return metaFile.startsWith(blockId)
588        && metaFile.endsWith(Block.METADATA_EXTENSION);
589  }
590
591  private static class ReportCompiler 
592  implements Callable<ScanInfoPerBlockPool> {
593    private final FsVolumeSpi volume;
594    private final DataNode datanode;
595
596    public ReportCompiler(DataNode datanode, FsVolumeSpi volume) {
597      this.datanode = datanode;
598      this.volume = volume;
599    }
600
601    @Override
602    public ScanInfoPerBlockPool call() throws Exception {
603      String[] bpList = volume.getBlockPoolList();
604      ScanInfoPerBlockPool result = new ScanInfoPerBlockPool(bpList.length);
605      for (String bpid : bpList) {
606        LinkedList<ScanInfo> report = new LinkedList<ScanInfo>();
607        File bpFinalizedDir = volume.getFinalizedDir(bpid);
608        result.put(bpid,
609            compileReport(volume, bpFinalizedDir, bpFinalizedDir, report));
610      }
611      return result;
612    }
613
614    /** Compile list {@link ScanInfo} for the blocks in the directory <dir> */
615    private LinkedList<ScanInfo> compileReport(FsVolumeSpi vol,
616        File bpFinalizedDir, File dir, LinkedList<ScanInfo> report) {
617      File[] files;
618      try {
619        files = FileUtil.listFiles(dir);
620      } catch (IOException ioe) {
621        LOG.warn("Exception occured while compiling report: ", ioe);
622        // Initiate a check on disk failure.
623        datanode.checkDiskErrorAsync();
624        // Ignore this directory and proceed.
625        return report;
626      }
627      Arrays.sort(files);
628      /*
629       * Assumption: In the sorted list of files block file appears immediately
630       * before block metadata file. This is true for the current naming
631       * convention for block file blk_<blockid> and meta file
632       * blk_<blockid>_<genstamp>.meta
633       */
634      for (int i = 0; i < files.length; i++) {
635        if (files[i].isDirectory()) {
636          compileReport(vol, bpFinalizedDir, files[i], report);
637          continue;
638        }
639        if (!Block.isBlockFilename(files[i])) {
640          if (isBlockMetaFile(Block.BLOCK_FILE_PREFIX, files[i].getName())) {
641            long blockId = Block.getBlockId(files[i].getName());
642            verifyFileLocation(files[i].getParentFile(), bpFinalizedDir,
643                blockId);
644            report.add(new ScanInfo(blockId, null, files[i], vol));
645          }
646          continue;
647        }
648        File blockFile = files[i];
649        long blockId = Block.filename2id(blockFile.getName());
650        File metaFile = null;
651
652        // Skip all the files that start with block name until
653        // getting to the metafile for the block
654        while (i + 1 < files.length && files[i + 1].isFile()
655            && files[i + 1].getName().startsWith(blockFile.getName())) {
656          i++;
657          if (isBlockMetaFile(blockFile.getName(), files[i].getName())) {
658            metaFile = files[i];
659            break;
660          }
661        }
662        verifyFileLocation(blockFile.getParentFile(), bpFinalizedDir,
663            blockId);
664        report.add(new ScanInfo(blockId, blockFile, metaFile, vol));
665      }
666      return report;
667    }
668
669    /**
670     * Verify whether the actual directory location of block file has the
671     * expected directory path computed using its block ID.
672     */
673    private void verifyFileLocation(File actualBlockDir,
674        File bpFinalizedDir, long blockId) {
675      File blockDir = DatanodeUtil.idToBlockDir(bpFinalizedDir, blockId);
676      if (actualBlockDir.compareTo(blockDir) != 0) {
677        LOG.warn("Block: " + blockId
678            + " has to be upgraded to block ID-based layout");
679      }
680    }
