065  /**
066   * JSP helper function that generates cluster health report.  When 
067   * encountering exception while getting Namenode status, the exception will 
068   * be listed on the page with corresponding stack trace.
069   */
070  ClusterStatus generateClusterHealthReport() {
071    ClusterStatus cs = new ClusterStatus();
072    Configuration conf = new Configuration();
073    List<ConfiguredNNAddress> nns = null;
074    try {
075      nns = DFSUtil.flattenAddressMap(
076          DFSUtil.getNNServiceRpcAddresses(conf));
077    } catch (Exception e) {
078      // Could not build cluster status
079      cs.setError(e);
080      return cs;
081    }
082    
083    // Process each namenode and add it to ClusterStatus
084    for (ConfiguredNNAddress cnn : nns) {
085      InetSocketAddress isa = cnn.getAddress();
086      NamenodeMXBeanHelper nnHelper = null;
087      try {
088        nnHelper = new NamenodeMXBeanHelper(isa, conf);
089        String mbeanProps= queryMbean(nnHelper.httpAddress, conf);
090        NamenodeStatus nn = nnHelper.getNamenodeStatus(mbeanProps);
091        if (cs.clusterid.isEmpty() || cs.clusterid.equals("")) { // Set clusterid only once
092          cs.clusterid = nnHelper.getClusterId(mbeanProps);
093        }
094        cs.addNamenodeStatus(nn);
095      } catch ( Exception e ) {
096        // track exceptions encountered when connecting to namenodes
097        cs.addException(isa.getHostName(), e);
098        continue;
099      } 
100    }
101    return cs;
102  }

104  /**
105   * Helper function that generates the decommissioning report.  Connect to each
106   * Namenode over http via JmxJsonServlet to collect the data nodes status.
107   */
108  DecommissionStatus generateDecommissioningReport() {
109    String clusterid = "";
110    Configuration conf = new Configuration();
111    List<ConfiguredNNAddress> cnns = null;
112    try {
113      cnns = DFSUtil.flattenAddressMap(
114          DFSUtil.getNNServiceRpcAddresses(conf));
115    } catch (Exception e) {
116      // catch any exception encountered other than connecting to namenodes
117      DecommissionStatus dInfo = new DecommissionStatus(clusterid, e);
118      return dInfo;
119    }
120    
121    // Outer map key is datanode. Inner map key is namenode and the value is 
122    // decom status of the datanode for the corresponding namenode
123    Map<String, Map<String, String>> statusMap = 
124      new HashMap<String, Map<String, String>>();
125    
126    // Map of exceptions encountered when connecting to namenode
127    // key is namenode and value is exception
128    Map<String, Exception> decommissionExceptions = 
129      new HashMap<String, Exception>();
130    
131    List<String> unreportedNamenode = new ArrayList<String>();
132    for (ConfiguredNNAddress cnn : cnns) {
133      InetSocketAddress isa = cnn.getAddress();
134      NamenodeMXBeanHelper nnHelper = null;
135      try {
136        nnHelper = new NamenodeMXBeanHelper(isa, conf);
137        String mbeanProps= queryMbean(nnHelper.httpAddress, conf);
138        if (clusterid.equals("")) {
139          clusterid = nnHelper.getClusterId(mbeanProps);
140        }
141        nnHelper.getDecomNodeInfoForReport(statusMap, mbeanProps);
142      } catch (Exception e) {
143        // catch exceptions encountered while connecting to namenodes
144        String nnHost = isa.getHostName();
145        decommissionExceptions.put(nnHost, e);
146        unreportedNamenode.add(nnHost);
147        continue;
148      } 
149    }
150    updateUnknownStatus(statusMap, unreportedNamenode);
151    getDecommissionNodeClusterState(statusMap);
152    return new DecommissionStatus(statusMap, clusterid,
153        getDatanodeHttpPort(conf), decommissionExceptions);
154  }

156  /**
157   * Based on the state of the datanode at each namenode, marks the overall
158   * state of the datanode across all the namenodes, to one of the following:
159   * <ol>
160   * <li>{@link DecommissionStates#DECOMMISSIONED}</li>
161   * <li>{@link DecommissionStates#DECOMMISSION_INPROGRESS}</li>
162   * <li>{@link DecommissionStates#PARTIALLY_DECOMMISSIONED}</li>
163   * <li>{@link DecommissionStates#UNKNOWN}</li>
164   * </ol>
165   * 
166   * @param statusMap
167   *          map whose key is datanode, value is an inner map with key being
168   *          namenode, value being decommission state.
169   */
170  private void getDecommissionNodeClusterState(
171      Map<String, Map<String, String>> statusMap) {
172    if (statusMap == null || statusMap.isEmpty()) {
173      return;
174    }
175    
176    // For each datanodes
177    Iterator<Entry<String, Map<String, String>>> it = 
178      statusMap.entrySet().iterator();
179    while (it.hasNext()) {
180      // Map entry for a datanode:
181      // key is namenode, value is datanode status at the namenode
182      Entry<String, Map<String, String>> entry = it.next();
183      Map<String, String> nnStatus = entry.getValue();
184      if (nnStatus == null || nnStatus.isEmpty()) {
185        continue;
186      }
187      
188      boolean isUnknown = false;
189      int unknown = 0;
190      int decommissioned = 0;
191      int decomInProg = 0;
192      int inservice = 0;
193      int dead = 0;
194      DecommissionStates overallState = DecommissionStates.UNKNOWN;
195      // Process a datanode state from each namenode
196      for (Map.Entry<String, String> m : nnStatus.entrySet()) {
197        String status = m.getValue();
198        if (status.equals(DecommissionStates.UNKNOWN.toString())) {
199          isUnknown = true;
200          unknown++;
201        } else 
202          if (status.equals(AdminStates.DECOMMISSION_INPROGRESS.toString())) {
203          decomInProg++;
204        } else if (status.equals(AdminStates.DECOMMISSIONED.toString())) {
205          decommissioned++;
206        } else if (status.equals(AdminStates.NORMAL.toString())) {
207          inservice++;
208        } else if (status.equals(DEAD)) {
209          // dead
210          dead++;
211        }
212      }
213      
214      // Consolidate all the states from namenode in to overall state
215      int nns = nnStatus.keySet().size();
216      if ((inservice + dead + unknown) == nns) {
217        // Do not display this data node. Remove this entry from status map.  
218        it.remove();
219      } else if (isUnknown) {
220        overallState = DecommissionStates.UNKNOWN;
221      } else if (decommissioned == nns) {
222        overallState = DecommissionStates.DECOMMISSIONED;
223      } else if ((decommissioned + decomInProg) == nns) {
224        overallState = DecommissionStates.DECOMMISSION_INPROGRESS;
225      } else if ((decommissioned + decomInProg < nns) 
226        && (decommissioned + decomInProg > 0)){
227        overallState = DecommissionStates.PARTIALLY_DECOMMISSIONED;
228      } else {
229        LOG.warn("Cluster console encounters a not handled situtation.");
230      }
231        
232      // insert overall state
233      nnStatus.put(OVERALL_STATUS, overallState.toString());
234    }
235  }

237  /**
238   * update unknown status in datanode status map for every unreported namenode
239   */
240  private void updateUnknownStatus(Map<String, Map<String, String>> statusMap,
241      List<String> unreportedNn) {
242    if (unreportedNn == null || unreportedNn.isEmpty()) {
243      // no unreported namenodes
244      return;
245    }
246    
247    for (Map.Entry<String, Map<String,String>> entry : statusMap.entrySet()) {
248      String dn = entry.getKey();
249      Map<String, String> nnStatus = entry.getValue();
250      for (String nn : unreportedNn) {
251        nnStatus.put(nn, DecommissionStates.UNKNOWN.toString());
252      }
253      statusMap.put(dn, nnStatus);
254    }
255  }

257  /**
258   * Get datanode http port from configration
259   */
260  private int getDatanodeHttpPort(Configuration conf) {
261    String address = conf.get(DFSConfigKeys.DFS_DATANODE_HTTP_ADDRESS_KEY, "");
262    if (address.equals("")) {
263      return -1;
264    }
265    return Integer.parseInt(address.split(":")[1]);
266  }

272  static class NamenodeMXBeanHelper {

274    private final String host;

277    NamenodeMXBeanHelper(InetSocketAddress addr, Configuration conf)
278        throws IOException, MalformedObjectNameException {
279      this.host = addr.getHostName();
280      this.httpAddress = DFSUtil.getInfoServer(addr, conf,

284    /** Get the map corresponding to the JSON string */
285    private static Map<String, Map<String, Object>> getNodeMap(String json)
286        throws IOException {
287      TypeReference<Map<String, Map<String, Object>>> type = 
288        new TypeReference<Map<String, Map<String, Object>>>() { };
289      return mapper.readValue(json, type);
290    }

292    /**
293     * Get the number of live datanodes.
294     * 
295     * @param json JSON string that contains live node status.
296     * @param nn namenode status to return information in
297     */
298    private static void getLiveNodeCount(String json, NamenodeStatus nn)
299        throws IOException {
300      // Map of datanode host to (map of attribute name to value)
301      Map<String, Map<String, Object>> nodeMap = getNodeMap(json);
302      if (nodeMap == null || nodeMap.isEmpty()) {
303        return;
304      }
305      
306      nn.liveDatanodeCount = nodeMap.size();
307      for (Entry<String, Map<String, Object>> entry : nodeMap.entrySet()) {
308        // Inner map of attribute name to value
309        Map<String, Object> innerMap = entry.getValue();
310        if (innerMap != null) {
311          if (innerMap.get("adminState")
312              .equals(AdminStates.DECOMMISSIONED.toString())) {
313            nn.liveDecomCount++;
314          }
315        }
316      }
317    }
318  
319    /**
320     * Count the number of dead datanode.
321     * 
322     * @param nn namenode
323     * @param json JSON string
324     */
325    private static void getDeadNodeCount(String json, NamenodeStatus nn)
326        throws IOException {
327      Map<String, Map<String, Object>> nodeMap = getNodeMap(json);
328      if (nodeMap == null || nodeMap.isEmpty()) {
329        return;
330      }
331      
332      nn.deadDatanodeCount = nodeMap.size();
333      for (Entry<String, Map<String, Object>> entry : nodeMap.entrySet()) {
334        Map<String, Object> innerMap = entry.getValue();
335        if (innerMap != null && !innerMap.isEmpty()) {
336          if (((Boolean) innerMap.get("decommissioned"))
337              .booleanValue() == true) {
338            nn.deadDecomCount++;
339          }
340        }
341      }
342    }

348    public NamenodeStatus getNamenodeStatus(String props) throws IOException,

350      NamenodeStatus nn = new NamenodeStatus();

356      nn.nonDfsUsed = getProperty(props, "NonDfsUsedSpace").getLongValue();
357      nn.blocksCount = getProperty(props, "TotalBlocks").getLongValue();

363      nn.softwareVersion = getProperty(props, "SoftwareVersion").getTextValue();

383    /**
384     * Store the live datanode status information into datanode status map and
385     * DecommissionNode.
386     * 
387     * @param statusMap Map of datanode status. Key is datanode, value
388     *          is an inner map whose key is namenode, value is datanode status.
389     *          reported by each namenode.
390     * @param namenodeHost host name of the namenode
391     * @param json JSON string contains datanode status
392     * @throws IOException
393     */
394    private static void getLiveNodeStatus(
395        Map<String, Map<String, String>> statusMap, String namenodeHost,
396        String json) throws IOException {
397      Map<String, Map<String, Object>> nodeMap = getNodeMap(json);
398      if (nodeMap != null && !nodeMap.isEmpty()) {
399        List<String> liveDecommed = new ArrayList<String>();
400        for (Map.Entry<String, Map<String, Object>> entry: nodeMap.entrySet()) {
401          Map<String, Object> innerMap = entry.getValue();
402          String dn = entry.getKey();
403          if (innerMap != null) {
404            if (innerMap.get("adminState").equals(
405                AdminStates.DECOMMISSIONED.toString())) {
406              liveDecommed.add(dn);
407            }
408            // the inner map key is namenode, value is datanode status.
409            Map<String, String> nnStatus = statusMap.get(dn);
410            if (nnStatus == null) {
411              nnStatus = new HashMap<String, String>();
412            }
413            nnStatus.put(namenodeHost, (String) innerMap.get("adminState"));
414            // map whose key is datanode, value is the inner map.
415            statusMap.put(dn, nnStatus);
416          }
417        }
418      }
419    }

421    /**
422     * Store the dead datanode information into datanode status map and
423     * DecommissionNode.
424     * 
425     * @param statusMap map with key being datanode, value being an
426     *          inner map (key:namenode, value:decommisionning state).
427     * @param host datanode hostname
428     * @param json String
429     * @throws IOException
430     */
431    private static void getDeadNodeStatus(
432        Map<String, Map<String, String>> statusMap, String host,
433        String json) throws IOException {
434      Map<String, Map<String, Object>> nodeMap = getNodeMap(json);
435      if (nodeMap == null || nodeMap.isEmpty()) {
436        return;
437      }
438      List<String> deadDn = new ArrayList<String>();
439      List<String> deadDecommed = new ArrayList<String>();
440      for (Entry<String, Map<String, Object>> entry : nodeMap.entrySet()) {
441        deadDn.add(entry.getKey());
442        Map<String, Object> deadNodeDetailMap = entry.getValue();
443        String dn = entry.getKey();
444        if (deadNodeDetailMap != null && !deadNodeDetailMap.isEmpty()) {
445          // NN - status
446          Map<String, String> nnStatus = statusMap.get(dn);
447          if (nnStatus == null) {
448            nnStatus = new HashMap<String, String>();
449          }
450          if (((Boolean) deadNodeDetailMap.get("decommissioned"))
451              .booleanValue() == true) {
452            deadDecommed.add(dn);
453            nnStatus.put(host, AdminStates.DECOMMISSIONED.toString());
454          } else {
455            nnStatus.put(host, DEAD);
456          }
457          // dn-nn-status
458          statusMap.put(dn, nnStatus);
459        }
460      }
461    }

463    /**
464     * Get the decommisioning datanode information.
465     * 
466     * @param dataNodeStatusMap map with key being datanode, value being an
467     *          inner map (key:namenode, value:decommisionning state).
468     * @param host datanode
469     * @param json String
470     */
471    private static void getDecommissionNodeStatus(
472        Map<String, Map<String, String>> dataNodeStatusMap, String host,
473        String json) throws IOException {
474      Map<String, Map<String, Object>> nodeMap = getNodeMap(json);
475      if (nodeMap == null || nodeMap.isEmpty()) {
476        return;
477      }
478      List<String> decomming = new ArrayList<String>();
479      for (Entry<String, Map<String, Object>> entry : nodeMap.entrySet()) {
480        String dn = entry.getKey();
481        decomming.add(dn);
482        // nn-status
483        Map<String, String> nnStatus = new HashMap<String, String>();
484        if (dataNodeStatusMap.containsKey(dn)) {
485          nnStatus = dataNodeStatusMap.get(dn);
486        }
487        nnStatus.put(host, AdminStates.DECOMMISSION_INPROGRESS.toString());
488        // dn-nn-status
489        dataNodeStatusMap.put(dn, nnStatus);
490      }
491    }

494  /**
495   * This class contains cluster statistics.
496   */
497  static class ClusterStatus {
498    /** Exception indicates failure to get cluster status */
499    Exception error = null;
500    
501    /** Cluster status information */
502    String clusterid = "";
503    long total_sum = 0;
504    long free_sum = 0;
505    long clusterDfsUsed = 0;
506    long nonDfsUsed_sum = 0;
507    long totalFilesAndDirectories = 0;
508    
509    /** List of namenodes in the cluster */
510    final List<NamenodeStatus> nnList = new ArrayList<NamenodeStatus>();
511    
512    /** Map of namenode host and exception encountered when getting status */
513    final Map<String, Exception> nnExceptions = new HashMap<String, Exception>();
514    
515    public void setError(Exception e) {
516      error = e;
517    }
518    
519    public void addNamenodeStatus(NamenodeStatus nn) {
520      nnList.add(nn);
521      
522      // Add namenode status to cluster status
523      totalFilesAndDirectories += nn.filesAndDirectories;
524      total_sum += nn.capacity;
525      free_sum += nn.free;
526      clusterDfsUsed += nn.bpUsed;
527      nonDfsUsed_sum += nn.nonDfsUsed;
528    }
529
530    public void addException(String host, Exception e) {
531      nnExceptions.put(host, e);
532    }
533
534    public void toXML(XMLOutputter doc) throws IOException {
535      if (error != null) {
536        // general exception, only print exception message onto web page.
537        createGeneralException(doc, clusterid,
538            StringUtils.stringifyException(error));
539        doc.getWriter().flush();
540        return;
541      }
542      
543      int size = nnList.size();
544      long total = 0L, free = 0L, nonDfsUsed = 0l;
545      float dfsUsedPercent = 0.0f, dfsRemainingPercent = 0.0f;
546      if (size > 0) {
547        total = total_sum / size;
548        free = free_sum / size;
549        nonDfsUsed = nonDfsUsed_sum / size;
550        dfsUsedPercent = DFSUtil.getPercentUsed(clusterDfsUsed, total);
551        dfsRemainingPercent = DFSUtil.getPercentRemaining(free, total);
552      }
553    
554      doc.startTag("cluster");
555      doc.attribute("clusterId", clusterid);
556    
557      doc.startTag("storage");
558    
559      toXmlItemBlock(doc, "Total Files And Directories",
560          Long.toString(totalFilesAndDirectories));
561    
562      toXmlItemBlock(doc, "Configured Capacity", StringUtils.byteDesc(total));
563    
564      toXmlItemBlock(doc, "DFS Used", StringUtils.byteDesc(clusterDfsUsed));
565    
566      toXmlItemBlock(doc, "Non DFS Used", StringUtils.byteDesc(nonDfsUsed));
567    
568      toXmlItemBlock(doc, "DFS Remaining", StringUtils.byteDesc(free));
569    
570      // dfsUsedPercent
571      toXmlItemBlock(doc, "DFS Used%", DFSUtil.percent2String(dfsUsedPercent));
572    
573      // dfsRemainingPercent
574      toXmlItemBlock(doc, "DFS Remaining%", DFSUtil.percent2String(dfsRemainingPercent));
575    
576      doc.endTag(); // storage
577    
578      doc.startTag("namenodes");
579      // number of namenodes
580      toXmlItemBlock(doc, "NamenodesCount", Integer.toString(size));
581    
582      for (NamenodeStatus nn : nnList) {
583        doc.startTag("node");
584        toXmlItemBlockWithLink(doc, nn.host, nn.httpAddress, "NameNode");
585        toXmlItemBlock(doc, "Blockpool Used",
586            StringUtils.byteDesc(nn.bpUsed));
587        toXmlItemBlock(doc, "Blockpool Used%",
588            DFSUtil.percent2String(DFSUtil.getPercentUsed(nn.bpUsed, total)));
589        toXmlItemBlock(doc, "Files And Directories",
590            Long.toString(nn.filesAndDirectories));
591        toXmlItemBlock(doc, "Blocks", Long.toString(nn.blocksCount));
592        toXmlItemBlock(doc, "Missing Blocks",
593            Long.toString(nn.missingBlocksCount));
594        toXmlItemBlockWithLink(doc, nn.liveDatanodeCount + " ("
595            + nn.liveDecomCount + ")", new URL(nn.httpAddress,
596            "/dfsnodelist.jsp?whatNodes=LIVE"),
597            "Live Datanode (Decommissioned)");
598        toXmlItemBlockWithLink(doc, nn.deadDatanodeCount + " ("
599            + nn.deadDecomCount + ")", new URL(nn.httpAddress,
600            "/dfsnodelist.jsp?whatNodes=DEAD"),
601            "Dead Datanode (Decommissioned)");
602        toXmlItemBlock(doc, "Software Version", nn.softwareVersion);
603        doc.endTag(); // node
604      }
605      doc.endTag(); // namenodes
606    
607      createNamenodeExceptionMsg(doc, nnExceptions);
608      doc.endTag(); // cluster
609      doc.getWriter().flush();
610    }
611  }

634  /**
635   * cluster-wide decommission state of a datanode
636   */
637  public enum DecommissionStates {
638    /*
639     * If datanode state is decommissioning at one or more namenodes and 
640     * decommissioned at the rest of the namenodes.
641     */
642    DECOMMISSION_INPROGRESS("Decommission In Progress"),
643    
644    /* If datanode state at all the namenodes is decommissioned */
645    DECOMMISSIONED("Decommissioned"),
646    
647    /*
648     * If datanode state is not decommissioning at one or more namenodes and 
649     * decommissioned/decommissioning at the rest of the namenodes.
650     */
651    PARTIALLY_DECOMMISSIONED("Partially Decommissioning"),
652    
653    /*
654     * If datanode state is not known at a namenode, due to problems in getting
655     * the datanode state from the namenode.
656     */
657    UNKNOWN("Unknown");
658
659    final String value;
660    
661    DecommissionStates(final String v) {
662      this.value = v;
663    }
664
665    @Override
666    public String toString() {
667      return value;
668    }

671  /**
672   * This class consolidates the decommissioning datanodes information in the
673   * cluster and generates decommissioning reports in XML.
674   */
675  static class DecommissionStatus {
676    /** Error when set indicates failure to get decomission status*/
677    final Exception error;
678    
679    /** Map of dn host <-> (Map of NN host <-> decommissioning state) */
680    final Map<String, Map<String, String>> statusMap;
681    final String clusterid;
682    final int httpPort;
683    int decommissioned = 0;   // total number of decommissioned nodes
684    int decommissioning = 0;  // total number of decommissioning datanodes
685    int partial = 0;          // total number of partially decommissioned nodes
686    
687    /** Map of namenode and exception encountered when getting decom status */
688    Map<String, Exception> exceptions = new HashMap<String, Exception>();
689
690    private DecommissionStatus(Map<String, Map<String, String>> statusMap,
691        String cid, int httpPort, Map<String, Exception> exceptions) {
692      this(statusMap, cid, httpPort, exceptions, null);
693    }
694
695    public DecommissionStatus(String cid, Exception e) {
696      this(null, cid, -1, null, e);
697    }
698    
699    private DecommissionStatus(Map<String, Map<String, String>> statusMap,
700        String cid, int httpPort, Map<String, Exception> exceptions,
701        Exception error) {
702      this.statusMap = statusMap;
703      this.clusterid = cid;
704      this.httpPort = httpPort;
705      this.exceptions = exceptions;
706      this.error = error;
707    }
708
709    /**
710     * Generate decommissioning datanode report in XML format
711     * 
712     * @param doc
713     *          , xmloutputter
714     * @throws IOException
715     */
716    public void toXML(XMLOutputter doc) throws IOException {
717      if (error != null) {
718        createGeneralException(doc, clusterid,
719            StringUtils.stringifyException(error));
720        doc.getWriter().flush();
721        return;
722      } 
723      if (statusMap == null || statusMap.isEmpty()) {
724        // none of the namenodes has reported, print exceptions from each nn.
725        doc.startTag("cluster");
726        createNamenodeExceptionMsg(doc, exceptions);
727        doc.endTag();
728        doc.getWriter().flush();
729        return;
730      }
731      doc.startTag("cluster");
732      doc.attribute("clusterId", clusterid);
733
734      doc.startTag("decommissioningReport");
735      countDecommissionDatanodes();
736      toXmlItemBlock(doc, DecommissionStates.DECOMMISSIONED.toString(),
737          Integer.toString(decommissioned));
738
739      toXmlItemBlock(doc,
740          DecommissionStates.DECOMMISSION_INPROGRESS.toString(),
741          Integer.toString(decommissioning));
742
743      toXmlItemBlock(doc,
744          DecommissionStates.PARTIALLY_DECOMMISSIONED.toString(),
745          Integer.toString(partial));
746
747      doc.endTag(); // decommissioningReport
748
749      doc.startTag("datanodes");
750      Set<String> dnSet = statusMap.keySet();
751      for (String dnhost : dnSet) {
752        Map<String, String> nnStatus = statusMap.get(dnhost);
753        if (nnStatus == null || nnStatus.isEmpty()) {
754          continue;
755        }
756        String overallStatus = nnStatus.get(OVERALL_STATUS);
757        // check if datanode is in decommission states
758        if (overallStatus != null
759            && (overallStatus.equals(AdminStates.DECOMMISSION_INPROGRESS
760                .toString())
761                || overallStatus.equals(AdminStates.DECOMMISSIONED.toString())
762                || overallStatus
763                    .equals(DecommissionStates.PARTIALLY_DECOMMISSIONED
764                        .toString()) || overallStatus
765                .equals(DecommissionStates.UNKNOWN.toString()))) {
766          doc.startTag("node");
767          // dn
768          toXmlItemBlockWithLink(doc, dnhost, new URL("http", dnhost, httpPort,
769              ""), "DataNode");
770
771          // overall status first
772          toXmlItemBlock(doc, OVERALL_STATUS, overallStatus);
773
774          for (Map.Entry<String, String> m : nnStatus.entrySet()) {
775            String nn = m.getKey();
776            if (nn.equals(OVERALL_STATUS)) {
777              continue;
778            }
779            // xml
780            toXmlItemBlock(doc, nn, nnStatus.get(nn));
781          }
782          doc.endTag(); // node
783        }
784      }
785      doc.endTag(); // datanodes
786
787      createNamenodeExceptionMsg(doc, exceptions);
788
789      doc.endTag();// cluster
790    } // toXML
791
792    /**
793     * Count the total number of decommissioned/decommission_inprogress/
794     * partially decommissioned datanodes.
795     */
796    private void countDecommissionDatanodes() {
797      for (String dn : statusMap.keySet()) {
798        Map<String, String> nnStatus = statusMap.get(dn);
799        String status = nnStatus.get(OVERALL_STATUS);
800        if (status.equals(DecommissionStates.DECOMMISSIONED.toString())) {
801          decommissioned++;
802        } else if (status.equals(DecommissionStates.DECOMMISSION_INPROGRESS
803            .toString())) {
804          decommissioning++;
805        } else if (status.equals(DecommissionStates.PARTIALLY_DECOMMISSIONED
806            .toString())) {
807          partial++;
808        }
809      }
810    }
811  }

813  /**
814   * Generate a XML block as such, <item label=key value=value/>
815   */
816  private static void toXmlItemBlock(XMLOutputter doc, String key, String value)
817      throws IOException {
818    doc.startTag("item");
819    doc.attribute("label", key);
820    doc.attribute("value", value);
821    doc.endTag();
822  }

825   * Generate a XML block as such, <item label="Node" value="hostname"
826   * link="http://hostname:50070" />
827   */
828  private static void toXmlItemBlockWithLink(XMLOutputter doc, String value,
829      URL url, String label) throws IOException {
830    doc.startTag("item");
831    doc.attribute("label", label);
832    doc.attribute("value", value);
833    doc.attribute("link", url.toString());
834    doc.endTag(); // item
835  }

837  /**
838   * create the XML for exceptions that we encountered when connecting to
839   * namenode.
840   */
841  private static void createNamenodeExceptionMsg(XMLOutputter doc,
842      Map<String, Exception> exceptionMsg) throws IOException {
843    if (exceptionMsg.size() > 0) {
844      doc.startTag("unreportedNamenodes");
845      for (Map.Entry<String, Exception> m : exceptionMsg.entrySet()) {
846        doc.startTag("node");
847        doc.attribute("name", m.getKey());
848        doc.attribute("exception",
849            StringUtils.stringifyException(m.getValue()));
850        doc.endTag();// node
851      }
852      doc.endTag(); // unreportedNamnodes
853    }
854  }

856  /**
857   * create XML block from general exception.
858   */
859  private static void createGeneralException(XMLOutputter doc,
860      String clusterid, String eMsg) throws IOException {
861    doc.startTag("cluster");
862    doc.attribute("clusterId", clusterid);
863    doc.startTag("message");
864    doc.startTag("item");
865    doc.attribute("msg", eMsg);
866    doc.endTag(); // item
867    doc.endTag(); // message
868    doc.endTag(); // cluster
869  }

871  /**
872   * Read in the content from a URL
873   * @param url URL To read
874   * @return the text from the output
875   * @throws IOException if something went wrong
876   */
877  private static String readOutput(URL url) throws IOException {
878    StringBuilder out = new StringBuilder();
879    URLConnection connection = url.openConnection();
880    BufferedReader in = new BufferedReader(
881                            new InputStreamReader(
882                            connection.getInputStream(), Charsets.UTF_8));
883    String inputLine;
884    while ((inputLine = in.readLine()) != null) {
885      out.append(inputLine);
886    }
887    in.close();
888    return out.toString();
889  }

891  private static String queryMbean(URI httpAddress, Configuration conf)
892    throws IOException {
893    /**
894     * Although the other namenode might support HTTPS, it is fundamentally
895     * broken to get the JMX via an HTTPS connection inside the namenode,
896     * because in HTTPS set up the principal of the client and the one of
897     * the namenode differs. Therefore, there is no guarantees that the
898     * HTTPS connection can be set up.
899     *
900     * As a result, we just hard code the connection as an HTTP connection.
901     */
902    URL url = new URL(httpAddress.toURL(), JMX_QRY);
903    return readOutput(url);
904  }
905  /**
906   * In order to query a namenode mxbean, a http connection in the form of
907   * "http://hostname/jmx?qry=Hadoop:service=NameNode,name=NameNodeInfo"
908   * is sent to namenode.  JMX attributes are exposed via JmxJsonServelet on 
909   * the namenode side.
910   */
911  private static JsonNode getProperty(String props, String propertyname)
912  throws IOException {
913    if (props == null || props.equals("") || propertyname == null 
914        || propertyname.equals("")) {
915      return null;
916    }
917    ObjectMapper m = new ObjectMapper();
918    JsonNode rootNode = m.readValue(props, JsonNode.class);
919    JsonNode jn = rootNode.get("beans").get(0).get(propertyname);
920    return jn;
921  }
