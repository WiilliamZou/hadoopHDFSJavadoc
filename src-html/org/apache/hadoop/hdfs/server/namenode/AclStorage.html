039/**
040 * AclStorage contains utility methods that define how ACL data is stored in the
041 * namespace.
042 *
043 * If an inode has an ACL, then the ACL bit is set in the inode's
044 * {@link FsPermission} and the inode also contains an {@link AclFeature}.  For
045 * the access ACL, the owner and other entries are identical to the owner and
046 * other bits stored in FsPermission, so we reuse those.  The access mask entry
047 * is stored into the group permission bits of FsPermission.  This is consistent
048 * with other file systems' implementations of ACLs and eliminates the need for
049 * special handling in various parts of the codebase.  For example, if a user
050 * calls chmod to change group permission bits on a file with an ACL, then the
051 * expected behavior is to change the ACL's mask entry.  By saving the mask entry
052 * into the group permission bits, chmod continues to work correctly without
053 * special handling.  All remaining access entries (named users and named groups)
054 * are stored as explicit {@link AclEntry} instances in a list inside the
055 * AclFeature.  Additionally, all default entries are stored in the AclFeature.
056 *
057 * The methods in this class encapsulate these rules for reading or writing the
058 * ACL entries to the appropriate location.
059 *
060 * The methods in this class assume that input ACL entry lists have already been
061 * validated and sorted according to the rules enforced by
062 * {@link AclTransformation}.
063 */
064@InterfaceAudience.Private
065public final class AclStorage {
066
067  private final static ReferenceCountMap<AclFeature> UNIQUE_ACL_FEATURES =
068      new ReferenceCountMap<AclFeature>();
069
070  /**
071   * If a default ACL is defined on a parent directory, then copies that default
072   * ACL to a newly created child file or directory.
073   *
074   * @param child INode newly created child
075   */
076  public static void copyINodeDefaultAcl(INode child) {
077    INodeDirectory parent = child.getParent();
078    AclFeature parentAclFeature = parent.getAclFeature();
079    if (parentAclFeature == null || !(child.isFile() || child.isDirectory())) {
080      return;
081    }
082
083    // Split parent's entries into access vs. default.
084    List<AclEntry> featureEntries = getEntriesFromAclFeature(parent
085        .getAclFeature());
086    ScopedAclEntries scopedEntries = new ScopedAclEntries(featureEntries);
087    List<AclEntry> parentDefaultEntries = scopedEntries.getDefaultEntries();
088
089    // The parent may have an access ACL but no default ACL.  If so, exit.
090    if (parentDefaultEntries.isEmpty()) {
091      return;
092    }
093
094    // Pre-allocate list size for access entries to copy from parent.
095    List<AclEntry> accessEntries = Lists.newArrayListWithCapacity(
096      parentDefaultEntries.size());
097
098    FsPermission childPerm = child.getFsPermission();
099
100    // Copy each default ACL entry from parent to new child's access ACL.
101    boolean parentDefaultIsMinimal = AclUtil.isMinimalAcl(parentDefaultEntries);
102    for (AclEntry entry: parentDefaultEntries) {
103      AclEntryType type = entry.getType();
104      String name = entry.getName();
105      AclEntry.Builder builder = new AclEntry.Builder()
106        .setScope(AclEntryScope.ACCESS)
107        .setType(type)
108        .setName(name);
109
110      // The child's initial permission bits are treated as the mode parameter,
111      // which can filter copied permission values for owner, mask and other.
112      final FsAction permission;
113      if (type == AclEntryType.USER && name == null) {
114        permission = entry.getPermission().and(childPerm.getUserAction());
115      } else if (type == AclEntryType.GROUP && parentDefaultIsMinimal) {
116        // This only happens if the default ACL is a minimal ACL: exactly 3
117        // entries corresponding to owner, group and other.  In this case,
118        // filter the group permissions.
119        permission = entry.getPermission().and(childPerm.getGroupAction());
120      } else if (type == AclEntryType.MASK) {
121        // Group bits from mode parameter filter permission of mask entry.
122        permission = entry.getPermission().and(childPerm.getGroupAction());
123      } else if (type == AclEntryType.OTHER) {
124        permission = entry.getPermission().and(childPerm.getOtherAction());
125      } else {
126        permission = entry.getPermission();
127      }
128
129      builder.setPermission(permission);
130      accessEntries.add(builder.build());
131    }
132
133    // A new directory also receives a copy of the parent's default ACL.
134    List<AclEntry> defaultEntries = child.isDirectory() ? parentDefaultEntries :
135      Collections.<AclEntry>emptyList();
136
137    final FsPermission newPerm;
138    if (!AclUtil.isMinimalAcl(accessEntries) || !defaultEntries.isEmpty()) {
139      // Save the new ACL to the child.
140      child.addAclFeature(createAclFeature(accessEntries, defaultEntries));
141      newPerm = createFsPermissionForExtendedAcl(accessEntries, childPerm);
142    } else {
143      // The child is receiving a minimal ACL.
144      newPerm = createFsPermissionForMinimalAcl(accessEntries, childPerm);
145    }
146
147    child.setPermission(newPerm);
148  }

150  /**
151   * Reads the existing extended ACL entries of an inode.  This method returns
152   * only the extended ACL entries stored in the AclFeature.  If the inode does
153   * not have an ACL, then this method returns an empty list.  This method
154   * supports querying by snapshot ID.
155   *
156   * @param inode INode to read
157   * @param snapshotId int ID of snapshot to read
158   * @return List<AclEntry> containing extended inode ACL entries
159   */
160  public static List<AclEntry> readINodeAcl(INode inode, int snapshotId) {
161    AclFeature f = inode.getAclFeature(snapshotId);
162    return getEntriesFromAclFeature(f);
163  }
164
165  /**
166   * Build list of AclEntries from the AclFeature
167   * @param aclFeature AclFeature
168   * @return List of entries
169   */
170  @VisibleForTesting
171  static ImmutableList<AclEntry> getEntriesFromAclFeature(AclFeature aclFeature) {
172    if (aclFeature == null) {
173      return ImmutableList.<AclEntry> of();
174    }
175    ImmutableList.Builder<AclEntry> b = new ImmutableList.Builder<AclEntry>();
176    for (int pos = 0, entry; pos < aclFeature.getEntriesSize(); pos++) {
177      entry = aclFeature.getEntryAt(pos);
178      b.add(AclEntryStatusFormat.toAclEntry(entry));
179    }
180    return b.build();
181  }

183  /**
184   * Reads the existing ACL of an inode.  This method always returns the full
185   * logical ACL of the inode after reading relevant data from the inode's
186   * {@link FsPermission} and {@link AclFeature}.  Note that every inode
187   * logically has an ACL, even if no ACL has been set explicitly.  If the inode
188   * does not have an extended ACL, then the result is a minimal ACL consising of
189   * exactly 3 entries that correspond to the owner, group and other permissions.
190   * This method always reads the inode's current state and does not support
191   * querying by snapshot ID.  This is because the method is intended to support
192   * ACL modification APIs, which always apply a delta on top of current state.
193   *
194   * @param inode INode to read
195   * @return List<AclEntry> containing all logical inode ACL entries
196   */
197  public static List<AclEntry> readINodeLogicalAcl(INode inode) {
198    FsPermission perm = inode.getFsPermission();
199    AclFeature f = inode.getAclFeature();
200    if (f == null) {
201      return AclUtil.getMinimalAcl(perm);
202    }
203
204    final List<AclEntry> existingAcl;
205    // Split ACL entries stored in the feature into access vs. default.
206    List<AclEntry> featureEntries = getEntriesFromAclFeature(f);
207    ScopedAclEntries scoped = new ScopedAclEntries(featureEntries);
208    List<AclEntry> accessEntries = scoped.getAccessEntries();
209    List<AclEntry> defaultEntries = scoped.getDefaultEntries();
210
211    // Pre-allocate list size for the explicit entries stored in the feature
212    // plus the 3 implicit entries (owner, group and other) from the permission
213    // bits.
214    existingAcl = Lists.newArrayListWithCapacity(featureEntries.size() + 3);
215
216    if (!accessEntries.isEmpty()) {
217      // Add owner entry implied from user permission bits.
218      existingAcl.add(new AclEntry.Builder().setScope(AclEntryScope.ACCESS)
219          .setType(AclEntryType.USER).setPermission(perm.getUserAction())
220          .build());
221
222      // Next add all named user and group entries taken from the feature.
223      existingAcl.addAll(accessEntries);
224
225      // Add mask entry implied from group permission bits.
226      existingAcl.add(new AclEntry.Builder().setScope(AclEntryScope.ACCESS)
227          .setType(AclEntryType.MASK).setPermission(perm.getGroupAction())
228          .build());
229
230      // Add other entry implied from other permission bits.
231      existingAcl.add(new AclEntry.Builder().setScope(AclEntryScope.ACCESS)
232          .setType(AclEntryType.OTHER).setPermission(perm.getOtherAction())
233          .build());
234    } else {
235      // It's possible that there is a default ACL but no access ACL. In this
236      // case, add the minimal access ACL implied by the permission bits.
237      existingAcl.addAll(AclUtil.getMinimalAcl(perm));
238    }
239
240    // Add all default entries after the access entries.
241    existingAcl.addAll(defaultEntries);
242
243    // The above adds entries in the correct order, so no need to sort here.
244    return existingAcl;
245  }

247  /**
248   * Updates an inode with a new ACL.  This method takes a full logical ACL and
249   * stores the entries to the inode's {@link FsPermission} and
250   * {@link AclFeature}.
251   *
252   * @param inode INode to update
253   * @param newAcl List<AclEntry> containing new ACL entries
254   * @param snapshotId int latest snapshot ID of inode
255   * @throws AclException if the ACL is invalid for the given inode
256   * @throws QuotaExceededException if quota limit is exceeded
257   */
258  public static void updateINodeAcl(INode inode, List<AclEntry> newAcl,
259      int snapshotId) throws AclException, QuotaExceededException {
260    assert newAcl.size() >= 3;
261    FsPermission perm = inode.getFsPermission();
262    final FsPermission newPerm;
263    if (!AclUtil.isMinimalAcl(newAcl)) {
264      // This is an extended ACL.  Split entries into access vs. default.
265      ScopedAclEntries scoped = new ScopedAclEntries(newAcl);
266      List<AclEntry> accessEntries = scoped.getAccessEntries();
267      List<AclEntry> defaultEntries = scoped.getDefaultEntries();
268
269      // Only directories may have a default ACL.
270      if (!defaultEntries.isEmpty() && !inode.isDirectory()) {
271        throw new AclException(
272          "Invalid ACL: only directories may have a default ACL.");
273      }
274
275      // Attach entries to the feature.
276      if (inode.getAclFeature() != null) {
277        inode.removeAclFeature(snapshotId);
278      }
279      inode.addAclFeature(createAclFeature(accessEntries, defaultEntries),
280        snapshotId);
281      newPerm = createFsPermissionForExtendedAcl(accessEntries, perm);
282    } else {
283      // This is a minimal ACL.  Remove the ACL feature if it previously had one.
284      if (inode.getAclFeature() != null) {
285        inode.removeAclFeature(snapshotId);
286      }
287      newPerm = createFsPermissionForMinimalAcl(newAcl, perm);
288    }
289
290    inode.setPermission(newPerm, snapshotId);
291  }

299  /**
300   * Creates an AclFeature from the given ACL entries.
301   *
302   * @param accessEntries List<AclEntry> access ACL entries
303   * @param defaultEntries List<AclEntry> default ACL entries
304   * @return AclFeature containing the required ACL entries
305   */
306  private static AclFeature createAclFeature(List<AclEntry> accessEntries,
307      List<AclEntry> defaultEntries) {
308    // Pre-allocate list size for the explicit entries stored in the feature,
309    // which is all entries minus the 3 entries implicitly stored in the
310    // permission bits.
311    List<AclEntry> featureEntries = Lists.newArrayListWithCapacity(
312      (accessEntries.size() - 3) + defaultEntries.size());
313
314    // For the access ACL, the feature only needs to hold the named user and
315    // group entries.  For a correctly sorted ACL, these will be in a
316    // predictable range.
317    if (!AclUtil.isMinimalAcl(accessEntries)) {
318      featureEntries.addAll(
319        accessEntries.subList(1, accessEntries.size() - 2));
320    }
321
322    // Add all default entries to the feature.
323    featureEntries.addAll(defaultEntries);
324    return new AclFeature(AclEntryStatusFormat.toInt(featureEntries));
325  }

327  /**
328   * Creates the new FsPermission for an inode that is receiving an extended
329   * ACL, based on its access ACL entries.  For a correctly sorted ACL, the
330   * first entry is the owner and the last 2 entries are the mask and other
331   * entries respectively.  Also preserve sticky bit and toggle ACL bit on.
332   * Note that this method intentionally copies the permissions of the mask
333   * entry into the FsPermission group permissions.  This is consistent with the
334   * POSIX ACLs model, which presents the mask as the permissions of the group
335   * class.
336   *
337   * @param accessEntries List<AclEntry> access ACL entries
338   * @param existingPerm FsPermission existing permissions
339   * @return FsPermission new permissions
340   */
341  private static FsPermission createFsPermissionForExtendedAcl(
342      List<AclEntry> accessEntries, FsPermission existingPerm) {
343    return new FsPermission(accessEntries.get(0).getPermission(),
344      accessEntries.get(accessEntries.size() - 2).getPermission(),
345      accessEntries.get(accessEntries.size() - 1).getPermission(),
346      existingPerm.getStickyBit());
347  }
348
349  /**
350   * Creates the new FsPermission for an inode that is receiving a minimal ACL,
351   * based on its access ACL entries.  For a correctly sorted ACL, the owner,
352   * group and other permissions are in order.  Also preserve sticky bit and
353   * toggle ACL bit off.
354   *
355   * @param accessEntries List<AclEntry> access ACL entries
356   * @param existingPerm FsPermission existing permissions
357   * @return FsPermission new permissions
358   */
359  private static FsPermission createFsPermissionForMinimalAcl(
360      List<AclEntry> accessEntries, FsPermission existingPerm) {
361    return new FsPermission(accessEntries.get(0).getPermission(),
362      accessEntries.get(1).getPermission(),
363      accessEntries.get(2).getPermission(),
364      existingPerm.getStickyBit());
365  }

367  @VisibleForTesting
368  public static ReferenceCountMap<AclFeature> getUniqueAclFeatures() {
369    return UNIQUE_ACL_FEATURES;
370  }

372  /**
373   * Add reference for the said AclFeature
374   * 
375   * @param aclFeature
376   * @return Referenced AclFeature
377   */
378  public static AclFeature addAclFeature(AclFeature aclFeature) {
379    return UNIQUE_ACL_FEATURES.put(aclFeature);
380  }

382  /**
383   * Remove reference to the AclFeature
384   * 
385   * @param aclFeature
386   */
387  public static void removeAclFeature(AclFeature aclFeature) {
388    UNIQUE_ACL_FEATURES.remove(aclFeature);
