046/**
047 * AclTransformation defines the operations that can modify an ACL.  All ACL
048 * modifications take as input an existing ACL and apply logic to add new
049 * entries, modify existing entries or remove old entries.  Some operations also
050 * accept an ACL spec: a list of entries that further describes the requested
051 * change.  Different operations interpret the ACL spec differently.  In the
052 * case of adding an ACL to an inode that previously did not have one, the
053 * existing ACL can be a "minimal ACL" containing exactly 3 entries for owner,
054 * group and other, all derived from the {@link FsPermission} bits.
055 *
056 * The algorithms implemented here require sorted lists of ACL entries.  For any
057 * existing ACL, it is assumed that the entries are sorted.  This is because all
058 * ACL creation and modification is intended to go through these methods, and
059 * they all guarantee correct sort order in their outputs.  However, an ACL spec
060 * is considered untrusted user input, so all operations pre-sort the ACL spec as
061 * the first step.
062 */
063@InterfaceAudience.Private
064final class AclTransformation {
065  private static final int MAX_ENTRIES = 32;
066
067  /**
068   * Filters (discards) any existing ACL entries that have the same scope, type
069   * and name of any entry in the ACL spec.  If necessary, recalculates the mask
070   * entries.  If necessary, default entries may be inferred by copying the
071   * permissions of the corresponding access entries.  It is invalid to request
072   * removal of the mask entry from an ACL that would otherwise require a mask
073   * entry, due to existing named entries or an unnamed group entry.
074   *
075   * @param existingAcl List<AclEntry> existing ACL
076   * @param inAclSpec List<AclEntry> ACL spec describing entries to filter
077   * @return List<AclEntry> new ACL
078   * @throws AclException if validation fails
079   */
080  public static List<AclEntry> filterAclEntriesByAclSpec(
081      List<AclEntry> existingAcl, List<AclEntry> inAclSpec) throws AclException {
082    ValidatedAclSpec aclSpec = new ValidatedAclSpec(inAclSpec);
083    ArrayList<AclEntry> aclBuilder = Lists.newArrayListWithCapacity(MAX_ENTRIES);
084    EnumMap<AclEntryScope, AclEntry> providedMask =
085      Maps.newEnumMap(AclEntryScope.class);
086    EnumSet<AclEntryScope> maskDirty = EnumSet.noneOf(AclEntryScope.class);
087    EnumSet<AclEntryScope> scopeDirty = EnumSet.noneOf(AclEntryScope.class);
088    for (AclEntry existingEntry: existingAcl) {
089      if (aclSpec.containsKey(existingEntry)) {
090        scopeDirty.add(existingEntry.getScope());
091        if (existingEntry.getType() == MASK) {
092          maskDirty.add(existingEntry.getScope());
093        }
094      } else {
095        if (existingEntry.getType() == MASK) {
096          providedMask.put(existingEntry.getScope(), existingEntry);
097        } else {
098          aclBuilder.add(existingEntry);
099        }
100      }
101    }
102    copyDefaultsIfNeeded(aclBuilder);
103    calculateMasks(aclBuilder, providedMask, maskDirty, scopeDirty);
104    return buildAndValidateAcl(aclBuilder);
105  }
106
107  /**
108   * Filters (discards) any existing default ACL entries.  The new ACL retains
109   * only the access ACL entries.
110   *
111   * @param existingAcl List<AclEntry> existing ACL
112   * @return List<AclEntry> new ACL
113   * @throws AclException if validation fails
114   */
115  public static List<AclEntry> filterDefaultAclEntries(
116      List<AclEntry> existingAcl) throws AclException {
117    ArrayList<AclEntry> aclBuilder = Lists.newArrayListWithCapacity(MAX_ENTRIES);
118    for (AclEntry existingEntry: existingAcl) {
119      if (existingEntry.getScope() == DEFAULT) {
120        // Default entries sort after access entries, so we can exit early.
121        break;
122      }
123      aclBuilder.add(existingEntry);
124    }
125    return buildAndValidateAcl(aclBuilder);
126  }

128  /**
129   * Merges the entries of the ACL spec into the existing ACL.  If necessary,
130   * recalculates the mask entries.  If necessary, default entries may be
131   * inferred by copying the permissions of the corresponding access entries.
132   *
133   * @param existingAcl List<AclEntry> existing ACL
134   * @param inAclSpec List<AclEntry> ACL spec containing entries to merge
135   * @return List<AclEntry> new ACL
136   * @throws AclException if validation fails
137   */
138  public static List<AclEntry> mergeAclEntries(List<AclEntry> existingAcl,
139      List<AclEntry> inAclSpec) throws AclException {
140    ValidatedAclSpec aclSpec = new ValidatedAclSpec(inAclSpec);
141    ArrayList<AclEntry> aclBuilder = Lists.newArrayListWithCapacity(MAX_ENTRIES);
142    List<AclEntry> foundAclSpecEntries =
143      Lists.newArrayListWithCapacity(MAX_ENTRIES);
144    EnumMap<AclEntryScope, AclEntry> providedMask =
145      Maps.newEnumMap(AclEntryScope.class);
146    EnumSet<AclEntryScope> maskDirty = EnumSet.noneOf(AclEntryScope.class);
147    EnumSet<AclEntryScope> scopeDirty = EnumSet.noneOf(AclEntryScope.class);
148    for (AclEntry existingEntry: existingAcl) {
149      AclEntry aclSpecEntry = aclSpec.findByKey(existingEntry);
150      if (aclSpecEntry != null) {
151        foundAclSpecEntries.add(aclSpecEntry);
152        scopeDirty.add(aclSpecEntry.getScope());
153        if (aclSpecEntry.getType() == MASK) {
154          providedMask.put(aclSpecEntry.getScope(), aclSpecEntry);
155          maskDirty.add(aclSpecEntry.getScope());
156        } else {
157          aclBuilder.add(aclSpecEntry);
158        }
159      } else {
160        if (existingEntry.getType() == MASK) {
161          providedMask.put(existingEntry.getScope(), existingEntry);
162        } else {
163          aclBuilder.add(existingEntry);
164        }
165      }
166    }
167    // ACL spec entries that were not replacements are new additions.
168    for (AclEntry newEntry: aclSpec) {
169      if (Collections.binarySearch(foundAclSpecEntries, newEntry,
170          ACL_ENTRY_COMPARATOR) < 0) {
171        scopeDirty.add(newEntry.getScope());
172        if (newEntry.getType() == MASK) {
173          providedMask.put(newEntry.getScope(), newEntry);
174          maskDirty.add(newEntry.getScope());
175        } else {
176          aclBuilder.add(newEntry);
177        }
178      }
179    }
180    copyDefaultsIfNeeded(aclBuilder);
181    calculateMasks(aclBuilder, providedMask, maskDirty, scopeDirty);
182    return buildAndValidateAcl(aclBuilder);
183  }
184
185  /**
186   * Completely replaces the ACL with the entries of the ACL spec.  If
187   * necessary, recalculates the mask entries.  If necessary, default entries
188   * are inferred by copying the permissions of the corresponding access
189   * entries.  Replacement occurs separately for each of the access ACL and the
190   * default ACL.  If the ACL spec contains only access entries, then the
191   * existing default entries are retained.  If the ACL spec contains only
192   * default entries, then the existing access entries are retained.  If the ACL
193   * spec contains both access and default entries, then both are replaced.
194   *
195   * @param existingAcl List<AclEntry> existing ACL
196   * @param inAclSpec List<AclEntry> ACL spec containing replacement entries
197   * @return List<AclEntry> new ACL
198   * @throws AclException if validation fails
199   */
200  public static List<AclEntry> replaceAclEntries(List<AclEntry> existingAcl,
201      List<AclEntry> inAclSpec) throws AclException {
202    ValidatedAclSpec aclSpec = new ValidatedAclSpec(inAclSpec);
203    ArrayList<AclEntry> aclBuilder = Lists.newArrayListWithCapacity(MAX_ENTRIES);
204    // Replacement is done separately for each scope: access and default.
205    EnumMap<AclEntryScope, AclEntry> providedMask =
206      Maps.newEnumMap(AclEntryScope.class);
207    EnumSet<AclEntryScope> maskDirty = EnumSet.noneOf(AclEntryScope.class);
208    EnumSet<AclEntryScope> scopeDirty = EnumSet.noneOf(AclEntryScope.class);
209    for (AclEntry aclSpecEntry: aclSpec) {
210      scopeDirty.add(aclSpecEntry.getScope());
211      if (aclSpecEntry.getType() == MASK) {
212        providedMask.put(aclSpecEntry.getScope(), aclSpecEntry);
213        maskDirty.add(aclSpecEntry.getScope());
214      } else {
215        aclBuilder.add(aclSpecEntry);
216      }
217    }
218    // Copy existing entries if the scope was not replaced.
219    for (AclEntry existingEntry: existingAcl) {
220      if (!scopeDirty.contains(existingEntry.getScope())) {
221        if (existingEntry.getType() == MASK) {
222          providedMask.put(existingEntry.getScope(), existingEntry);
223        } else {
224          aclBuilder.add(existingEntry);
225        }
226      }
227    }
228    copyDefaultsIfNeeded(aclBuilder);
229    calculateMasks(aclBuilder, providedMask, maskDirty, scopeDirty);
230    return buildAndValidateAcl(aclBuilder);
231  }

239  /**
240   * Comparator that enforces required ordering for entries within an ACL:
241   * -owner entry (unnamed user)
242   * -all named user entries (internal ordering undefined)
243   * -owning group entry (unnamed group)
244   * -all named group entries (internal ordering undefined)
245   * -mask entry
246   * -other entry
247   * All access ACL entries sort ahead of all default ACL entries.
248   */
249  static final Comparator<AclEntry> ACL_ENTRY_COMPARATOR =
250    new Comparator<AclEntry>() {
251      @Override
252      public int compare(AclEntry entry1, AclEntry entry2) {
253        return ComparisonChain.start()
254          .compare(entry1.getScope(), entry2.getScope(),
255            Ordering.explicit(ACCESS, DEFAULT))
256          .compare(entry1.getType(), entry2.getType(),
257            Ordering.explicit(USER, GROUP, MASK, OTHER))
258          .compare(entry1.getName(), entry2.getName(),
259            Ordering.natural().nullsFirst())
260          .result();
261      }
262    };
263
264  /**
265   * Builds the final list of ACL entries to return by trimming, sorting and
266   * validating the ACL entries that have been added.
267   *
268   * @param aclBuilder ArrayList<AclEntry> containing entries to build
269   * @return List<AclEntry> unmodifiable, sorted list of ACL entries
270   * @throws AclException if validation fails
271   */
272  private static List<AclEntry> buildAndValidateAcl(
273      ArrayList<AclEntry> aclBuilder) throws AclException {
274    if (aclBuilder.size() > MAX_ENTRIES) {
275      throw new AclException("Invalid ACL: ACL has " + aclBuilder.size() +
276        " entries, which exceeds maximum of " + MAX_ENTRIES + ".");
277    }
278    aclBuilder.trimToSize();
279    Collections.sort(aclBuilder, ACL_ENTRY_COMPARATOR);
280    // Full iteration to check for duplicates and invalid named entries.
281    AclEntry prevEntry = null;
282    for (AclEntry entry: aclBuilder) {
283      if (prevEntry != null &&
284          ACL_ENTRY_COMPARATOR.compare(prevEntry, entry) == 0) {
285        throw new AclException(
286          "Invalid ACL: multiple entries with same scope, type and name.");
287      }
288      if (entry.getName() != null && (entry.getType() == MASK ||
289          entry.getType() == OTHER)) {
290        throw new AclException(
291          "Invalid ACL: this entry type must not have a name: " + entry + ".");
292      }
293      prevEntry = entry;
294    }
295    // Search for the required base access entries.  If there is a default ACL,
296    // then do the same check on the default entries.
297    ScopedAclEntries scopedEntries = new ScopedAclEntries(aclBuilder);
298    for (AclEntryType type: EnumSet.of(USER, GROUP, OTHER)) {
299      AclEntry accessEntryKey = new AclEntry.Builder().setScope(ACCESS)
300        .setType(type).build();
301      if (Collections.binarySearch(scopedEntries.getAccessEntries(),
302          accessEntryKey, ACL_ENTRY_COMPARATOR) < 0) {
303        throw new AclException(
304          "Invalid ACL: the user, group and other entries are required.");
305      }
306      if (!scopedEntries.getDefaultEntries().isEmpty()) {
307        AclEntry defaultEntryKey = new AclEntry.Builder().setScope(DEFAULT)
308          .setType(type).build();
309        if (Collections.binarySearch(scopedEntries.getDefaultEntries(),
310            defaultEntryKey, ACL_ENTRY_COMPARATOR) < 0) {
311          throw new AclException(
312            "Invalid default ACL: the user, group and other entries are required.");
313        }
314      }
315    }
316    return Collections.unmodifiableList(aclBuilder);
317  }

319  /**
320   * Calculates mask entries required for the ACL.  Mask calculation is performed
321   * separately for each scope: access and default.  This method is responsible
322   * for handling the following cases of mask calculation:
323   * 1. Throws an exception if the caller attempts to remove the mask entry of an
324   *   existing ACL that requires it.  If the ACL has any named entries, then a
325   *   mask entry is required.
326   * 2. If the caller supplied a mask in the ACL spec, use it.
327   * 3. If the caller did not supply a mask, but there are ACL entry changes in
328   *   this scope, then automatically calculate a new mask.  The permissions of
329   *   the new mask are the union of the permissions on the group entry and all
330   *   named entries.
331   *
332   * @param aclBuilder ArrayList<AclEntry> containing entries to build
333   * @param providedMask EnumMap<AclEntryScope, AclEntry> mapping each scope to
334   *   the mask entry that was provided for that scope (if provided)
335   * @param maskDirty EnumSet<AclEntryScope> which contains a scope if the mask
336   *   entry is dirty (added or deleted) in that scope
337   * @param scopeDirty EnumSet<AclEntryScope> which contains a scope if any entry
338   *   is dirty (added or deleted) in that scope
339   * @throws AclException if validation fails
340   */
341  private static void calculateMasks(List<AclEntry> aclBuilder,
342      EnumMap<AclEntryScope, AclEntry> providedMask,
343      EnumSet<AclEntryScope> maskDirty, EnumSet<AclEntryScope> scopeDirty)
344      throws AclException {
345    EnumSet<AclEntryScope> scopeFound = EnumSet.noneOf(AclEntryScope.class);
346    EnumMap<AclEntryScope, FsAction> unionPerms =
347      Maps.newEnumMap(AclEntryScope.class);
348    EnumSet<AclEntryScope> maskNeeded = EnumSet.noneOf(AclEntryScope.class);
349    // Determine which scopes are present, which scopes need a mask, and the
350    // union of group class permissions in each scope.
351    for (AclEntry entry: aclBuilder) {
352      scopeFound.add(entry.getScope());
353      if (entry.getType() == GROUP || entry.getName() != null) {
354        FsAction scopeUnionPerms = Objects.firstNonNull(
355          unionPerms.get(entry.getScope()), FsAction.NONE);
356        unionPerms.put(entry.getScope(),
357          scopeUnionPerms.or(entry.getPermission()));
358      }
359      if (entry.getName() != null) {
360        maskNeeded.add(entry.getScope());
361      }
362    }
363    // Add mask entry if needed in each scope.
364    for (AclEntryScope scope: scopeFound) {
365      if (!providedMask.containsKey(scope) && maskNeeded.contains(scope) &&
366          maskDirty.contains(scope)) {
367        // Caller explicitly removed mask entry, but it's required.
368        throw new AclException(
369          "Invalid ACL: mask is required and cannot be deleted.");
370      } else if (providedMask.containsKey(scope) &&
371          (!scopeDirty.contains(scope) || maskDirty.contains(scope))) {
372        // Caller explicitly provided new mask, or we are preserving the existing
373        // mask in an unchanged scope.
374        aclBuilder.add(providedMask.get(scope));
375      } else if (maskNeeded.contains(scope) || providedMask.containsKey(scope)) {
376        // Otherwise, if there are maskable entries present, or the ACL
377        // previously had a mask, then recalculate a mask automatically.
378        aclBuilder.add(new AclEntry.Builder()
379          .setScope(scope)
380          .setType(MASK)
381          .setPermission(unionPerms.get(scope))
382          .build());
383      }
384    }
385  }

387  /**
388   * Adds unspecified default entries by copying permissions from the
389   * corresponding access entries.
390   *
391   * @param aclBuilder ArrayList<AclEntry> containing entries to build
392   */
393  private static void copyDefaultsIfNeeded(List<AclEntry> aclBuilder) {
394    Collections.sort(aclBuilder, ACL_ENTRY_COMPARATOR);
395    ScopedAclEntries scopedEntries = new ScopedAclEntries(aclBuilder);
396    if (!scopedEntries.getDefaultEntries().isEmpty()) {
397      List<AclEntry> accessEntries = scopedEntries.getAccessEntries();
398      List<AclEntry> defaultEntries = scopedEntries.getDefaultEntries();
399      List<AclEntry> copiedEntries = Lists.newArrayListWithCapacity(3);
400      for (AclEntryType type: EnumSet.of(USER, GROUP, OTHER)) {
401        AclEntry defaultEntryKey = new AclEntry.Builder().setScope(DEFAULT)
402          .setType(type).build();
403        int defaultEntryIndex = Collections.binarySearch(defaultEntries,
404          defaultEntryKey, ACL_ENTRY_COMPARATOR);
405        if (defaultEntryIndex < 0) {
406          AclEntry accessEntryKey = new AclEntry.Builder().setScope(ACCESS)
407            .setType(type).build();
408          int accessEntryIndex = Collections.binarySearch(accessEntries,
409            accessEntryKey, ACL_ENTRY_COMPARATOR);
410          if (accessEntryIndex >= 0) {
411            copiedEntries.add(new AclEntry.Builder()
412              .setScope(DEFAULT)
413              .setType(type)
414              .setPermission(accessEntries.get(accessEntryIndex).getPermission())
415              .build());
416          }
417        }
418      }
419      // Add all copied entries when done to prevent potential issues with binary
420      // search on a modified aclBulider during the main loop.
421      aclBuilder.addAll(copiedEntries);
422    }
423  }

425  /**
426   * An ACL spec that has been pre-validated and sorted.
427   */
428  private static final class ValidatedAclSpec implements Iterable<AclEntry> {
429    private final List<AclEntry> aclSpec;
430
431    /**
432     * Creates a ValidatedAclSpec by pre-validating and sorting the given ACL
433     * entries.  Pre-validation checks that it does not exceed the maximum
434     * entries.  This check is performed before modifying the ACL, and it's
435     * actually insufficient for enforcing the maximum number of entries.
436     * Transformation logic can create additional entries automatically,such as
437     * the mask and some of the default entries, so we also need additional
438     * checks during transformation.  The up-front check is still valuable here
439     * so that we don't run a lot of expensive transformation logic while
440     * holding the namesystem lock for an attacker who intentionally sent a huge
441     * ACL spec.
442     *
443     * @param aclSpec List<AclEntry> containing unvalidated input ACL spec
444     * @throws AclException if validation fails
445     */
446    public ValidatedAclSpec(List<AclEntry> aclSpec) throws AclException {
447      if (aclSpec.size() > MAX_ENTRIES) {
448        throw new AclException("Invalid ACL: ACL spec has " + aclSpec.size() +
449          " entries, which exceeds maximum of " + MAX_ENTRIES + ".");
450      }
451      Collections.sort(aclSpec, ACL_ENTRY_COMPARATOR);
452      this.aclSpec = aclSpec;
453    }
454
455    /**
456     * Returns true if this contains an entry matching the given key.  An ACL
457     * entry's key consists of scope, type and name (but not permission).
458     *
459     * @param key AclEntry search key
460     * @return boolean true if found
461     */
462    public boolean containsKey(AclEntry key) {
463      return Collections.binarySearch(aclSpec, key, ACL_ENTRY_COMPARATOR) >= 0;
464    }
465
466    /**
467     * Returns the entry matching the given key or null if not found.  An ACL
468     * entry's key consists of scope, type and name (but not permission).
469     *
470     * @param key AclEntry search key
471     * @return AclEntry entry matching the given key or null if not found
472     */
473    public AclEntry findByKey(AclEntry key) {
474      int index = Collections.binarySearch(aclSpec, key, ACL_ENTRY_COMPARATOR);
475      if (index >= 0) {
476        return aclSpec.get(index);
477      }
478      return null;
479    }
480
481    @Override
482    public Iterator<AclEntry> iterator() {
483      return aclSpec.iterator();
484    }
485  }
