108  @Override
109  synchronized public EditLogOutputStream startLogSegment(long txid,
110      int layoutVersion) throws IOException {
111    try {
112      currentInProgress = NNStorage.getInProgressEditsFile(sd, txid);
113      EditLogOutputStream stm = new EditLogFileOutputStream(conf,
114          currentInProgress, outputBufferCapacity);
115      stm.create(layoutVersion);
116      return stm;
117    } catch (IOException e) {
118      LOG.warn("Unable to start log segment " + txid +
119          " at " + currentInProgress + ": " +
120          e.getLocalizedMessage());
121      errorReporter.reportErrorOnFile(currentInProgress);
122      throw e;
123    }
124  }

126  @Override
127  synchronized public void finalizeLogSegment(long firstTxId, long lastTxId)
128      throws IOException {
129    File inprogressFile = NNStorage.getInProgressEditsFile(sd, firstTxId);
130
131    File dstFile = NNStorage.getFinalizedEditsFile(
132        sd, firstTxId, lastTxId);
133    LOG.info("Finalizing edits file " + inprogressFile + " -> " + dstFile);
134    
135    Preconditions.checkState(!dstFile.exists(),
136        "Can't finalize edits file " + inprogressFile + " since finalized file " +
137        "already exists");
138
139    try {
140      NativeIO.renameTo(inprogressFile, dstFile);
141    } catch (IOException e) {
142      errorReporter.reportErrorOnFile(dstFile);
143      throw new IllegalStateException("Unable to finalize edits file " + inprogressFile, e);
144    }
145
146    if (inprogressFile.equals(currentInProgress)) {
147      currentInProgress = null;
148    }
149  }

161  @Override
162  public void purgeLogsOlderThan(long minTxIdToKeep)
163      throws IOException {
164    LOG.info("Purging logs older than " + minTxIdToKeep);
165    File[] files = FileUtil.listFiles(sd.getCurrentDir());
166    List<EditLogFile> editLogs = matchEditLogs(files, true);
167    for (EditLogFile log : editLogs) {
168      if (log.getFirstTxId() < minTxIdToKeep &&
169          log.getLastTxId() < minTxIdToKeep) {
170        purger.purgeLog(log);
171      }
172    }
173  }

175  /**
176   * Find all editlog segments starting at or above the given txid.
177   * @param firstTxId the txnid which to start looking
178   * @param inProgressOk whether or not to include the in-progress edit log 
179   *        segment       
180   * @return a list of remote edit logs
181   * @throws IOException if edit logs cannot be listed.
182   */
183  public List<RemoteEditLog> getRemoteEditLogs(long firstTxId,
184      boolean inProgressOk) throws IOException {
185    File currentDir = sd.getCurrentDir();
186    List<EditLogFile> allLogFiles = matchEditLogs(currentDir);
187    List<RemoteEditLog> ret = Lists.newArrayListWithCapacity(
188        allLogFiles.size());
189    for (EditLogFile elf : allLogFiles) {
190      if (elf.hasCorruptHeader() || (!inProgressOk && elf.isInProgress())) {
191        continue;
192      }
193      if (elf.isInProgress()) {
194        try {
195          elf.validateLog();
196        } catch (IOException e) {
197          LOG.error("got IOException while trying to validate header of " +
198              elf + ".  Skipping.", e);
199          continue;
200        }
201      }
202      if (elf.getFirstTxId() >= firstTxId) {
203        ret.add(new RemoteEditLog(elf.firstTxId, elf.lastTxId,
204            elf.isInProgress()));
205      } else if (elf.getFirstTxId() < firstTxId && firstTxId <= elf.getLastTxId()) {
206        // If the firstTxId is in the middle of an edit log segment. Return this
207        // anyway and let the caller figure out whether it wants to use it.
208        ret.add(new RemoteEditLog(elf.firstTxId, elf.lastTxId,
209            elf.isInProgress()));
210      }
211    }
212    
213    Collections.sort(ret);
214    
215    return ret;
216  }

218  /**
219   * Discard all editlog segments whose first txid is greater than or equal to
220   * the given txid, by renaming them with suffix ".trash".
221   */
222  private void discardEditLogSegments(long startTxId) throws IOException {
223    File currentDir = sd.getCurrentDir();
224    List<EditLogFile> allLogFiles = matchEditLogs(currentDir);
225    List<EditLogFile> toTrash = Lists.newArrayList();
226    LOG.info("Discard the EditLog files, the given start txid is " + startTxId);
227    // go through the editlog files to make sure the startTxId is right at the
228    // segment boundary
229    for (EditLogFile elf : allLogFiles) {
230      if (elf.getFirstTxId() >= startTxId) {
231        toTrash.add(elf);
232      } else {
233        Preconditions.checkState(elf.getLastTxId() < startTxId);
234      }
235    }
236
237    for (EditLogFile elf : toTrash) {
238      // rename these editlog file as .trash
239      elf.moveAsideTrashFile(startTxId);
240      LOG.info("Trash the EditLog file " + elf);
241    }
242  }

258  static List<EditLogFile> matchEditLogs(File[] filesInStorage) {
259    return matchEditLogs(filesInStorage, false);

262  private static List<EditLogFile> matchEditLogs(File[] filesInStorage,
263      boolean forPurging) {
264    List<EditLogFile> ret = Lists.newArrayList();
265    for (File f : filesInStorage) {
266      String name = f.getName();
267      // Check for edits
268      Matcher editsMatch = EDITS_REGEX.matcher(name);
269      if (editsMatch.matches()) {
270        try {
271          long startTxId = Long.parseLong(editsMatch.group(1));
272          long endTxId = Long.parseLong(editsMatch.group(2));
273          ret.add(new EditLogFile(f, startTxId, endTxId));
274          continue;
275        } catch (NumberFormatException nfe) {
276          LOG.error("Edits file " + f + " has improperly formatted " +
277                    "transaction ID");
278          // skip
279        }
280      }
281      
282      // Check for in-progress edits
283      Matcher inProgressEditsMatch = EDITS_INPROGRESS_REGEX.matcher(name);
284      if (inProgressEditsMatch.matches()) {
285        try {
286          long startTxId = Long.parseLong(inProgressEditsMatch.group(1));
287          ret.add(
288              new EditLogFile(f, startTxId, HdfsConstants.INVALID_TXID, true));
289          continue;
290        } catch (NumberFormatException nfe) {
291          LOG.error("In-progress edits file " + f + " has improperly " +
292                    "formatted transaction ID");
293          // skip
294        }
295      }
296      if (forPurging) {
297        // Check for in-progress stale edits
298        Matcher staleInprogressEditsMatch = EDITS_INPROGRESS_STALE_REGEX
299            .matcher(name);
300        if (staleInprogressEditsMatch.matches()) {
301          try {
302            long startTxId = Long.parseLong(staleInprogressEditsMatch.group(1));
303            ret.add(new EditLogFile(f, startTxId, HdfsConstants.INVALID_TXID,
304                true));
305            continue;
306          } catch (NumberFormatException nfe) {
307            LOG.error("In-progress stale edits file " + f + " has improperly "
308                + "formatted transaction ID");
309            // skip
310    }
311        }
312      }
313    }
314    return ret;
315  }

317  @Override
318  synchronized public void selectInputStreams(
319      Collection<EditLogInputStream> streams, long fromTxId,
320      boolean inProgressOk) throws IOException {
321    List<EditLogFile> elfs = matchEditLogs(sd.getCurrentDir());
322    LOG.debug(this + ": selecting input streams starting at " + fromTxId + 
323        (inProgressOk ? " (inProgress ok) " : " (excluding inProgress) ") +
324        "from among " + elfs.size() + " candidate file(s)");
325    addStreamsToCollectionFromFiles(elfs, streams, fromTxId, inProgressOk);
326  }
327  
328  static void addStreamsToCollectionFromFiles(Collection<EditLogFile> elfs,
329      Collection<EditLogInputStream> streams, long fromTxId, boolean inProgressOk) {
330    for (EditLogFile elf : elfs) {
331      if (elf.isInProgress()) {
332        if (!inProgressOk) {
333          LOG.debug("passing over " + elf + " because it is in progress " +
334              "and we are ignoring in-progress logs.");
335          continue;
336        }
337        try {
338          elf.validateLog();
339        } catch (IOException e) {
340          LOG.error("got IOException while trying to validate header of " +
341              elf + ".  Skipping.", e);
342          continue;
343        }
344      }
345      if (elf.lastTxId < fromTxId) {
346        assert elf.lastTxId != HdfsConstants.INVALID_TXID;
347        LOG.debug("passing over " + elf + " because it ends at " +
348            elf.lastTxId + ", but we only care about transactions " +
349            "as new as " + fromTxId);
350        continue;
351      }
352      EditLogFileInputStream elfis = new EditLogFileInputStream(elf.getFile(),
353            elf.getFirstTxId(), elf.getLastTxId(), elf.isInProgress());
354      LOG.debug("selecting edit log stream " + elf);
355      streams.add(elfis);
356    }
357  }

359  @Override
360  synchronized public void recoverUnfinalizedSegments() throws IOException {
361    File currentDir = sd.getCurrentDir();
362    LOG.info("Recovering unfinalized segments in " + currentDir);
363    List<EditLogFile> allLogFiles = matchEditLogs(currentDir);
364
365    for (EditLogFile elf : allLogFiles) {
366      if (elf.getFile().equals(currentInProgress)) {
367        continue;
368      }
369      if (elf.isInProgress()) {
370        // If the file is zero-length, we likely just crashed after opening the
371        // file, but before writing anything to it. Safe to delete it.
372        if (elf.getFile().length() == 0) {
373          LOG.info("Deleting zero-length edit log file " + elf);
374          if (!elf.getFile().delete()) {
375            throw new IOException("Unable to delete file " + elf.getFile());
376          }
377          continue;
378        }
379
380        elf.validateLog();
381
382        if (elf.hasCorruptHeader()) {
383          elf.moveAsideCorruptFile();
384          throw new CorruptionException("In-progress edit log file is corrupt: "
385              + elf);
386        }
387        if (elf.getLastTxId() == HdfsConstants.INVALID_TXID) {
388          // If the file has a valid header (isn't corrupt) but contains no
389          // transactions, we likely just crashed after opening the file and
390          // writing the header, but before syncing any transactions. Safe to
391          // delete the file.
392          LOG.info("Moving aside edit log file that seems to have zero " +
393              "transactions " + elf);
394          elf.moveAsideEmptyFile();
395          continue;
396        }
397        finalizeLogSegment(elf.getFirstTxId(), elf.getLastTxId());
398      }
399    }
400  }

402  public List<EditLogFile> getLogFiles(long fromTxId) throws IOException {
403    File currentDir = sd.getCurrentDir();
404    List<EditLogFile> allLogFiles = matchEditLogs(currentDir);
405    List<EditLogFile> logFiles = Lists.newArrayList();
406    
407    for (EditLogFile elf : allLogFiles) {
408      if (fromTxId <= elf.getFirstTxId() ||
409          elf.containsTxId(fromTxId)) {
410        logFiles.add(elf);
411      }
412    }
413    
414    Collections.sort(logFiles, EditLogFile.COMPARE_BY_START_TXID);
415
416    return logFiles;
417  }

419  public EditLogFile getLogFile(long startTxId) throws IOException {
420    return getLogFile(sd.getCurrentDir(), startTxId);
421  }

423  public static EditLogFile getLogFile(File dir, long startTxId)
424      throws IOException {
425    List<EditLogFile> files = matchEditLogs(dir);
426    List<EditLogFile> ret = Lists.newLinkedList();
427    for (EditLogFile elf : files) {
428      if (elf.getFirstTxId() == startTxId) {
429        ret.add(elf);
430      }
431    }
432    
433    if (ret.isEmpty()) {
434      // no matches
435      return null;
436    } else if (ret.size() == 1) {
437      return ret.get(0);
438    } else {
439      throw new IllegalStateException("More than one log segment in " + 
440          dir + " starting at txid " + startTxId + ": " +
441          Joiner.on(", ").join(ret));
442    }
443  }

450  /**
451   * Record of an edit log that has been located and had its filename parsed.
452   */
453  @InterfaceAudience.Private
454  public static class EditLogFile {
455    private File file;
456    private final long firstTxId;
457    private long lastTxId;
458
459    private boolean hasCorruptHeader = false;
460    private final boolean isInProgress;
461
462    final static Comparator<EditLogFile> COMPARE_BY_START_TXID 
463      = new Comparator<EditLogFile>() {
464      @Override
465      public int compare(EditLogFile a, EditLogFile b) {
466        return ComparisonChain.start()
467        .compare(a.getFirstTxId(), b.getFirstTxId())
468        .compare(a.getLastTxId(), b.getLastTxId())
469        .result();
470      }
471    };
472
473    EditLogFile(File file,
474        long firstTxId, long lastTxId) {
475      this(file, firstTxId, lastTxId, false);
476      assert (lastTxId != HdfsConstants.INVALID_TXID)
477        && (lastTxId >= firstTxId);
478    }
479    
480    EditLogFile(File file, long firstTxId, 
481                long lastTxId, boolean isInProgress) { 
482      assert (lastTxId == HdfsConstants.INVALID_TXID && isInProgress)
483        || (lastTxId != HdfsConstants.INVALID_TXID && lastTxId >= firstTxId);
484      assert (firstTxId > 0) || (firstTxId == HdfsConstants.INVALID_TXID);
485      assert file != null;
486      
487      Preconditions.checkArgument(!isInProgress ||
488          lastTxId == HdfsConstants.INVALID_TXID);
489      
490      this.firstTxId = firstTxId;
491      this.lastTxId = lastTxId;
492      this.file = file;
493      this.isInProgress = isInProgress;
494    }
495    
496    public long getFirstTxId() {
497      return firstTxId;
498    }
499    
500    public long getLastTxId() {
501      return lastTxId;
502    }
503    
504    boolean containsTxId(long txId) {
505      return firstTxId <= txId && txId <= lastTxId;
506    }
507
508    /** 
509     * Find out where the edit log ends.
510     * This will update the lastTxId of the EditLogFile or
511     * mark it as corrupt if it is.
512     */
513    public void validateLog() throws IOException {
514      EditLogValidation val = EditLogFileInputStream.validateEditLog(file);
515      this.lastTxId = val.getEndTxId();
516      this.hasCorruptHeader = val.hasCorruptHeader();
517    }
518
519    public void scanLog() throws IOException {
520      EditLogValidation val = EditLogFileInputStream.scanEditLog(file);
521      this.lastTxId = val.getEndTxId();
522      this.hasCorruptHeader = val.hasCorruptHeader();
523    }
524
525    public boolean isInProgress() {
526      return isInProgress;
527    }
528
529    public File getFile() {
530      return file;
531    }
532    
533    boolean hasCorruptHeader() {
534      return hasCorruptHeader;
535    }
536
537    void moveAsideCorruptFile() throws IOException {
538      assert hasCorruptHeader;
539      renameSelf(".corrupt");
540    }
541
542    void moveAsideTrashFile(long markerTxid) throws IOException {
543      assert this.getFirstTxId() >= markerTxid;
544      renameSelf(".trash");
545    }
546
547    public void moveAsideEmptyFile() throws IOException {
548      assert lastTxId == HdfsConstants.INVALID_TXID;
549      renameSelf(".empty");
550    }
551      
552    private void renameSelf(String newSuffix) throws IOException {
553      File src = file;
554      File dst = new File(src.getParent(), src.getName() + newSuffix);
555      // renameTo fails on Windows if the destination file already exists.
556      try {
557        if (dst.exists()) {
558          if (!dst.delete()) {
559            throw new IOException("Couldn't delete " + dst);
560          }
561        }
562        NativeIO.renameTo(src, dst);
563      } catch (IOException e) {
564        throw new IOException(
565            "Couldn't rename log " + src + " to " + dst, e);
566      }
567      file = dst;
568    }
569
570    @Override
571    public String toString() {
572      return String.format("EditLogFile(file=%s,first=%019d,last=%019d,"
573                           +"inProgress=%b,hasCorruptHeader=%b)",
574                           file.toString(), firstTxId, lastTxId,
575                           isInProgress(), hasCorruptHeader);
576    }
577  }
