065  public static INodeFile valueOf(INode inode, String path, boolean acceptNull)
066      throws FileNotFoundException {
067    if (inode == null) {
068      if (acceptNull) {
069        return null;
070      } else {
071        throw new FileNotFoundException("File does not exist: " + path);

075      throw new FileNotFoundException("Path is not a file: " + path);
076    }
077    return inode.asFile();
078  }

080  /** 
081   * Bit format:
082   * [4-bit storagePolicyID][12-bit replication][48-bit preferredBlockSize]
083   */
084  static enum HeaderFormat {
085    PREFERRED_BLOCK_SIZE(null, 48, 1),
086    REPLICATION(PREFERRED_BLOCK_SIZE.BITS, 12, 1),
087    STORAGE_POLICY_ID(REPLICATION.BITS, BlockStoragePolicySuite.ID_BIT_LENGTH,
088        0);
089
090    private final LongBitFormat BITS;
091
092    private HeaderFormat(LongBitFormat previous, int length, long min) {
093      BITS = new LongBitFormat(name(), previous, length, min);
094    }
095
096    static short getReplication(long header) {
097      return (short)REPLICATION.BITS.retrieve(header);
098    }
099
100    static long getPreferredBlockSize(long header) {
101      return PREFERRED_BLOCK_SIZE.BITS.retrieve(header);
102    }
103
104    static byte getStoragePolicyID(long header) {
105      return (byte)STORAGE_POLICY_ID.BITS.retrieve(header);
106    }
107
108    static long toLong(long preferredBlockSize, short replication,
109        byte storagePolicyID) {
110      long h = 0;
111      if (preferredBlockSize == 0) {
112        preferredBlockSize = PREFERRED_BLOCK_SIZE.BITS.getMin();
113      }
114      h = PREFERRED_BLOCK_SIZE.BITS.combine(preferredBlockSize, h);
115      h = REPLICATION.BITS.combine(replication, h);
116      h = STORAGE_POLICY_ID.BITS.combine(storagePolicyID, h);
117      return h;
118    }
119  }

205    Preconditions.checkState(isUnderConstruction(),
206        "file is no longer under construction");
207    FileUnderConstructionFeature uc = getFileUnderConstructionFeature();
208    if (uc != null) {

210      removeFeature(uc);

216  /** Assert all blocks are complete. */
217  private void assertAllBlocksComplete() {
218    if (blocks == null) {
219      return;
220    }
221    for (int i = 0; i < blocks.length; i++) {
222      Preconditions.checkState(blocks[i].isComplete(), "Failed to finalize"
223          + " %s %s since blocks[%s] is non-complete, where blocks=%s.",
224          getClass().getSimpleName(), this, i, Arrays.asList(blocks));
225    }
226  }

233  @Override // BlockCollection, the file should be under construction
234  public BlockInfoContiguousUnderConstruction setLastBlock(
235      BlockInfoContiguous lastBlock, DatanodeStorageInfo[] locations)
236      throws IOException {
237    Preconditions.checkState(isUnderConstruction(),
238        "file is no longer under construction");
239
240    if (numBlocks() == 0) {
241      throw new IOException("Failed to set last block: File is empty.");
242    }
243    BlockInfoContiguousUnderConstruction ucBlock =
244      lastBlock.convertToBlockUnderConstruction(
245          BlockUCState.UNDER_CONSTRUCTION, locations);
246    setBlock(numBlocks() - 1, ucBlock);
247    return ucBlock;
248  }

250  /**
251   * Remove a block from the block list. This block should be
252   * the last one on the list.
253   */
254  boolean removeLastBlock(Block oldblock) {
255    Preconditions.checkState(isUnderConstruction(),
256        "file is no longer under construction");
257    if (blocks == null || blocks.length == 0) {
258      return false;
259    }
260    int size_1 = blocks.length - 1;
261    if (!blocks[size_1].equals(oldblock)) {
262      return false;
263    }
264
265    //copy to a new list
266    BlockInfoContiguous[] newlist = new BlockInfoContiguous[size_1];
267    System.arraycopy(blocks, 0, newlist, 0, size_1);
268    setBlocks(newlist);
269    return true;
270  }

318  public void recordModification(final int latestSnapshotId, boolean withBlocks) {
319    if (isInLatestSnapshot(latestSnapshotId)
320        && !shouldRecordInSrcSnapshot(latestSnapshotId)) {
321      // the file is in snapshot, create a snapshot feature if it does not have
322      FileWithSnapshotFeature sf = this.getFileWithSnapshotFeature();
323      if (sf == null) {
324        sf = addSnapshotFeature(null);
325      }
326      // record self in the diff list if necessary
327      sf.getDiffs().saveSelf2Snapshot(latestSnapshotId, this, null, withBlocks);
328    }
329  }

355  @Override // BlockCollection
356  public short getBlockReplication() {
357    short max = getFileReplication(CURRENT_STATE_ID);
358    FileWithSnapshotFeature sf = this.getFileWithSnapshotFeature();
359    if (sf != null) {
360      short maxInSnapshot = sf.getMaxBlockRepInDiffs();
361      if (sf.isCurrentFileDeleted()) {
362        return maxInSnapshot;
363      }
364      max = maxInSnapshot > max ? maxInSnapshot : max;
365    }
366    return max;
367  }

430  /** @return blocks of the file corresponding to the snapshot. */
431  public BlockInfoContiguous[] getBlocks(int snapshot) {
432    if(snapshot == CURRENT_STATE_ID || getDiffs() == null)
433      return getBlocks();
434    FileDiff diff = getDiffs().getDiffById(snapshot);
435    BlockInfoContiguous[] snapshotBlocks =
436        diff == null ? getBlocks() : diff.getBlocks();
437    if(snapshotBlocks != null)
438      return snapshotBlocks;
439    // Blocks are not in the current snapshot
440    // Find next snapshot with blocks present or return current file blocks
441    snapshotBlocks = getDiffs().findLaterSnapshotBlocks(snapshot);
442    return (snapshotBlocks == null) ? getBlocks() : snapshotBlocks;
443  }

453  /**
454   * append array of blocks to this.blocks
455   */
456  void concatBlocks(INodeFile[] inodes) {
457    int size = this.blocks.length;
458    int totalAddedBlocks = 0;
459    for(INodeFile f : inodes) {
460      totalAddedBlocks += f.blocks.length;
461    }
462    
463    BlockInfoContiguous[] newlist =
464        new BlockInfoContiguous[size + totalAddedBlocks];
465    System.arraycopy(this.blocks, 0, newlist, 0, size);
466    
467    for(INodeFile in: inodes) {
468      System.arraycopy(in.blocks, 0, newlist, size, in.blocks.length);
469      size += in.blocks.length;
470    }
471
472    setBlocks(newlist);
473    updateBlockCollection();
474  }

484      BlockInfoContiguous[] newlist = new BlockInfoContiguous[size + 1];

486      newlist[size] = newblock;
487      this.setBlocks(newlist);
488    }
489  }

496  @Override
497  public QuotaCounts cleanSubtree(BlockStoragePolicySuite bsps, final int snapshot,
498                                  int priorSnapshotId,
499      final BlocksMapUpdateInfo collectedBlocks,
500      final List<INode> removedINodes) {
501    FileWithSnapshotFeature sf = getFileWithSnapshotFeature();
502    if (sf != null) {
503      return sf.cleanFile(bsps, this, snapshot, priorSnapshotId, collectedBlocks,
504          removedINodes);
505    }
506    QuotaCounts counts = new QuotaCounts.Builder().build();
507    if (snapshot == CURRENT_STATE_ID) {
508      if (priorSnapshotId == NO_SNAPSHOT_ID) {
509        // this only happens when deleting the current file and the file is not
510        // in any snapshot
511        computeQuotaUsage(bsps, counts, false);
512        destroyAndCollectBlocks(bsps, collectedBlocks, removedINodes);
513      } else {
514        // when deleting the current file and the file is in snapshot, we should
515        // clean the 0-sized block if the file is UC
516        FileUnderConstructionFeature uc = getFileUnderConstructionFeature();
517        if (uc != null) {
518          uc.cleanZeroSizeBlock(this, collectedBlocks);
519        }
520      }
521    }
522    return counts;
523  }

525  @Override
526  public void destroyAndCollectBlocks(BlockStoragePolicySuite bsps,
527      BlocksMapUpdateInfo collectedBlocks, final List<INode> removedINodes) {
528    if (blocks != null && collectedBlocks != null) {
529      for (BlockInfoContiguous blk : blocks) {
530        collectedBlocks.addDeleteBlock(blk);
531        blk.setBlockCollection(null);
532      }
533    }
534    setBlocks(null);
535    if (getAclFeature() != null) {
536      AclStorage.removeAclFeature(getAclFeature());
537    }
538    clear();
539    removedINodes.add(this);
540    FileWithSnapshotFeature sf = getFileWithSnapshotFeature();
541    if (sf != null) {
542      sf.getDiffs().destroyAndCollectSnapshotBlocks(collectedBlocks);
543      sf.clearDiffs();
544    }
545  }

553  // This is the only place that needs to use the BlockStoragePolicySuite to
554  // derive the intended storage type usage for quota by storage type
555  @Override
556  public final QuotaCounts computeQuotaUsage(
557      BlockStoragePolicySuite bsps, byte blockStoragePolicyId,
558      QuotaCounts counts, boolean useCache,
559      int lastSnapshotId) {
560    long nsDelta = 1;
561    final long ssDeltaNoReplication;
562    short replication;
563    FileWithSnapshotFeature sf = getFileWithSnapshotFeature();
564    if (sf != null) {
565      FileDiffList fileDiffList = sf.getDiffs();
566      int last = fileDiffList.getLastSnapshotId();
567
568      if (lastSnapshotId == Snapshot.CURRENT_STATE_ID
569          || last == Snapshot.CURRENT_STATE_ID) {
570        ssDeltaNoReplication = storagespaceConsumedNoReplication();
571        replication = getBlockReplication();
572      } else if (last < lastSnapshotId) {
573        ssDeltaNoReplication = computeFileSize(true, false);
574        replication = getFileReplication();
575      } else {
576        int sid = fileDiffList.getSnapshotById(lastSnapshotId);
577        ssDeltaNoReplication = storagespaceConsumedNoReplication(sid);
578        replication = getReplication(sid);
579      }
580    } else {
581      ssDeltaNoReplication = storagespaceConsumedNoReplication();
582      replication = getBlockReplication();
583    }
584    counts.addNameSpace(nsDelta);
585    counts.addStorageSpace(ssDeltaNoReplication * replication);
586
587    if (blockStoragePolicyId != ID_UNSPECIFIED){
588      BlockStoragePolicy bsp = bsps.getPolicy(blockStoragePolicyId);
589      List<StorageType> storageTypes = bsp.chooseStorageTypes(replication);
590      for (StorageType t : storageTypes) {
591        if (!t.supportTypeQuota()) {
592          continue;
593        }
594        counts.addTypeSpace(t, ssDeltaNoReplication);
595      }
596    }
597    return counts;
598  }

600  @Override
601  public final ContentSummaryComputationContext computeContentSummary(
602      final ContentSummaryComputationContext summary) {
603    final ContentCounts counts = summary.getCounts();
604    FileWithSnapshotFeature sf = getFileWithSnapshotFeature();
605    long fileLen = 0;
606    if (sf == null) {
607      fileLen = computeFileSize();
608      counts.addContent(Content.FILE, 1);
609    } else {
610      final FileDiffList diffs = sf.getDiffs();
611      final int n = diffs.asList().size();
612      counts.addContent(Content.FILE, n);
613      if (n > 0 && sf.isCurrentFileDeleted()) {
614        fileLen =  diffs.getLast().getFileSize();
615      } else {
616        fileLen = computeFileSize();
617      }
618    }
619    counts.addContent(Content.LENGTH, fileLen);
620    counts.addContent(Content.DISKSPACE, storagespaceConsumed());
621
622    if (getStoragePolicyID() != ID_UNSPECIFIED){
623      BlockStoragePolicy bsp = summary.getBlockStoragePolicySuite().
624          getPolicy(getStoragePolicyID());
625      List<StorageType> storageTypes = bsp.chooseStorageTypes(getFileReplication());
626      for (StorageType t : storageTypes) {
627        if (!t.supportTypeQuota()) {
628          continue;
629        }
630        counts.addTypeSpace(t, fileLen);
631      }
632    }
633    return summary;
634  }

664  /**
665   * Compute file size of the current file.
666   * 
667   * @param includesLastUcBlock
668   *          If the last block is under construction, should it be included?
669   * @param usePreferredBlockSize4LastUcBlock
670   *          If the last block is under construction, should we use actual
671   *          block size or preferred block size?
672   *          Note that usePreferredBlockSize4LastUcBlock is ignored
673   *          if includesLastUcBlock == false.
674   * @return file size
675   */
676  public final long computeFileSize(boolean includesLastUcBlock,
677      boolean usePreferredBlockSize4LastUcBlock) {
678    if (blocks == null || blocks.length == 0) {
679      return 0;
680    }
681    final int last = blocks.length - 1;
682    //check if the last block is BlockInfoUnderConstruction
683    long size = blocks[last].getNumBytes();
684    if (blocks[last] instanceof BlockInfoContiguousUnderConstruction) {
685       if (!includesLastUcBlock) {
686         size = 0;
687       } else if (usePreferredBlockSize4LastUcBlock) {
688         size = getPreferredBlockSize();
689       }
690    }
691    //sum other blocks
692    for(int i = 0; i < last; i++) {
693      size += blocks[i].getNumBytes();
694    }
695    return size;
696  }

698  /**

700   * including blocks in its snapshots.
701   * Use preferred block size for the last block if it is under construction.
702   */

704    return storagespaceConsumedNoReplication() * getBlockReplication();

707  public final long storagespaceConsumedNoReplication() {
708    FileWithSnapshotFeature sf = getFileWithSnapshotFeature();
709    if(sf == null) {
710      return computeFileSize(true, true);
711    }
712
713    // Collect all distinct blocks
714    long size = 0;
715    Set<Block> allBlocks = new HashSet<Block>(Arrays.asList(getBlocks()));
716    List<FileDiff> diffs = sf.getDiffs().asList();
717    for(FileDiff diff : diffs) {
718      BlockInfoContiguous[] diffBlocks = diff.getBlocks();
719      if (diffBlocks != null) {
720        allBlocks.addAll(Arrays.asList(diffBlocks));
721      }
722    }
723    for(Block block : allBlocks) {
724      size += block.getNumBytes();
725    }
726    // check if the last block is under construction
727    BlockInfoContiguous lastBlock = getLastBlock();
728    if(lastBlock != null &&
729        lastBlock instanceof BlockInfoContiguousUnderConstruction) {
730      size += getPreferredBlockSize() - lastBlock.getNumBytes();
731    }
732    return size;
733  }

736    if (lastSnapshotId != CURRENT_STATE_ID) {
737      return computeFileSize(lastSnapshotId)
738        * getFileReplication(lastSnapshotId);
739    } else {
740      return storagespaceConsumed();

792  /**
793   * Remove full blocks at the end file up to newLength
794   * @return sum of sizes of the remained blocks
795   */
796  public long collectBlocksBeyondMax(final long max,
797      final BlocksMapUpdateInfo collectedBlocks) {
798    final BlockInfoContiguous[] oldBlocks = getBlocks();
799    if (oldBlocks == null)
800      return 0;
801    // find the minimum n such that the size of the first n blocks > max
802    int n = 0;
803    long size = 0;
804    for(; n < oldBlocks.length && max > size; n++) {
805      size += oldBlocks[n].getNumBytes();
806    }
807    if (n >= oldBlocks.length)
808      return size;
809
810    // starting from block n, the data is beyond max.
811    // resize the array.
812    truncateBlocksTo(n);
813
814    // collect the blocks beyond max
815    if (collectedBlocks != null) {
816      for(; n < oldBlocks.length; n++) {
817        collectedBlocks.addDeleteBlock(oldBlocks[n]);
818      }
819    }
820    return size;
821  }

823  /**
824   * compute the quota usage change for a truncate op
825   * @param newLength the length for truncation
826   * @return the quota usage delta (not considering replication factor)
827   */
828  long computeQuotaDeltaForTruncate(final long newLength) {
829    final BlockInfoContiguous[] blocks = getBlocks();
830    if (blocks == null || blocks.length == 0) {
831      return 0;
832    }
833
834    int n = 0;
835    long size = 0;
836    for (; n < blocks.length && newLength > size; n++) {
837      size += blocks[n].getNumBytes();
838    }
839    final boolean onBoundary = size == newLength;
840
841    long truncateSize = 0;
842    for (int i = (onBoundary ? n : n - 1); i < blocks.length; i++) {
843      truncateSize += blocks[i].getNumBytes();
844    }
845
846    FileWithSnapshotFeature sf = getFileWithSnapshotFeature();
847    if (sf != null) {
848      FileDiff diff = sf.getDiffs().getLast();
849      BlockInfoContiguous[] sblocks = diff != null ? diff.getBlocks() : null;
850      if (sblocks != null) {
851        for (int i = (onBoundary ? n : n-1); i < blocks.length
852            && i < sblocks.length && blocks[i].equals(sblocks[i]); i++) {
853          truncateSize -= blocks[i].getNumBytes();
854        }
855      }
856    }
857    return onBoundary ? -truncateSize : (getPreferredBlockSize() - truncateSize);
858  }

860  void truncateBlocksTo(int n) {
861    final BlockInfoContiguous[] newBlocks;
862    if (n == 0) {
863      newBlocks = BlockInfoContiguous.EMPTY_ARRAY;
864    } else {
865      newBlocks = new BlockInfoContiguous[n];
866      System.arraycopy(getBlocks(), 0, newBlocks, 0, n);
867    }
868    // set new blocks
869    setBlocks(newBlocks);
870  }

872  public void collectBlocksBeyondSnapshot(BlockInfoContiguous[] snapshotBlocks,
873                                          BlocksMapUpdateInfo collectedBlocks) {
874    BlockInfoContiguous[] oldBlocks = getBlocks();
875    if(snapshotBlocks == null || oldBlocks == null)
876      return;
877    // Skip blocks in common between the file and the snapshot
878    int n = 0;
879    while(n < oldBlocks.length && n < snapshotBlocks.length &&
880          oldBlocks[n] == snapshotBlocks[n]) {
881      n++;
882    }
883    truncateBlocksTo(n);
884    // Collect the remaining blocks of the file
885    while(n < oldBlocks.length) {
886      collectedBlocks.addDeleteBlock(oldBlocks[n++]);
887    }
888  }

890  /** Exclude blocks collected for deletion that belong to a snapshot. */
891  void excludeSnapshotBlocks(int snapshotId,
892                             BlocksMapUpdateInfo collectedBlocks) {
893    if(collectedBlocks == null || collectedBlocks.getToDeleteList().isEmpty())
894      return;
895    FileWithSnapshotFeature sf = getFileWithSnapshotFeature();
896    if(sf == null)
897      return;
898    BlockInfoContiguous[] snapshotBlocks =
899        getDiffs().findEarlierSnapshotBlocks(snapshotId);
900    if(snapshotBlocks == null)
901      return;
902    List<Block> toDelete = collectedBlocks.getToDeleteList();
903    for(Block blk : snapshotBlocks) {
904      if(toDelete.contains(blk))
905        collectedBlocks.removeDeleteBlock(blk);
906    }
907  }

909  /**
910   * @return true if the block is contained in a snapshot or false otherwise.
911   */
912  boolean isBlockInLatestSnapshot(BlockInfoContiguous block) {
913    FileWithSnapshotFeature sf = this.getFileWithSnapshotFeature();
914    if (sf == null || sf.getDiffs() == null) {
915      return false;
916    }
917    BlockInfoContiguous[] snapshotBlocks = getDiffs()
918        .findEarlierSnapshotBlocks(getDiffs().getLastSnapshotId());
919    return snapshotBlocks != null &&
920        Arrays.asList(snapshotBlocks).contains(block);
921  }
