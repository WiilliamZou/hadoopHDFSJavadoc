049  /**
050   * Set xattr for a file or directory.
051   *
052   * @param src
053   *          - path on which it sets the xattr
054   * @param xAttr
055   *          - xAttr details to set
056   * @param flag
057   *          - xAttrs flags
058   * @throws IOException
059   */
060  static HdfsFileStatus setXAttr(
061      FSDirectory fsd, String src, XAttr xAttr, EnumSet<XAttrSetFlag> flag,
062      boolean logRetryCache)
063      throws IOException {
064    checkXAttrsConfigFlag(fsd);
065    checkXAttrSize(fsd, xAttr);
066    FSPermissionChecker pc = fsd.getPermissionChecker();
067    XAttrPermissionFilter.checkPermissionForApi(
068        pc, xAttr, FSDirectory.isReservedRawName(src));
069    byte[][] pathComponents = FSDirectory.getPathComponentsForReservedPath(src);
070    src = fsd.resolvePath(pc, src, pathComponents);
071    List<XAttr> xAttrs = Lists.newArrayListWithCapacity(1);
072    xAttrs.add(xAttr);
073    INodesInPath iip;
074    fsd.writeLock();
075    try {
076      iip = fsd.getINodesInPath4Write(src);
077      checkXAttrChangeAccess(fsd, iip, xAttr, pc);
078      unprotectedSetXAttrs(fsd, src, xAttrs, flag);
079    } finally {
080      fsd.writeUnlock();
081    }
082    fsd.getEditLog().logSetXAttrs(src, xAttrs, logRetryCache);
083    return fsd.getAuditFileInfo(iip);
084  }

086  static List<XAttr> getXAttrs(FSDirectory fsd, final String srcArg,
087                               List<XAttr> xAttrs)
088      throws IOException {
089    String src = srcArg;
090    checkXAttrsConfigFlag(fsd);
091    FSPermissionChecker pc = fsd.getPermissionChecker();
092    final boolean isRawPath = FSDirectory.isReservedRawName(src);
093    boolean getAll = xAttrs == null || xAttrs.isEmpty();
094    if (!getAll) {
095      XAttrPermissionFilter.checkPermissionForApi(pc, xAttrs, isRawPath);
096    }
097    byte[][] pathComponents = FSDirectory.getPathComponentsForReservedPath(src);
098    src = fsd.resolvePath(pc, src, pathComponents);
099    final INodesInPath iip = fsd.getINodesInPath(src, true);
100    if (fsd.isPermissionEnabled()) {
101      fsd.checkPathAccess(pc, iip, FsAction.READ);
102    }
103    List<XAttr> all = FSDirXAttrOp.getXAttrs(fsd, src);
104    List<XAttr> filteredAll = XAttrPermissionFilter.
105        filterXAttrsForApi(pc, all, isRawPath);
106
107    if (getAll) {
108      return filteredAll;
109    }
110    if (filteredAll == null || filteredAll.isEmpty()) {
111      return null;
112    }
113    List<XAttr> toGet = Lists.newArrayListWithCapacity(xAttrs.size());
114    for (XAttr xAttr : xAttrs) {
115      boolean foundIt = false;
116      for (XAttr a : filteredAll) {
117        if (xAttr.getNameSpace() == a.getNameSpace() && xAttr.getName().equals(
118            a.getName())) {
119          toGet.add(a);
120          foundIt = true;
121          break;
122        }
123      }
124      if (!foundIt) {
125        throw new IOException(
126            "At least one of the attributes provided was not found.");
127      }
128    }
129    return toGet;
130  }

132  static List<XAttr> listXAttrs(
133      FSDirectory fsd, String src) throws IOException {
134    FSDirXAttrOp.checkXAttrsConfigFlag(fsd);
135    final FSPermissionChecker pc = fsd.getPermissionChecker();
136    final boolean isRawPath = FSDirectory.isReservedRawName(src);
137    byte[][] pathComponents = FSDirectory.getPathComponentsForReservedPath(src);
138    src = fsd.resolvePath(pc, src, pathComponents);
139    final INodesInPath iip = fsd.getINodesInPath(src, true);
140    if (fsd.isPermissionEnabled()) {
141      /* To access xattr names, you need EXECUTE in the owning directory. */
142      fsd.checkParentAccess(pc, iip, FsAction.EXECUTE);
143    }
144    final List<XAttr> all = FSDirXAttrOp.getXAttrs(fsd, src);
145    return XAttrPermissionFilter.
146        filterXAttrsForApi(pc, all, isRawPath);
147  }

149  /**
150   * Remove an xattr for a file or directory.
151   *
152   * @param src
153   *          - path to remove the xattr from
154   * @param xAttr
155   *          - xAttr to remove
156   * @throws IOException
157   */
158  static HdfsFileStatus removeXAttr(
159      FSDirectory fsd, String src, XAttr xAttr, boolean logRetryCache)
160      throws IOException {
161    FSDirXAttrOp.checkXAttrsConfigFlag(fsd);
162    FSPermissionChecker pc = fsd.getPermissionChecker();
163    XAttrPermissionFilter.checkPermissionForApi(
164        pc, xAttr, FSDirectory.isReservedRawName(src));
165    byte[][] pathComponents = FSDirectory.getPathComponentsForReservedPath(
166        src);
167
168    List<XAttr> xAttrs = Lists.newArrayListWithCapacity(1);
169    xAttrs.add(xAttr);
170    INodesInPath iip;
171    fsd.writeLock();
172    try {
173      src = fsd.resolvePath(pc, src, pathComponents);
174      iip = fsd.getINodesInPath4Write(src);
175      checkXAttrChangeAccess(fsd, iip, xAttr, pc);
176
177      List<XAttr> removedXAttrs = unprotectedRemoveXAttrs(fsd, src, xAttrs);
178      if (removedXAttrs != null && !removedXAttrs.isEmpty()) {
179        fsd.getEditLog().logRemoveXAttrs(src, removedXAttrs, logRetryCache);
180      } else {
181        throw new IOException(
182            "No matching attributes found for remove operation");
183      }
184    } finally {
185      fsd.writeUnlock();
186    }
187    return fsd.getAuditFileInfo(iip);
188  }

190  static List<XAttr> unprotectedRemoveXAttrs(
191      FSDirectory fsd, final String src, final List<XAttr> toRemove)
192      throws IOException {
193    assert fsd.hasWriteLock();
194    INodesInPath iip = fsd.getINodesInPath4Write(
195        FSDirectory.normalizePath(src), true);
196    INode inode = FSDirectory.resolveLastINode(iip);
197    int snapshotId = iip.getLatestSnapshotId();
198    List<XAttr> existingXAttrs = XAttrStorage.readINodeXAttrs(inode);
199    List<XAttr> removedXAttrs = Lists.newArrayListWithCapacity(toRemove.size());
200    List<XAttr> newXAttrs = filterINodeXAttrs(existingXAttrs, toRemove,
201                                              removedXAttrs);
202    if (existingXAttrs.size() != newXAttrs.size()) {
203      XAttrStorage.updateINodeXAttrs(inode, newXAttrs, snapshotId);
204      return removedXAttrs;
205    }
206    return null;
207  }

209  /**
210   * Filter XAttrs from a list of existing XAttrs. Removes matched XAttrs from
211   * toFilter and puts them into filtered. Upon completion,
212   * toFilter contains the filter XAttrs that were not found, while
213   * fitleredXAttrs contains the XAttrs that were found.
214   *
215   * @param existingXAttrs Existing XAttrs to be filtered
216   * @param toFilter XAttrs to filter from the existing XAttrs
217   * @param filtered Return parameter, XAttrs that were filtered
218   * @return List of XAttrs that does not contain filtered XAttrs
219   */
220  @VisibleForTesting
221  static List<XAttr> filterINodeXAttrs(
222      final List<XAttr> existingXAttrs, final List<XAttr> toFilter,
223      final List<XAttr> filtered)
224    throws AccessControlException {
225    if (existingXAttrs == null || existingXAttrs.isEmpty() ||
226        toFilter == null || toFilter.isEmpty()) {
227      return existingXAttrs;
228    }
229
230    // Populate a new list with XAttrs that pass the filter
231    List<XAttr> newXAttrs =
232        Lists.newArrayListWithCapacity(existingXAttrs.size());
233    for (XAttr a : existingXAttrs) {
234      boolean add = true;
235      for (ListIterator<XAttr> it = toFilter.listIterator(); it.hasNext()
236          ;) {
237        XAttr filter = it.next();
238        Preconditions.checkArgument(
239            !KEYID_XATTR.equalsIgnoreValue(filter),
240            "The encryption zone xattr should never be deleted.");
241        if (UNREADABLE_BY_SUPERUSER_XATTR.equalsIgnoreValue(filter)) {
242          throw new AccessControlException("The xattr '" +
243              SECURITY_XATTR_UNREADABLE_BY_SUPERUSER + "' can not be deleted.");
244        }
245        if (a.equalsIgnoreValue(filter)) {
246          add = false;
247          it.remove();
248          filtered.add(filter);
249          break;
250        }
251      }
252      if (add) {
253        newXAttrs.add(a);
254      }
255    }
256
257    return newXAttrs;
258  }

260  static INode unprotectedSetXAttrs(
261      FSDirectory fsd, final String src, final List<XAttr> xAttrs,
262      final EnumSet<XAttrSetFlag> flag)
263      throws IOException {
264    assert fsd.hasWriteLock();
265    INodesInPath iip = fsd.getINodesInPath4Write(FSDirectory.normalizePath(src),
266        true);
267    INode inode = FSDirectory.resolveLastINode(iip);
268    int snapshotId = iip.getLatestSnapshotId();
269    List<XAttr> existingXAttrs = XAttrStorage.readINodeXAttrs(inode);
270    List<XAttr> newXAttrs = setINodeXAttrs(fsd, existingXAttrs, xAttrs, flag);
271    final boolean isFile = inode.isFile();
272
273    for (XAttr xattr : newXAttrs) {
274      final String xaName = XAttrHelper.getPrefixName(xattr);
275
276      /*
277       * If we're adding the encryption zone xattr, then add src to the list
278       * of encryption zones.
279       */
280      if (CRYPTO_XATTR_ENCRYPTION_ZONE.equals(xaName)) {
281        final HdfsProtos.ZoneEncryptionInfoProto ezProto =
282            HdfsProtos.ZoneEncryptionInfoProto.parseFrom(xattr.getValue());
283        fsd.ezManager.addEncryptionZone(inode.getId(),
284                                        PBHelper.convert(ezProto.getSuite()),
285                                        PBHelper.convert(
286                                            ezProto.getCryptoProtocolVersion()),
287                                        ezProto.getKeyName());
288      }
289
290      if (!isFile && SECURITY_XATTR_UNREADABLE_BY_SUPERUSER.equals(xaName)) {
291        throw new IOException("Can only set '" +
292            SECURITY_XATTR_UNREADABLE_BY_SUPERUSER + "' on a file.");
293      }
294    }
295
296    XAttrStorage.updateINodeXAttrs(inode, newXAttrs, snapshotId);
297    return inode;
298  }

300  static List<XAttr> setINodeXAttrs(
301      FSDirectory fsd, final List<XAttr> existingXAttrs,
302      final List<XAttr> toSet, final EnumSet<XAttrSetFlag> flag)
303      throws IOException {
304    // Check for duplicate XAttrs in toSet
305    // We need to use a custom comparator, so using a HashSet is not suitable
306    for (int i = 0; i < toSet.size(); i++) {
307      for (int j = i + 1; j < toSet.size(); j++) {
308        if (toSet.get(i).equalsIgnoreValue(toSet.get(j))) {
309          throw new IOException("Cannot specify the same XAttr to be set " +
310              "more than once");
311        }
312      }
313    }
314
315    // Count the current number of user-visible XAttrs for limit checking
316    int userVisibleXAttrsNum = 0; // Number of user visible xAttrs
317
318    // The XAttr list is copied to an exactly-sized array when it's stored,
319    // so there's no need to size it precisely here.
320    int newSize = (existingXAttrs != null) ? existingXAttrs.size() : 0;
321    newSize += toSet.size();
322    List<XAttr> xAttrs = Lists.newArrayListWithCapacity(newSize);
323
324    // Check if the XAttr already exists to validate with the provided flag
325    for (XAttr xAttr: toSet) {
326      boolean exist = false;
327      if (existingXAttrs != null) {
328        for (XAttr a : existingXAttrs) {
329          if (a.equalsIgnoreValue(xAttr)) {
330            exist = true;
331            break;
332          }
333        }
334      }
335      XAttrSetFlag.validate(xAttr.getName(), exist, flag);
336      // add the new XAttr since it passed validation
337      xAttrs.add(xAttr);
338      if (isUserVisible(xAttr)) {
339        userVisibleXAttrsNum++;
340      }
341    }
342
343    // Add the existing xattrs back in, if they weren't already set
344    if (existingXAttrs != null) {
345      for (XAttr existing : existingXAttrs) {
346        boolean alreadySet = false;
347        for (XAttr set : toSet) {
348          if (set.equalsIgnoreValue(existing)) {
349            alreadySet = true;
350            break;
351          }
352        }
353        if (!alreadySet) {
354          xAttrs.add(existing);
355          if (isUserVisible(existing)) {
356            userVisibleXAttrsNum++;
357          }
358        }
359      }
360    }
361
362    if (userVisibleXAttrsNum > fsd.getInodeXAttrsLimit()) {
363      throw new IOException("Cannot add additional XAttr to inode, "
364          + "would exceed limit of " + fsd.getInodeXAttrsLimit());
365    }
366
367    return xAttrs;
368  }

370  static List<XAttr> getXAttrs(FSDirectory fsd, INode inode, int snapshotId)
371      throws IOException {
372    fsd.readLock();
373    try {
374      return XAttrStorage.readINodeXAttrs(inode, snapshotId);
375    } finally {
376      fsd.readUnlock();
377    }
378  }

380  static XAttr unprotectedGetXAttrByName(
381      INode inode, int snapshotId, String xAttrName)
382      throws IOException {
383    List<XAttr> xAttrs = XAttrStorage.readINodeXAttrs(inode, snapshotId);
384    if (xAttrs == null) {
385      return null;
386    }
387    for (XAttr x : xAttrs) {
388      if (XAttrHelper.getPrefixName(x)
389          .equals(xAttrName)) {
390        return x;
391      }
392    }
393    return null;
394  }

396  private static void checkXAttrChangeAccess(
397      FSDirectory fsd, INodesInPath iip, XAttr xAttr,
398      FSPermissionChecker pc)
399      throws AccessControlException {
400    if (fsd.isPermissionEnabled() && xAttr.getNameSpace() == XAttr.NameSpace
401        .USER) {
402      final INode inode = iip.getLastINode();
403      if (inode != null &&
404          inode.isDirectory() &&
405          inode.getFsPermission().getStickyBit()) {
406        if (!pc.isSuperUser()) {
407          fsd.checkOwner(pc, iip);
408        }
409      } else {
410        fsd.checkPathAccess(pc, iip, FsAction.WRITE);
411      }
412    }
413  }

415  /**
416   * Verifies that the combined size of the name and value of an xattr is within
417   * the configured limit. Setting a limit of zero disables this check.
418   */
419  private static void checkXAttrSize(FSDirectory fsd, XAttr xAttr) {
420    if (fsd.getXattrMaxSize() == 0) {
421      return;
422    }
423    int size = xAttr.getName().getBytes(Charsets.UTF_8).length;
424    if (xAttr.getValue() != null) {
425      size += xAttr.getValue().length;
426    }
427    if (size > fsd.getXattrMaxSize()) {
428      throw new HadoopIllegalArgumentException(
429          "The XAttr is too big. The maximum combined size of the"
430          + " name and value is " + fsd.getXattrMaxSize()
431          + ", but the total size is " + size);
432    }
433  }

435  private static void checkXAttrsConfigFlag(FSDirectory fsd) throws
436                                                             IOException {
437    if (!fsd.isXattrsEnabled()) {
438      throw new IOException(String.format(
439          "The XAttr operation has been rejected.  "
440              + "Support for XAttrs has been disabled by setting %s to false.",
441          DFSConfigKeys.DFS_NAMENODE_XATTRS_ENABLED_KEY));
442    }
443  }

445  private static List<XAttr> getXAttrs(FSDirectory fsd,
446                                String src) throws IOException {
447    String srcs = FSDirectory.normalizePath(src);
448    fsd.readLock();
449    try {
450      INodesInPath iip = fsd.getINodesInPath(srcs, true);
451      INode inode = FSDirectory.resolveLastINode(iip);
452      int snapshotId = iip.getPathSnapshotId();
453      return XAttrStorage.readINodeXAttrs(inode, snapshotId);
454    } finally {
455      fsd.readUnlock();
456    }
457  }
458
459  private static boolean isUserVisible(XAttr xAttr) {
460    XAttr.NameSpace ns = xAttr.getNameSpace();
461    return ns == XAttr.NameSpace.USER || ns == XAttr.NameSpace.TRUSTED;
462  }
