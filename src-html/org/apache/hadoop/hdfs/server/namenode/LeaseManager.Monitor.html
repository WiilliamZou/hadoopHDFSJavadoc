047/**
048 * LeaseManager does the lease housekeeping for writing on files.   
049 * This class also provides useful static methods for lease recovery.
050 * 
051 * Lease Recovery Algorithm
052 * 1) Namenode retrieves lease information
053 * 2) For each file f in the lease, consider the last block b of f
054 * 2.1) Get the datanodes which contains b
055 * 2.2) Assign one of the datanodes as the primary datanode p
056
057 * 2.3) p obtains a new generation stamp from the namenode
058 * 2.4) p gets the block info from each datanode
059 * 2.5) p computes the minimum block length
060 * 2.6) p updates the datanodes, which have a valid generation stamp,
061 *      with the new generation stamp and the minimum block length 
062 * 2.7) p acknowledges the namenode the update results
063
064 * 2.8) Namenode updates the BlockInfo
065 * 2.9) Namenode removes f from the lease
066 *      and removes the lease once all files have been removed
067 * 2.10) Namenode commit changes to edit log
068 */
069@InterfaceAudience.Private
070public class LeaseManager {
071  public static final Log LOG = LogFactory.getLog(LeaseManager.class);

104  /**
105   * This method iterates through all the leases and counts the number of blocks
106   * which are not COMPLETE. The FSNamesystem read lock MUST be held before
107   * calling this method.
108   * @return
109   */
110  synchronized long getNumUnderConstructionBlocks() {
111    assert this.fsnamesystem.hasReadLock() : "The FSNamesystem read lock wasn't"
112      + "acquired before counting under construction blocks";
113    long numUCBlocks = 0;
114    for (Lease lease : sortedLeases) {
115      for (String path : lease.getPaths()) {
116        final INodeFile cons;
117        try {
118          cons = this.fsnamesystem.getFSDirectory().getINode(path).asFile();
119          Preconditions.checkState(cons.isUnderConstruction());
120        } catch (UnresolvedLinkException e) {
121          throw new AssertionError("Lease files should reside on this FS");
122        }
123        BlockInfoContiguous[] blocks = cons.getBlocks();
124        if(blocks == null)
125          continue;
126        for(BlockInfoContiguous b : blocks) {
127          if(!b.isComplete())
128            numUCBlocks++;
129        }
130      }
131    }
132    LOG.info("Number of blocks under construction: " + numUCBlocks);
133    return numUCBlocks;
134  }

151  /**
152   * Adds (or re-adds) the lease for the specified file.
153   */
154  synchronized Lease addLease(String holder, String src) {
155    Lease lease = getLease(holder);
156    if (lease == null) {
157      lease = new Lease(holder);
158      leases.put(holder, lease);
159      sortedLeases.add(lease);
160    } else {
161      renewLease(lease);
162    }
163    sortedLeasesByPath.put(src, lease);
164    lease.paths.add(src);
165    return lease;
166  }
167
168  /**
169   * Remove the specified lease and src.
170   */
171  synchronized void removeLease(Lease lease, String src) {
172    sortedLeasesByPath.remove(src);
173    if (!lease.removePath(src)) {
174      if (LOG.isDebugEnabled()) {
175        LOG.debug(src + " not found in lease.paths (=" + lease.paths + ")");
176      }
177    }
178
179    if (!lease.hasPath()) {
180      leases.remove(lease.holder);
181      if (!sortedLeases.remove(lease)) {
182        LOG.error(lease + " not found in sortedLeases");
183      }
184    }
185  }
186
187  /**
188   * Remove the lease for the specified holder and src
189   */
190  synchronized void removeLease(String holder, String src) {
191    Lease lease = getLease(holder);
192    if (lease != null) {
193      removeLease(lease, src);
194    } else {
195      LOG.warn("Removing non-existent lease! holder=" + holder +
196          " src=" + src);
197    }
198  }
199
200  synchronized void removeAllLeases() {
201    sortedLeases.clear();
202    sortedLeasesByPath.clear();
203    leases.clear();
204  }
205
206  /**
207   * Reassign lease for file src to the new holder.
208   */
209  synchronized Lease reassignLease(Lease lease, String src, String newHolder) {
210    assert newHolder != null : "new lease holder is null";
211    if (lease != null) {
212      removeLease(lease, src);
213    }
214    return addLease(newHolder, src);
215  }

217  /**
218   * Renew the lease(s) held by the given client
219   */
220  synchronized void renewLease(String holder) {
221    renewLease(getLease(holder));
222  }
223  synchronized void renewLease(Lease lease) {
224    if (lease != null) {
225      sortedLeases.remove(lease);
226      lease.renew();
227      sortedLeases.add(lease);
228    }
229  }

231  /**
232   * Renew all of the currently open leases.
233   */
234  synchronized void renewAllLeases() {
235    for (Lease l : leases.values()) {
236      renewLease(l);
237    }
238  }

240  /************************************************************
241   * A Lease governs all the locks held by a single client.
242   * For each client there's a corresponding lease, whose
243   * timestamp is updated when the client periodically
244   * checks in.  If the client dies and allows its lease to
245   * expire, all the corresponding locks can be released.
246   *************************************************************/
247  class Lease implements Comparable<Lease> {
248    private final String holder;
249    private long lastUpdate;
250    private final Collection<String> paths = new TreeSet<String>();
251  
252    /** Only LeaseManager object can create a lease */
253    private Lease(String holder) {
254      this.holder = holder;
255      renew();
256    }
257    /** Only LeaseManager object can renew a lease */
258    private void renew() {
259      this.lastUpdate = monotonicNow();
260    }
261
262    /** @return true if the Hard Limit Timer has expired */
263    public boolean expiredHardLimit() {
264      return monotonicNow() - lastUpdate > hardLimit;
265    }
266
267    /** @return true if the Soft Limit Timer has expired */
268    public boolean expiredSoftLimit() {
269      return monotonicNow() - lastUpdate > softLimit;
270    }
271
272    /** Does this lease contain any path? */
273    boolean hasPath() {return !paths.isEmpty();}
274
275    boolean removePath(String src) {
276      return paths.remove(src);
277    }
278
279    @Override
280    public String toString() {
281      return "[Lease.  Holder: " + holder
282          + ", pendingcreates: " + paths.size() + "]";
283    }
284  
285    @Override
286    public int compareTo(Lease o) {
287      Lease l1 = this;
288      Lease l2 = o;
289      long lu1 = l1.lastUpdate;
290      long lu2 = l2.lastUpdate;
291      if (lu1 < lu2) {
292        return -1;
293      } else if (lu1 > lu2) {
294        return 1;
295      } else {
296        return l1.holder.compareTo(l2.holder);
297      }
298    }
299  
300    @Override
301    public boolean equals(Object o) {
302      if (!(o instanceof Lease)) {
303        return false;
304      }
305      Lease obj = (Lease) o;
306      if (lastUpdate == obj.lastUpdate &&
307          holder.equals(obj.holder)) {
308        return true;
309      }
310      return false;
311    }
312  
313    @Override
314    public int hashCode() {
315      return holder.hashCode();
316    }
317    
318    Collection<String> getPaths() {
319      return paths;
320    }
321
322    String getHolder() {
323      return holder;
324    }
325
326    void replacePath(String oldpath, String newpath) {
327      paths.remove(oldpath);
328      paths.add(newpath);
329    }
330    
331    @VisibleForTesting
332    long getLastUpdate() {
333      return lastUpdate;
334    }
335  }
336
337  synchronized void changeLease(String src, String dst) {
338    if (LOG.isDebugEnabled()) {
339      LOG.debug(getClass().getSimpleName() + ".changelease: " +
340               " src=" + src + ", dest=" + dst);
341    }
342
343    final int len = src.length();
344    for(Map.Entry<String, Lease> entry
345        : findLeaseWithPrefixPath(src, sortedLeasesByPath).entrySet()) {
346      final String oldpath = entry.getKey();
347      final Lease lease = entry.getValue();
348      // replace stem of src with new destination
349      final String newpath = dst + oldpath.substring(len);
350      if (LOG.isDebugEnabled()) {
351        LOG.debug("changeLease: replacing " + oldpath + " with " + newpath);
352      }
353      lease.replacePath(oldpath, newpath);
354      sortedLeasesByPath.remove(oldpath);
355      sortedLeasesByPath.put(newpath, lease);
356    }
357  }
358
359  synchronized void removeLeaseWithPrefixPath(String prefix) {
360    for(Map.Entry<String, Lease> entry
361        : findLeaseWithPrefixPath(prefix, sortedLeasesByPath).entrySet()) {
362      if (LOG.isDebugEnabled()) {
363        LOG.debug(LeaseManager.class.getSimpleName()
364            + ".removeLeaseWithPrefixPath: entry=" + entry);
365      }
366      removeLease(entry.getValue(), entry.getKey());    
367    }
368  }
369
370  static private Map<String, Lease> findLeaseWithPrefixPath(
371      String prefix, SortedMap<String, Lease> path2lease) {
372    if (LOG.isDebugEnabled()) {
373      LOG.debug(LeaseManager.class.getSimpleName() + ".findLease: prefix=" + prefix);
374    }
375
376    final Map<String, Lease> entries = new HashMap<String, Lease>();
377    int srclen = prefix.length();
378    
379    // prefix may ended with '/'
380    if (prefix.charAt(srclen - 1) == Path.SEPARATOR_CHAR) {
381      srclen -= 1;
382    }
383
384    for(Map.Entry<String, Lease> entry : path2lease.tailMap(prefix).entrySet()) {
385      final String p = entry.getKey();
386      if (!p.startsWith(prefix)) {
387        return entries;
388      }
389      if (p.length() == srclen || p.charAt(srclen) == Path.SEPARATOR_CHAR) {
390        entries.put(entry.getKey(), entry.getValue());
391      }
392    }
393    return entries;
394  }

401  /******************************************************
402   * Monitor checks for leases that have expired,
403   * and disposes of them.
404   ******************************************************/
405  class Monitor implements Runnable {
406    final String name = getClass().getSimpleName();
407
408    /** Check leases periodically. */
409    @Override
410    public void run() {
411      for(; shouldRunMonitor && fsnamesystem.isRunning(); ) {
412        boolean needSync = false;
413        try {
414          fsnamesystem.writeLockInterruptibly();
415          try {
416            if (!fsnamesystem.isInSafeMode()) {
417              needSync = checkLeases();
418            }
419          } finally {
420            fsnamesystem.writeUnlock();
421            // lease reassignments should to be sync'ed.
422            if (needSync) {
423              fsnamesystem.getEditLog().logSync();
424            }
425          }
426  
427          Thread.sleep(HdfsServerConstants.NAMENODE_LEASE_RECHECK_INTERVAL);
428        } catch(InterruptedException ie) {
429          if (LOG.isDebugEnabled()) {
430            LOG.debug(name + " is interrupted", ie);
431          }
432        }
433      }
434    }
435  }

437  /**
438   * Get the list of inodes corresponding to valid leases.
439   * @return list of inodes
440   */
441  Map<String, INodeFile> getINodesUnderConstruction() {
442    Map<String, INodeFile> inodes = new TreeMap<String, INodeFile>();
443    for (String p : sortedLeasesByPath.keySet()) {
444      // verify that path exists in namespace
445      try {
446        INodeFile node = INodeFile.valueOf(fsnamesystem.dir.getINode(p), p);
447        Preconditions.checkState(node.isUnderConstruction());
448        inodes.put(p, node);
449      } catch (IOException ioe) {
450        LOG.error(ioe);
451      }
452    }
453    return inodes;
454  }

456  /** Check the leases beginning from the oldest.
457   *  @return true is sync is needed.
458   */
459  @VisibleForTesting
460  synchronized boolean checkLeases() {
461    boolean needSync = false;
462    assert fsnamesystem.hasWriteLock();
463    Lease leaseToCheck = null;
464    try {
465      leaseToCheck = sortedLeases.first();
466    } catch(NoSuchElementException e) {}
467
468    while(leaseToCheck != null) {
469      if (!leaseToCheck.expiredHardLimit()) {
470        break;
471      }
472
473      LOG.info(leaseToCheck + " has expired hard limit");
474
475      final List<String> removing = new ArrayList<String>();
476      // need to create a copy of the oldest lease paths, because 
477      // internalReleaseLease() removes paths corresponding to empty files,
478      // i.e. it needs to modify the collection being iterated over
479      // causing ConcurrentModificationException
480      String[] leasePaths = new String[leaseToCheck.getPaths().size()];
481      leaseToCheck.getPaths().toArray(leasePaths);
482      for(String p : leasePaths) {
483        try {
484          INodesInPath iip = fsnamesystem.getFSDirectory().getINodesInPath(p,
485              true);
486          boolean completed = fsnamesystem.internalReleaseLease(leaseToCheck, p,
487              iip, HdfsServerConstants.NAMENODE_LEASE_HOLDER);
488          if (LOG.isDebugEnabled()) {
489            if (completed) {
490              LOG.debug("Lease recovery for " + p + " is complete. File closed.");
491            } else {
492              LOG.debug("Started block recovery " + p + " lease " + leaseToCheck);
493            }
494          }
495          // If a lease recovery happened, we need to sync later.
496          if (!needSync && !completed) {
497            needSync = true;
498          }
499        } catch (IOException e) {
500          LOG.error("Cannot release the path " + p + " in the lease "
501              + leaseToCheck, e);
502          removing.add(p);
503        }
504      }
505
506      for(String p : removing) {
507        removeLease(leaseToCheck, p);
508      }
509      leaseToCheck = sortedLeases.higher(leaseToCheck);
510    }
511
512    try {
513      if(leaseToCheck != sortedLeases.first()) {
514        LOG.warn("Unable to release hard-limit expired lease: "
515          + sortedLeases.first());
516      }
517    } catch(NoSuchElementException e) {}
518    return needSync;
519  }

530  void startMonitor() {
531    Preconditions.checkState(lmthread == null,
532        "Lease Monitor already running");
533    shouldRunMonitor = true;
534    lmthread = new Daemon(new Monitor());
535    lmthread.start();
536  }
537  
538  void stopMonitor() {
539    if (lmthread != null) {
540      shouldRunMonitor = false;
541      try {
542        lmthread.interrupt();
543        lmthread.join(3000);
544      } catch (InterruptedException ie) {
545        LOG.warn("Encountered exception ", ie);
546      }
547      lmthread = null;
548    }
549  }
