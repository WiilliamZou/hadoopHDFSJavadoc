017package org.apache.hadoop.hdfs.server.namenode.startupprogress;
018
019import org.apache.hadoop.classification.InterfaceAudience;
020
021/**
022 * Abstract base of internal data structures used for tracking progress.  For
023 * primitive long properties, {@link Long#MIN_VALUE} is used as a sentinel value
024 * to indicate that the property is undefined.
025 */
026@InterfaceAudience.Private
027abstract class AbstractTracking implements Cloneable {
028  long beginTime = Long.MIN_VALUE;
029  long endTime = Long.MIN_VALUE;
030
031  /**
032   * Subclass instances may call this method during cloning to copy the values of
033   * all properties stored in this base class.
034   * 
035   * @param dest AbstractTracking destination for copying properties
036   */
037  protected void copy(AbstractTracking dest) {
038    dest.beginTime = beginTime;
039    dest.endTime = endTime;
040  }
041}
