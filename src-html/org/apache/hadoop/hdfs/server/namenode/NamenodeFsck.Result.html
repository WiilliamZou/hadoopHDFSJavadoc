084/**
085 * This class provides rudimentary checking of DFS volumes for errors and
086 * sub-optimal conditions.
087 * <p>The tool scans all files and directories, starting from an indicated
088 *  root path. The following abnormal conditions are detected and handled:</p>
089 * <ul>
090 * <li>files with blocks that are completely missing from all datanodes.<br/>
091 * In this case the tool can perform one of the following actions:
092 *  <ul>
093 *      <li>none ({@link #FIXING_NONE})</li>
094 *      <li>move corrupted files to /lost+found directory on DFS
095 *      ({@link #FIXING_MOVE}). Remaining data blocks are saved as a
096 *      block chains, representing longest consecutive series of valid blocks.</li>
097 *      <li>delete corrupted files ({@link #FIXING_DELETE})</li>
098 *  </ul>
099 *  </li>
100 *  <li>detect files with under-replicated or over-replicated blocks</li>
101 *  </ul>
102 *  Additionally, the tool collects a detailed overall DFS statistics, and
103 *  optionally can print detailed statistics on block locations and replication
104 *  factors of each file.
105 */
106@InterfaceAudience.Private

171  /**
172   * Filesystem checker.
173   * @param conf configuration (namenode config)
174   * @param namenode namenode that this fsck is going to use
175   * @param pmap key=value[] map passed to the http servlet as url parameters
176   * @param out output stream to write the fsck output
177   * @param totalDatanodes number of live datanodes
178   * @param minReplication minimum replication
179   * @param remoteAddress source address of the fsck request
180   */
181  NamenodeFsck(Configuration conf, NameNode namenode,
182      NetworkTopology networktopology, 
183      Map<String,String[]> pmap, PrintWriter out,
184      int totalDatanodes, short minReplication, InetAddress remoteAddress) {
185    this.conf = conf;
186    this.namenode = namenode;
187    this.networktopology = networktopology;
188    this.out = out;
189    this.totalDatanodes = totalDatanodes;
190    this.minReplication = minReplication;
191    this.remoteAddress = remoteAddress;
192    this.bpPolicy = BlockPlacementPolicy.getInstance(conf, null,
193        networktopology,
194        namenode.getNamesystem().getBlockManager().getDatanodeManager()
195        .getHost2DatanodeMap());
196    
197    for (Iterator<String> it = pmap.keySet().iterator(); it.hasNext();) {
198      String key = it.next();
199      if (key.equals("path")) { this.path = pmap.get("path")[0]; }
200      else if (key.equals("move")) { this.doMove = true; }
201      else if (key.equals("delete")) { this.doDelete = true; }
202      else if (key.equals("files")) { this.showFiles = true; }
203      else if (key.equals("blocks")) { this.showBlocks = true; }
204      else if (key.equals("locations")) { this.showLocations = true; }
205      else if (key.equals("racks")) { this.showRacks = true; }
206      else if (key.equals("storagepolicies")) { this.showStoragePolcies = true; }
207      else if (key.equals("openforwrite")) {this.showOpenFiles = true; }
208      else if (key.equals("listcorruptfileblocks")) {
209        this.showCorruptFileBlocks = true;
210      } else if (key.equals("startblockafter")) {
211        this.currentCookie[0] = pmap.get("startblockafter")[0];
212      } else if (key.equals("includeSnapshots")) {
213        this.snapshottableDirs = new ArrayList<String>();
214      } else if (key.equals("blockId")) {
215        this.blockIds = pmap.get("blockId")[0];
216      }
217    }
218  }

220  /**
221   * Check block information given a blockId number
222   *
223  */
224  public void blockIdCK(String blockId) {
225
226    if(blockId == null) {
227      out.println("Please provide valid blockId!");
228      return;
229    }
230
231    BlockManager bm = namenode.getNamesystem().getBlockManager();
232    try {
233      //get blockInfo
234      Block block = new Block(Block.getBlockId(blockId));
235      //find which file this block belongs to
236      BlockInfoContiguous blockInfo = bm.getStoredBlock(block);
237      if(blockInfo == null) {
238        out.println("Block "+ blockId +" " + NONEXISTENT_STATUS);
239        LOG.warn("Block "+ blockId + " " + NONEXISTENT_STATUS);
240        return;
241      }
242      BlockCollection bc = bm.getBlockCollection(blockInfo);
243      INode iNode = (INode) bc;
244      NumberReplicas numberReplicas= bm.countNodes(block);
245      out.println("Block Id: " + blockId);
246      out.println("Block belongs to: "+iNode.getFullPathName());
247      out.println("No. of Expected Replica: " + bc.getBlockReplication());
248      out.println("No. of live Replica: " + numberReplicas.liveReplicas());
249      out.println("No. of excess Replica: " + numberReplicas.excessReplicas());
250      out.println("No. of stale Replica: " + numberReplicas.replicasOnStaleNodes());
251      out.println("No. of decommission Replica: "
252          + numberReplicas.decommissionedReplicas());
253      out.println("No. of corrupted Replica: " + numberReplicas.corruptReplicas());
254      //record datanodes that have corrupted block replica
255      Collection<DatanodeDescriptor> corruptionRecord = null;
256      if (bm.getCorruptReplicas(block) != null) {
257        corruptionRecord = bm.getCorruptReplicas(block);
258      }
259
260      //report block replicas status on datanodes
261      for(int idx = (blockInfo.numNodes()-1); idx >= 0; idx--) {
262        DatanodeDescriptor dn = blockInfo.getDatanode(idx);
263        out.print("Block replica on datanode/rack: " + dn.getHostName() +
264            dn.getNetworkLocation() + " ");
265        if (corruptionRecord != null && corruptionRecord.contains(dn)) {
266          out.print(CORRUPT_STATUS+"\t ReasonCode: "+
267            bm.getCorruptReason(block,dn));
268        } else if (dn.isDecommissioned() ){
269          out.print(DECOMMISSIONED_STATUS);
270        } else if (dn.isDecommissionInProgress()) {
271          out.print(DECOMMISSIONING_STATUS);
272        } else {
273          out.print(HEALTHY_STATUS);
274        }
275        out.print("\n");
276      }
277    } catch (Exception e){
278      String errMsg = "Fsck on blockId '" + blockId;
279      LOG.warn(errMsg, e);
280      out.println(e.getMessage());
281      out.print("\n\n" + errMsg);
282      LOG.warn("Error in looking up block", e);
283    }
284  }

286  /**
287   * Check files on DFS, starting from the indicated path.
288   */
289  public void fsck() {
290    final long startTime = Time.monotonicNow();
291    try {
292      if(blockIds != null) {
293
294        String[] blocks = blockIds.split(" ");
295        StringBuilder sb = new StringBuilder();
296        sb.append("FSCK started by " +
297            UserGroupInformation.getCurrentUser() + " from " +
298            remoteAddress + " at " + new Date());
299        out.println(sb.toString());
300        sb.append(" for blockIds: \n");
301        for (String blk: blocks) {
302          if(blk == null || !blk.contains(Block.BLOCK_FILE_PREFIX)) {
303            out.println("Incorrect blockId format: " + blk);
304            continue;
305          }
306          out.print("\n");
307          blockIdCK(blk);
308          sb.append(blk + "\n");
309        }
310        LOG.info(sb.toString());
311        namenode.getNamesystem().logFsckEvent("/", remoteAddress);
312        out.flush();
313        return;
314      }
315
316      String msg = "FSCK started by " + UserGroupInformation.getCurrentUser()
317          + " from " + remoteAddress + " for path " + path + " at " + new Date();
318      LOG.info(msg);
319      out.println(msg);
320      namenode.getNamesystem().logFsckEvent(path, remoteAddress);
321
322      if (snapshottableDirs != null) {
323        SnapshottableDirectoryStatus[] snapshotDirs = namenode.getRpcServer()
324            .getSnapshottableDirListing();
325        if (snapshotDirs != null) {
326          for (SnapshottableDirectoryStatus dir : snapshotDirs) {
327            snapshottableDirs.add(dir.getFullPath().toString());
328          }
329        }
330      }
331
332      final HdfsFileStatus file = namenode.getRpcServer().getFileInfo(path);
333      if (file != null) {
334
335        if (showCorruptFileBlocks) {
336          listCorruptFileBlocks();
337          return;
338        }
339        
340        if (this.showStoragePolcies) {
341          storageTypeSummary = new StoragePolicySummary(
342              namenode.getNamesystem().getBlockManager().getStoragePolicies());
343        }
344
345        Result res = new Result(conf);
346
347        check(path, file, res);
348
349        out.println(res);
350        out.println(" Number of data-nodes:\t\t" + totalDatanodes);
351        out.println(" Number of racks:\t\t" + networktopology.getNumOfRacks());
352
353        if (this.showStoragePolcies) {
354          out.print(storageTypeSummary.toString());
355        }
356
357        out.println("FSCK ended at " + new Date() + " in "
358            + (Time.monotonicNow() - startTime + " milliseconds"));
359
360        // If there were internal errors during the fsck operation, we want to
361        // return FAILURE_STATUS, even if those errors were not immediately
362        // fatal.  Otherwise many unit tests will pass even when there are bugs.
363        if (internalError) {
364          throw new IOException("fsck encountered internal errors!");
365        }
366
367        // DFSck client scans for the string HEALTHY/CORRUPT to check the status
368        // of file system and return appropriate code. Changing the output
369        // string might break testcases. Also note this must be the last line 
370        // of the report.
371        if (res.isHealthy()) {
372          out.print("\n\nThe filesystem under path '" + path + "' " + HEALTHY_STATUS);
373        } else {
374          out.print("\n\nThe filesystem under path '" + path + "' " + CORRUPT_STATUS);
375        }
376
377      } else {
378        out.print("\n\nPath '" + path + "' " + NONEXISTENT_STATUS);
379      }
380    } catch (Exception e) {
381      String errMsg = "Fsck on path '" + path + "' " + FAILURE_STATUS;
382      LOG.warn(errMsg, e);
383      out.println("FSCK ended at " + new Date() + " in "
384          + (Time.monotonicNow() - startTime + " milliseconds"));
385      out.println(e.getMessage());
386      out.print("\n\n" + errMsg);
387    } finally {
388      out.close();
389    }
390  }

392  private void listCorruptFileBlocks() throws IOException {
393    Collection<FSNamesystem.CorruptFileBlockInfo> corruptFiles = namenode.
394      getNamesystem().listCorruptFileBlocks(path, currentCookie);
395    int numCorruptFiles = corruptFiles.size();
396    String filler;
397    if (numCorruptFiles > 0) {
398      filler = Integer.toString(numCorruptFiles);
399    } else if (currentCookie[0].equals("0")) {
400      filler = "no";
401    } else {
402      filler = "no more";
403    }
404    out.println("Cookie:\t" + currentCookie[0]);
405    for (FSNamesystem.CorruptFileBlockInfo c : corruptFiles) {
406      out.println(c.toString());
407    }
408    out.println("\n\nThe filesystem under path '" + path + "' has " + filler
409        + " CORRUPT files");
410    out.println();
411  }

413  @VisibleForTesting
414  void check(String parent, HdfsFileStatus file, Result res) throws IOException {
415    String path = file.getFullName(parent);
416    boolean isOpen = false;
417
418    if (file.isDir()) {
419      if (snapshottableDirs != null && snapshottableDirs.contains(path)) {
420        String snapshotPath = (path.endsWith(Path.SEPARATOR) ? path : path
421            + Path.SEPARATOR)
422            + HdfsConstants.DOT_SNAPSHOT_DIR;
423        HdfsFileStatus snapshotFileInfo = namenode.getRpcServer().getFileInfo(
424            snapshotPath);
425        check(snapshotPath, snapshotFileInfo, res);
426      }
427      byte[] lastReturnedName = HdfsFileStatus.EMPTY_NAME;
428      DirectoryListing thisListing;
429      if (showFiles) {
430        out.println(path + " <dir>");
431      }
432      res.totalDirs++;
433      do {
434        assert lastReturnedName != null;
435        thisListing = namenode.getRpcServer().getListing(
436            path, lastReturnedName, false);
437        if (thisListing == null) {
438          return;
439        }
440        HdfsFileStatus[] files = thisListing.getPartialListing();
441        for (int i = 0; i < files.length; i++) {
442          check(path, files[i], res);
443        }
444        lastReturnedName = thisListing.getLastName();
445      } while (thisListing.hasMore());
446      return;
447    }
448    if (file.isSymlink()) {
449      if (showFiles) {
450        out.println(path + " <symlink>");
451      }
452      res.totalSymlinks++;
453      return;
454    }
455    long fileLen = file.getLen();
456    // Get block locations without updating the file access time 
457    // and without block access tokens
458    LocatedBlocks blocks = null;
459    FSNamesystem fsn = namenode.getNamesystem();
460    fsn.readLock();
461    try {
462      blocks = fsn.getBlockLocations(path, 0, fileLen, false, false).blocks;
463    } catch (FileNotFoundException fnfe) {
464      blocks = null;
465    } finally {
466      fsn.readUnlock();
467    }
468    if (blocks == null) { // the file is deleted
469      return;
470    }
471    isOpen = blocks.isUnderConstruction();
472    if (isOpen && !showOpenFiles) {
473      // We collect these stats about open files to report with default options
474      res.totalOpenFilesSize += fileLen;
475      res.totalOpenFilesBlocks += blocks.locatedBlockCount();
476      res.totalOpenFiles++;
477      return;
478    }
479    res.totalFiles++;
480    res.totalSize += fileLen;
481    res.totalBlocks += blocks.locatedBlockCount();
482    if (showOpenFiles && isOpen) {
483      out.print(path + " " + fileLen + " bytes, " +
484        blocks.locatedBlockCount() + " block(s), OPENFORWRITE: ");
485    } else if (showFiles) {
486      out.print(path + " " + fileLen + " bytes, " +
487        blocks.locatedBlockCount() + " block(s): ");
488    } else {
489      out.print('.');
490    }
491    if (res.totalFiles % 100 == 0) { out.println(); out.flush(); }
492    int missing = 0;
493    int corrupt = 0;
494    long missize = 0;
495    int underReplicatedPerFile = 0;
496    int misReplicatedPerFile = 0;
497    StringBuilder report = new StringBuilder();
498    int i = 0;
499    for (LocatedBlock lBlk : blocks.getLocatedBlocks()) {
500      ExtendedBlock block = lBlk.getBlock();
501      boolean isCorrupt = lBlk.isCorrupt();
502      String blkName = block.toString();
503      DatanodeInfo[] locs = lBlk.getLocations();
504      NumberReplicas numberReplicas =
505          namenode.getNamesystem().getBlockManager().countNodes(block.getLocalBlock());
506      int liveReplicas = numberReplicas.liveReplicas();
507      res.totalReplicas += liveReplicas;
508      short targetFileReplication = file.getReplication();
509      res.numExpectedReplicas += targetFileReplication;
510      if(liveReplicas<minReplication){
511        res.numUnderMinReplicatedBlocks++;
512      }
513      if (liveReplicas > targetFileReplication) {
514        res.excessiveReplicas += (liveReplicas - targetFileReplication);
515        res.numOverReplicatedBlocks += 1;
516      }
517      //keep track of storage tier counts
518      if (this.showStoragePolcies && lBlk.getStorageTypes() != null) {
519        StorageType[] storageTypes = lBlk.getStorageTypes();
520        storageTypeSummary.add(Arrays.copyOf(storageTypes, storageTypes.length),
521            fsn.getBlockManager().getStoragePolicy(file.getStoragePolicy()));
522      }
523      // Check if block is Corrupt
524      if (isCorrupt) {
525        corrupt++;
526        res.corruptBlocks++;
527        out.print("\n" + path + ": CORRUPT blockpool " + block.getBlockPoolId() + 
528            " block " + block.getBlockName()+"\n");
529      }
530      if (liveReplicas >= minReplication)
531        res.numMinReplicatedBlocks++;
532      if (liveReplicas < targetFileReplication && liveReplicas > 0) {
533        res.missingReplicas += (targetFileReplication - liveReplicas);
534        res.numUnderReplicatedBlocks += 1;
535        underReplicatedPerFile++;
536        if (!showFiles) {
537          out.print("\n" + path + ": ");
538        }
539        out.println(" Under replicated " + block +
540                    ". Target Replicas is " +
541                    targetFileReplication + " but found " +
542                    liveReplicas + " replica(s).");
543      }
544      // verify block placement policy
545      BlockPlacementStatus blockPlacementStatus = bpPolicy
546          .verifyBlockPlacement(path, lBlk, targetFileReplication);
547      if (!blockPlacementStatus.isPlacementPolicySatisfied()) {
548        res.numMisReplicatedBlocks++;
549        misReplicatedPerFile++;
550        if (!showFiles) {
551          if(underReplicatedPerFile == 0)
552            out.println();
553          out.print(path + ": ");
554        }
555        out.println(" Replica placement policy is violated for " + 
556                    block + ". " + blockPlacementStatus.getErrorDescription());
557      }
558      report.append(i + ". " + blkName + " len=" + block.getNumBytes());
559      if (liveReplicas == 0) {
560        report.append(" MISSING!");
561        res.addMissing(block.toString(), block.getNumBytes());
562        missing++;
563        missize += block.getNumBytes();
564      } else {
565        report.append(" repl=" + liveReplicas);
566        if (showLocations || showRacks) {
567          StringBuilder sb = new StringBuilder("[");
568          for (int j = 0; j < locs.length; j++) {
569            if (j > 0) { sb.append(", "); }
570            if (showRacks)
571              sb.append(NodeBase.getPath(locs[j]));
572            else
573              sb.append(locs[j]);
574          }
575          sb.append(']');
576          report.append(" " + sb.toString());
577        }
578      }
579      report.append('\n');
580      i++;
581    }
582    if ((missing > 0) || (corrupt > 0)) {
583      if (!showFiles && (missing > 0)) {
584        out.print("\n" + path + ": MISSING " + missing
585            + " blocks of total size " + missize + " B.");
586      }
587      res.corruptFiles++;
588      if (isOpen) {
589        LOG.info("Fsck: ignoring open file " + path);
590      } else {
591        if (doMove) copyBlocksToLostFound(parent, file, blocks);
592        if (doDelete) deleteCorruptedFile(path);
593      }
594    }
595    if (showFiles) {
596      if (missing > 0) {
597        out.print(" MISSING " + missing + " blocks of total size " + missize + " B\n");
598      }  else if (underReplicatedPerFile == 0 && misReplicatedPerFile == 0) {
599        out.print(" OK\n");
600      }
601      if (showBlocks) {
602        out.print(report.toString() + "\n");
603      }
604    }
605  }

607  private void deleteCorruptedFile(String path) {
608    try {
609      namenode.getRpcServer().delete(path, true);
610      LOG.info("Fsck: deleted corrupt file " + path);
611    } catch (Exception e) {
612      LOG.error("Fsck: error deleting corrupted file " + path, e);
613      internalError = true;
614    }
615  }

627  private void copyBlocksToLostFound(String parent, HdfsFileStatus file,
628        LocatedBlocks blocks) throws IOException {
629    final DFSClient dfs = new DFSClient(NameNode.getAddress(conf), conf);
630    final String fullName = file.getFullName(parent);
631    OutputStream fos = null;
632    try {
633      if (!lfInited) {
634        lostFoundInit(dfs);
635      }
636      if (!lfInitedOk) {
637        throw new IOException("failed to initialize lost+found");
638      }
639      String target = lostFound + fullName;
640      if (hdfsPathExists(target)) {
641        LOG.warn("Fsck: can't copy the remains of " + fullName + " to " +
642          "lost+found, because " + target + " already exists.");
643        return;
644      }
645      if (!namenode.getRpcServer().mkdirs(
646          target, file.getPermission(), true)) {
647        throw new IOException("failed to create directory " + target);
648      }
649      // create chains
650      int chain = 0;
651      boolean copyError = false;
652      for (LocatedBlock lBlk : blocks.getLocatedBlocks()) {
653        LocatedBlock lblock = lBlk;
654        DatanodeInfo[] locs = lblock.getLocations();
655        if (locs == null || locs.length == 0) {
656          if (fos != null) {
657            fos.flush();
658            fos.close();
659            fos = null;
660          }
661          continue;
662        }
663        if (fos == null) {
664          fos = dfs.create(target + "/" + chain, true);
665          chain++;
666        }
667        
668        // copy the block. It's a pity it's not abstracted from DFSInputStream ...
669        try {
670          copyBlock(dfs, lblock, fos);
671        } catch (Exception e) {
672          LOG.error("Fsck: could not copy block " + lblock.getBlock() +
673              " to " + target, e);
674          fos.flush();
675          fos.close();
676          fos = null;
677          internalError = true;
678          copyError = true;
679        }
680      }
681      if (copyError) {
682        LOG.warn("Fsck: there were errors copying the remains of the " +
683          "corrupted file " + fullName + " to /lost+found");
684      } else {
685        LOG.info("Fsck: copied the remains of the corrupted file " + 
686          fullName + " to /lost+found");
687      }
688    } catch (Exception e) {
689      LOG.error("copyBlocksToLostFound: error processing " + fullName, e);
690      internalError = true;
691    } finally {
692      if (fos != null) fos.close();
693      dfs.close();
694    }
695  }

697  /*
698   * XXX (ab) Bulk of this method is copied verbatim from {@link DFSClient}, which is
699   * bad. Both places should be refactored to provide a method to copy blocks
700   * around.
701   */
702  private void copyBlock(final DFSClient dfs, LocatedBlock lblock,
703                         OutputStream fos) throws Exception {
704    int failures = 0;
705    InetSocketAddress targetAddr = null;
706    TreeSet<DatanodeInfo> deadNodes = new TreeSet<DatanodeInfo>();
707    BlockReader blockReader = null; 
708    ExtendedBlock block = lblock.getBlock(); 
709
710    while (blockReader == null) {
711      DatanodeInfo chosenNode;
712      
713      try {
714        chosenNode = bestNode(dfs, lblock.getLocations(), deadNodes);
715        targetAddr = NetUtils.createSocketAddr(chosenNode.getXferAddr());
716      }  catch (IOException ie) {
717        if (failures >= DFSConfigKeys.DFS_CLIENT_MAX_BLOCK_ACQUIRE_FAILURES_DEFAULT) {
718          throw new IOException("Could not obtain block " + lblock, ie);
719        }
720        LOG.info("Could not obtain block from any node:  " + ie);
721        try {
722          Thread.sleep(10000);
723        }  catch (InterruptedException iex) {
724        }
725        deadNodes.clear();
726        failures++;
727        continue;
728      }
729      try {
730        String file = BlockReaderFactory.getFileName(targetAddr,
731            block.getBlockPoolId(), block.getBlockId());
732        blockReader = new BlockReaderFactory(dfs.getConf()).
733            setFileName(file).
734            setBlock(block).
735            setBlockToken(lblock.getBlockToken()).
736            setStartOffset(0).
737            setLength(-1).
738            setVerifyChecksum(true).
739            setClientName("fsck").
740            setDatanodeInfo(chosenNode).
741            setInetSocketAddress(targetAddr).
742            setCachingStrategy(CachingStrategy.newDropBehind()).
743            setClientCacheContext(dfs.getClientContext()).
744            setConfiguration(namenode.conf).
745            setRemotePeerFactory(new RemotePeerFactory() {
746              @Override
747              public Peer newConnectedPeer(InetSocketAddress addr,
748                  Token<BlockTokenIdentifier> blockToken, DatanodeID datanodeId)
749                  throws IOException {
750                Peer peer = null;
751                Socket s = NetUtils.getDefaultSocketFactory(conf).createSocket();
752                try {
753                  s.connect(addr, HdfsServerConstants.READ_TIMEOUT);
754                  s.setSoTimeout(HdfsServerConstants.READ_TIMEOUT);
755                  peer = TcpPeerServer.peerFromSocketAndKey(
756                        dfs.getSaslDataTransferClient(), s, NamenodeFsck.this,
757                        blockToken, datanodeId);
758                } finally {
759                  if (peer == null) {
760                    IOUtils.closeQuietly(s);
761                  }
762                }
763                return peer;
764              }
765            }).
766            build();
767      }  catch (IOException ex) {
768        // Put chosen node into dead list, continue
769        LOG.info("Failed to connect to " + targetAddr + ":" + ex);
770        deadNodes.add(chosenNode);
771      }
772    }
773    byte[] buf = new byte[1024];
774    int cnt = 0;
775    boolean success = true;
776    long bytesRead = 0;
777    try {
778      while ((cnt = blockReader.read(buf, 0, buf.length)) > 0) {
779        fos.write(buf, 0, cnt);
780        bytesRead += cnt;
781      }
782      if ( bytesRead != block.getNumBytes() ) {
783        throw new IOException("Recorded block size is " + block.getNumBytes() + 
784                              ", but datanode returned " +bytesRead+" bytes");
785      }
786    } catch (Exception e) {
787      LOG.error("Error reading block", e);
788      success = false;
789    } finally {
790      blockReader.close();
791    }
792    if (!success) {
793      throw new Exception("Could not copy block data for " + lblock.getBlock());
794    }
795  }

802  /*
803   * XXX (ab) See comment above for copyBlock().
804   *
805   * Pick the best node from which to stream the data.
806   * That's the local one, if available.
807   */
808  private DatanodeInfo bestNode(DFSClient dfs, DatanodeInfo[] nodes,
809                                TreeSet<DatanodeInfo> deadNodes) throws IOException {
810    if ((nodes == null) ||
811        (nodes.length - deadNodes.size() < 1)) {
812      throw new IOException("No live nodes contain current block");
813    }
814    DatanodeInfo chosenNode;
815    do {
816      chosenNode = nodes[DFSUtil.getRandom().nextInt(nodes.length)];
817    } while (deadNodes.contains(chosenNode));
818    return chosenNode;
819  }

821  private void lostFoundInit(DFSClient dfs) {
822    lfInited = true;
823    try {
824      String lfName = "/lost+found";
825      
826      final HdfsFileStatus lfStatus = dfs.getFileInfo(lfName);
827      if (lfStatus == null) { // not exists
828        lfInitedOk = dfs.mkdirs(lfName, null, true);
829        lostFound = lfName;
830      } else if (!lfStatus.isDir()) { // exists but not a directory
831        LOG.warn("Cannot use /lost+found : a regular file with this name exists.");
832        lfInitedOk = false;
833      }  else { // exists and is a directory
834        lostFound = lfName;
835        lfInitedOk = true;
836      }
837    }  catch (Exception e) {
838      e.printStackTrace();
839      lfInitedOk = false;
840    }
841    if (lostFound == null) {
842      LOG.warn("Cannot initialize /lost+found .");
843      lfInitedOk = false;
844      internalError = true;
845    }
846  }

848  /**
849   * FsckResult of checking, plus overall DFS statistics.
850   */
851  @VisibleForTesting
852  static class Result {
853    final List<String> missingIds = new ArrayList<String>();
854    long missingSize = 0L;
855    long corruptFiles = 0L;
856    long corruptBlocks = 0L;
857    long excessiveReplicas = 0L;
858    long missingReplicas = 0L;
859    long numUnderMinReplicatedBlocks=0L;
860    long numOverReplicatedBlocks = 0L;
861    long numUnderReplicatedBlocks = 0L;
862    long numMisReplicatedBlocks = 0L;  // blocks that do not satisfy block placement policy
863    long numMinReplicatedBlocks = 0L;  // minimally replicatedblocks
864    long totalBlocks = 0L;
865    long numExpectedReplicas = 0L;
866    long totalOpenFilesBlocks = 0L;
867    long totalFiles = 0L;
868    long totalOpenFiles = 0L;
869    long totalDirs = 0L;
870    long totalSymlinks = 0L;
871    long totalSize = 0L;
872    long totalOpenFilesSize = 0L;
873    long totalReplicas = 0L;
874
875    final short replication;
876    final int minReplication;
877    
878    Result(Configuration conf) {
879      this.replication = (short)conf.getInt(DFSConfigKeys.DFS_REPLICATION_KEY, 
880                                            DFSConfigKeys.DFS_REPLICATION_DEFAULT);
881      this.minReplication = (short)conf.getInt(DFSConfigKeys.DFS_NAMENODE_REPLICATION_MIN_KEY,
882                                            DFSConfigKeys.DFS_NAMENODE_REPLICATION_MIN_DEFAULT);
883    }
884    
885    /**
886     * DFS is considered healthy if there are no missing blocks.
887     */
888    boolean isHealthy() {
889      return ((missingIds.size() == 0) && (corruptBlocks == 0));
890    }
891    
892    /** Add a missing block name, plus its size. */
893    void addMissing(String id, long size) {
894      missingIds.add(id);
895      missingSize += size;
896    }
897    
898    /** Return the actual replication factor. */
899    float getReplicationFactor() {
900      if (totalBlocks == 0)
901        return 0.0f;
902      return (float) (totalReplicas) / (float) totalBlocks;
903    }
904    
905    @Override
906    public String toString() {
907      StringBuilder res = new StringBuilder();
908      res.append("Status: ").append((isHealthy() ? "HEALTHY" : "CORRUPT"))
909          .append("\n Total size:\t").append(totalSize).append(" B");
910      if (totalOpenFilesSize != 0) {
911        res.append(" (Total open files size: ").append(totalOpenFilesSize)
912            .append(" B)");
913      }
914      res.append("\n Total dirs:\t").append(totalDirs).append(
915          "\n Total files:\t").append(totalFiles);
916      res.append("\n Total symlinks:\t\t").append(totalSymlinks);
917      if (totalOpenFiles != 0) {
918        res.append(" (Files currently being written: ").append(totalOpenFiles)
919            .append(")");
920      }
921      res.append("\n Total blocks (validated):\t").append(totalBlocks);
922      if (totalBlocks > 0) {
923        res.append(" (avg. block size ").append((totalSize / totalBlocks))
924            .append(" B)");
925      }
926      if (totalOpenFilesBlocks != 0) {
927        res.append(" (Total open file blocks (not validated): ").append(
928            totalOpenFilesBlocks).append(")");
929      }
930      if (corruptFiles > 0 || numUnderMinReplicatedBlocks>0) {
931        res.append("\n  ********************************");
932        if(numUnderMinReplicatedBlocks>0){
933          res.append("\n  UNDER MIN REPL'D BLOCKS:\t").append(numUnderMinReplicatedBlocks);
934          if(totalBlocks>0){
935            res.append(" (").append(
936                ((float) (numUnderMinReplicatedBlocks * 100) / (float) totalBlocks))
937                .append(" %)");
938          }
939          res.append("\n  ").append("DFSConfigKeys.DFS_NAMENODE_REPLICATION_MIN_KEY:\t")
940             .append(minReplication);
941        }
942        if(corruptFiles>0) {
943          res.append(
944              "\n  CORRUPT FILES:\t").append(corruptFiles);
945          if (missingSize > 0) {
946            res.append("\n  MISSING BLOCKS:\t").append(missingIds.size()).append(
947                "\n  MISSING SIZE:\t\t").append(missingSize).append(" B");
948          }
949          if (corruptBlocks > 0) {
950            res.append("\n  CORRUPT BLOCKS: \t").append(corruptBlocks);
951          }
952        }
953        res.append("\n  ********************************");
954      }
955      res.append("\n Minimally replicated blocks:\t").append(
956          numMinReplicatedBlocks);
957      if (totalBlocks > 0) {
958        res.append(" (").append(
959            ((float) (numMinReplicatedBlocks * 100) / (float) totalBlocks))
960            .append(" %)");
961      }
962      res.append("\n Over-replicated blocks:\t")
963          .append(numOverReplicatedBlocks);
964      if (totalBlocks > 0) {
965        res.append(" (").append(
966            ((float) (numOverReplicatedBlocks * 100) / (float) totalBlocks))
967            .append(" %)");
968      }
969      res.append("\n Under-replicated blocks:\t").append(
970          numUnderReplicatedBlocks);
971      if (totalBlocks > 0) {
972        res.append(" (").append(
973            ((float) (numUnderReplicatedBlocks * 100) / (float) totalBlocks))
974            .append(" %)");
975      }
976      res.append("\n Mis-replicated blocks:\t\t")
977          .append(numMisReplicatedBlocks);
978      if (totalBlocks > 0) {
979        res.append(" (").append(
980            ((float) (numMisReplicatedBlocks * 100) / (float) totalBlocks))
981            .append(" %)");
982      }
983      res.append("\n Default replication factor:\t").append(replication)
984          .append("\n Average block replication:\t").append(
985              getReplicationFactor()).append("\n Corrupt blocks:\t\t").append(
986              corruptBlocks).append("\n Missing replicas:\t\t").append(
987              missingReplicas);
988      if (totalReplicas > 0) {
989        res.append(" (").append(
990            ((float) (missingReplicas * 100) / (float) numExpectedReplicas)).append(
991            " %)");
992      }
993      return res.toString();
994    }
