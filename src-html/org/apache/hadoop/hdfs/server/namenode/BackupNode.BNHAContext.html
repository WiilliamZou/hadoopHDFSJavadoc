056/**
057 * BackupNode.
058 * <p>
059 * Backup node can play two roles.
060 * <ol>
061 * <li>{@link NamenodeRole#CHECKPOINT} node periodically creates checkpoints, 
062 * that is downloads image and edits from the active node, merges them, and
063 * uploads the new image back to the active.</li>
064 * <li>{@link NamenodeRole#BACKUP} node keeps its namespace in sync with the
065 * active node, and periodically creates checkpoints by simply saving the
066 * namespace image to local disk(s).</li>
067 * </ol>
068 */
069@InterfaceAudience.Private
070public class BackupNode extends NameNode {

137    namesystem.dir.disableQuotaChecks();
138    bnImage.setNamesystem(namesystem);

142  @Override // NameNode
143  protected void initialize(Configuration conf) throws IOException {
144    // Trash is disabled in BackupNameNode,
145    // but should be turned back on if it ever becomes active.
146    conf.setLong(CommonConfigurationKeys.FS_TRASH_INTERVAL_KEY, 
147                 CommonConfigurationKeys.FS_TRASH_INTERVAL_DEFAULT);
148    NamespaceInfo nsInfo = handshake(conf);
149    super.initialize(conf);
150    namesystem.setBlockPoolId(nsInfo.getBlockPoolID());
151
152    if (false == namesystem.isInSafeMode()) {
153      namesystem.setSafeMode(SafeModeAction.SAFEMODE_ENTER);
154    }
155
156    // Backup node should never do lease recovery,
157    // therefore lease hard limit should never expire.
158    namesystem.leaseManager.setLeasePeriod(
159        HdfsConstants.LEASE_SOFTLIMIT_PERIOD, Long.MAX_VALUE);
160
161    // register with the active name-node 
162    registerWith(nsInfo);
163    // Checkpoint daemon should start after the rpc server started
164    runCheckpointDaemon(conf);
165    InetSocketAddress addr = getHttpAddress();
166    if (addr != null) {
167      conf.set(BN_HTTP_ADDRESS_NAME_KEY, NetUtils.getHostPortString(getHttpAddress()));
168    }
169  }

177  @Override // NameNode
178  public void stop() {
179    if(checkpointManager != null) {
180      // Prevent from starting a new checkpoint.
181      // Checkpoints that has already been started may proceed until 
182      // the error reporting to the name-node is complete.
183      // Checkpoint manager should not be interrupted yet because it will
184      // close storage file channels and the checkpoint may fail with 
185      // ClosedByInterruptException.
186      checkpointManager.shouldRun = false;
187    }
188    if(namenode != null && getRegistration() != null) {
189      // Exclude this node from the list of backup streams on the name-node
190      try {
191        namenode.errorReport(getRegistration(), NamenodeProtocol.FATAL,
192            "Shutting down.");
193      } catch(IOException e) {
194        LOG.error("Failed to report to name-node.", e);
195      }
196    }
197    // Stop the RPC client
198    if (namenode != null) {
199      RPC.stopProxy(namenode);
200    }
201    namenode = null;
202    // Stop the checkpoint manager
203    if(checkpointManager != null) {
204      checkpointManager.interrupt();
205      checkpointManager = null;
206    }
207
208    // Abort current log segment - otherwise the NN shutdown code
209    // will close it gracefully, which is incorrect.
210    getFSImage().getEditLog().abortCurrentLogSegment();
211
212    // Stop name-node threads
213    super.stop();
214  }

235    /** 
236     * Verifies a journal request
237     */
238    private void verifyJournalRequest(JournalInfo journalInfo)
239        throws IOException {
240      verifyLayoutVersion(journalInfo.getLayoutVersion());
241      String errorMsg = null;
242      int expectedNamespaceID = namesystem.getNamespaceInfo().getNamespaceID();
243      if (journalInfo.getNamespaceId() != expectedNamespaceID) {
244        errorMsg = "Invalid namespaceID in journal request - expected " + expectedNamespaceID
245            + " actual " + journalInfo.getNamespaceId();
246        LOG.warn(errorMsg);
247        throw new UnregisteredNodeException(journalInfo);
248      } 
249      if (!journalInfo.getClusterId().equals(namesystem.getClusterId())) {
250        errorMsg = "Invalid clusterId in journal request - expected "
251            + journalInfo.getClusterId() + " actual " + namesystem.getClusterId();
252        LOG.warn(errorMsg);
253        throw new UnregisteredNodeException(journalInfo);
254      }
255    }

292  boolean shouldCheckpointAtStartup() {
293    FSImage fsImage = getFSImage();
294    if(isRole(NamenodeRole.CHECKPOINT)) {
295      assert fsImage.getStorage().getNumStorageDirs() > 0;
296      return ! fsImage.getStorage().getStorageDir(0).getVersionFile().exists();
297    }
298    
299    // BN always checkpoints on startup in order to get in sync with namespace
300    return true;
301  }

303  private NamespaceInfo handshake(Configuration conf) throws IOException {
304    // connect to name node
305    InetSocketAddress nnAddress = NameNode.getServiceAddress(conf, true);
306    this.namenode = NameNodeProxies.createNonHAProxy(conf, nnAddress,
307        NamenodeProtocol.class, UserGroupInformation.getCurrentUser(),
308        true).getProxy();
309    this.nnRpcAddress = NetUtils.getHostPortString(nnAddress);
310    this.nnHttpAddress = DFSUtil.getInfoServer(nnAddress, conf,
311        DFSUtil.getHttpClientScheme(conf)).toURL();
312    // get version and id info from the name-node
313    NamespaceInfo nsInfo = null;
314    while(!isStopRequested()) {
315      try {
316        nsInfo = handshake(namenode);
317        break;
318      } catch(SocketTimeoutException e) {  // name-node is busy
319        LOG.info("Problem connecting to server: " + nnAddress);
320        try {
321          Thread.sleep(1000);
322        } catch (InterruptedException ie) {
323          LOG.warn("Encountered exception ", e);
324        }
325      }
326    }
327    return nsInfo;
328  }

347  /**
348   * Register this backup node with the active name-node.
349   * @param nsInfo namespace information
350   * @throws IOException
351   */
352  private void registerWith(NamespaceInfo nsInfo) throws IOException {
353    BackupImage bnImage = (BackupImage)getFSImage();
354    NNStorage storage = bnImage.getStorage();
355    // verify namespaceID
356    if (storage.getNamespaceID() == 0) { // new backup storage
357      storage.setStorageInfo(nsInfo);
358      storage.setBlockPoolID(nsInfo.getBlockPoolID());
359      storage.setClusterID(nsInfo.getClusterID());
360    } else {
361      nsInfo.validateStorage(storage);
362    }
363    bnImage.initEditLog(StartupOption.REGULAR);
364    setRegistration();
365    NamenodeRegistration nnReg = null;
366    while(!isStopRequested()) {
367      try {
368        nnReg = namenode.registerSubordinateNamenode(getRegistration());
369        break;
370      } catch(SocketTimeoutException e) {  // name-node is busy
371        LOG.info("Problem connecting to name-node: " + nnRpcAddress);
372        try {
373          Thread.sleep(1000);
374        } catch (InterruptedException ie) {
375          LOG.warn("Encountered exception ", e);
376        }
377      }
378    }
379
380    String msg = null;
381    if(nnReg == null) // consider as a rejection
382      msg = "Registration rejected by " + nnRpcAddress;
383    else if(!nnReg.isRole(NamenodeRole.NAMENODE)) {
384      msg = "Name-node " + nnRpcAddress + " is not active";
385    }
386    if(msg != null) {
387      msg += ". Shutting down.";
388      LOG.error(msg);
389      throw new IOException(msg); // stop the node
390    }
391    nnRpcAddress = nnReg.getAddress();
392  }

394  // TODO: move to a common with DataNode util class
395  private static NamespaceInfo handshake(NamenodeProtocol namenode)
396  throws IOException, SocketTimeoutException {
397    NamespaceInfo nsInfo;
398    nsInfo = namenode.versionRequest();  // throws SocketTimeoutException 
399    String errorMsg = null;
400    // verify build version
401    if( ! nsInfo.getBuildVersion().equals( Storage.getBuildVersion())) {
402      errorMsg = "Incompatible build versions: active name-node BV = " 
403        + nsInfo.getBuildVersion() + "; backup node BV = "
404        + Storage.getBuildVersion();
405      LOG.error(errorMsg);
406      throw new IOException(errorMsg);
407    }
408    assert HdfsConstants.NAMENODE_LAYOUT_VERSION == nsInfo.getLayoutVersion() :
409      "Active and backup node layout versions must be the same. Expected: "
410      + HdfsConstants.NAMENODE_LAYOUT_VERSION + " actual "+ nsInfo.getLayoutVersion();
411    return nsInfo;
412  }

430    @Override // NameNodeHAContext
431    public void checkOperation(OperationCategory op)
432        throws StandbyException {
433      if (op == OperationCategory.UNCHECKED ||
434          op == OperationCategory.CHECKPOINT) {
435        return;
436      }
437      if (OperationCategory.JOURNAL != op &&
438          !(OperationCategory.READ == op && !isRole(NamenodeRole.CHECKPOINT))) {
439        String msg = "Operation category " + op
440            + " is not supported at " + getRole();
441        throw new StandbyException(msg);
442      }
443    }
