151  void format(FSNamesystem fsn, String clusterId) throws IOException {
152    long fileCount = fsn.getTotalFiles();
153    // Expect 1 file, which is the root inode
154    Preconditions.checkState(fileCount == 1,
155        "FSImage.format should be called with an uninitialized namesystem, has " +
156        fileCount + " files");
157    NamespaceInfo ns = NNStorage.newNamespaceInfo();
158    LOG.info("Allocated new BlockPoolId: " + ns.getBlockPoolID());
159    ns.clusterID = clusterId;
160    
161    storage.format(ns);
162    editLog.formatNonFileJournals(ns);
163    saveFSImageInAllDirs(fsn, 0);
164  }

187  /**
188   * Analyze storage directories.
189   * Recover from previous transitions if required. 
190   * Perform fs state transition if necessary depending on the namespace info.
191   * Read storage info. 
192   * 
193   * @throws IOException
194   * @return true if the image needs to be saved or false otherwise
195   */
196  boolean recoverTransitionRead(StartupOption startOpt, FSNamesystem target,
197      MetaRecoveryContext recovery)
198      throws IOException {
199    assert startOpt != StartupOption.FORMAT : 
200      "NameNode formatting should be performed before reading the image";
201    
202    Collection<URI> imageDirs = storage.getImageDirectories();
203    Collection<URI> editsDirs = editLog.getEditURIs();
204
205    // none of the data dirs exist
206    if((imageDirs.size() == 0 || editsDirs.size() == 0) 
207                             && startOpt != StartupOption.IMPORT)  
208      throw new IOException(
209          "All specified directories are not accessible or do not exist.");
210    
211    // 1. For each data directory calculate its state and 
212    // check whether all is consistent before transitioning.
213    Map<StorageDirectory, StorageState> dataDirStates = 
214             new HashMap<StorageDirectory, StorageState>();
215    boolean isFormatted = recoverStorageDirs(startOpt, dataDirStates);
216
217    if (LOG.isTraceEnabled()) {
218      LOG.trace("Data dir states:\n  " +
219        Joiner.on("\n  ").withKeyValueSeparator(": ")
220        .join(dataDirStates));
221    }
222    
223    if (!isFormatted && startOpt != StartupOption.ROLLBACK 
224                     && startOpt != StartupOption.IMPORT) {
225      throw new IOException("NameNode is not formatted.");      
226    }
227
228
229    int layoutVersion = storage.getLayoutVersion();
230    if (startOpt == StartupOption.METADATAVERSION) {
231      System.out.println("HDFS Image Version: " + layoutVersion);
232      System.out.println("Software format version: " +
233        HdfsConstants.NAMENODE_LAYOUT_VERSION);
234      return false;
235    }
236
237    if (layoutVersion < Storage.LAST_PRE_UPGRADE_LAYOUT_VERSION) {
238      NNStorage.checkVersionUpgradable(storage.getLayoutVersion());
239    }
240    if (startOpt != StartupOption.UPGRADE
241        && startOpt != StartupOption.UPGRADEONLY
242        && !RollingUpgradeStartupOption.STARTED.matches(startOpt)
243        && layoutVersion < Storage.LAST_PRE_UPGRADE_LAYOUT_VERSION
244        && layoutVersion != HdfsConstants.NAMENODE_LAYOUT_VERSION) {
245      throw new IOException(
246          "\nFile system image contains an old layout version " 
247          + storage.getLayoutVersion() + ".\nAn upgrade to version "
248          + HdfsConstants.NAMENODE_LAYOUT_VERSION + " is required.\n"
249          + "Please restart NameNode with the \""
250          + RollingUpgradeStartupOption.STARTED.getOptionString()
251          + "\" option if a rolling upgrade is already started;"
252          + " or restart NameNode with the \""
253          + StartupOption.UPGRADE.getName() + "\" option to start"
254          + " a new upgrade.");
255    }
256    
257    storage.processStartupOptionsForUpgrade(startOpt, layoutVersion);
258
259    // 2. Format unformatted dirs.
260    for (Iterator<StorageDirectory> it = storage.dirIterator(); it.hasNext();) {
261      StorageDirectory sd = it.next();
262      StorageState curState = dataDirStates.get(sd);
263      switch(curState) {
264      case NON_EXISTENT:
265        throw new IOException(StorageState.NON_EXISTENT + 
266                              " state cannot be here");
267      case NOT_FORMATTED:
268        LOG.info("Storage directory " + sd.getRoot() + " is not formatted.");
269        LOG.info("Formatting ...");
270        sd.clearDirectory(); // create empty currrent dir
271        break;
272      default:
273        break;
274      }
275    }
276
277    // 3. Do transitions
278    switch(startOpt) {
279    case UPGRADE:
280    case UPGRADEONLY:
281      doUpgrade(target);
282      return false; // upgrade saved image already
283    case IMPORT:
284      doImportCheckpoint(target);
285      return false; // import checkpoint saved image already
286    case ROLLBACK:
287      throw new AssertionError("Rollback is now a standalone command, "
288          + "NameNode should not be starting with this option.");
289    case REGULAR:
290    default:
291      // just load the image
292    }
293    
294    return loadFSImage(target, startOpt, recovery);
295  }

297  /**
298   * For each storage directory, performs recovery of incomplete transitions
299   * (eg. upgrade, rollback, checkpoint) and inserts the directory's storage
300   * state into the dataDirStates map.
301   * @param dataDirStates output of storage directory states
302   * @return true if there is at least one valid formatted storage directory
303   */
304  private boolean recoverStorageDirs(StartupOption startOpt,
305      Map<StorageDirectory, StorageState> dataDirStates) throws IOException {
306    boolean isFormatted = false;
307    // This loop needs to be over all storage dirs, even shared dirs, to make
308    // sure that we properly examine their state, but we make sure we don't
309    // mutate the shared dir below in the actual loop.
310    for (Iterator<StorageDirectory> it = 
311                      storage.dirIterator(); it.hasNext();) {
312      StorageDirectory sd = it.next();
313      StorageState curState;
314      if (startOpt == StartupOption.METADATAVERSION) {
315        /* All we need is the layout version. */
316        storage.readProperties(sd);
317        return true;
318      }
319
320      try {
321        curState = sd.analyzeStorage(startOpt, storage);
322        // sd is locked but not opened
323        switch(curState) {
324        case NON_EXISTENT:
325          // name-node fails if any of the configured storage dirs are missing
326          throw new InconsistentFSStateException(sd.getRoot(),
327                      "storage directory does not exist or is not accessible.");
328        case NOT_FORMATTED:
329          break;
330        case NORMAL:
331          break;
332        default:  // recovery is possible
333          sd.doRecover(curState);
334        }
335        if (curState != StorageState.NOT_FORMATTED 
336            && startOpt != StartupOption.ROLLBACK) {
337          // read and verify consistency with other directories
338          storage.readProperties(sd, startOpt);
339          isFormatted = true;
340        }
341        if (startOpt == StartupOption.IMPORT && isFormatted)
342          // import of a checkpoint is allowed only into empty image directories
343          throw new IOException("Cannot import image from a checkpoint. " 
344              + " NameNode already contains an image in " + sd.getRoot());
345      } catch (IOException ioe) {
346        sd.unlock();
347        throw ioe;
348      }
349      dataDirStates.put(sd,curState);
350    }
351    return isFormatted;
352  }

354  /** Check if upgrade is in progress. */
355  void checkUpgrade(FSNamesystem target) throws IOException {
356    // Upgrade or rolling upgrade is allowed only if there are 
357    // no previous fs states in any of the directories
358    for (Iterator<StorageDirectory> it = storage.dirIterator(false); it.hasNext();) {
359      StorageDirectory sd = it.next();
360      if (sd.getPreviousDir().exists())
361        throw new InconsistentFSStateException(sd.getRoot(),
362            "previous fs state should not exist during upgrade. "
363            + "Finalize or rollback first.");
364    }
365  }

383  void doUpgrade(FSNamesystem target) throws IOException {
384    checkUpgrade(target);
385
386    // load the latest image
387    this.loadFSImage(target, StartupOption.UPGRADE, null);
388
389    // Do upgrade for each directory
390    target.checkRollingUpgrade("upgrade namenode");
391    
392    long oldCTime = storage.getCTime();
393    storage.cTime = now();  // generate new cTime for the state
394    int oldLV = storage.getLayoutVersion();
395    storage.layoutVersion = HdfsConstants.NAMENODE_LAYOUT_VERSION;
396    
397    List<StorageDirectory> errorSDs =
398      Collections.synchronizedList(new ArrayList<StorageDirectory>());
399    assert !editLog.isSegmentOpen() : "Edits log must not be open.";
400    LOG.info("Starting upgrade of local storage directories."
401        + "\n   old LV = " + oldLV
402        + "; old CTime = " + oldCTime
403        + ".\n   new LV = " + storage.getLayoutVersion()
404        + "; new CTime = " + storage.getCTime());
405    // Do upgrade for each directory
406    for (Iterator<StorageDirectory> it = storage.dirIterator(false); it.hasNext();) {
407      StorageDirectory sd = it.next();
408      try {
409        NNUpgradeUtil.doPreUpgrade(conf, sd);
410      } catch (Exception e) {
411        LOG.error("Failed to move aside pre-upgrade storage " +
412            "in image directory " + sd.getRoot(), e);
413        errorSDs.add(sd);
414        continue;
415      }
416    }
417    if (target.isHaEnabled()) {
418      editLog.doPreUpgradeOfSharedLog();
419    }
420    storage.reportErrorsOnDirectories(errorSDs);
421    errorSDs.clear();
422
423    saveFSImageInAllDirs(target, editLog.getLastWrittenTxId());
424
425    // upgrade shared edit storage first
426    if (target.isHaEnabled()) {
427      editLog.doUpgradeOfSharedLog();
428    }
429    for (Iterator<StorageDirectory> it = storage.dirIterator(false); it.hasNext();) {
430      StorageDirectory sd = it.next();
431      try {
432        NNUpgradeUtil.doUpgrade(sd, storage);
433      } catch (IOException ioe) {
434        errorSDs.add(sd);
435        continue;
436      }
437    }
438    storage.reportErrorsOnDirectories(errorSDs);
439    
440    isUpgradeFinalized = false;
441    if (!storage.getRemovedStorageDirs().isEmpty()) {
442      // during upgrade, it's a fatal error to fail any storage directory
443      throw new IOException("Upgrade failed in "
444          + storage.getRemovedStorageDirs().size()
445          + " storage directory(ies), previously logged.");
446    }
447  }

449  void doRollback(FSNamesystem fsns) throws IOException {
450    // Rollback is allowed only if there is 
451    // a previous fs states in at least one of the storage directories.
452    // Directories that don't have previous state do not rollback
453    boolean canRollback = false;
454    FSImage prevState = new FSImage(conf);
455    try {
456      prevState.getStorage().layoutVersion = HdfsConstants.NAMENODE_LAYOUT_VERSION;
457      for (Iterator<StorageDirectory> it = storage.dirIterator(false); it.hasNext();) {
458        StorageDirectory sd = it.next();
459        if (!NNUpgradeUtil.canRollBack(sd, storage, prevState.getStorage(),
460            HdfsConstants.NAMENODE_LAYOUT_VERSION)) {
461          continue;
462        }
463        LOG.info("Can perform rollback for " + sd);
464        canRollback = true;
465      }
466      
467      if (fsns.isHaEnabled()) {
468        // If HA is enabled, check if the shared log can be rolled back as well.
469        editLog.initJournalsForWrite();
470        boolean canRollBackSharedEditLog = editLog.canRollBackSharedLog(
471            prevState.getStorage(), HdfsConstants.NAMENODE_LAYOUT_VERSION);
472        if (canRollBackSharedEditLog) {
473          LOG.info("Can perform rollback for shared edit log.");
474          canRollback = true;
475        }
476      }
477      
478      if (!canRollback)
479        throw new IOException("Cannot rollback. None of the storage "
480            + "directories contain previous fs state.");
481  
482      // Now that we know all directories are going to be consistent
483      // Do rollback for each directory containing previous state
484      for (Iterator<StorageDirectory> it = storage.dirIterator(false); it.hasNext();) {
485        StorageDirectory sd = it.next();
486        LOG.info("Rolling back storage directory " + sd.getRoot()
487                 + ".\n   new LV = " + prevState.getStorage().getLayoutVersion()
488                 + "; new CTime = " + prevState.getStorage().getCTime());
489        NNUpgradeUtil.doRollBack(sd);
490      }
491      if (fsns.isHaEnabled()) {
492        // If HA is enabled, try to roll back the shared log as well.
493        editLog.doRollback();
494      }
495      
496      isUpgradeFinalized = true;
497    } finally {
498      prevState.close();
499    }
500  }

502  /**
503   * Load image from a checkpoint directory and save it into the current one.
504   * @param target the NameSystem to import into
505   * @throws IOException
506   */
507  void doImportCheckpoint(FSNamesystem target) throws IOException {
508    Collection<URI> checkpointDirs =
509      FSImage.getCheckpointDirs(conf, null);
510    List<URI> checkpointEditsDirs =
511      FSImage.getCheckpointEditsDirs(conf, null);
512
513    if (checkpointDirs == null || checkpointDirs.isEmpty()) {
514      throw new IOException("Cannot import image from a checkpoint. "
515                            + "\"dfs.namenode.checkpoint.dir\" is not set." );
516    }
517    
518    if (checkpointEditsDirs == null || checkpointEditsDirs.isEmpty()) {
519      throw new IOException("Cannot import image from a checkpoint. "
520                            + "\"dfs.namenode.checkpoint.dir\" is not set." );
521    }
522
523    FSImage realImage = target.getFSImage();
524    FSImage ckptImage = new FSImage(conf, 
525                                    checkpointDirs, checkpointEditsDirs);
526    // load from the checkpoint dirs
527    try {
528      ckptImage.recoverTransitionRead(StartupOption.REGULAR, target, null);
529    } finally {
530      ckptImage.close();
531    }
532    // return back the real image
533    realImage.getStorage().setStorageInfo(ckptImage.getStorage());
534    realImage.getEditLog().setNextTxId(ckptImage.getEditLog().getLastWrittenTxId()+1);
535    realImage.initEditLog(StartupOption.IMPORT);
536
537    realImage.getStorage().setBlockPoolID(ckptImage.getBlockPoolID());
538
539    // and save it but keep the same checkpointTime
540    saveNamespace(target);
541    getStorage().writeAll();
542  }

544  void finalizeUpgrade(boolean finalizeEditLog) throws IOException {
545    LOG.info("Finalizing upgrade for local dirs. " +
546        (storage.getLayoutVersion() == 0 ? "" : 
547          "\n   cur LV = " + storage.getLayoutVersion()
548          + "; cur CTime = " + storage.getCTime()));
549    for (Iterator<StorageDirectory> it = storage.dirIterator(false); it.hasNext();) {
550      StorageDirectory sd = it.next();
551      NNUpgradeUtil.doFinalize(sd);
552    }
553    if (finalizeEditLog) {
554      // We only do this in the case that HA is enabled and we're active. In any
555      // other case the NN will have done the upgrade of the edits directories
556      // already by virtue of the fact that they're local.
557      editLog.doFinalizeOfSharedLog();
558    }
559    isUpgradeFinalized = true;
560  }

586  /**
587   * Choose latest image from one of the directories,
588   * load it and merge with the edits.
589   * 
590   * Saving and loading fsimage should never trigger symlink resolution. 
591   * The paths that are persisted do not have *intermediate* symlinks 
592   * because intermediate symlinks are resolved at the time files, 
593   * directories, and symlinks are created. All paths accessed while 
594   * loading or saving fsimage should therefore only see symlinks as 
595   * the final path component, and the functions called below do not
596   * resolve symlinks that are the final path component.
597   *
598   * @return whether the image should be saved
599   * @throws IOException
600   */
601  private boolean loadFSImage(FSNamesystem target, StartupOption startOpt,
602      MetaRecoveryContext recovery)
603      throws IOException {
604    final boolean rollingRollback
605        = RollingUpgradeStartupOption.ROLLBACK.matches(startOpt);
606    final EnumSet<NameNodeFile> nnfs;
607    if (rollingRollback) {
608      // if it is rollback of rolling upgrade, only load from the rollback image
609      nnfs = EnumSet.of(NameNodeFile.IMAGE_ROLLBACK);
610    } else {
611      // otherwise we can load from both IMAGE and IMAGE_ROLLBACK
612      nnfs = EnumSet.of(NameNodeFile.IMAGE, NameNodeFile.IMAGE_ROLLBACK);
613    }
614    final FSImageStorageInspector inspector = storage
615        .readAndInspectDirs(nnfs, startOpt);
616
617    isUpgradeFinalized = inspector.isUpgradeFinalized();
618    List<FSImageFile> imageFiles = inspector.getLatestImages();
619
620    StartupProgress prog = NameNode.getStartupProgress();
621    prog.beginPhase(Phase.LOADING_FSIMAGE);
622    File phaseFile = imageFiles.get(0).getFile();
623    prog.setFile(Phase.LOADING_FSIMAGE, phaseFile.getAbsolutePath());
624    prog.setSize(Phase.LOADING_FSIMAGE, phaseFile.length());
625    boolean needToSave = inspector.needToSave();
626
627    Iterable<EditLogInputStream> editStreams = null;
628
629    initEditLog(startOpt);
630
631    if (NameNodeLayoutVersion.supports(
632        LayoutVersion.Feature.TXID_BASED_LAYOUT, getLayoutVersion())) {
633      // If we're open for write, we're either non-HA or we're the active NN, so
634      // we better be able to load all the edits. If we're the standby NN, it's
635      // OK to not be able to read all of edits right now.
636      // In the meanwhile, for HA upgrade, we will still write editlog thus need
637      // this toAtLeastTxId to be set to the max-seen txid
638      // For rollback in rolling upgrade, we need to set the toAtLeastTxId to
639      // the txid right before the upgrade marker.  
640      long toAtLeastTxId = editLog.isOpenForWrite() ? inspector
641          .getMaxSeenTxId() : 0;
642      if (rollingRollback) {
643        // note that the first image in imageFiles is the special checkpoint
644        // for the rolling upgrade
645        toAtLeastTxId = imageFiles.get(0).getCheckpointTxId() + 2;
646      }
647      editStreams = editLog.selectInputStreams(
648          imageFiles.get(0).getCheckpointTxId() + 1,
649          toAtLeastTxId, recovery, false);
650    } else {
651      editStreams = FSImagePreTransactionalStorageInspector
652        .getEditLogStreams(storage);
653    }
654    int maxOpSize = conf.getInt(DFSConfigKeys.DFS_NAMENODE_MAX_OP_SIZE_KEY,
655        DFSConfigKeys.DFS_NAMENODE_MAX_OP_SIZE_DEFAULT);
656    for (EditLogInputStream elis : editStreams) {
657      elis.setMaxOpSize(maxOpSize);
658    }
659 
660    for (EditLogInputStream l : editStreams) {
661      LOG.debug("Planning to load edit log stream: " + l);
662    }
663    if (!editStreams.iterator().hasNext()) {
664      LOG.info("No edit log streams selected.");
665    }
666    
667    FSImageFile imageFile = null;
668    for (int i = 0; i < imageFiles.size(); i++) {
669      try {
670        imageFile = imageFiles.get(i);
671        loadFSImageFile(target, recovery, imageFile, startOpt);
672        break;
673      } catch (IOException ioe) {
674        LOG.error("Failed to load image from " + imageFile, ioe);
675        target.clear();
676        imageFile = null;
677      }
678    }
679    // Failed to load any images, error out
680    if (imageFile == null) {
681      FSEditLog.closeAllStreams(editStreams);
682      throw new IOException("Failed to load an FSImage file!");
683    }
684    prog.endPhase(Phase.LOADING_FSIMAGE);
685    
686    if (!rollingRollback) {
687      long txnsAdvanced = loadEdits(editStreams, target, startOpt, recovery);
688      needToSave |= needsResaveBasedOnStaleCheckpoint(imageFile.getFile(),
689          txnsAdvanced);
690      if (RollingUpgradeStartupOption.DOWNGRADE.matches(startOpt)) {
691        // rename rollback image if it is downgrade
692        renameCheckpoint(NameNodeFile.IMAGE_ROLLBACK, NameNodeFile.IMAGE);
693      }
694    } else {
695      // Trigger the rollback for rolling upgrade. Here lastAppliedTxId equals
696      // to the last txid in rollback fsimage.
697      rollingRollback(lastAppliedTxId + 1, imageFiles.get(0).getCheckpointTxId());
698      needToSave = false;
699    }
700    editLog.setNextTxId(lastAppliedTxId + 1);
701    return needToSave;
702  }
703
704  /** rollback for rolling upgrade. */
705  private void rollingRollback(long discardSegmentTxId, long ckptId)
706      throws IOException {
707    // discard discard unnecessary editlog segments starting from the given id
708    this.editLog.discardSegments(discardSegmentTxId);
709    // rename the special checkpoint
710    renameCheckpoint(ckptId, NameNodeFile.IMAGE_ROLLBACK, NameNodeFile.IMAGE,
711        true);
712    // purge all the checkpoints after the marker
713    archivalManager.purgeCheckpoinsAfter(NameNodeFile.IMAGE, ckptId);
714    String nameserviceId = DFSUtil.getNamenodeNameServiceId(conf);
715    if (HAUtil.isHAEnabled(conf, nameserviceId)) {
716      // close the editlog since it is currently open for write
717      this.editLog.close();
718      // reopen the editlog for read
719      this.editLog.initSharedJournalsForRead();
720    }
721  }

723  void loadFSImageFile(FSNamesystem target, MetaRecoveryContext recovery,
724      FSImageFile imageFile, StartupOption startupOption) throws IOException {
725    LOG.debug("Planning to load image :\n" + imageFile);
726    StorageDirectory sdForProperties = imageFile.sd;
727    storage.readProperties(sdForProperties, startupOption);
728
729    if (NameNodeLayoutVersion.supports(
730        LayoutVersion.Feature.TXID_BASED_LAYOUT, getLayoutVersion())) {
731      // For txid-based layout, we should have a .md5 file
732      // next to the image file
733      boolean isRollingRollback = RollingUpgradeStartupOption.ROLLBACK
734          .matches(startupOption);
735      loadFSImage(imageFile.getFile(), target, recovery, isRollingRollback);
736    } else if (NameNodeLayoutVersion.supports(
737        LayoutVersion.Feature.FSIMAGE_CHECKSUM, getLayoutVersion())) {
738      // In 0.22, we have the checksum stored in the VERSION file.
739      String md5 = storage.getDeprecatedProperty(
740          NNStorage.DEPRECATED_MESSAGE_DIGEST_PROPERTY);
741      if (md5 == null) {
742        throw new InconsistentFSStateException(sdForProperties.getRoot(),
743            "Message digest property " +
744            NNStorage.DEPRECATED_MESSAGE_DIGEST_PROPERTY +
745            " not set for storage directory " + sdForProperties.getRoot());
746      }
747      loadFSImage(imageFile.getFile(), new MD5Hash(md5), target, recovery,
748          false);
749    } else {
750      // We don't have any record of the md5sum
751      loadFSImage(imageFile.getFile(), null, target, recovery, false);
752    }
753  }

755  public void initEditLog(StartupOption startOpt) throws IOException {
756    Preconditions.checkState(getNamespaceID() != 0,
757        "Must know namespace ID before initting edit log");
758    String nameserviceId = DFSUtil.getNamenodeNameServiceId(conf);
759    if (!HAUtil.isHAEnabled(conf, nameserviceId)) {
760      // If this NN is not HA
761      editLog.initJournalsForWrite();
762      editLog.recoverUnclosedStreams();
763    } else if (HAUtil.isHAEnabled(conf, nameserviceId)
764        && (startOpt == StartupOption.UPGRADE
765            || startOpt == StartupOption.UPGRADEONLY
766            || RollingUpgradeStartupOption.ROLLBACK.matches(startOpt))) {
767      // This NN is HA, but we're doing an upgrade or a rollback of rolling
768      // upgrade so init the edit log for write.
769      editLog.initJournalsForWrite();
770      if (startOpt == StartupOption.UPGRADE
771          || startOpt == StartupOption.UPGRADEONLY) {
772        long sharedLogCTime = editLog.getSharedLogCTime();
773        if (this.storage.getCTime() < sharedLogCTime) {
774          throw new IOException("It looks like the shared log is already " +
775              "being upgraded but this NN has not been upgraded yet. You " +
776              "should restart this NameNode with the '" +
777              StartupOption.BOOTSTRAPSTANDBY.getName() + "' option to bring " +
778              "this NN in sync with the other.");
779        }
780      }
781      editLog.recoverUnclosedStreams();
782    } else {
783      // This NN is HA and we're not doing an upgrade.
784      editLog.initSharedJournalsForRead();
785    }
786  }

788  /**
789   * @param imageFile the image file that was loaded
790   * @param numEditsLoaded the number of edits loaded from edits logs
791   * @return true if the NameNode should automatically save the namespace
792   * when it is started, due to the latest checkpoint being too old.
793   */
794  private boolean needsResaveBasedOnStaleCheckpoint(
795      File imageFile, long numEditsLoaded) {
796    final long checkpointPeriod = conf.getLong(
797        DFSConfigKeys.DFS_NAMENODE_CHECKPOINT_PERIOD_KEY, 
798        DFSConfigKeys.DFS_NAMENODE_CHECKPOINT_PERIOD_DEFAULT);
799    final long checkpointTxnCount = conf.getLong(
800        DFSConfigKeys.DFS_NAMENODE_CHECKPOINT_TXNS_KEY, 
801        DFSConfigKeys.DFS_NAMENODE_CHECKPOINT_TXNS_DEFAULT);
802    long checkpointAge = Time.now() - imageFile.lastModified();
803
804    return (checkpointAge > checkpointPeriod * 1000) ||
805           (numEditsLoaded > checkpointTxnCount);
806  }

816  private long loadEdits(Iterable<EditLogInputStream> editStreams,
817      FSNamesystem target, StartupOption startOpt, MetaRecoveryContext recovery)
818      throws IOException {
819    LOG.debug("About to load edits:\n  " + Joiner.on("\n  ").join(editStreams));
820    StartupProgress prog = NameNode.getStartupProgress();
821    prog.beginPhase(Phase.LOADING_EDITS);
822    
823    long prevLastAppliedTxId = lastAppliedTxId;  
824    try {    
825      FSEditLogLoader loader = new FSEditLogLoader(target, lastAppliedTxId);
826      
827      // Load latest edits
828      for (EditLogInputStream editIn : editStreams) {
829        LOG.info("Reading " + editIn + " expecting start txid #" +
830              (lastAppliedTxId + 1));
831        try {
832          loader.loadFSEdits(editIn, lastAppliedTxId + 1, startOpt, recovery);
833        } finally {
834          // Update lastAppliedTxId even in case of error, since some ops may
835          // have been successfully applied before the error.
836          lastAppliedTxId = loader.getLastAppliedTxId();
837        }
838        // If we are in recovery mode, we may have skipped over some txids.
839        if (editIn.getLastTxId() != HdfsConstants.INVALID_TXID) {
840          lastAppliedTxId = editIn.getLastTxId();
841        }
842      }
843    } finally {
844      FSEditLog.closeAllStreams(editStreams);
845      // update the counts
846      updateCountForQuota(target.getBlockManager().getStoragePolicySuite(),
847          target.dir.rootDir);
848    }
849    prog.endPhase(Phase.LOADING_EDITS);
850    return lastAppliedTxId - prevLastAppliedTxId;
851  }

853  /**
854   * Update the count of each directory with quota in the namespace.
855   * A directory's count is defined as the total number inodes in the tree
856   * rooted at the directory.
857   * 
858   * This is an update of existing state of the filesystem and does not
859   * throw QuotaExceededException.
860   */
861  static void updateCountForQuota(BlockStoragePolicySuite bsps,
862                                  INodeDirectory root) {
863    updateCountForQuotaRecursively(bsps, root.getStoragePolicyID(), root,
864        new QuotaCounts.Builder().build());
865 }
866
867  private static void updateCountForQuotaRecursively(BlockStoragePolicySuite bsps,
868      byte blockStoragePolicyId, INodeDirectory dir, QuotaCounts counts) {
869    final long parentNamespace = counts.getNameSpace();
870    final long parentStoragespace = counts.getStorageSpace();
871    final EnumCounters<StorageType> parentTypeSpaces = counts.getTypeSpaces();
872
873    dir.computeQuotaUsage4CurrentDirectory(bsps, blockStoragePolicyId, counts);
874    
875    for (INode child : dir.getChildrenList(Snapshot.CURRENT_STATE_ID)) {
876      final byte childPolicyId = child.getStoragePolicyIDForQuota(blockStoragePolicyId);
877      if (child.isDirectory()) {
878        updateCountForQuotaRecursively(bsps, childPolicyId,
879            child.asDirectory(), counts);
880      } else {
881        // file or symlink: count here to reduce recursive calls.
882        child.computeQuotaUsage(bsps, childPolicyId, counts, false,
883            Snapshot.CURRENT_STATE_ID);
884      }
885    }
886      
887    if (dir.isQuotaSet()) {
888      // check if quota is violated. It indicates a software bug.
889      final QuotaCounts q = dir.getQuotaCounts();
890
891      final long namespace = counts.getNameSpace() - parentNamespace;
892      final long nsQuota = q.getNameSpace();
893      if (Quota.isViolated(nsQuota, namespace)) {
894        LOG.warn("Namespace quota violation in image for "
895            + dir.getFullPathName()
896            + " quota = " + nsQuota + " < consumed = " + namespace);
897      }
898
899      final long ssConsumed = counts.getStorageSpace() - parentStoragespace;
900      final long ssQuota = q.getStorageSpace();
901      if (Quota.isViolated(ssQuota, ssConsumed)) {
902        LOG.warn("Storagespace quota violation in image for "
903            + dir.getFullPathName()
904            + " quota = " + ssQuota + " < consumed = " + ssConsumed);
905      }
906
907      final EnumCounters<StorageType> typeSpaces =
908          new EnumCounters<StorageType>(StorageType.class);
909      for (StorageType t : StorageType.getTypesSupportingQuota()) {
910        final long typeSpace = counts.getTypeSpaces().get(t) -
911            parentTypeSpaces.get(t);
912        final long typeQuota = q.getTypeSpaces().get(t);
913        if (Quota.isViolated(typeQuota, typeSpace)) {
914          LOG.warn("Storage type quota violation in image for "
915              + dir.getFullPathName()
916              + " type = " + t.toString() + " quota = "
917              + typeQuota + " < consumed " + typeSpace);
918        }
919      }
920
921      dir.getDirectoryWithQuotaFeature().setSpaceConsumed(namespace, ssConsumed,
922          typeSpaces);
923    }
924  }

927   * Load the image namespace from the given image file, verifying
928   * it against the MD5 sum stored in its associated .md5 file.
929   */
930  private void loadFSImage(File imageFile, FSNamesystem target,
931      MetaRecoveryContext recovery, boolean requireSameLayoutVersion)
932      throws IOException {
933    MD5Hash expectedMD5 = MD5FileUtils.readStoredMd5ForFile(imageFile);
934    if (expectedMD5 == null) {
935      throw new IOException("No MD5 file found corresponding to image file "
936          + imageFile);
937    }
938    loadFSImage(imageFile, expectedMD5, target, recovery,

942  /**
943   * Load in the filesystem image from file. It's a big list of
944   * filenames and blocks.
945   */
946  private void loadFSImage(File curFile, MD5Hash expectedMd5,
947      FSNamesystem target, MetaRecoveryContext recovery,
948      boolean requireSameLayoutVersion) throws IOException {
949    // BlockPoolId is required when the FsImageLoader loads the rolling upgrade
950    // information. Make sure the ID is properly set.
951    target.setBlockPoolId(this.getBlockPoolID());
952
953    FSImageFormat.LoaderDelegator loader = FSImageFormat.newLoader(conf, target);
954    loader.load(curFile, requireSameLayoutVersion);
955
956    // Check that the image digest we loaded matches up with what
957    // we expected
958    MD5Hash readImageMd5 = loader.getLoadedImageMd5();
959    if (expectedMd5 != null &&
960        !expectedMd5.equals(readImageMd5)) {
961      throw new IOException("Image file " + curFile +
962          " is corrupt with MD5 checksum of " + readImageMd5 +
963          " but expecting " + expectedMd5);
964    }
965
966    long txId = loader.getLoadedImageTxId();
967    LOG.info("Loaded image for txid " + txId + " from " + curFile);
968    lastAppliedTxId = txId;
969    storage.setMostRecentCheckpointInfo(txId, curFile.lastModified());
970  }

1008  /**
1009   * FSImageSaver is being run in a separate thread when saving
1010   * FSImage. There is one thread per each copy of the image.
1011   *
1012   * FSImageSaver assumes that it was launched from a thread that holds
1013   * FSNamesystem lock and waits for the execution of FSImageSaver thread
1014   * to finish.
1015   * This way we are guaranteed that the namespace is not being updated
1016   * while multiple instances of FSImageSaver are traversing it
1017   * and writing it out.
1018   */
1019  private class FSImageSaver implements Runnable {
1020    private final SaveNamespaceContext context;
1021    private final StorageDirectory sd;
1022    private final NameNodeFile nnf;
1023
1024    public FSImageSaver(SaveNamespaceContext context, StorageDirectory sd,
1025        NameNodeFile nnf) {
1026      this.context = context;
1027      this.sd = sd;
1028      this.nnf = nnf;
1029    }
1030
1031    @Override
1032    public void run() {
1033      try {
1034        saveFSImage(context, sd, nnf);
1035      } catch (SaveNamespaceCancelledException snce) {
1036        LOG.info("Cancelled image saving for " + sd.getRoot() +
1037            ": " + snce.getMessage());
1038        // don't report an error on the storage dir!
1039      } catch (Throwable t) {
1040        LOG.error("Unable to save image for " + sd.getRoot(), t);
1041        context.reportErrorOnStorageDirectory(sd);
1042      }
1043    }
1044    
1045    @Override
1046    public String toString() {
1047      return "FSImageSaver for " + sd.getRoot() +
1048             " of type " + sd.getStorageDirType();
1049    }
1050  }
1051  
1052  private void waitForThreads(List<Thread> threads) {
1053    for (Thread thread : threads) {
1054      while (thread.isAlive()) {
1055        try {
1056          thread.join();
1057        } catch (InterruptedException iex) {
1058          LOG.error("Caught interrupted exception while waiting for thread " +
1059                    thread.getName() + " to finish. Retrying join");
1060        }        
1061      }
1062    }
1063  }

1080  /**
1081   * Save the contents of the FS image to a new image file in each of the
1082   * current storage directories.
1083   */
1084  public synchronized void saveNamespace(FSNamesystem source, NameNodeFile nnf,
1085      Canceler canceler) throws IOException {
1086    assert editLog != null : "editLog must be initialized";
1087    LOG.info("Save namespace ...");
1088    storage.attemptRestoreRemovedStorage();
1089
1090    boolean editLogWasOpen = editLog.isSegmentOpen();
1091    
1092    if (editLogWasOpen) {
1093      editLog.endCurrentLogSegment(true);
1094    }
1095    long imageTxId = getLastAppliedOrWrittenTxId();
1096    if (!addToCheckpointing(imageTxId)) {
1097      throw new IOException(
1098          "FS image is being downloaded from another NN at txid " + imageTxId);
1099    }
1100    try {
1101      try {
1102        saveFSImageInAllDirs(source, nnf, imageTxId, canceler);
1103        storage.writeAll();
1104      } finally {
1105        if (editLogWasOpen) {
1106          editLog.startLogSegment(imageTxId + 1, true);
1107          // Take this opportunity to note the current transaction.
1108          // Even if the namespace save was cancelled, this marker
1109          // is only used to determine what transaction ID is required
1110          // for startup. So, it doesn't hurt to update it unnecessarily.
1111          storage.writeTransactionIdFileToStorage(imageTxId + 1);
1112        }
1113      }
1114    } finally {
1115      removeFromCheckpointing(imageTxId);
1116    }
1117  }

1121   */
1122  protected synchronized void saveFSImageInAllDirs(FSNamesystem source, long txid)
1123      throws IOException {
1124    if (!addToCheckpointing(txid)) {
1125      throw new IOException(("FS image is being downloaded from another NN"));
1126    }
1127    try {
1128      saveFSImageInAllDirs(source, NameNodeFile.IMAGE, txid, null);
1129    } finally {
1130      removeFromCheckpointing(txid);
1131    }
1132  }

1142  private synchronized void saveFSImageInAllDirs(FSNamesystem source,
1143      NameNodeFile nnf, long txid, Canceler canceler) throws IOException {
1144    StartupProgress prog = NameNode.getStartupProgress();
1145    prog.beginPhase(Phase.SAVING_CHECKPOINT);
1146    if (storage.getNumStorageDirs(NameNodeDirType.IMAGE) == 0) {
1147      throw new IOException("No image directories available!");
1148    }
1149    if (canceler == null) {
1150      canceler = new Canceler();
1151    }
1152    SaveNamespaceContext ctx = new SaveNamespaceContext(
1153        source, txid, canceler);
1154    
1155    try {
1156      List<Thread> saveThreads = new ArrayList<Thread>();
1157      // save images into current
1158      for (Iterator<StorageDirectory> it
1159             = storage.dirIterator(NameNodeDirType.IMAGE); it.hasNext();) {
1160        StorageDirectory sd = it.next();
1161        FSImageSaver saver = new FSImageSaver(ctx, sd, nnf);
1162        Thread saveThread = new Thread(saver, saver.toString());
1163        saveThreads.add(saveThread);
1164        saveThread.start();
1165      }
1166      waitForThreads(saveThreads);
1167      saveThreads.clear();
1168      storage.reportErrorsOnDirectories(ctx.getErrorSDs());
1169  
1170      if (storage.getNumStorageDirs(NameNodeDirType.IMAGE) == 0) {
1171        throw new IOException(
1172          "Failed to save in any storage directories while saving namespace.");
1173      }
1174      if (canceler.isCancelled()) {
1175        deleteCancelledCheckpoint(txid);
1176        ctx.checkCancelled(); // throws
1177        assert false : "should have thrown above!";
1178      }
1179  
1180      renameCheckpoint(txid, NameNodeFile.IMAGE_NEW, nnf, false);
1181  
1182      // Since we now have a new checkpoint, we can clean up some
1183      // old edit logs and checkpoints.
1184      purgeOldStorage(nnf);
1185    } finally {
1186      // Notify any threads waiting on the checkpoint to be canceled
1187      // that it is complete.
1188      ctx.markComplete();
1189      ctx = null;
1190    }
1191    prog.endPhase(Phase.SAVING_CHECKPOINT);
1192  }

1227  /**
1228   * Rename all the fsimage files with the specific NameNodeFile type. The
1229   * associated checksum files will also be renamed.
1230   */
1231  void renameCheckpoint(NameNodeFile fromNnf, NameNodeFile toNnf)
1232      throws IOException {
1233    ArrayList<StorageDirectory> al = null;
1234    FSImageTransactionalStorageInspector inspector =
1235        new FSImageTransactionalStorageInspector(EnumSet.of(fromNnf));
1236    storage.inspectStorageDirs(inspector);
1237    for (FSImageFile image : inspector.getFoundImages()) {
1238      try {
1239        renameImageFileInDir(image.sd, fromNnf, toNnf, image.txId, true);
1240      } catch (IOException ioe) {
1241        LOG.warn("Unable to rename checkpoint in " + image.sd, ioe);
1242        if (al == null) {
1243          al = Lists.newArrayList();
1244        }
1245        al.add(image.sd);
1246      }
1247    }
1248    if(al != null) {
1249      storage.reportErrorsOnDirectories(al);
1250    }
1251  }

1253  /**
1254   * Deletes the checkpoint file in every storage directory,
1255   * since the checkpoint was cancelled.
1256   */
1257  private void deleteCancelledCheckpoint(long txid) throws IOException {
1258    ArrayList<StorageDirectory> al = Lists.newArrayList();
1259
1260    for (StorageDirectory sd : storage.dirIterable(NameNodeDirType.IMAGE)) {
1261      File ckpt = NNStorage.getStorageFile(sd, NameNodeFile.IMAGE_NEW, txid);
1262      if (ckpt.exists() && !ckpt.delete()) {
1263        LOG.warn("Unable to delete cancelled checkpoint in " + sd);
1264        al.add(sd);            
1265      }
1266    }
1267    storage.reportErrorsOnDirectories(al);
1268  }

1270  private void renameImageFileInDir(StorageDirectory sd, NameNodeFile fromNnf,
1271      NameNodeFile toNnf, long txid, boolean renameMD5) throws IOException {
1272    final File fromFile = NNStorage.getStorageFile(sd, fromNnf, txid);
1273    final File toFile = NNStorage.getStorageFile(sd, toNnf, txid);
1274    // renameTo fails on Windows if the destination file already exists.
1275    if(LOG.isDebugEnabled()) {
1276      LOG.debug("renaming  " + fromFile.getAbsolutePath() 
1277                + " to " + toFile.getAbsolutePath());
1278    }
1279    if (!fromFile.renameTo(toFile)) {
1280      if (!toFile.delete() || !fromFile.renameTo(toFile)) {
1281        throw new IOException("renaming  " + fromFile.getAbsolutePath() + " to "  + 
1282            toFile.getAbsolutePath() + " FAILED");
1283      }
1284    }
1285    if (renameMD5) {
1286      MD5FileUtils.renameMD5File(fromFile, toFile);
1287    }
1288  }

1292    // Record this log segment ID in all of the storage directories, so
1293    // we won't miss this log segment on a restart if the edits directories

1296    return new CheckpointSignature(this);

1299  /**
1300   * Start checkpoint.
1301   * <p>
1302   * If backup storage contains image that is newer than or incompatible with 
1303   * what the active name-node has, then the backup node should shutdown.<br>
1304   * If the backup image is older than the active one then it should 
1305   * be discarded and downloaded from the active node.<br>
1306   * If the images are the same then the backup image will be used as current.
1307   * 
1308   * @param bnReg the backup node registration.
1309   * @param nnReg this (active) name-node registration.
1310   * @return {@link NamenodeCommand} if backup node should shutdown or
1311   * {@link CheckpointCommand} prescribing what backup node should 
1312   *         do with its image.
1313   * @throws IOException
1314   */
1315  NamenodeCommand startCheckpoint(NamenodeRegistration bnReg, // backup node
1316                                  NamenodeRegistration nnReg) // active name-node
1317  throws IOException {
1318    LOG.info("Start checkpoint at txid " + getEditLog().getLastWrittenTxId());
1319    String msg = null;
1320    // Verify that checkpoint is allowed
1321    if(bnReg.getNamespaceID() != storage.getNamespaceID())
1322      msg = "Name node " + bnReg.getAddress()
1323            + " has incompatible namespace id: " + bnReg.getNamespaceID()
1324            + " expected: " + storage.getNamespaceID();
1325    else if(bnReg.isRole(NamenodeRole.NAMENODE))
1326      msg = "Name node " + bnReg.getAddress()
1327            + " role " + bnReg.getRole() + ": checkpoint is not allowed.";
1328    else if(bnReg.getLayoutVersion() < storage.getLayoutVersion()
1329        || (bnReg.getLayoutVersion() == storage.getLayoutVersion()
1330            && bnReg.getCTime() > storage.getCTime()))
1331      // remote node has newer image age
1332      msg = "Name node " + bnReg.getAddress()
1333            + " has newer image layout version: LV = " +bnReg.getLayoutVersion()
1334            + " cTime = " + bnReg.getCTime()
1335            + ". Current version: LV = " + storage.getLayoutVersion()
1336            + " cTime = " + storage.getCTime();
1337    if(msg != null) {
1338      LOG.error(msg);
1339      return new NamenodeCommand(NamenodeProtocol.ACT_SHUTDOWN);
1340    }
1341    boolean needToReturnImg = true;
1342    if(storage.getNumStorageDirs(NameNodeDirType.IMAGE) == 0)
1343      // do not return image if there are no image directories
1344      needToReturnImg = false;
1345    CheckpointSignature sig = rollEditLog();
1346    return new CheckpointCommand(sig, needToReturnImg);
1347  }

1350   * End checkpoint.
1351   * <p>
1352   * Validate the current storage info with the given signature.
1353   * 
1354   * @param sig to validate the current storage info against
1355   * @throws IOException if the checkpoint fields are inconsistent

1362  /**
1363   * This is called by the 2NN after having downloaded an image, and by
1364   * the NN after having received a new image from the 2NN. It
1365   * renames the image from fsimage_N.ckpt to fsimage_N and also
1366   * saves the related .md5 file into place.
1367   */
1368  public synchronized void saveDigestAndRenameCheckpointImage(NameNodeFile nnf,
1369      long txid, MD5Hash digest) throws IOException {
1370    // Write and rename MD5 file
1371    List<StorageDirectory> badSds = Lists.newArrayList();
1372    
1373    for (StorageDirectory sd : storage.dirIterable(NameNodeDirType.IMAGE)) {
1374      File imageFile = NNStorage.getImageFile(sd, nnf, txid);
1375      try {
1376        MD5FileUtils.saveMD5File(imageFile, digest);
1377      } catch (IOException ioe) {
1378        badSds.add(sd);
1379      }
1380    }
1381    storage.reportErrorsOnDirectories(badSds);
1382    
1383    CheckpointFaultInjector.getInstance().afterMD5Rename();
1384    
1385    // Rename image from tmp file
1386    renameCheckpoint(txid, NameNodeFile.IMAGE_NEW, nnf, false);
1387    // So long as this is the newest image available,
1388    // advertise it as such to other checkpointers
1389    // from now on
1390    if (txid > storage.getMostRecentCheckpointTxId()) {
1391      storage.setMostRecentCheckpointInfo(txid, Time.now());
1392    }
1393  }
