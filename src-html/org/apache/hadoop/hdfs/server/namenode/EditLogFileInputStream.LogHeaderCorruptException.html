053/**
054 * An implementation of the abstract class {@link EditLogInputStream}, which
055 * reads edits from a file. That file may be either on the local disk or
056 * accessible via a URL.
057 */
058@InterfaceAudience.Private
059public class EditLogFileInputStream extends EditLogInputStream {
060  private final LogSource log;
061  private final long firstTxId;
062  private final long lastTxId;
063  private final boolean isInProgress;
064  private int maxOpSize;
065  static private enum State {
066    UNINIT,
067    OPEN,
068    CLOSED
069  }
070  private State state = State.UNINIT;
071  private InputStream fStream = null;
072  private int logVersion = 0;
073  private FSEditLogOp.Reader reader = null;
074  private FSEditLogLoader.PositionTrackingInputStream tracker = null;
075  private DataInputStream dataIn = null;
076  static final Log LOG = LogFactory.getLog(EditLogInputStream.class);

125  private EditLogFileInputStream(LogSource log,
126      long firstTxId, long lastTxId,
127      boolean isInProgress) {
128      
129    this.log = log;
130    this.firstTxId = firstTxId;
131    this.lastTxId = lastTxId;
132    this.isInProgress = isInProgress;
133    this.maxOpSize = DFSConfigKeys.DFS_NAMENODE_MAX_OP_SIZE_DEFAULT;
134  }
135
136  private void init(boolean verifyLayoutVersion)
137      throws LogHeaderCorruptException, IOException {
138    Preconditions.checkState(state == State.UNINIT);
139    BufferedInputStream bin = null;
140    try {
141      fStream = log.getInputStream();
142      bin = new BufferedInputStream(fStream);
143      tracker = new FSEditLogLoader.PositionTrackingInputStream(bin);
144      dataIn = new DataInputStream(tracker);
145      try {
146        logVersion = readLogVersion(dataIn, verifyLayoutVersion);
147      } catch (EOFException eofe) {
148        throw new LogHeaderCorruptException("No header found in log");
149      }
150      // We assume future layout will also support ADD_LAYOUT_FLAGS
151      if (NameNodeLayoutVersion.supports(
152          LayoutVersion.Feature.ADD_LAYOUT_FLAGS, logVersion) ||
153          logVersion < NameNodeLayoutVersion.CURRENT_LAYOUT_VERSION) {
154        try {
155          LayoutFlags.read(dataIn);
156        } catch (EOFException eofe) {
157          throw new LogHeaderCorruptException("EOF while reading layout " +
158              "flags from log");
159        }
160      }
161      reader = new FSEditLogOp.Reader(dataIn, tracker, logVersion);
162      reader.setMaxOpSize(maxOpSize);
163      state = State.OPEN;
164    } finally {
165      if (reader == null) {
166        IOUtils.cleanup(LOG, dataIn, tracker, bin, fStream);
167        state = State.CLOSED;
168      }
169    }
170  }

187  private FSEditLogOp nextOpImpl(boolean skipBrokenEdits) throws IOException {
188    FSEditLogOp op = null;
189    switch (state) {
190    case UNINIT:
191      try {
192        init(true);
193      } catch (Throwable e) {
194        LOG.error("caught exception initializing " + this, e);
195        if (skipBrokenEdits) {
196          return null;
197        }
198        Throwables.propagateIfPossible(e, IOException.class);
199      }
200      Preconditions.checkState(state != State.UNINIT);
201      return nextOpImpl(skipBrokenEdits);
202    case OPEN:
203      op = reader.readOp(skipBrokenEdits);
204      if ((op != null) && (op.hasTransactionId())) {
205        long txId = op.getTransactionId();
206        if ((txId >= lastTxId) &&
207            (lastTxId != HdfsConstants.INVALID_TXID)) {
208          //
209          // Sometimes, the NameNode crashes while it's writing to the
210          // edit log.  In that case, you can end up with an unfinalized edit log
211          // which has some garbage at the end.
212          // JournalManager#recoverUnfinalizedSegments will finalize these
213          // unfinished edit logs, giving them a defined final transaction 
214          // ID.  Then they will be renamed, so that any subsequent
215          // readers will have this information.
216          //
217          // Since there may be garbage at the end of these "cleaned up"
218          // logs, we want to be sure to skip it here if we've read everything
219          // we were supposed to read out of the stream.
220          // So we force an EOF on all subsequent reads.
221          //
222          long skipAmt = log.length() - tracker.getPos();
223          if (skipAmt > 0) {
224            if (LOG.isDebugEnabled()) {
225                LOG.debug("skipping " + skipAmt + " bytes at the end " +
226                  "of edit log  '" + getName() + "': reached txid " + txId +
227                  " out of " + lastTxId);
228            }
229            tracker.clearLimit();
230            IOUtils.skipFully(tracker, skipAmt);
231          }
232        }
233      }
234      break;
235      case CLOSED:
236        break; // return null
237    }
238    return op;
239  }

241  @Override
242  protected long scanNextOp() throws IOException {
243    Preconditions.checkState(state == State.OPEN);
244    FSEditLogOp cachedNext = getCachedOp();
245    return cachedNext == null ? reader.scanOp() : cachedNext.txid;
246  }
247
248  @Override
249  protected FSEditLogOp nextOp() throws IOException {
250    return nextOpImpl(false);
251  }
252
253  @Override
254  protected FSEditLogOp nextValidOp() {
255    try {
256      return nextOpImpl(true);
257    } catch (Throwable e) {
258      LOG.error("nextValidOp: got exception while reading " + this, e);
259      return null;
260    }
261  }

263  @Override
264  public int getVersion(boolean verifyVersion) throws IOException {
265    if (state == State.UNINIT) {
266      init(verifyVersion);
267    }
268    return logVersion;
269  }

271  @Override
272  public long getPosition() {
273    if (state == State.OPEN) {
274      return tracker.getPos();
275    } else {
276      return 0;
277    }
278  }

280  @Override
281  public void close() throws IOException {
282    if (state == State.OPEN) {
283      dataIn.close();
284    }
285    state = State.CLOSED;
286  }

304  static FSEditLogLoader.EditLogValidation validateEditLog(File file)
305      throws IOException {
306    EditLogFileInputStream in;
307    try {
308      in = new EditLogFileInputStream(file);
309      in.getVersion(true); // causes us to read the header
310    } catch (LogHeaderCorruptException e) {
311      // If the header is malformed or the wrong value, this indicates a corruption
312      LOG.warn("Log file " + file + " has no valid header", e);
313      return new FSEditLogLoader.EditLogValidation(0,
314          HdfsConstants.INVALID_TXID, true);
315    }
316    
317    try {
318      return FSEditLogLoader.validateEditLog(in);
319    } finally {
320      IOUtils.closeStream(in);
321    }
322  }
323
324  static FSEditLogLoader.EditLogValidation scanEditLog(File file)
325      throws IOException {
326    EditLogFileInputStream in;
327    try {
328      in = new EditLogFileInputStream(file);
329      // read the header, initialize the inputstream, but do not check the
330      // layoutversion
331      in.getVersion(false);
332    } catch (LogHeaderCorruptException e) {
333      LOG.warn("Log file " + file + " has no valid header", e);
334      return new FSEditLogLoader.EditLogValidation(0,
335          HdfsConstants.INVALID_TXID, true);
336    }
337
338    long lastPos = 0;
339    long lastTxId = HdfsConstants.INVALID_TXID;
340    long numValid = 0;
341    try {
342      while (true) {
343        long txid = HdfsConstants.INVALID_TXID;
344        lastPos = in.getPosition();
345        try {
346          if ((txid = in.scanNextOp()) == HdfsConstants.INVALID_TXID) {
347            break;
348          }
349        } catch (Throwable t) {
350          FSImage.LOG.warn("Caught exception after scanning through "
351              + numValid + " ops from " + in
352              + " while determining its valid length. Position was "
353              + lastPos, t);
354          in.resync();
355          FSImage.LOG.warn("After resync, position is " + in.getPosition());
356          continue;
357        }
358        if (lastTxId == HdfsConstants.INVALID_TXID || txid > lastTxId) {
359          lastTxId = txid;
360        }
361        numValid++;
362      }
363      return new EditLogValidation(lastPos, lastTxId, false);
364    } finally {
365      IOUtils.closeStream(in);
366    }
367  }

369  /**
370   * Read the header of fsedit log
371   * @param in fsedit stream
372   * @return the edit log version number
373   * @throws IOException if error occurs
374   */
375  @VisibleForTesting
376  static int readLogVersion(DataInputStream in, boolean verifyLayoutVersion)
377      throws IOException, LogHeaderCorruptException {
378    int logVersion;
379    try {
380      logVersion = in.readInt();
381    } catch (EOFException eofe) {
382      throw new LogHeaderCorruptException(
383          "Reached EOF when reading log header");
384    }
385    if (verifyLayoutVersion &&
386        (logVersion < HdfsConstants.NAMENODE_LAYOUT_VERSION || // future version
387         logVersion > Storage.LAST_UPGRADABLE_LAYOUT_VERSION)) { // unsupported
388      throw new LogHeaderCorruptException(
389          "Unexpected version of the file system log file: "
390          + logVersion + ". Current version = "
391          + HdfsConstants.NAMENODE_LAYOUT_VERSION + ".");
392    }
393    return logVersion;
394  }

396  /**
397   * Exception indicating that the header of an edits log file is
398   * corrupted. This can be because the header is not present,
399   * or because the header data is invalid (eg claims to be
400   * over a newer version than the running NameNode)
401   */
402  static class LogHeaderCorruptException extends IOException {
403    private static final long serialVersionUID = 1L;

406      super(msg);
407    }

439  private static class URLLog implements LogSource {
440    private final URL url;
441    private long advertisedSize = -1;
442
443    private final static String CONTENT_LENGTH = "Content-Length";
444    private final URLConnectionFactory connectionFactory;
445    private final boolean isSpnegoEnabled;
446
447    public URLLog(URLConnectionFactory connectionFactory, URL url) {
448      this.connectionFactory = connectionFactory;
449      this.isSpnegoEnabled = UserGroupInformation.isSecurityEnabled();
450      this.url = url;
451    }
452
453    @Override
454    public InputStream getInputStream() throws IOException {
455      return SecurityUtil.doAsCurrentUser(
456          new PrivilegedExceptionAction<InputStream>() {
457            @Override
458            public InputStream run() throws IOException {
459              HttpURLConnection connection;
460              try {
461                connection = (HttpURLConnection)
462                    connectionFactory.openConnection(url, isSpnegoEnabled);
463              } catch (AuthenticationException e) {
464                throw new IOException(e);
465              }
466              
467              if (connection.getResponseCode() != HttpURLConnection.HTTP_OK) {
468                throw new HttpGetFailedException(
469                    "Fetch of " + url +
470                    " failed with status code " + connection.getResponseCode() +
471                    "\nResponse message:\n" + connection.getResponseMessage(),
472                    connection);
473              }
474        
475              String contentLength = connection.getHeaderField(CONTENT_LENGTH);
476              if (contentLength != null) {
477                advertisedSize = Long.parseLong(contentLength);
478                if (advertisedSize <= 0) {
479                  throw new IOException("Invalid " + CONTENT_LENGTH + " header: " +
480                      contentLength);
481                }
482              } else {
483                throw new IOException(CONTENT_LENGTH + " header is not provided " +
484                                      "by the server when trying to fetch " + url);
485              }
486        
487              return connection.getInputStream();
488            }
489          });
490    }
491
492    @Override
493    public long length() {
494      return advertisedSize;
495    }
496
497    @Override
498    public String getName() {
499      return url.toString();
500    }
501  }
