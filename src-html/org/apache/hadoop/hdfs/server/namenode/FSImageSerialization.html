102  private static void writeBlocks(final Block[] blocks,
103      final DataOutput out) throws IOException {
104    if (blocks == null) {
105      out.writeInt(0);
106    } else {
107      out.writeInt(blocks.length);
108      for (Block blk : blocks) {
109        blk.write(out);
110      }
111    }
112  }

114  // Helper function that reads in an INodeUnderConstruction
115  // from the input stream
116  //
117  static INodeFile readINodeUnderConstruction(
118      DataInput in, FSNamesystem fsNamesys, int imgVersion)
119      throws IOException {
120    byte[] name = readBytes(in);
121    long inodeId = NameNodeLayoutVersion.supports(
122        LayoutVersion.Feature.ADD_INODE_ID, imgVersion) ? in.readLong()
123        : fsNamesys.dir.allocateNewInodeId();
124    short blockReplication = in.readShort();
125    long modificationTime = in.readLong();
126    long preferredBlockSize = in.readLong();
127  
128    int numBlocks = in.readInt();
129    BlockInfoContiguous[] blocks = new BlockInfoContiguous[numBlocks];
130    Block blk = new Block();
131    int i = 0;
132    for (; i < numBlocks-1; i++) {
133      blk.readFields(in);
134      blocks[i] = new BlockInfoContiguous(blk, blockReplication);
135    }
136    // last block is UNDER_CONSTRUCTION
137    if(numBlocks > 0) {
138      blk.readFields(in);
139      blocks[i] = new BlockInfoContiguousUnderConstruction(
140        blk, blockReplication, BlockUCState.UNDER_CONSTRUCTION, null);
141    }
142    PermissionStatus perm = PermissionStatus.read(in);
143    String clientName = readString(in);
144    String clientMachine = readString(in);
145
146    // We previously stored locations for the last block, now we
147    // just record that there are none
148    int numLocs = in.readInt();
149    assert numLocs == 0 : "Unexpected block locations";
150
151    // Images in the pre-protobuf format will not have the lazyPersist flag,
152    // so it is safe to pass false always.
153    INodeFile file = new INodeFile(inodeId, name, perm, modificationTime,
154        modificationTime, blocks, blockReplication, preferredBlockSize, (byte)0);
155    file.toUnderConstruction(clientName, clientMachine);
156    return file;
157  }

159  // Helper function that writes an INodeUnderConstruction
160  // into the output stream
161  //
162  static void writeINodeUnderConstruction(DataOutputStream out, INodeFile cons,
163      String path) throws IOException {
164    writeString(path, out);
165    out.writeLong(cons.getId());
166    out.writeShort(cons.getFileReplication());
167    out.writeLong(cons.getModificationTime());
168    out.writeLong(cons.getPreferredBlockSize());
169
170    writeBlocks(cons.getBlocks(), out);
171    cons.getPermissionStatus().write(out);
172
173    FileUnderConstructionFeature uc = cons.getFileUnderConstructionFeature();
174    writeString(uc.getClientName(), out);
175    writeString(uc.getClientMachine(), out);
176
177    out.writeInt(0); //  do not store locations of last block
178  }

180  /**
181   * Serialize a {@link INodeFile} node
182   * @param node The node to write
183   * @param out The {@link DataOutputStream} where the fields are written
184   * @param writeBlock Whether to write block information
185   */
186  public static void writeINodeFile(INodeFile file, DataOutput out,
187      boolean writeUnderConstruction) throws IOException {
188    writeLocalName(file, out);
189    out.writeLong(file.getId());
190    out.writeShort(file.getFileReplication());
191    out.writeLong(file.getModificationTime());
192    out.writeLong(file.getAccessTime());
193    out.writeLong(file.getPreferredBlockSize());
194
195    writeBlocks(file.getBlocks(), out);
196    SnapshotFSImageFormat.saveFileDiffList(file, out);
197
198    if (writeUnderConstruction) {
199      if (file.isUnderConstruction()) {
200        out.writeBoolean(true);
201        final FileUnderConstructionFeature uc = file.getFileUnderConstructionFeature();
202        writeString(uc.getClientName(), out);
203        writeString(uc.getClientMachine(), out);
204      } else {
205        out.writeBoolean(false);
206      }
207    }
208
209    writePermissionStatus(file, out);
210  }

231   * Serialize a {@link INodeDirectory}
232   * @param node The node to write
233   * @param out The {@link DataOutput} where the fields are written
234   */
235  public static void writeINodeDirectory(INodeDirectory node, DataOutput out)
236      throws IOException {
237    writeLocalName(node, out);
238    out.writeLong(node.getId());
239    out.writeShort(0);  // replication
240    out.writeLong(node.getModificationTime());
241    out.writeLong(0);   // access time
242    out.writeLong(0);   // preferred block size
243    out.writeInt(-1);   // # of blocks
244
245    writeQuota(node.getQuotaCounts(), out);
246
247    if (node.isSnapshottable()) {
248      out.writeBoolean(true);
249    } else {
250      out.writeBoolean(false);
251      out.writeBoolean(node.isWithSnapshot());
252    }
253
254    writePermissionStatus(node, out);
255  }

289  /** Serialize a {@link INodeReference} node */
290  private static void writeINodeReference(INodeReference ref, DataOutput out,
291      boolean writeUnderConstruction, ReferenceMap referenceMap
292      ) throws IOException {
293    writeLocalName(ref, out);
294    out.writeLong(ref.getId());
295    out.writeShort(0);  // replication
296    out.writeLong(0);   // modification time
297    out.writeLong(0);   // access time
298    out.writeLong(0);   // preferred block size
299    out.writeInt(-3);   // # of blocks
300
301    final boolean isWithName = ref instanceof INodeReference.WithName;
302    out.writeBoolean(isWithName);
303
304    if (!isWithName) {
305      Preconditions.checkState(ref instanceof INodeReference.DstReference);
306      // dst snapshot id
307      out.writeInt(((INodeReference.DstReference) ref).getDstSnapshotId());
308    } else {
309      out.writeInt(((INodeReference.WithName) ref).getLastSnapshotId());
310    }
311
312    final INodeReference.WithCount withCount
313        = (INodeReference.WithCount)ref.getReferredINode();
314    referenceMap.writeINodeReferenceWithCount(withCount, out,
315        writeUnderConstruction);
316  }

318  /**
319   * Save one inode's attributes to the image.
320   */
321  public static void saveINode2Image(INode node, DataOutput out,
322      boolean writeUnderConstruction, ReferenceMap referenceMap)
323      throws IOException {
324    if (node.isReference()) {
325      writeINodeReference(node.asReference(), out, writeUnderConstruction,
326          referenceMap);
327    } else if (node.isDirectory()) {
328      writeINodeDirectory(node.asDirectory(), out);
329    } else if (node.isSymlink()) {
330      writeINodeSymlink(node.asSymlink(), out);
331    } else if (node.isFile()) {
332      writeINodeFile(node.asFile(), out, writeUnderConstruction);
333    }
334  }

473  /**
474   * Write an array of blocks as compactly as possible. This uses
475   * delta-encoding for the generation stamp and size, following
476   * the principle that genstamp increases relatively slowly,
477   * and size is equal for all but the last block of a file.
478   */
479  public static void writeCompactBlockArray(
480      Block[] blocks, DataOutputStream out) throws IOException {
481    WritableUtils.writeVInt(out, blocks.length);
482    Block prev = null;
483    for (Block b : blocks) {
484      long szDelta = b.getNumBytes() -
485          (prev != null ? prev.getNumBytes() : 0);
486      long gsDelta = b.getGenerationStamp() -
487          (prev != null ? prev.getGenerationStamp() : 0);
488      out.writeLong(b.getBlockId()); // blockid is random
489      WritableUtils.writeVLong(out, szDelta);
490      WritableUtils.writeVLong(out, gsDelta);
491      prev = b;
492    }
493  }
494  
495  public static Block[] readCompactBlockArray(
496      DataInput in, int logVersion) throws IOException {
497    int num = WritableUtils.readVInt(in);
498    if (num < 0) {
499      throw new IOException("Invalid block array length: " + num);
500    }
501    Block prev = null;
502    Block[] ret = new Block[num];
503    for (int i = 0; i < num; i++) {
504      long id = in.readLong();
505      long sz = WritableUtils.readVLong(in) +
506          ((prev != null) ? prev.getNumBytes() : 0);
507      long gs = WritableUtils.readVLong(in) +
508          ((prev != null) ? prev.getGenerationStamp() : 0);
509      ret[i] = new Block(id, sz, gs);
510      prev = ret[i];
511    }
512    return ret;
513  }

515  public static void writeCacheDirectiveInfo(DataOutputStream out,
516      CacheDirectiveInfo directive) throws IOException {
517    writeLong(directive.getId(), out);
518    int flags =
519        ((directive.getPath() != null) ? 0x1 : 0) |
520        ((directive.getReplication() != null) ? 0x2 : 0) |
521        ((directive.getPool() != null) ? 0x4 : 0) |
522        ((directive.getExpiration() != null) ? 0x8 : 0);
523    out.writeInt(flags);
524    if (directive.getPath() != null) {
525      writeString(directive.getPath().toUri().getPath(), out);
526    }
527    if (directive.getReplication() != null) {
528      writeShort(directive.getReplication(), out);
529    }
530    if (directive.getPool() != null) {
531      writeString(directive.getPool(), out);
532    }
533    if (directive.getExpiration() != null) {
534      writeLong(directive.getExpiration().getMillis(), out);
535    }
536  }

538  public static CacheDirectiveInfo readCacheDirectiveInfo(DataInput in)
539      throws IOException {
540    CacheDirectiveInfo.Builder builder =
541        new CacheDirectiveInfo.Builder();
542    builder.setId(readLong(in));
543    int flags = in.readInt();
544    if ((flags & 0x1) != 0) {
545      builder.setPath(new Path(readString(in)));
546    }
547    if ((flags & 0x2) != 0) {
548      builder.setReplication(readShort(in));
549    }
550    if ((flags & 0x4) != 0) {
551      builder.setPool(readString(in));
552    }
553    if ((flags & 0x8) != 0) {
554      builder.setExpiration(
555          CacheDirectiveInfo.Expiration.newAbsolute(readLong(in)));
556    }
557    if ((flags & ~0xF) != 0) {
558      throw new IOException("unknown flags set in " +
559          "ModifyCacheDirectiveInfoOp: " + flags);
560    }
561    return builder.build();
562  }

564  public static CacheDirectiveInfo readCacheDirectiveInfo(Stanza st)
565      throws InvalidXmlException {
566    CacheDirectiveInfo.Builder builder =
567        new CacheDirectiveInfo.Builder();
568    builder.setId(Long.parseLong(st.getValue("ID")));
569    String path = st.getValueOrNull("PATH");
570    if (path != null) {
571      builder.setPath(new Path(path));
572    }
573    String replicationString = st.getValueOrNull("REPLICATION");
574    if (replicationString != null) {
575      builder.setReplication(Short.parseShort(replicationString));
576    }
577    String pool = st.getValueOrNull("POOL");
578    if (pool != null) {
579      builder.setPool(pool);
580    }
581    String expiryTime = st.getValueOrNull("EXPIRATION");
582    if (expiryTime != null) {
583      builder.setExpiration(CacheDirectiveInfo.Expiration.newAbsolute(
584          Long.parseLong(expiryTime)));
585    }
586    return builder.build();
587  }

610  public static void writeCachePoolInfo(DataOutputStream out, CachePoolInfo info)
611      throws IOException {
612    writeString(info.getPoolName(), out);
613
614    final String ownerName = info.getOwnerName();
615    final String groupName = info.getGroupName();
616    final Long limit = info.getLimit();
617    final FsPermission mode = info.getMode();
618    final Long maxRelativeExpiry = info.getMaxRelativeExpiryMs();
619
620    boolean hasOwner, hasGroup, hasMode, hasLimit, hasMaxRelativeExpiry;
621    hasOwner = ownerName != null;
622    hasGroup = groupName != null;
623    hasMode = mode != null;
624    hasLimit = limit != null;
625    hasMaxRelativeExpiry = maxRelativeExpiry != null;
626
627    int flags =
628        (hasOwner ? 0x1 : 0) |
629        (hasGroup ? 0x2 : 0) |
630        (hasMode  ? 0x4 : 0) |
631        (hasLimit ? 0x8 : 0) |
632        (hasMaxRelativeExpiry ? 0x10 : 0);
633
634    writeInt(flags, out);
635
636    if (hasOwner) {
637      writeString(ownerName, out);
638    }
639    if (hasGroup) {
640      writeString(groupName, out);
641    }
642    if (hasMode) {
643      mode.write(out);
644    }
645    if (hasLimit) {
646      writeLong(limit, out);
647    }
648    if (hasMaxRelativeExpiry) {
649      writeLong(maxRelativeExpiry, out);
650    }
651  }

653  public static CachePoolInfo readCachePoolInfo(DataInput in)
654      throws IOException {
655    String poolName = readString(in);
656    CachePoolInfo info = new CachePoolInfo(poolName);
657    int flags = readInt(in);
658    if ((flags & 0x1) != 0) {
659      info.setOwnerName(readString(in));
660    }
661    if ((flags & 0x2) != 0)  {
662      info.setGroupName(readString(in));
663    }
664    if ((flags & 0x4) != 0) {
665      info.setMode(FsPermission.read(in));
666    }
667    if ((flags & 0x8) != 0) {
668      info.setLimit(readLong(in));
669    }
670    if ((flags & 0x10) != 0) {
671      info.setMaxRelativeExpiryMs(readLong(in));
672    }
673    if ((flags & ~0x1F) != 0) {
674      throw new IOException("Unknown flag in CachePoolInfo: " + flags);
675    }
676    return info;
677  }

679  public static void writeCachePoolInfo(ContentHandler contentHandler,
680      CachePoolInfo info) throws SAXException {
681    XMLUtils.addSaxString(contentHandler, "POOLNAME", info.getPoolName());
682
683    final String ownerName = info.getOwnerName();
684    final String groupName = info.getGroupName();
685    final Long limit = info.getLimit();
686    final FsPermission mode = info.getMode();
687    final Long maxRelativeExpiry = info.getMaxRelativeExpiryMs();
688
689    if (ownerName != null) {
690      XMLUtils.addSaxString(contentHandler, "OWNERNAME", ownerName);
691    }
692    if (groupName != null) {
693      XMLUtils.addSaxString(contentHandler, "GROUPNAME", groupName);
694    }
695    if (mode != null) {
696      FSEditLogOp.fsPermissionToXml(contentHandler, mode);
697    }
698    if (limit != null) {
699      XMLUtils.addSaxString(contentHandler, "LIMIT",
700          Long.toString(limit));
701    }
702    if (maxRelativeExpiry != null) {
703      XMLUtils.addSaxString(contentHandler, "MAXRELATIVEEXPIRY",
704          Long.toString(maxRelativeExpiry));
705    }
706  }

708  public static CachePoolInfo readCachePoolInfo(Stanza st)
709      throws InvalidXmlException {
710    String poolName = st.getValue("POOLNAME");
711    CachePoolInfo info = new CachePoolInfo(poolName);
712    if (st.hasChildren("OWNERNAME")) {
713      info.setOwnerName(st.getValue("OWNERNAME"));
714    }
715    if (st.hasChildren("GROUPNAME")) {
716      info.setGroupName(st.getValue("GROUPNAME"));
717    }
718    if (st.hasChildren("MODE")) {
719      info.setMode(FSEditLogOp.fsPermissionFromXml(st));
720    }
721    if (st.hasChildren("LIMIT")) {
722      info.setLimit(Long.parseLong(st.getValue("LIMIT")));
723    }
724    if (st.hasChildren("MAXRELATIVEEXPIRY")) {
725      info.setMaxRelativeExpiryMs(
726          Long.parseLong(st.getValue("MAXRELATIVEEXPIRY")));
727    }
728    return info;
