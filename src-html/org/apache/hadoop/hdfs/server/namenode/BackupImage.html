037/**
038 * Extension of FSImage for the backup node.
039 * This class handles the setup of the journaling 
040 * spool on the backup namenode.
041 */
042@InterfaceAudience.Private
043public class BackupImage extends FSImage {
044  /** Backup input stream for loading edits into memory */
045  private final EditLogBackupInputStream backupInputStream =
046    new EditLogBackupInputStream("Data from remote NameNode");
047  
048  /**
049   * Current state of the BackupNode. The BackupNode's state
050   * transitions are as follows:
051   * 
052   * Initial: DROP_UNTIL_NEXT_ROLL
053   * - Transitions to JOURNAL_ONLY the next time the log rolls
054   * - Transitions to IN_SYNC in convergeJournalSpool
055   * - Transitions back to JOURNAL_ONLY if the log rolls while
056   *   stopApplyingOnNextRoll is true.
057   */
058  volatile BNState bnState;
059  static enum BNState {
060    /**
061     * Edits from the NN should be dropped. On the next log roll,
062     * transition to JOURNAL_ONLY state
063     */
064    DROP_UNTIL_NEXT_ROLL,
065    /**
066     * Edits from the NN should be written to the local edits log
067     * but not applied to the namespace.
068     */
069    JOURNAL_ONLY,
070    /**
071     * Edits should be written to the local edits log and applied
072     * to the local namespace.
073     */
074    IN_SYNC;
075  }
076
077  /**
078   * Flag to indicate that the next time the NN rolls, the BN
079   * should transition from to JOURNAL_ONLY state.
080   * {@see #freezeNamespaceAtNextRoll()}
081   */
082  private boolean stopApplyingEditsOnNextRoll = false;
083  
084  private FSNamesystem namesystem;
085
086  /**
087   * Construct a backup image.
088   * @param conf Configuration
089   * @throws IOException if storage cannot be initialised.
090   */
091  BackupImage(Configuration conf) throws IOException {
092    super(conf);
093    storage.setDisablePreUpgradableLayoutCheck(true);
094    bnState = BNState.DROP_UNTIL_NEXT_ROLL;
095  }
096
097  synchronized FSNamesystem getNamesystem() {
098    return namesystem;
099  }

101  synchronized void setNamesystem(FSNamesystem fsn) {
102    // Avoids overriding this.namesystem object
103    if (namesystem == null) {
104      this.namesystem = fsn;
105    }
106  }

108  /**
109   * Analyze backup storage directories for consistency.<br>
110   * Recover from incomplete checkpoints if required.<br>
111   * Read VERSION and fstime files if exist.<br>
112   * Do not load image or edits.
113   *
114   * @throws IOException if the node should shutdown.
115   */
116  void recoverCreateRead() throws IOException {
117    for (Iterator<StorageDirectory> it = storage.dirIterator(); it.hasNext();) {
118      StorageDirectory sd = it.next();
119      StorageState curState;
120      try {
121        curState = sd.analyzeStorage(HdfsServerConstants.StartupOption.REGULAR, storage);
122        // sd is locked but not opened
123        switch(curState) {
124        case NON_EXISTENT:
125          // fail if any of the configured storage dirs are inaccessible
126          throw new InconsistentFSStateException(sd.getRoot(),
127                "checkpoint directory does not exist or is not accessible.");
128        case NOT_FORMATTED:
129          // for backup node all directories may be unformatted initially
130          LOG.info("Storage directory " + sd.getRoot() + " is not formatted.");
131          LOG.info("Formatting ...");
132          sd.clearDirectory(); // create empty current
133          break;
134        case NORMAL:
135          break;
136        default:  // recovery is possible
137          sd.doRecover(curState);
138        }
139        if(curState != StorageState.NOT_FORMATTED) {
140          // read and verify consistency with other directories
141          storage.readProperties(sd);
142        }
143      } catch(IOException ioe) {
144        sd.unlock();
145        throw ioe;
146      }
147    }
148  }
149
150  /**
151   * Receive a batch of edits from the NameNode.
152   * 
153   * Depending on bnState, different actions are taken. See
154   * {@link BackupImage.BNState}
155   * 
156   * @param firstTxId first txid in batch
157   * @param numTxns number of transactions
158   * @param data serialized journal records.
159   * @throws IOException
160   * @see #convergeJournalSpool()
161   */
162  synchronized void journal(long firstTxId, int numTxns, byte[] data) throws IOException {
163    if (LOG.isTraceEnabled()) {
164      LOG.trace("Got journal, " +
165          "state = " + bnState +
166          "; firstTxId = " + firstTxId +
167          "; numTxns = " + numTxns);
168    }
169    
170    switch(bnState) {
171      case DROP_UNTIL_NEXT_ROLL:
172        return;
173
174      case IN_SYNC:
175        // update NameSpace in memory
176        applyEdits(firstTxId, numTxns, data);
177        break;
178      
179      case JOURNAL_ONLY:
180        break;
181      
182      default:
183        throw new AssertionError("Unhandled state: " + bnState);
184    }
185    
186    // write to BN's local edit log.
187    logEditsLocally(firstTxId, numTxns, data);
188  }
189
190  /**
191   * Write the batch of edits to the local copy of the edit logs.
192   */
193  private void logEditsLocally(long firstTxId, int numTxns, byte[] data) {
194    long expectedTxId = editLog.getLastWrittenTxId() + 1;
195    Preconditions.checkState(firstTxId == expectedTxId,
196        "received txid batch starting at %s but expected txn %s",
197        firstTxId, expectedTxId);
198    editLog.setNextTxId(firstTxId + numTxns - 1);
199    editLog.logEdit(data.length, data);
200    editLog.logSync();
201  }
202
203  /**
204   * Apply the batch of edits to the local namespace.
205   */
206  private synchronized void applyEdits(long firstTxId, int numTxns, byte[] data)
207      throws IOException {
208    Preconditions.checkArgument(firstTxId == lastAppliedTxId + 1,
209        "Received txn batch starting at %s but expected %s",
210        firstTxId, lastAppliedTxId + 1);
211    assert backupInputStream.length() == 0 : "backup input stream is not empty";
212    try {
213      if (LOG.isTraceEnabled()) {
214        LOG.debug("data:" + StringUtils.byteToHexString(data));
215      }
216
217      FSEditLogLoader logLoader =
218          new FSEditLogLoader(getNamesystem(), lastAppliedTxId);
219      int logVersion = storage.getLayoutVersion();
220      backupInputStream.setBytes(data, logVersion);
221
222      long numTxnsAdvanced = logLoader.loadEditRecords(
223          backupInputStream, true, lastAppliedTxId + 1, null, null);
224      if (numTxnsAdvanced != numTxns) {
225        throw new IOException("Batch of txns starting at txnid " +
226            firstTxId + " was supposed to contain " + numTxns +
227            " transactions, but we were only able to advance by " +
228            numTxnsAdvanced);
229      }
230      lastAppliedTxId = logLoader.getLastAppliedTxId();
231
232      FSImage.updateCountForQuota(
233          getNamesystem().dir.getBlockStoragePolicySuite(),
234          getNamesystem().dir.rootDir); // inefficient!
235    } finally {
236      backupInputStream.clear();
237    }
238  }

240  /**
241   * Transition the BackupNode from JOURNAL_ONLY state to IN_SYNC state.
242   * This is done by repeated invocations of tryConvergeJournalSpool until
243   * we are caught up to the latest in-progress edits file.
244   */
245  void convergeJournalSpool() throws IOException {
246    Preconditions.checkState(bnState == BNState.JOURNAL_ONLY,
247        "bad state: %s", bnState);
248
249    while (!tryConvergeJournalSpool()) {
250      ;
251    }
252    assert bnState == BNState.IN_SYNC;
253  }
254  
255  private boolean tryConvergeJournalSpool() throws IOException {
256    Preconditions.checkState(bnState == BNState.JOURNAL_ONLY,
257        "bad state: %s", bnState);
258    
259    // This section is unsynchronized so we can continue to apply
260    // ahead of where we're reading, concurrently. Since the state
261    // is JOURNAL_ONLY at this point, we know that lastAppliedTxId
262    // doesn't change, and curSegmentTxId only increases
263
264    while (lastAppliedTxId < editLog.getCurSegmentTxId() - 1) {
265      long target = editLog.getCurSegmentTxId();
266      LOG.info("Loading edits into backupnode to try to catch up from txid "
267          + lastAppliedTxId + " to " + target);
268      FSImageTransactionalStorageInspector inspector =
269        new FSImageTransactionalStorageInspector();
270      
271      storage.inspectStorageDirs(inspector);
272
273      editLog.recoverUnclosedStreams();
274      Iterable<EditLogInputStream> editStreamsAll 
275        = editLog.selectInputStreams(lastAppliedTxId, target - 1);
276      // remove inprogress
277      List<EditLogInputStream> editStreams = Lists.newArrayList();
278      for (EditLogInputStream s : editStreamsAll) {
279        if (s.getFirstTxId() != editLog.getCurSegmentTxId()) {
280          editStreams.add(s);
281        }
282      }
283      loadEdits(editStreams, getNamesystem());
284    }
285    
286    // now, need to load the in-progress file
287    synchronized (this) {
288      if (lastAppliedTxId != editLog.getCurSegmentTxId() - 1) {
289        LOG.debug("Logs rolled while catching up to current segment");
290        return false; // drop lock and try again to load local logs
291      }
292      
293      EditLogInputStream stream = null;
294      Collection<EditLogInputStream> editStreams
295        = getEditLog().selectInputStreams(
296            getEditLog().getCurSegmentTxId(),
297            getEditLog().getCurSegmentTxId());
298      
299      for (EditLogInputStream s : editStreams) {
300        if (s.getFirstTxId() == getEditLog().getCurSegmentTxId()) {
301          stream = s;
302        }
303        break;
304      }
305      if (stream == null) {
306        LOG.warn("Unable to find stream starting with " + editLog.getCurSegmentTxId()
307                 + ". This indicates that there is an error in synchronization in BackupImage");
308        return false;
309      }
310
311      try {
312        long remainingTxns = getEditLog().getLastWrittenTxId() - lastAppliedTxId;
313        
314        LOG.info("Going to finish converging with remaining " + remainingTxns
315            + " txns from in-progress stream " + stream);
316        
317        FSEditLogLoader loader =
318            new FSEditLogLoader(getNamesystem(), lastAppliedTxId);
319        loader.loadFSEdits(stream, lastAppliedTxId + 1);
320        lastAppliedTxId = loader.getLastAppliedTxId();
321        assert lastAppliedTxId == getEditLog().getLastWrittenTxId();
322      } finally {
323        FSEditLog.closeAllStreams(editStreams);
324      }
325
326      LOG.info("Successfully synced BackupNode with NameNode at txnid " +
327          lastAppliedTxId);
328      setState(BNState.IN_SYNC);
329    }
330    return true;
331  }

333  /**
334   * Transition edit log to a new state, logging as necessary.
335   */
336  private synchronized void setState(BNState newState) {
337    if (LOG.isDebugEnabled()) {
338      LOG.debug("State transition " + bnState + " -> " + newState);
339    }
340    bnState = newState;
341  }

343  /**
344   * Receive a notification that the NameNode has begun a new edit log.
345   * This causes the BN to also start the new edit log in its local
346   * directories.
347   */
348  synchronized void namenodeStartedLogSegment(long txid)
349      throws IOException {
350    LOG.info("NameNode started a new log segment at txid " + txid);
351    if (editLog.isSegmentOpen()) {
352      if (editLog.getLastWrittenTxId() == txid - 1) {
353        // We are in sync with the NN, so end and finalize the current segment
354        editLog.endCurrentLogSegment(false);
355      } else {
356        // We appear to have missed some transactions -- the NN probably
357        // lost contact with us temporarily. So, mark the current segment
358        // as aborted.
359        LOG.warn("NN started new log segment at txid " + txid +
360            ", but BN had only written up to txid " +
361            editLog.getLastWrittenTxId() +
362            "in the log segment starting at " + 
363                        editLog.getCurSegmentTxId() + ". Aborting this " +
364                        "log segment.");
365        editLog.abortCurrentLogSegment();
366      }
367    }
368    editLog.setNextTxId(txid);
369    editLog.startLogSegment(txid, false);
370    if (bnState == BNState.DROP_UNTIL_NEXT_ROLL) {
371      setState(BNState.JOURNAL_ONLY);
372    }
373    
374    if (stopApplyingEditsOnNextRoll) {
375      if (bnState == BNState.IN_SYNC) {
376        LOG.info("Stopped applying edits to prepare for checkpoint.");
377        setState(BNState.JOURNAL_ONLY);
378      }
379      stopApplyingEditsOnNextRoll = false;
380      notifyAll();
381    }
382  }

384  /**
385   * Request that the next time the BN receives a log roll, it should
386   * stop applying the edits log to the local namespace. This is
387   * typically followed on by a call to {@link #waitUntilNamespaceFrozen()}
388   */
389  synchronized void freezeNamespaceAtNextRoll() {
390    stopApplyingEditsOnNextRoll = true;
391  }
392
393  /**
394   * After {@link #freezeNamespaceAtNextRoll()} has been called, wait until
395   * the BN receives notification of the next log roll.
396   */
397  synchronized void waitUntilNamespaceFrozen() throws IOException {
398    if (bnState != BNState.IN_SYNC) return;
399
400    LOG.info("Waiting until the NameNode rolls its edit logs in order " +
401        "to freeze the BackupNode namespace.");
402    while (bnState == BNState.IN_SYNC) {
403      Preconditions.checkState(stopApplyingEditsOnNextRoll,
404        "If still in sync, we should still have the flag set to " +
405        "freeze at next roll");
406      try {
407        wait();
408      } catch (InterruptedException ie) {
409        LOG.warn("Interrupted waiting for namespace to freeze", ie);
410        throw new IOException(ie);
411      }
412    }
413    LOG.info("BackupNode namespace frozen.");
414  }
415
416  /**
417   * Override close() so that we don't finalize edit logs.
418   */
419  @Override
420  public synchronized void close() throws IOException {
421    editLog.abortCurrentLogSegment();
422    storage.close();
423  }
