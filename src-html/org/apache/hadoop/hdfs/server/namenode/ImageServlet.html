084  @Override
085  public void doGet(final HttpServletRequest request,
086      final HttpServletResponse response) throws ServletException, IOException {
087    try {
088      final ServletContext context = getServletContext();
089      final FSImage nnImage = NameNodeHttpServer.getFsImageFromContext(context);
090      final GetImageParams parsedParams = new GetImageParams(request, response);
091      final Configuration conf = (Configuration) context
092          .getAttribute(JspHelper.CURRENT_CONF);
093      final NameNodeMetrics metrics = NameNode.getNameNodeMetrics();
094
095      validateRequest(context, conf, request, response, nnImage,
096          parsedParams.getStorageInfoString());
097
098      UserGroupInformation.getCurrentUser().doAs(new PrivilegedExceptionAction<Void>() {
099        @Override
100        public Void run() throws Exception {
101          if (parsedParams.isGetImage()) {
102            long txid = parsedParams.getTxId();
103            File imageFile = null;
104            String errorMessage = "Could not find image";
105            if (parsedParams.shouldFetchLatest()) {
106              imageFile = nnImage.getStorage().getHighestFsImageName();
107            } else {
108              errorMessage += " with txid " + txid;
109              imageFile = nnImage.getStorage().getFsImage(txid,
110                  EnumSet.of(NameNodeFile.IMAGE, NameNodeFile.IMAGE_ROLLBACK));
111            }
112            if (imageFile == null) {
113              throw new IOException(errorMessage);
114            }
115            CheckpointFaultInjector.getInstance().beforeGetImageSetsHeaders();
116            long start = monotonicNow();
117            serveFile(imageFile);
118
119            if (metrics != null) { // Metrics non-null only when used inside name node
120              long elapsed = monotonicNow() - start;
121              metrics.addGetImage(elapsed);
122            }
123          } else if (parsedParams.isGetEdit()) {
124            long startTxId = parsedParams.getStartTxId();
125            long endTxId = parsedParams.getEndTxId();
126            
127            File editFile = nnImage.getStorage()
128                .findFinalizedEditsFile(startTxId, endTxId);
129            long start = monotonicNow();
130            serveFile(editFile);
131
132            if (metrics != null) { // Metrics non-null only when used inside name node
133              long elapsed = monotonicNow() - start;
134              metrics.addGetEdit(elapsed);
135            }
136          }
137          return null;
138        }
139
140        private void serveFile(File file) throws IOException {
141          FileInputStream fis = new FileInputStream(file);
142          try {
143            setVerificationHeadersForGet(response, file);
144            setFileNameHeaders(response, file);
145            if (!file.exists()) {
146              // Potential race where the file was deleted while we were in the
147              // process of setting headers!
148              throw new FileNotFoundException(file.toString());
149              // It's possible the file could be deleted after this point, but
150              // we've already opened the 'fis' stream.
151              // It's also possible length could change, but this would be
152              // detected by the client side as an inaccurate length header.
153            }
154            // send file
155            TransferFsImage.copyFileToStream(response.getOutputStream(),
156               file, fis, getThrottler(conf));
157          } finally {
158            IOUtils.closeStream(fis);
159          }
160        }
161      });
162      
163    } catch (Throwable t) {
164      String errMsg = "GetImage failed. " + StringUtils.stringifyException(t);
165      response.sendError(HttpServletResponse.SC_GONE, errMsg);
166      throw new IOException(errMsg);
167    } finally {
168      response.getOutputStream().close();
169    }
170  }

172  private void validateRequest(ServletContext context, Configuration conf,
173      HttpServletRequest request, HttpServletResponse response,
174      FSImage nnImage, String theirStorageInfoString) throws IOException {
175
176    if (UserGroupInformation.isSecurityEnabled()
177        && !isValidRequestor(context, request.getUserPrincipal().getName(),
178            conf)) {
179      String errorMsg = "Only Namenode, Secondary Namenode, and administrators may access "
180          + "this servlet";
181      response.sendError(HttpServletResponse.SC_FORBIDDEN, errorMsg);
182      LOG.warn("Received non-NN/SNN/administrator request for image or edits from "
183          + request.getUserPrincipal().getName()
184          + " at "
185          + request.getRemoteHost());
186      throw new IOException(errorMsg);
187    }
188
189    String myStorageInfoString = nnImage.getStorage().toColonSeparatedString();
190    if (theirStorageInfoString != null
191        && !myStorageInfoString.equals(theirStorageInfoString)) {
192      String errorMsg = "This namenode has storage info " + myStorageInfoString
193          + " but the secondary expected " + theirStorageInfoString;
194      response.sendError(HttpServletResponse.SC_FORBIDDEN, errorMsg);
195      LOG.warn("Received an invalid request file transfer request "
196          + "from a secondary with storage info " + theirStorageInfoString);
197      throw new IOException(errorMsg);
198    }
199  }

201  public static void setFileNameHeaders(HttpServletResponse response,
202      File file) {
203    response.setHeader(CONTENT_DISPOSITION, "attachment; filename=" +
204        file.getName());
205    response.setHeader(HADOOP_IMAGE_EDITS_HEADER, file.getName());

208  /**
209   * Construct a throttler from conf
210   * @param conf configuration
211   * @return a data transfer throttler
212   */
213  public final static DataTransferThrottler getThrottler(Configuration conf) {
214    long transferBandwidth = 
215      conf.getLong(DFSConfigKeys.DFS_IMAGE_TRANSFER_RATE_KEY,
216                   DFSConfigKeys.DFS_IMAGE_TRANSFER_RATE_DEFAULT);
217    DataTransferThrottler throttler = null;
218    if (transferBandwidth > 0) {
219      throttler = new DataTransferThrottler(transferBandwidth);
220    }
221    return throttler;
222  }

224  @VisibleForTesting
225  static boolean isValidRequestor(ServletContext context, String remoteUser,
226      Configuration conf) throws IOException {
227    if (remoteUser == null) { // This really shouldn't happen...
228      LOG.warn("Received null remoteUser while authorizing access to getImage servlet");
229      return false;
230    }
231
232    Set<String> validRequestors = new HashSet<String>();
233
234    validRequestors.add(SecurityUtil.getServerPrincipal(conf
235        .get(DFSConfigKeys.DFS_NAMENODE_KERBEROS_PRINCIPAL_KEY),
236        NameNode.getAddress(conf).getHostName()));
237    try {
238      validRequestors.add(
239          SecurityUtil.getServerPrincipal(conf
240              .get(DFSConfigKeys.DFS_SECONDARY_NAMENODE_KERBEROS_PRINCIPAL_KEY),
241              SecondaryNameNode.getHttpAddress(conf).getHostName()));
242    } catch (Exception e) {
243      // Don't halt if SecondaryNameNode principal could not be added.
244      LOG.debug("SecondaryNameNode principal could not be added", e);
245      String msg = String.format(
246        "SecondaryNameNode principal not considered, %s = %s, %s = %s",
247        DFSConfigKeys.DFS_SECONDARY_NAMENODE_KERBEROS_PRINCIPAL_KEY,
248        conf.get(DFSConfigKeys.DFS_SECONDARY_NAMENODE_KERBEROS_PRINCIPAL_KEY),
249        DFSConfigKeys.DFS_NAMENODE_SECONDARY_HTTP_ADDRESS_KEY,
250        conf.getTrimmed(DFSConfigKeys.DFS_NAMENODE_SECONDARY_HTTP_ADDRESS_KEY,
251          DFSConfigKeys.DFS_NAMENODE_SECONDARY_HTTP_ADDRESS_DEFAULT));
252      LOG.warn(msg);
253    }
254
255    if (HAUtil.isHAEnabled(conf, DFSUtil.getNamenodeNameServiceId(conf))) {
256      Configuration otherNnConf = HAUtil.getConfForOtherNode(conf);
257      validRequestors.add(SecurityUtil.getServerPrincipal(otherNnConf
258          .get(DFSConfigKeys.DFS_NAMENODE_KERBEROS_PRINCIPAL_KEY),
259          NameNode.getAddress(otherNnConf).getHostName()));
260    }
261
262    for (String v : validRequestors) {
263      if (v != null && v.equals(remoteUser)) {
264        LOG.info("ImageServlet allowing checkpointer: " + remoteUser);
265        return true;
266      }
267    }
268
269    if (HttpServer2.userHasAdministratorAccess(context, remoteUser)) {
270      LOG.info("ImageServlet allowing administrator: " + remoteUser);
271      return true;
272    }
273
274    LOG.info("ImageServlet rejecting: " + remoteUser);
275    return false;
276  }

278  /**
279   * Set headers for content length, and, if available, md5.
280   * @throws IOException 
281   */
282  public static void setVerificationHeadersForGet(HttpServletResponse response,
283      File file) throws IOException {
284    response.setHeader(TransferFsImage.CONTENT_LENGTH,
285        String.valueOf(file.length()));
286    MD5Hash hash = MD5FileUtils.readStoredMd5ForFile(file);
287    if (hash != null) {
288      response.setHeader(TransferFsImage.MD5_HEADER, hash.toString());
289    }

314  static class GetImageParams {
315    private boolean isGetImage;
316    private boolean isGetEdit;
317    private NameNodeFile nnf;
318    private long startTxId, endTxId, txId;
319    private String storageInfoString;
320    private boolean fetchLatest;
321
322    /**
323     * @param request the object from which this servlet reads the url contents
324     * @param response the object into which this servlet writes the url contents
325     * @throws IOException if the request is bad
326     */
327    public GetImageParams(HttpServletRequest request,
328                          HttpServletResponse response
329                           ) throws IOException {
330      @SuppressWarnings("unchecked")
331      Map<String, String[]> pmap = request.getParameterMap();
332      isGetImage = isGetEdit = fetchLatest = false;
333
334      for (Map.Entry<String, String[]> entry : pmap.entrySet()) {
335        String key = entry.getKey();
336        String[] val = entry.getValue();
337        if (key.equals("getimage")) { 
338          isGetImage = true;
339          try {
340            txId = ServletUtil.parseLongParam(request, TXID_PARAM);
341            String imageType = ServletUtil.getParameter(request, IMAGE_FILE_TYPE);
342            nnf = imageType == null ? NameNodeFile.IMAGE : NameNodeFile
343                .valueOf(imageType);
344          } catch (NumberFormatException nfe) {
345            if (request.getParameter(TXID_PARAM).equals(LATEST_FSIMAGE_VALUE)) {
346              fetchLatest = true;
347            } else {
348              throw nfe;
349            }
350          }
351        } else if (key.equals("getedit")) { 
352          isGetEdit = true;
353          startTxId = ServletUtil.parseLongParam(request, START_TXID_PARAM);
354          endTxId = ServletUtil.parseLongParam(request, END_TXID_PARAM);
355        } else if (key.equals(STORAGEINFO_PARAM)) {
356          storageInfoString = val[0];
357        }
358      }
359
360      int numGets = (isGetImage?1:0) + (isGetEdit?1:0);
361      if ((numGets > 1) || (numGets == 0)) {
362        throw new IOException("Illegal parameters to TransferFsImage");
363      }
364    }
365
366    public String getStorageInfoString() {
367      return storageInfoString;
368    }
369
370    public long getTxId() {
371      Preconditions.checkState(isGetImage);
372      return txId;
373    }
374
375    public NameNodeFile getNameNodeFile() {
376      Preconditions.checkState(isGetImage);
377      return nnf;
378    }
379
380    public long getStartTxId() {
381      Preconditions.checkState(isGetEdit);
382      return startTxId;
383    }
384    
385    public long getEndTxId() {
386      Preconditions.checkState(isGetEdit);
387      return endTxId;
388    }
389
390    boolean isGetEdit() {
391      return isGetEdit;
392    }
393
394    boolean isGetImage() {
395      return isGetImage;
396    }
397
398    boolean shouldFetchLatest() {
399      return fetchLatest;
400    }
401    
402  }

420  /**
421   * Set the required parameters for uploading image
422   * 
423   * @param httpMethod instance of method to set the parameters
424   * @param storage colon separated storageInfo string
425   * @param txid txid of the image
426   * @param imageFileSize size of the imagefile to be uploaded
427   * @param nnf NameNodeFile Type
428   * @return Returns map of parameters to be used with PUT request.
429   */
430  static Map<String, String> getParamsForPutImage(Storage storage, long txid,
431      long imageFileSize, NameNodeFile nnf) {
432    Map<String, String> params = new HashMap<String, String>();
433    params.put(TXID_PARAM, Long.toString(txid));
434    params.put(STORAGEINFO_PARAM, storage.toColonSeparatedString());
435    // setting the length of the file to be uploaded in separate property as
436    // Content-Length only supports up to 2GB
437    params.put(TransferFsImage.FILE_LENGTH, Long.toString(imageFileSize));
438    params.put(IMAGE_FILE_TYPE, nnf.name());
439    return params;
440  }

442  @Override
443  protected void doPut(final HttpServletRequest request,
444      final HttpServletResponse response) throws ServletException, IOException {
445    try {
446      ServletContext context = getServletContext();
447      final FSImage nnImage = NameNodeHttpServer.getFsImageFromContext(context);
448      final Configuration conf = (Configuration) getServletContext()
449          .getAttribute(JspHelper.CURRENT_CONF);
450      final PutImageParams parsedParams = new PutImageParams(request, response,
451          conf);
452      final NameNodeMetrics metrics = NameNode.getNameNodeMetrics();
453
454      validateRequest(context, conf, request, response, nnImage,
455          parsedParams.getStorageInfoString());
456
457      UserGroupInformation.getCurrentUser().doAs(
458          new PrivilegedExceptionAction<Void>() {
459
460            @Override
461            public Void run() throws Exception {
462
463              final long txid = parsedParams.getTxId();
464
465              final NameNodeFile nnf = parsedParams.getNameNodeFile();
466
467              if (!nnImage.addToCheckpointing(txid)) {
468                response.sendError(HttpServletResponse.SC_CONFLICT,
469                    "Either current namenode is checkpointing or another"
470                        + " checkpointer is already in the process of "
471                        + "uploading a checkpoint made at transaction ID "
472                        + txid);
473                return null;
474              }
475              try {
476                if (nnImage.getStorage().findImageFile(nnf, txid) != null) {
477                  response.sendError(HttpServletResponse.SC_CONFLICT,
478                      "Either current namenode has checkpointed or "
479                          + "another checkpointer already uploaded an "
480                          + "checkpoint for txid " + txid);
481                  return null;
482                }
483
484                InputStream stream = request.getInputStream();
485                try {
486                  long start = monotonicNow();
487                  MD5Hash downloadImageDigest = TransferFsImage
488                      .handleUploadImageRequest(request, txid,
489                          nnImage.getStorage(), stream,
490                          parsedParams.getFileSize(), getThrottler(conf));
491                  nnImage.saveDigestAndRenameCheckpointImage(nnf, txid,
492                      downloadImageDigest);
493                  // Metrics non-null only when used inside name node
494                  if (metrics != null) {
495                    long elapsed = monotonicNow() - start;
496                    metrics.addPutImage(elapsed);
497                  }
498                  // Now that we have a new checkpoint, we might be able to
499                  // remove some old ones.
500                  nnImage.purgeOldStorage(nnf);
501                } finally {
502                  stream.close();
503                }
504              } finally {
505                nnImage.removeFromCheckpointing(txid);
506              }
507              return null;
508            }
509
510          });
511    } catch (Throwable t) {
512      String errMsg = "PutImage failed. " + StringUtils.stringifyException(t);
513      response.sendError(HttpServletResponse.SC_GONE, errMsg);
514      throw new IOException(errMsg);
515    }
516  }

518  /*
519   * Params required to handle put image request
520   */
521  static class PutImageParams {
522    private long txId = -1;
523    private String storageInfoString = null;
524    private long fileSize = 0L;
525    private NameNodeFile nnf;
526
527    public PutImageParams(HttpServletRequest request,
528        HttpServletResponse response, Configuration conf) throws IOException {
529      txId = ServletUtil.parseLongParam(request, TXID_PARAM);
530      storageInfoString = ServletUtil.getParameter(request, STORAGEINFO_PARAM);
531      fileSize = ServletUtil.parseLongParam(request,
532          TransferFsImage.FILE_LENGTH);
533      String imageType = ServletUtil.getParameter(request, IMAGE_FILE_TYPE);
534      nnf = imageType == null ? NameNodeFile.IMAGE : NameNodeFile
535          .valueOf(imageType);
536      if (fileSize == 0 || txId == -1 || storageInfoString == null
537          || storageInfoString.isEmpty()) {
538        throw new IOException("Illegal parameters to TransferFsImage");
539      }
540    }

543      return txId;
544    }
545
546    public String getStorageInfoString() {
547      return storageInfoString;
548    }
549
550    public long getFileSize() {
551      return fileSize;
552    }
553
554    public NameNodeFile getNameNodeFile() {
555      return nnf;
556    }
