064/** <p>The balancer is a tool that balances disk space usage on an HDFS cluster
065 * when some datanodes become full or when new empty nodes join the cluster.
066 * The tool is deployed as an application program that can be run by the 
067 * cluster administrator on a live HDFS cluster while applications
068 * adding and deleting files.
069 * 
070 * <p>SYNOPSIS
071 * <pre>
072 * To start:
073 *      bin/start-balancer.sh [-threshold <threshold>]
074 *      Example: bin/ start-balancer.sh 
075 *                     start the balancer with a default threshold of 10%
076 *               bin/ start-balancer.sh -threshold 5
077 *                     start the balancer with a threshold of 5%
078 *               bin/ start-balancer.sh -idleiterations 20
079 *                     start the balancer with maximum 20 consecutive idle iterations
080 *               bin/ start-balancer.sh -idleiterations -1
081 *                     run the balancer with default threshold infinitely
082 * To stop:
083 *      bin/ stop-balancer.sh
084 * </pre>
085 * 
086 * <p>DESCRIPTION
087 * <p>The threshold parameter is a fraction in the range of (1%, 100%) with a 
088 * default value of 10%. The threshold sets a target for whether the cluster 
089 * is balanced. A cluster is balanced if for each datanode, the utilization 
090 * of the node (ratio of used space at the node to total capacity of the node) 
091 * differs from the utilization of the (ratio of used space in the cluster 
092 * to total capacity of the cluster) by no more than the threshold value. 
093 * The smaller the threshold, the more balanced a cluster will become. 
094 * It takes more time to run the balancer for small threshold values. 
095 * Also for a very small threshold the cluster may not be able to reach the 
096 * balanced state when applications write and delete files concurrently.
097 * 
098 * <p>The tool moves blocks from highly utilized datanodes to poorly 
099 * utilized datanodes iteratively. In each iteration a datanode moves or 
100 * receives no more than the lesser of 10G bytes or the threshold fraction 
101 * of its capacity. Each iteration runs no more than 20 minutes.
102 * At the end of each iteration, the balancer obtains updated datanodes
103 * information from the namenode.
104 * 
105 * <p>A system property that limits the balancer's use of bandwidth is 
106 * defined in the default configuration file:
107 * <pre>
108 * <property>
109 *   <name>dfs.balance.bandwidthPerSec</name>
110 *   <value>1048576</value>
111 * <description>  Specifies the maximum bandwidth that each datanode 
112 * can utilize for the balancing purpose in term of the number of bytes 
113 * per second. </description>
114 * </property>
115 * </pre>
116 * 
117 * <p>This property determines the maximum speed at which a block will be 
118 * moved from one datanode to another. The default value is 1MB/s. The higher 
119 * the bandwidth, the faster a cluster can reach the balanced state, 
120 * but with greater competition with application processes. If an 
121 * administrator changes the value of this property in the configuration 
122 * file, the change is observed when HDFS is next restarted.
123 * 
124 * <p>MONITERING BALANCER PROGRESS
125 * <p>After the balancer is started, an output file name where the balancer 
126 * progress will be recorded is printed on the screen.  The administrator 
127 * can monitor the running of the balancer by reading the output file. 
128 * The output shows the balancer's status iteration by iteration. In each 
129 * iteration it prints the starting time, the iteration number, the total 
130 * number of bytes that have been moved in the previous iterations, 
131 * the total number of bytes that are left to move in order for the cluster 
132 * to be balanced, and the number of bytes that are being moved in this 
133 * iteration. Normally "Bytes Already Moved" is increasing while "Bytes Left 
134 * To Move" is decreasing.
135 * 
136 * <p>Running multiple instances of the balancer in an HDFS cluster is 
137 * prohibited by the tool.
138 * 
139 * <p>The balancer automatically exits when any of the following five 
140 * conditions is satisfied:
141 * <ol>
142 * <li>The cluster is balanced;
143 * <li>No block can be moved;
144 * <li>No block has been moved for specified consecutive iterations (5 by default);
145 * <li>An IOException occurs while communicating with the namenode;
146 * <li>Another balancer is running.
147 * </ol>
148 * 
149 * <p>Upon exit, a balancer returns an exit code and prints one of the 
150 * following messages to the output file in corresponding to the above exit 
151 * reasons:
152 * <ol>
153 * <li>The cluster is balanced. Exiting
154 * <li>No block can be moved. Exiting...
155 * <li>No block has been moved for specified iterations (5 by default). Exiting...
156 * <li>Received an IO exception: failure reason. Exiting...
157 * <li>Another balancer is running. Exiting...
158 * </ol>
159 * 
160 * <p>The administrator can interrupt the execution of the balancer at any 
161 * time by running the command "stop-balancer.sh" on the machine where the 
162 * balancer is running.
163 */

217  Balancer(NameNodeConnector theblockpool, Parameters p, Configuration conf) {
218    final long movedWinWidth = conf.getLong(
219        DFSConfigKeys.DFS_BALANCER_MOVEDWINWIDTH_KEY,
220        DFSConfigKeys.DFS_BALANCER_MOVEDWINWIDTH_DEFAULT);
221    final int moverThreads = conf.getInt(
222        DFSConfigKeys.DFS_BALANCER_MOVERTHREADS_KEY,
223        DFSConfigKeys.DFS_BALANCER_MOVERTHREADS_DEFAULT);
224    final int dispatcherThreads = conf.getInt(
225        DFSConfigKeys.DFS_BALANCER_DISPATCHERTHREADS_KEY,
226        DFSConfigKeys.DFS_BALANCER_DISPATCHERTHREADS_DEFAULT);
227    final int maxConcurrentMovesPerNode = conf.getInt(
228        DFSConfigKeys.DFS_DATANODE_BALANCE_MAX_NUM_CONCURRENT_MOVES_KEY,
229        DFSConfigKeys.DFS_DATANODE_BALANCE_MAX_NUM_CONCURRENT_MOVES_DEFAULT);
230
231    this.dispatcher = new Dispatcher(theblockpool, p.nodesToBeIncluded,
232        p.nodesToBeExcluded, movedWinWidth, moverThreads, dispatcherThreads,
233        maxConcurrentMovesPerNode, conf);
234    this.threshold = p.threshold;
235    this.policy = p.policy;
236  }

238  private static long getCapacity(DatanodeStorageReport report, StorageType t) {
239    long capacity = 0L;
240    for(StorageReport r : report.getStorageReports()) {
241      if (r.getStorage().getStorageType() == t) {
242        capacity += r.getCapacity();
243      }
244    }
245    return capacity;
246  }
247
248  private static long getRemaining(DatanodeStorageReport report, StorageType t) {
249    long remaining = 0L;
250    for(StorageReport r : report.getStorageReports()) {
251      if (r.getStorage().getStorageType() == t) {
252        remaining += r.getRemaining();
253      }
254    }
255    return remaining;
256  }

258  /**
259   * Given a datanode storage set, build a network topology and decide
260   * over-utilized storages, above average utilized storages, 
261   * below average utilized storages, and underutilized storages. 
262   * The input datanode storage set is shuffled in order to randomize
263   * to the storage matching later on.
264   *
265   * @return the number of bytes needed to move in order to balance the cluster.
266   */
267  private long init(List<DatanodeStorageReport> reports) {
268    // compute average utilization
269    for (DatanodeStorageReport r : reports) {
270      policy.accumulateSpaces(r);
271    }
272    policy.initAvgUtilization();
273
274    // create network topology and classify utilization collections: 
275    //   over-utilized, above-average, below-average and under-utilized.
276    long overLoadedBytes = 0L, underLoadedBytes = 0L;
277    for(DatanodeStorageReport r : reports) {
278      final DDatanode dn = dispatcher.newDatanode(r.getDatanodeInfo());
279      for(StorageType t : StorageType.getMovableTypes()) {
280        final Double utilization = policy.getUtilization(r, t);
281        if (utilization == null) { // datanode does not have such storage type 
282          continue;
283        }
284        
285        final long capacity = getCapacity(r, t);
286        final double utilizationDiff = utilization - policy.getAvgUtilization(t);
287        final double thresholdDiff = Math.abs(utilizationDiff) - threshold;
288        final long maxSize2Move = computeMaxSize2Move(capacity,
289            getRemaining(r, t), utilizationDiff, threshold);
290
291        final StorageGroup g;
292        if (utilizationDiff > 0) {
293          final Source s = dn.addSource(t, maxSize2Move, dispatcher);
294          if (thresholdDiff <= 0) { // within threshold
295            aboveAvgUtilized.add(s);
296          } else {
297            overLoadedBytes += precentage2bytes(thresholdDiff, capacity);
298            overUtilized.add(s);
299          }
300          g = s;
301        } else {
302          g = dn.addTarget(t, maxSize2Move);
303          if (thresholdDiff <= 0) { // within threshold
304            belowAvgUtilized.add(g);
305          } else {
306            underLoadedBytes += precentage2bytes(thresholdDiff, capacity);
307            underUtilized.add(g);
308          }
309        }
310        dispatcher.getStorageGroupMap().put(g);
311      }
312    }
313
314    logUtilizationCollections();
315    
316    Preconditions.checkState(dispatcher.getStorageGroupMap().size()
317        == overUtilized.size() + underUtilized.size() + aboveAvgUtilized.size()
318           + belowAvgUtilized.size(),
319        "Mismatched number of storage groups");
320    
321    // return number of bytes to be moved in order to make the cluster balanced
322    return Math.max(overLoadedBytes, underLoadedBytes);
323  }
324
325  private static long computeMaxSize2Move(final long capacity, final long remaining,
326      final double utilizationDiff, final double threshold) {
327    final double diff = Math.min(threshold, Math.abs(utilizationDiff));
328    long maxSizeToMove = precentage2bytes(diff, capacity);
329    if (utilizationDiff < 0) {
330      maxSizeToMove = Math.min(remaining, maxSizeToMove);
331    }
332    return Math.min(MAX_SIZE_TO_MOVE, maxSizeToMove);
333  }
334
335  private static long precentage2bytes(double precentage, long capacity) {
336    Preconditions.checkArgument(precentage >= 0,
337        "precentage = " + precentage + " < 0");
338    return (long)(precentage * capacity / 100.0);
339  }

356  /**
357   * Decide all <source, target> pairs and
358   * the number of bytes to move from a source to a target
359   * Maximum bytes to be moved per storage group is
360   * min(1 Band worth of bytes,  MAX_SIZE_TO_MOVE).
361   * @return total number of bytes to move in this iteration
362   */
363  private long chooseStorageGroups() {
364    // First, match nodes on the same node group if cluster is node group aware
365    if (dispatcher.getCluster().isNodeGroupAware()) {
366      chooseStorageGroups(Matcher.SAME_NODE_GROUP);
367    }
368    
369    // Then, match nodes on the same rack
370    chooseStorageGroups(Matcher.SAME_RACK);
371    // At last, match all remaining nodes
372    chooseStorageGroups(Matcher.ANY_OTHER);
373    
374    return dispatcher.bytesToMove();
375  }
376
377  /** Decide all <source, target> pairs according to the matcher. */
378  private void chooseStorageGroups(final Matcher matcher) {
379    /* first step: match each overUtilized datanode (source) to
380     * one or more underUtilized datanodes (targets).
381     */
382    chooseStorageGroups(overUtilized, underUtilized, matcher);
383    
384    /* match each remaining overutilized datanode (source) to 
385     * below average utilized datanodes (targets).
386     * Note only overutilized datanodes that haven't had that max bytes to move
387     * satisfied in step 1 are selected
388     */
389    chooseStorageGroups(overUtilized, belowAvgUtilized, matcher);
390
391    /* match each remaining underutilized datanode (target) to 
392     * above average utilized datanodes (source).
393     * Note only underutilized datanodes that have not had that max bytes to
394     * move satisfied in step 1 are selected.
395     */
396    chooseStorageGroups(underUtilized, aboveAvgUtilized, matcher);
397  }

399  /**
400   * For each datanode, choose matching nodes from the candidates. Either the
401   * datanodes or the candidates are source nodes with (utilization > Avg), and
402   * the others are target nodes with (utilization < Avg).
403   */
404  private <G extends StorageGroup, C extends StorageGroup>
405      void chooseStorageGroups(Collection<G> groups, Collection<C> candidates,
406          Matcher matcher) {
407    for(final Iterator<G> i = groups.iterator(); i.hasNext();) {
408      final G g = i.next();
409      for(; choose4One(g, candidates, matcher); );
410      if (!g.hasSpaceForScheduling()) {
411        i.remove();
412      }
413    }
414  }
415
416  /**
417   * For the given datanode, choose a candidate and then schedule it.
418   * @return true if a candidate is chosen; false if no candidates is chosen.
419   */
420  private <C extends StorageGroup> boolean choose4One(StorageGroup g,
421      Collection<C> candidates, Matcher matcher) {
422    final Iterator<C> i = candidates.iterator();
423    final C chosen = chooseCandidate(g, i, matcher);
424  
425    if (chosen == null) {
426      return false;
427    }
428    if (g instanceof Source) {
429      matchSourceWithTargetToMove((Source)g, chosen);
430    } else {
431      matchSourceWithTargetToMove((Source)chosen, g);
432    }
433    if (!chosen.hasSpaceForScheduling()) {
434      i.remove();
435    }
436    return true;
437  }
438  
439  private void matchSourceWithTargetToMove(Source source, StorageGroup target) {
440    long size = Math.min(source.availableSizeToMove(), target.availableSizeToMove());
441    final Task task = new Task(target, size);
442    source.addTask(task);
443    target.incScheduledSize(task.getSize());
444    dispatcher.add(source, target);
445    LOG.info("Decided to move "+StringUtils.byteDesc(size)+" bytes from "
446        + source.getDisplayName() + " to " + target.getDisplayName());
447  }
448  
449  /** Choose a candidate for the given datanode. */
450  private <G extends StorageGroup, C extends StorageGroup>
451      C chooseCandidate(G g, Iterator<C> candidates, Matcher matcher) {
452    if (g.hasSpaceForScheduling()) {
453      for(; candidates.hasNext(); ) {
454        final C c = candidates.next();
455        if (!c.hasSpaceForScheduling()) {
456          candidates.remove();
457        } else if (matcher.match(dispatcher.getCluster(),
458            g.getDatanodeInfo(), c.getDatanodeInfo())) {
459          return c;
460        }
461      }
462    }
463    return null;
464  }

476  static class Result {
477    final ExitStatus exitStatus;
478    final long bytesLeftToMove;
479    final long bytesBeingMoved;
480    final long bytesAlreadyMoved;
481
482    Result(ExitStatus exitStatus, long bytesLeftToMove, long bytesBeingMoved,
483        long bytesAlreadyMoved) {
484      this.exitStatus = exitStatus;
485      this.bytesLeftToMove = bytesLeftToMove;
486      this.bytesBeingMoved = bytesBeingMoved;
487      this.bytesAlreadyMoved = bytesAlreadyMoved;
488    }
489
490    void print(int iteration, PrintStream out) {
491      out.printf("%-24s %10d  %19s  %18s  %17s%n",
492          DateFormat.getDateTimeInstance().format(new Date()), iteration,
493          StringUtils.byteDesc(bytesAlreadyMoved),
494          StringUtils.byteDesc(bytesLeftToMove),
495          StringUtils.byteDesc(bytesBeingMoved));
496    }
497  }

499  Result newResult(ExitStatus exitStatus, long bytesLeftToMove, long bytesBeingMoved) {
500    return new Result(exitStatus, bytesLeftToMove, bytesBeingMoved,
501        dispatcher.getBytesMoved());
502  }

504  Result newResult(ExitStatus exitStatus) {
505    return new Result(exitStatus, -1, -1, dispatcher.getBytesMoved());
506  }

508  /** Run an iteration for all datanodes. */
509  Result runOneIteration() {
510    try {
511      final List<DatanodeStorageReport> reports = dispatcher.init();
512      final long bytesLeftToMove = init(reports);
513      if (bytesLeftToMove == 0) {
514        System.out.println("The cluster is balanced. Exiting...");
515        return newResult(ExitStatus.SUCCESS, bytesLeftToMove, -1);
516      } else {
517        LOG.info( "Need to move "+ StringUtils.byteDesc(bytesLeftToMove)
518            + " to make the cluster balanced." );
519      }
520      
521      /* Decide all the nodes that will participate in the block move and
522       * the number of bytes that need to be moved from one node to another
523       * in this iteration. Maximum bytes to be moved per node is
524       * Min(1 Band worth of bytes,  MAX_SIZE_TO_MOVE).
525       */
526      final long bytesBeingMoved = chooseStorageGroups();
527      if (bytesBeingMoved == 0) {
528        System.out.println("No block can be moved. Exiting...");
529        return newResult(ExitStatus.NO_MOVE_BLOCK, bytesLeftToMove, bytesBeingMoved);
530      } else {
531        LOG.info( "Will move " + StringUtils.byteDesc(bytesBeingMoved) +
532            " in this iteration");
533      }
534      
535      /* For each pair of <source, target>, start a thread that repeatedly 
536       * decide a block to be moved and its proxy source, 
537       * then initiates the move until all bytes are moved or no more block
538       * available to move.
539       * Exit no byte has been moved for 5 consecutive iterations.
540       */
541      if (!dispatcher.dispatchAndCheckContinue()) {
542        return newResult(ExitStatus.NO_MOVE_PROGRESS, bytesLeftToMove, bytesBeingMoved);
543      }
544
545      return newResult(ExitStatus.IN_PROGRESS, bytesLeftToMove, bytesBeingMoved);
546    } catch (IllegalArgumentException e) {
547      System.out.println(e + ".  Exiting ...");
548      return newResult(ExitStatus.ILLEGAL_ARGUMENTS);
549    } catch (IOException e) {
550      System.out.println(e + ".  Exiting ...");
551      return newResult(ExitStatus.IO_EXCEPTION);
552    } catch (InterruptedException e) {
553      System.out.println(e + ".  Exiting ...");
554      return newResult(ExitStatus.INTERRUPTED);
555    } finally {
556      dispatcher.shutdownNow();
557    }
558  }

560  /**
561   * Balance all namenodes.
562   * For each iteration,
563   * for each namenode,
564   * execute a {@link Balancer} to work through all datanodes once.  
565   */
566  static int run(Collection<URI> namenodes, final Parameters p,
567      Configuration conf) throws IOException, InterruptedException {
568    final long sleeptime =
569        conf.getLong(DFSConfigKeys.DFS_HEARTBEAT_INTERVAL_KEY,
570            DFSConfigKeys.DFS_HEARTBEAT_INTERVAL_DEFAULT) * 2000 +
571        conf.getLong(DFSConfigKeys.DFS_NAMENODE_REPLICATION_INTERVAL_KEY,
572            DFSConfigKeys.DFS_NAMENODE_REPLICATION_INTERVAL_DEFAULT) * 1000;
573    LOG.info("namenodes  = " + namenodes);
574    LOG.info("parameters = " + p);
575    
576    System.out.println("Time Stamp               Iteration#  Bytes Already Moved  Bytes Left To Move  Bytes Being Moved");
577    
578    List<NameNodeConnector> connectors = Collections.emptyList();
579    try {
580      connectors = NameNodeConnector.newNameNodeConnectors(namenodes, 
581            Balancer.class.getSimpleName(), BALANCER_ID_PATH, conf, p.maxIdleIteration);
582    
583      boolean done = false;
584      for(int iteration = 0; !done; iteration++) {
585        done = true;
586        Collections.shuffle(connectors);
587        for(NameNodeConnector nnc : connectors) {
588          final Balancer b = new Balancer(nnc, p, conf);
589          final Result r = b.runOneIteration();
590          r.print(iteration, System.out);
591
592          // clean all lists
593          b.resetData(conf);
594          if (r.exitStatus == ExitStatus.IN_PROGRESS) {
595            done = false;
596          } else if (r.exitStatus != ExitStatus.SUCCESS) {
597            //must be an error statue, return.
598            return r.exitStatus.getExitCode();
599          }
600        }
601
602        if (!done) {
603          Thread.sleep(sleeptime);
604        }
605      }
606    } finally {
607      for(NameNodeConnector nnc : connectors) {
608        IOUtils.cleanup(LOG, nnc);
609      }
610    }
611    return ExitStatus.SUCCESS.getExitCode();
612  }

614  /* Given elaspedTime in ms, return a printable string */
615  private static String time2Str(long elapsedTime) {
616    String unit;
617    double time = elapsedTime;
618    if (elapsedTime < 1000) {
619      unit = "milliseconds";
620    } else if (elapsedTime < 60*1000) {
621      unit = "seconds";
622      time = time/1000;
623    } else if (elapsedTime < 3600*1000) {
624      unit = "minutes";
625      time = time/(60*1000);
626    } else {
627      unit = "hours";
628      time = time/(3600*1000);
629    }
630
631    return time+" "+unit;
632  }

667  static class Cli extends Configured implements Tool {
668    /**
669     * Parse arguments and then run Balancer.
670     * 
671     * @param args command specific arguments.
672     * @return exit code. 0 indicates success, non-zero indicates failure.
673     */
674    @Override
675    public int run(String[] args) {
676      final long startTime = Time.monotonicNow();
677      final Configuration conf = getConf();
678
679      try {
680        checkReplicationPolicyCompatibility(conf);
681
682        final Collection<URI> namenodes = DFSUtil.getNsServiceRpcUris(conf);
683        return Balancer.run(namenodes, parse(args), conf);
684      } catch (IOException e) {
685        System.out.println(e + ".  Exiting ...");
686        return ExitStatus.IO_EXCEPTION.getExitCode();
687      } catch (InterruptedException e) {
688        System.out.println(e + ".  Exiting ...");
689        return ExitStatus.INTERRUPTED.getExitCode();
690      } finally {
691        System.out.format("%-24s ",
692            DateFormat.getDateTimeInstance().format(new Date()));
693        System.out.println("Balancing took "
694            + time2Str(Time.monotonicNow() - startTime));
695      }
696    }
697
698    /** parse command line arguments */
699    static Parameters parse(String[] args) {
700      BalancingPolicy policy = Parameters.DEFAULT.policy;
701      double threshold = Parameters.DEFAULT.threshold;
702      int maxIdleIteration = Parameters.DEFAULT.maxIdleIteration;
703      Set<String> nodesTobeExcluded = Parameters.DEFAULT.nodesToBeExcluded;
704      Set<String> nodesTobeIncluded = Parameters.DEFAULT.nodesToBeIncluded;
705
706      if (args != null) {
707        try {
708          for(int i = 0; i < args.length; i++) {
709            if ("-threshold".equalsIgnoreCase(args[i])) {
710              checkArgument(++i < args.length,
711                "Threshold value is missing: args = " + Arrays.toString(args));
712              try {
713                threshold = Double.parseDouble(args[i]);
714                if (threshold < 1 || threshold > 100) {
715                  throw new IllegalArgumentException(
716                      "Number out of range: threshold = " + threshold);
717                }
718                LOG.info( "Using a threshold of " + threshold );
719              } catch(IllegalArgumentException e) {
720                System.err.println(
721                    "Expecting a number in the range of [1.0, 100.0]: "
722                    + args[i]);
723                throw e;
724              }
725            } else if ("-policy".equalsIgnoreCase(args[i])) {
726              checkArgument(++i < args.length,
727                "Policy value is missing: args = " + Arrays.toString(args));
728              try {
729                policy = BalancingPolicy.parse(args[i]);
730              } catch(IllegalArgumentException e) {
731                System.err.println("Illegal policy name: " + args[i]);
732                throw e;
733              }
734            } else if ("-exclude".equalsIgnoreCase(args[i])) {
735              checkArgument(++i < args.length,
736                  "List of nodes to exclude | -f <filename> is missing: args = "
737                  + Arrays.toString(args));
738              if ("-f".equalsIgnoreCase(args[i])) {
739                checkArgument(++i < args.length,
740                    "File containing nodes to exclude is not specified: args = "
741                    + Arrays.toString(args));
742                nodesTobeExcluded = Util.getHostListFromFile(args[i], "exclude");
743              } else {
744                nodesTobeExcluded = Util.parseHostList(args[i]);
745              }
746            } else if ("-include".equalsIgnoreCase(args[i])) {
747              checkArgument(++i < args.length,
748                "List of nodes to include | -f <filename> is missing: args = "
749                + Arrays.toString(args));
750              if ("-f".equalsIgnoreCase(args[i])) {
751                checkArgument(++i < args.length,
752                    "File containing nodes to include is not specified: args = "
753                    + Arrays.toString(args));
754                nodesTobeIncluded = Util.getHostListFromFile(args[i], "include");
755               } else {
756                nodesTobeIncluded = Util.parseHostList(args[i]);
757              }
758            } else if ("-idleiterations".equalsIgnoreCase(args[i])) {
759              checkArgument(++i < args.length,
760                  "idleiterations value is missing: args = " + Arrays.toString(args));
761              maxIdleIteration = Integer.parseInt(args[i]);
762              LOG.info("Using a idleiterations of " + maxIdleIteration);
763            } else {
764              throw new IllegalArgumentException("args = "
765                  + Arrays.toString(args));
766            }
767          }
768          checkArgument(nodesTobeExcluded.isEmpty() || nodesTobeIncluded.isEmpty(),
769              "-exclude and -include options cannot be specified together.");
770        } catch(RuntimeException e) {
771          printUsage(System.err);
772          throw e;
773        }
774      }
775      
776      return new Parameters(policy, threshold, maxIdleIteration, nodesTobeExcluded, nodesTobeIncluded);
777    }
