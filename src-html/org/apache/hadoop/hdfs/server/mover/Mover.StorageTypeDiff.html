114  Mover(NameNodeConnector nnc, Configuration conf) {
115    final long movedWinWidth = conf.getLong(
116        DFSConfigKeys.DFS_MOVER_MOVEDWINWIDTH_KEY,
117        DFSConfigKeys.DFS_MOVER_MOVEDWINWIDTH_DEFAULT);
118    final int moverThreads = conf.getInt(
119        DFSConfigKeys.DFS_MOVER_MOVERTHREADS_KEY,
120        DFSConfigKeys.DFS_MOVER_MOVERTHREADS_DEFAULT);
121    final int maxConcurrentMovesPerNode = conf.getInt(
122        DFSConfigKeys.DFS_DATANODE_BALANCE_MAX_NUM_CONCURRENT_MOVES_KEY,
123        DFSConfigKeys.DFS_DATANODE_BALANCE_MAX_NUM_CONCURRENT_MOVES_DEFAULT);
124
125    this.dispatcher = new Dispatcher(nnc, Collections.<String> emptySet(),
126        Collections.<String> emptySet(), movedWinWidth, moverThreads, 0,
127        maxConcurrentMovesPerNode, conf);
128    this.storages = new StorageMap();
129    this.targetPaths = nnc.getTargetPaths();
130    this.blockStoragePolicies = new BlockStoragePolicy[1 <<
131        BlockStoragePolicySuite.ID_BIT_LENGTH];
132  }
133
134  void init() throws IOException {
135    initStoragePolicies();
136    final List<DatanodeStorageReport> reports = dispatcher.init();
137    for(DatanodeStorageReport r : reports) {
138      final DDatanode dn = dispatcher.newDatanode(r.getDatanodeInfo());
139      for(StorageType t : StorageType.getMovableTypes()) {
140        final Source source = dn.addSource(t, Long.MAX_VALUE, dispatcher);
141        final long maxRemaining = getMaxRemaining(r, t);
142        final StorageGroup target = maxRemaining > 0L ? dn.addTarget(t,
143            maxRemaining) : null;
144        storages.add(source, target);
145      }
146    }
147  }
148
149  private void initStoragePolicies() throws IOException {
150    BlockStoragePolicy[] policies = dispatcher.getDistributedFileSystem()
151        .getStoragePolicies();
152    for (BlockStoragePolicy policy : policies) {
153      this.blockStoragePolicies[policy.getId()] = policy;
154    }
155  }
156
157  private ExitStatus run() {
158    try {
159      init();
160      boolean hasRemaining = new Processor().processNamespace();
161      return hasRemaining ? ExitStatus.IN_PROGRESS : ExitStatus.SUCCESS;
162    } catch (IllegalArgumentException e) {
163      System.out.println(e + ".  Exiting ...");
164      return ExitStatus.ILLEGAL_ARGUMENTS;
165    } catch (IOException e) {
166      System.out.println(e + ".  Exiting ...");
167      return ExitStatus.IO_EXCEPTION;
168    } finally {
169      dispatcher.shutdownNow();
170    }
171  }

173  DBlock newDBlock(Block block, List<MLocation> locations) {
174    final DBlock db = new DBlock(block);
175    for(MLocation ml : locations) {
176      StorageGroup source = storages.getSource(ml);
177      if (source != null) {
178        db.addLocation(source);
179      }
180    }
181    return db;
182  }

184  private static long getMaxRemaining(DatanodeStorageReport report, StorageType t) {
185    long max = 0L;
186    for(StorageReport r : report.getStorageReports()) {
187      if (r.getStorage().getStorageType() == t) {
188        if (r.getRemaining() > max) {
189          max = r.getRemaining();
190        }
191      }
192    }
193    return max;
194  }

196  /**
197   * convert a snapshot path to non-snapshot path. E.g.,
198   * /foo/.snapshot/snapshot-name/bar --> /foo/bar
199   */
200  private static String convertSnapshotPath(String[] pathComponents) {
201    StringBuilder sb = new StringBuilder(Path.SEPARATOR);
202    for (int i = 0; i < pathComponents.length; i++) {
203      if (pathComponents[i].equals(HdfsConstants.DOT_SNAPSHOT_DIR)) {
204        i++;
205      } else {
206        sb.append(pathComponents[i]);
207      }
208    }
209    return sb.toString();
210  }

212  class Processor {
213    private final DFSClient dfs;
214    private final List<String> snapshottableDirs = new ArrayList<String>();
215
216    Processor() {
217      dfs = dispatcher.getDistributedFileSystem().getClient();
218    }
219
220    private void getSnapshottableDirs() {
221      SnapshottableDirectoryStatus[] dirs = null;
222      try {
223        dirs = dfs.getSnapshottableDirListing();
224      } catch (IOException e) {
225        LOG.warn("Failed to get snapshottable directories."
226            + " Ignore and continue.", e);
227      }
228      if (dirs != null) {
229        for (SnapshottableDirectoryStatus dir : dirs) {
230          snapshottableDirs.add(dir.getFullPath().toString());
231        }
232      }
233    }
234
235    /**
236     * @return true if the given path is a snapshot path and the corresponding
237     * INode is still in the current fsdirectory.
238     */
239    private boolean isSnapshotPathInCurrent(String path) throws IOException {
240      // if the parent path contains "/.snapshot/", this is a snapshot path
241      if (path.contains(HdfsConstants.SEPARATOR_DOT_SNAPSHOT_DIR_SEPARATOR)) {
242        String[] pathComponents = INode.getPathNames(path);
243        if (HdfsConstants.DOT_SNAPSHOT_DIR
244            .equals(pathComponents[pathComponents.length - 2])) {
245          // this is a path for a specific snapshot (e.g., /foo/.snapshot/s1)
246          return false;
247        }
248        String nonSnapshotPath = convertSnapshotPath(pathComponents);
249        return dfs.getFileInfo(nonSnapshotPath) != null;
250      } else {
251        return false;
252      }
253    }
254
255    /**
256     * @return whether there is still remaining migration work for the next
257     *         round
258     */
259    private boolean processNamespace() {
260      getSnapshottableDirs();
261      boolean hasRemaining = false;
262      for (Path target : targetPaths) {
263        hasRemaining |= processPath(target.toUri().getPath());
264      }
265      // wait for pending move to finish and retry the failed migration
266      hasRemaining |= Dispatcher.waitForMoveCompletion(storages.targets.values());
267      return hasRemaining;
268    }
269
270    /**
271     * @return whether there is still remaing migration work for the next
272     *         round
273     */
274    private boolean processPath(String fullPath) {
275      boolean hasRemaining = false;
276      for (byte[] lastReturnedName = HdfsFileStatus.EMPTY_NAME;;) {
277        final DirectoryListing children;
278        try {
279          children = dfs.listPaths(fullPath, lastReturnedName, true);
280        } catch(IOException e) {
281          LOG.warn("Failed to list directory " + fullPath
282              + ". Ignore the directory and continue.", e);
283          return hasRemaining;
284        }
285        if (children == null) {
286          return hasRemaining;
287        }
288        for (HdfsFileStatus child : children.getPartialListing()) {
289          hasRemaining |= processRecursively(fullPath, child);
290        }
291        if (children.hasMore()) {
292          lastReturnedName = children.getLastName();
293        } else {
294          return hasRemaining;
295        }
296      }
297    }
298
299    /** @return whether the migration requires next round */
300    private boolean processRecursively(String parent, HdfsFileStatus status) {
301      String fullPath = status.getFullName(parent);
302      boolean hasRemaining = false;
303      if (status.isDir()) {
304        if (!fullPath.endsWith(Path.SEPARATOR)) {
305          fullPath = fullPath + Path.SEPARATOR;
306        }
307
308        hasRemaining = processPath(fullPath);
309        // process snapshots if this is a snapshottable directory
310        if (snapshottableDirs.contains(fullPath)) {
311          final String dirSnapshot = fullPath + HdfsConstants.DOT_SNAPSHOT_DIR;
312          hasRemaining |= processPath(dirSnapshot);
313        }
314      } else if (!status.isSymlink()) { // file
315        try {
316          if (!isSnapshotPathInCurrent(fullPath)) {
317            // the full path is a snapshot path but it is also included in the
318            // current directory tree, thus ignore it.
319            hasRemaining = processFile(fullPath, (HdfsLocatedFileStatus)status);
320          }
321        } catch (IOException e) {
322          LOG.warn("Failed to check the status of " + parent
323              + ". Ignore it and continue.", e);
324          return false;
325        }
326      }
327      return hasRemaining;
328    }
329
330    /** @return true if it is necessary to run another round of migration */
331    private boolean processFile(String fullPath, HdfsLocatedFileStatus status) {
332      final byte policyId = status.getStoragePolicy();
333      // currently we ignore files with unspecified storage policy
334      if (policyId == BlockStoragePolicySuite.ID_UNSPECIFIED) {
335        return false;
336      }
337      final BlockStoragePolicy policy = blockStoragePolicies[policyId];
338      if (policy == null) {
339        LOG.warn("Failed to get the storage policy of file " + fullPath);
340        return false;
341      }
342      final List<StorageType> types = policy.chooseStorageTypes(
343          status.getReplication());
344
345      final LocatedBlocks locatedBlocks = status.getBlockLocations();
346      boolean hasRemaining = false;
347      final boolean lastBlkComplete = locatedBlocks.isLastBlockComplete();
348      List<LocatedBlock> lbs = locatedBlocks.getLocatedBlocks();
349      for(int i = 0; i < lbs.size(); i++) {
350        if (i == lbs.size() - 1 && !lastBlkComplete) {
351          // last block is incomplete, skip it
352          continue;
353        }
354        LocatedBlock lb = lbs.get(i);
355        final StorageTypeDiff diff = new StorageTypeDiff(types,
356            lb.getStorageTypes());
357        if (!diff.removeOverlap(true)) {
358          if (scheduleMoves4Block(diff, lb)) {
359            hasRemaining |= (diff.existing.size() > 1 &&
360                diff.expected.size() > 1);
361          }
362        }
363      }
364      return hasRemaining;
365    }
366
367    boolean scheduleMoves4Block(StorageTypeDiff diff, LocatedBlock lb) {
368      final List<MLocation> locations = MLocation.toLocations(lb);
369      Collections.shuffle(locations);
370      final DBlock db = newDBlock(lb.getBlock().getLocalBlock(), locations);
371
372      for (final StorageType t : diff.existing) {
373        for (final MLocation ml : locations) {
374          final Source source = storages.getSource(ml);
375          if (ml.storageType == t && source != null) {
376            // try to schedule one replica move.
377            if (scheduleMoveReplica(db, source, diff.expected)) {
378              return true;
379            }
380          }
381        }
382      }
383      return false;
384    }
385
386    @VisibleForTesting
387    boolean scheduleMoveReplica(DBlock db, MLocation ml,
388                                List<StorageType> targetTypes) {
389      final Source source = storages.getSource(ml);
390      return source == null ? false : scheduleMoveReplica(db, source,
391          targetTypes);
392    }
393
394    boolean scheduleMoveReplica(DBlock db, Source source,
395        List<StorageType> targetTypes) {
396      // Match storage on the same node
397      if (chooseTargetInSameNode(db, source, targetTypes)) {
398        return true;
399      }
400
401      if (dispatcher.getCluster().isNodeGroupAware()) {
402        if (chooseTarget(db, source, targetTypes, Matcher.SAME_NODE_GROUP)) {
403          return true;
404        }
405      }
406      
407      // Then, match nodes on the same rack
408      if (chooseTarget(db, source, targetTypes, Matcher.SAME_RACK)) {
409        return true;
410      }
411      // At last, match all remaining nodes
412      return chooseTarget(db, source, targetTypes, Matcher.ANY_OTHER);
413    }
414
415    /**
416     * Choose the target storage within same Datanode if possible.
417     */
418    boolean chooseTargetInSameNode(DBlock db, Source source,
419        List<StorageType> targetTypes) {
420      for (StorageType t : targetTypes) {
421        StorageGroup target = storages.getTarget(source.getDatanodeInfo()
422            .getDatanodeUuid(), t);
423        if (target == null) {
424          continue;
425        }
426        final PendingMove pm = source.addPendingMove(db, target);
427        if (pm != null) {
428          dispatcher.executePendingMove(pm);
429          return true;
430        }
431      }
432      return false;
433    }
434
435    boolean chooseTarget(DBlock db, Source source,
436        List<StorageType> targetTypes, Matcher matcher) {
437      final NetworkTopology cluster = dispatcher.getCluster(); 
438      for (StorageType t : targetTypes) {
439        for(StorageGroup target : storages.getTargetStorages(t)) {
440          if (matcher.match(cluster, source.getDatanodeInfo(),
441              target.getDatanodeInfo())) {
442            final PendingMove pm = source.addPendingMove(db, target);
443            if (pm != null) {
444              dispatcher.executePendingMove(pm);
445              return true;
446            }
447          }
448        }
449      }
450      return false;
451    }

454  static class MLocation {
455    final DatanodeInfo datanode;
456    final StorageType storageType;
457    final long size;

459    MLocation(DatanodeInfo datanode, StorageType storageType, long size) {
460      this.datanode = datanode;
461      this.storageType = storageType;
462      this.size = size;
463    }
464    
465    static List<MLocation> toLocations(LocatedBlock lb) {
466      final DatanodeInfo[] datanodeInfos = lb.getLocations();
467      final StorageType[] storageTypes = lb.getStorageTypes();
468      final long size = lb.getBlockSize();
469      final List<MLocation> locations = new LinkedList<MLocation>();
470      for(int i = 0; i < datanodeInfos.length; i++) {
471        locations.add(new MLocation(datanodeInfos[i], storageTypes[i], size));
472      }
473      return locations;
474    }

477  @VisibleForTesting
478  static class StorageTypeDiff {
479    final List<StorageType> expected;
480    final List<StorageType> existing;
481
482    StorageTypeDiff(List<StorageType> expected, StorageType[] existing) {
483      this.expected = new LinkedList<StorageType>(expected);
484      this.existing = new LinkedList<StorageType>(Arrays.asList(existing));
485    }
486
487    /**
488     * Remove the overlap between the expected types and the existing types.
489     * @param  ignoreNonMovable ignore non-movable storage types
490     *         by removing them from both expected and existing storage type list
491     *         to prevent non-movable storage from being moved.
492     * @returns if the existing types or the expected types is empty after
493     *         removing the overlap.
494     */
495    boolean removeOverlap(boolean ignoreNonMovable) {
496      for(Iterator<StorageType> i = existing.iterator(); i.hasNext(); ) {
497        final StorageType t = i.next();
498        if (expected.remove(t)) {
499          i.remove();
500        }
501      }
502      if (ignoreNonMovable) {
503        removeNonMovable(existing);
504        removeNonMovable(expected);
505      }
506      return expected.isEmpty() || existing.isEmpty();
507    }
508
509    void removeNonMovable(List<StorageType> types) {
510      for (Iterator<StorageType> i = types.iterator(); i.hasNext(); ) {
511        final StorageType t = i.next();
512        if (!t.isMovable()) {
513          i.remove();
514        }
515      }
516    }
517
518    @Override
519    public String toString() {
520      return getClass().getSimpleName() + "{expected=" + expected
521          + ", existing=" + existing + "}";
522    }
523  }

525  static int run(Map<URI, List<Path>> namenodes, Configuration conf)
526      throws IOException, InterruptedException {
527    final long sleeptime =
528        conf.getLong(DFSConfigKeys.DFS_HEARTBEAT_INTERVAL_KEY,
529            DFSConfigKeys.DFS_HEARTBEAT_INTERVAL_DEFAULT) * 2000 +
530        conf.getLong(DFSConfigKeys.DFS_NAMENODE_REPLICATION_INTERVAL_KEY,
531            DFSConfigKeys.DFS_NAMENODE_REPLICATION_INTERVAL_DEFAULT) * 1000;
532    LOG.info("namenodes = " + namenodes);
533    
534    List<NameNodeConnector> connectors = Collections.emptyList();
535    try {
536      connectors = NameNodeConnector.newNameNodeConnectors(namenodes,
537          Mover.class.getSimpleName(), MOVER_ID_PATH, conf,
538          NameNodeConnector.DEFAULT_MAX_IDLE_ITERATIONS);
539
540      while (connectors.size() > 0) {
541        Collections.shuffle(connectors);
542        Iterator<NameNodeConnector> iter = connectors.iterator();
543        while (iter.hasNext()) {
544          NameNodeConnector nnc = iter.next();
545          final Mover m = new Mover(nnc, conf);
546          final ExitStatus r = m.run();
547
548          if (r == ExitStatus.SUCCESS) {
549            IOUtils.cleanup(LOG, nnc);
550            iter.remove();
551          } else if (r != ExitStatus.IN_PROGRESS) {
552            // must be an error statue, return
553            return r.getExitCode();
554          }
555        }
556        Thread.sleep(sleeptime);
557      }
558      return ExitStatus.SUCCESS.getExitCode();
559    } finally {
560      for (NameNodeConnector nnc : connectors) {
561        IOUtils.cleanup(LOG, nnc);
562      }
563    }
564  }

587    private static String[] readPathFile(String file) throws IOException {
588      List<String> list = Lists.newArrayList();
589      BufferedReader reader = new BufferedReader(
590          new InputStreamReader(new FileInputStream(file), "UTF-8"));
591      try {
592        String line;
593        while ((line = reader.readLine()) != null) {
594          if (!line.trim().isEmpty()) {
595            list.add(line);
596          }
597        }
598      } finally {
599        IOUtils.cleanup(LOG, reader);
600      }
601      return list.toArray(new String[list.size()]);
602    }

604    private static Map<URI, List<Path>> getNameNodePaths(CommandLine line,
605        Configuration conf) throws Exception {
606      Map<URI, List<Path>> map = Maps.newHashMap();
607      String[] paths = null;
608      if (line.hasOption("f")) {
609        paths = readPathFile(line.getOptionValue("f"));
610      } else if (line.hasOption("p")) {
611        paths = line.getOptionValues("p");
612      }
613      Collection<URI> namenodes = DFSUtil.getNsServiceRpcUris(conf);
614      if (paths == null || paths.length == 0) {
615        for (URI namenode : namenodes) {
616          map.put(namenode, null);
617        }
618        return map;
619      }
620      final URI singleNs = namenodes.size() == 1 ?
621          namenodes.iterator().next() : null;
622      for (String path : paths) {
623        Path target = new Path(path);
624        if (!target.isUriPathAbsolute()) {
625          throw new IllegalArgumentException("The path " + target
626              + " is not absolute");
627        }
628        URI targetUri = target.toUri();
629        if ((targetUri.getAuthority() == null || targetUri.getScheme() ==
630            null) && singleNs == null) {
631          // each path must contains both scheme and authority information
632          // unless there is only one name service specified in the
633          // configuration
634          throw new IllegalArgumentException("The path " + target
635              + " does not contain scheme and authority thus cannot identify"
636              + " its name service");
637        }
638        URI key = singleNs;
639        if (singleNs == null) {
640          key = new URI(targetUri.getScheme(), targetUri.getAuthority(),
641              null, null, null);
642          if (!namenodes.contains(key)) {
643            throw new IllegalArgumentException("Cannot resolve the path " +
644                target + ". The namenode services specified in the " +
645                "configuration: " + namenodes);
646          }
647        }
648        List<Path> targets = map.get(key);
649        if (targets == null) {
650          targets = Lists.newArrayList();
651          map.put(key, targets);
652        }
653        targets.add(Path.getPathWithoutSchemeAndAuthority(target));
654      }
655      return map;
656    }

658    @VisibleForTesting
659    static Map<URI, List<Path>> getNameNodePathsToMove(Configuration conf,
660        String... args) throws Exception {
661      final Options opts = buildCliOptions();
662      CommandLineParser parser = new GnuParser();
663      CommandLine commandLine = parser.parse(opts, args, true);
664      return getNameNodePaths(commandLine, conf);
665    }

667    @Override
668    public int run(String[] args) throws Exception {
669      final long startTime = Time.monotonicNow();
670      final Configuration conf = getConf();
671
672      try {
673        final Map<URI, List<Path>> map = getNameNodePathsToMove(conf, args);
674        return Mover.run(map, conf);
675      } catch (IOException e) {
676        System.out.println(e + ".  Exiting ...");
677        return ExitStatus.IO_EXCEPTION.getExitCode();
678      } catch (InterruptedException e) {
679        System.out.println(e + ".  Exiting ...");
680        return ExitStatus.INTERRUPTED.getExitCode();
681      } catch (ParseException e) {
682        System.out.println(e + ".  Exiting ...");
683        return ExitStatus.ILLEGAL_ARGUMENTS.getExitCode();
684      } catch (IllegalArgumentException e) {
685        System.out.println(e + ".  Exiting ...");
686        return ExitStatus.ILLEGAL_ARGUMENTS.getExitCode();
687      } finally {
688        System.out.format("%-24s ", DateFormat.getDateTimeInstance().format(new Date()));
689        System.out.println("Mover took " + StringUtils.formatTime(Time.monotonicNow()-startTime));
690      }
691    }
692  }
693
694  /**
695   * Run a Mover in command line.
696   *
697   * @param args Command line arguments
698   */
699  public static void main(String[] args) {
700    if (DFSUtil.parseHelpArgument(args, Cli.USAGE, System.out, true)) {
701      System.exit(0);
702    }
703
704    try {
705      System.exit(ToolRunner.run(new HdfsConfiguration(), new Cli(), args));
706    } catch (Throwable e) {
707      LOG.error("Exiting " + Mover.class.getSimpleName()
708          + " due to an exception", e);
709      System.exit(-1);
710    }
711  }
