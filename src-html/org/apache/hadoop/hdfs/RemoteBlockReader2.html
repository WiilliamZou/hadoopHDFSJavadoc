059/**
060 * This is a wrapper around connection to datanode
061 * and understands checksum, offset etc.
062 *
063 * Terminology:
064 * <dl>
065 * <dt>block</dt>
066 *   <dd>The hdfs block, typically large (~64MB).
067 *   </dd>
068 * <dt>chunk</dt>
069 *   <dd>A block is divided into chunks, each comes with a checksum.
070 *       We want transfers to be chunk-aligned, to be able to
071 *       verify checksums.
072 *   </dd>
073 * <dt>packet</dt>
074 *   <dd>A grouping of chunks used for transport. It contains a
075 *       header, followed by checksum data, followed by real data.
076 *   </dd>
077 * </dl>
078 * Please see DataNode for the RPC specification.
079 *
080 * This is a new implementation introduced in Hadoop 0.23 which
081 * is more efficient and simpler than the older BlockReader
082 * implementation. It should be renamed to RemoteBlockReader
083 * once we are confident in it.
084 */
085@InterfaceAudience.Private
086public class RemoteBlockReader2  implements BlockReader {

136  @Override
137  public synchronized int read(byte[] buf, int off, int len) 
138                               throws IOException {
139
140    UUID randomId = null;
141    if (LOG.isTraceEnabled()) {
142      randomId = UUID.randomUUID();
143      LOG.trace(String.format("Starting read #%s file %s from datanode %s",
144        randomId.toString(), this.filename,
145        this.datanodeID.getHostName()));
146    }
147
148    if (curDataSlice == null || curDataSlice.remaining() == 0 && bytesNeededToFinish > 0) {
149      TraceScope scope = Trace.startSpan(
150          "RemoteBlockReader2#readNextPacket(" + blockId + ")", Sampler.NEVER);
151      try {
152        readNextPacket();
153      } finally {
154        scope.close();
155      }
156    }
157
158    if (LOG.isTraceEnabled()) {
159      LOG.trace(String.format("Finishing read #" + randomId));
160    }
161
162    if (curDataSlice.remaining() == 0) {
163      // we're at EOF now
164      return -1;
165    }
166    
167    int nRead = Math.min(curDataSlice.remaining(), len);
168    curDataSlice.get(buf, off, nRead);
169    
170    return nRead;
171  }

174  @Override
175  public int read(ByteBuffer buf) throws IOException {
176    if (curDataSlice == null || curDataSlice.remaining() == 0 && bytesNeededToFinish > 0) {
177      TraceScope scope = Trace.startSpan(
178          "RemoteBlockReader2#readNextPacket(" + blockId + ")", Sampler.NEVER);
179      try {
180        readNextPacket();
181      } finally {
182        scope.close();
183      }
184    }
185    if (curDataSlice.remaining() == 0) {
186      // we're at EOF now
187      return -1;
188    }
189
190    int nRead = Math.min(curDataSlice.remaining(), buf.remaining());
191    ByteBuffer writeSlice = curDataSlice.duplicate();
192    writeSlice.limit(writeSlice.position() + nRead);
193    buf.put(writeSlice);
194    curDataSlice.position(writeSlice.position());
195
196    return nRead;
197  }
198
199  private void readNextPacket() throws IOException {
200    //Read packet headers.
201    packetReceiver.receiveNextPacket(in);
202
203    PacketHeader curHeader = packetReceiver.getHeader();
204    curDataSlice = packetReceiver.getDataSlice();
205    assert curDataSlice.capacity() == curHeader.getDataLen();
206    
207    if (LOG.isTraceEnabled()) {
208      LOG.trace("DFSClient readNextPacket got header " + curHeader);
209    }
210
211    // Sanity check the lengths
212    if (!curHeader.sanityCheck(lastSeqNo)) {
213         throw new IOException("BlockReader: error in packet header " +
214                               curHeader);
215    }
216    
217    if (curHeader.getDataLen() > 0) {
218      int chunks = 1 + (curHeader.getDataLen() - 1) / bytesPerChecksum;
219      int checksumsLen = chunks * checksumSize;
220
221      assert packetReceiver.getChecksumSlice().capacity() == checksumsLen :
222        "checksum slice capacity=" + packetReceiver.getChecksumSlice().capacity() + 
223          " checksumsLen=" + checksumsLen;
224      
225      lastSeqNo = curHeader.getSeqno();
226      if (verifyChecksum && curDataSlice.remaining() > 0) {
227        // N.B.: the checksum error offset reported here is actually
228        // relative to the start of the block, not the start of the file.
229        // This is slightly misleading, but preserves the behavior from
230        // the older BlockReader.
231        checksum.verifyChunkedSums(curDataSlice,
232            packetReceiver.getChecksumSlice(),
233            filename, curHeader.getOffsetInBlock());
234      }
235      bytesNeededToFinish -= curHeader.getDataLen();
236    }    
237    
238    // First packet will include some data prior to the first byte
239    // the user requested. Skip it.
240    if (curHeader.getOffsetInBlock() < startOffset) {
241      int newPos = (int) (startOffset - curHeader.getOffsetInBlock());
242      curDataSlice.position(newPos);
243    }
244
245    // If we've now satisfied the whole client read, read one last packet
246    // header, which should be empty
247    if (bytesNeededToFinish <= 0) {
248      readTrailingEmptyPacket();
249      if (verifyChecksum) {
250        sendReadResult(Status.CHECKSUM_OK);
251      } else {
252        sendReadResult(Status.SUCCESS);
253      }
254    }
255  }

257  @Override
258  public synchronized long skip(long n) throws IOException {
259    /* How can we make sure we don't throw a ChecksumException, at least
260     * in majority of the cases?. This one throws. */  
261    if ( skipBuf == null ) {
262      skipBuf = new byte[bytesPerChecksum]; 
263    }
264
265    long nSkipped = 0;
266    while ( nSkipped < n ) {
267      int toSkip = (int)Math.min(n-nSkipped, skipBuf.length);
268      int ret = read(skipBuf, 0, toSkip);
269      if ( ret <= 0 ) {
270        return nSkipped;
271      }
272      nSkipped += ret;
273    }
274    return nSkipped;
275  }
276
277  private void readTrailingEmptyPacket() throws IOException {
278    if (LOG.isTraceEnabled()) {
279      LOG.trace("Reading empty packet at end of read");
280    }
281    
282    packetReceiver.receiveNextPacket(in);
283
284    PacketHeader trailer = packetReceiver.getHeader();
285    if (!trailer.isLastPacketInBlock() ||
286       trailer.getDataLen() != 0) {
287      throw new IOException("Expected empty end-of-read packet! Header: " +
288                            trailer);
289    }
290  }

292  protected RemoteBlockReader2(String file, String bpid, long blockId,
293      DataChecksum checksum, boolean verifyChecksum,
294      long startOffset, long firstChunkOffset, long bytesToRead, Peer peer,
295      DatanodeID datanodeID, PeerCache peerCache) {
296    this.isLocal = DFSClient.isLocalAddress(NetUtils.
297        createSocketAddr(datanodeID.getXferAddr()));
298    // Path is used only for printing block and file information in debug
299    this.peer = peer;
300    this.datanodeID = datanodeID;
301    this.in = peer.getInputStreamChannel();
302    this.checksum = checksum;
303    this.verifyChecksum = verifyChecksum;
304    this.startOffset = Math.max( startOffset, 0 );
305    this.filename = file;
306    this.peerCache = peerCache;
307    this.blockId = blockId;
308
309    // The total number of bytes that we need to transfer from the DN is
310    // the amount that the user wants (bytesToRead), plus the padding at
311    // the beginning in order to chunk-align. Note that the DN may elect
312    // to send more than this amount if the read starts/ends mid-chunk.
313    this.bytesNeededToFinish = bytesToRead + (startOffset - firstChunkOffset);
314    bytesPerChecksum = this.checksum.getBytesPerChecksum();
315    checksumSize = this.checksum.getChecksumSize();
316  }

319  @Override
320  public synchronized void close() throws IOException {
321    packetReceiver.close();
322    startOffset = -1;
323    checksum = null;
324    if (peerCache != null && sentStatusCode) {
325      peerCache.put(datanodeID, peer);
326    } else {
327      peer.close();
328    }
329
330    // in will be closed when its Socket is closed.
331  }
332  
333  /**
334   * When the reader reaches end of the read, it sends a status response
335   * (e.g. CHECKSUM_OK) to the DN. Failure to do so could lead to the DN
336   * closing our connection (which we will re-open), but won't affect
337   * data correctness.
338   */
339  void sendReadResult(Status statusCode) {
340    assert !sentStatusCode : "already sent status code to " + peer;
341    try {
342      writeReadResult(peer.getOutputStream(), statusCode);
343      sentStatusCode = true;
344    } catch (IOException e) {
345      // It's ok not to be able to send this. But something is probably wrong.
346      LOG.info("Could not send read status (" + statusCode + ") to datanode " +
347               peer.getRemoteAddressString() + ": " + e.getMessage());
348    }
349  }
350
351  /**
352   * Serialize the actual read result on the wire.
353   */
354  static void writeReadResult(OutputStream out, Status statusCode)
355      throws IOException {
356    
357    ClientReadStatusProto.newBuilder()
358      .setStatus(statusCode)
359      .build()
360      .writeDelimitedTo(out);
361
362    out.flush();
363  }

387  /**
388   * Create a new BlockReader specifically to satisfy a read.
389   * This method also sends the OP_READ_BLOCK request.
390   *
391   * @param file  File location
392   * @param block  The block object
393   * @param blockToken  The block token for security
394   * @param startOffset  The read offset, relative to block head
395   * @param len  The number of bytes to read
396   * @param verifyChecksum  Whether to verify checksum
397   * @param clientName  Client name
398   * @param peer  The Peer to use
399   * @param datanodeID  The DatanodeID this peer is connected to
400   * @return New BlockReader instance, or null on error.
401   */
402  public static BlockReader newBlockReader(String file,
403                                     ExtendedBlock block,
404                                     Token<BlockTokenIdentifier> blockToken,
405                                     long startOffset, long len,
406                                     boolean verifyChecksum,
407                                     String clientName,
408                                     Peer peer, DatanodeID datanodeID,
409                                     PeerCache peerCache,
410                                     CachingStrategy cachingStrategy) throws IOException {
411    // in and out will be closed when sock is closed (by the caller)
412    final DataOutputStream out = new DataOutputStream(new BufferedOutputStream(
413          peer.getOutputStream()));
414    new Sender(out).readBlock(block, blockToken, clientName, startOffset, len,
415        verifyChecksum, cachingStrategy);
416
417    //
418    // Get bytes in block
419    //
420    DataInputStream in = new DataInputStream(peer.getInputStream());
421
422    BlockOpResponseProto status = BlockOpResponseProto.parseFrom(
423        PBHelper.vintPrefixed(in));
424    checkSuccess(status, peer, block, file);
425    ReadOpChecksumInfoProto checksumInfo =
426      status.getReadOpChecksumInfo();
427    DataChecksum checksum = DataTransferProtoUtil.fromProto(
428        checksumInfo.getChecksum());
429    //Warning when we get CHECKSUM_NULL?
430
431    // Read the first chunk offset.
432    long firstChunkOffset = checksumInfo.getChunkOffset();
433
434    if ( firstChunkOffset < 0 || firstChunkOffset > startOffset ||
435        firstChunkOffset <= (startOffset - checksum.getBytesPerChecksum())) {
436      throw new IOException("BlockReader: error in first chunk offset (" +
437                            firstChunkOffset + ") startOffset is " +
438                            startOffset + " for file " + file);
439    }
440
441    return new RemoteBlockReader2(file, block.getBlockPoolId(), block.getBlockId(),
442        checksum, verifyChecksum, startOffset, firstChunkOffset, len, peer,
443        datanodeID, peerCache);
444  }
445
446  static void checkSuccess(
447      BlockOpResponseProto status, Peer peer,
448      ExtendedBlock block, String file)
449      throws IOException {
450    String logInfo = "for OP_READ_BLOCK"
451      + ", self=" + peer.getLocalAddressString()
452      + ", remote=" + peer.getRemoteAddressString()
453      + ", for file " + file
454      + ", for pool " + block.getBlockPoolId()
455      + " block " + block.getBlockId() + "_" + block.getGenerationStamp();
456    DataTransferProtoUtil.checkBlockOpStatus(status, logInfo);
457  }
