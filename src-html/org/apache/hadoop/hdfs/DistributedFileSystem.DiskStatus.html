565  /**
566   * Move blocks from srcs to trg and delete srcs afterwards.
567   * The file block sizes must be the same.
568   * 
569   * @param trg existing file to append to
570   * @param psrcs list of files (same block size, same replication)
571   * @throws IOException
572   */
573  @Override
574  public void concat(Path trg, Path [] psrcs) throws IOException {
575    statistics.incrementWriteOps(1);
576    // Make target absolute
577    Path absF = fixRelativePart(trg);
578    // Make all srcs absolute
579    Path[] srcs = new Path[psrcs.length];
580    for (int i=0; i<psrcs.length; i++) {
581      srcs[i] = fixRelativePart(psrcs[i]);
582    }
583    // Try the concat without resolving any links
584    String[] srcsStr = new String[psrcs.length];
585    try {
586      for (int i=0; i<psrcs.length; i++) {
587        srcsStr[i] = getPathName(srcs[i]);
588      }
589      dfs.concat(getPathName(trg), srcsStr);
590    } catch (UnresolvedLinkException e) {
591      // Exception could be from trg or any src.
592      // Fully resolve trg and srcs. Fail if any of them are a symlink.
593      FileStatus stat = getFileLinkStatus(absF);
594      if (stat.isSymlink()) {
595        throw new IOException("Cannot concat with a symlink target: "
596            + trg + " -> " + stat.getPath());
597      }
598      absF = fixRelativePart(stat.getPath());
599      for (int i=0; i<psrcs.length; i++) {
600        stat = getFileLinkStatus(srcs[i]);
601        if (stat.isSymlink()) {
602          throw new IOException("Cannot concat with a symlink src: "
603              + psrcs[i] + " -> " + stat.getPath());
604        }
605        srcs[i] = fixRelativePart(stat.getPath());
606      }
607      // Try concat again. Can still race with another symlink.
608      for (int i=0; i<psrcs.length; i++) {
609        srcsStr[i] = getPathName(srcs[i]);
610      }
611      dfs.concat(getPathName(absF), srcsStr);
612    }
613  }

616  @SuppressWarnings("deprecation")
617  @Override
618  public boolean rename(Path src, Path dst) throws IOException {
619    statistics.incrementWriteOps(1);
620
621    final Path absSrc = fixRelativePart(src);
622    final Path absDst = fixRelativePart(dst);
623
624    // Try the rename without resolving first
625    try {
626      return dfs.rename(getPathName(absSrc), getPathName(absDst));
627    } catch (UnresolvedLinkException e) {
628      // Fully resolve the source
629      final Path source = getFileLinkStatus(absSrc).getPath();
630      // Keep trying to resolve the destination
631      return new FileSystemLinkResolver<Boolean>() {
632        @Override
633        public Boolean doCall(final Path p)
634            throws IOException, UnresolvedLinkException {
635          return dfs.rename(getPathName(source), getPathName(p));
636        }
637        @Override
638        public Boolean next(final FileSystem fs, final Path p)
639            throws IOException {
640          // Should just throw an error in FileSystem#checkPath
641          return doCall(p);
642        }
643      }.resolve(this, absDst);
644    }
645  }

787  private FileStatus[] listStatusInternal(Path p) throws IOException {
788    String src = getPathName(p);
789
790    // fetch the first batch of entries in the directory
791    DirectoryListing thisListing = dfs.listPaths(
792        src, HdfsFileStatus.EMPTY_NAME);
793
794    if (thisListing == null) { // the directory does not exist
795      throw new FileNotFoundException("File " + p + " does not exist.");
796    }
797    
798    HdfsFileStatus[] partialListing = thisListing.getPartialListing();
799    if (!thisListing.hasMore()) { // got all entries of the directory
800      FileStatus[] stats = new FileStatus[partialListing.length];
801      for (int i = 0; i < partialListing.length; i++) {
802        stats[i] = partialListing[i].makeQualified(getUri(), p);
803      }
804      statistics.incrementReadOps(1);
805      return stats;
806    }
807
808    // The directory size is too big that it needs to fetch more
809    // estimate the total number of entries in the directory
810    int totalNumEntries =
811      partialListing.length + thisListing.getRemainingEntries();
812    ArrayList<FileStatus> listing =
813      new ArrayList<FileStatus>(totalNumEntries);
814    // add the first batch of entries to the array list
815    for (HdfsFileStatus fileStatus : partialListing) {
816      listing.add(fileStatus.makeQualified(getUri(), p));
817    }
818    statistics.incrementLargeReadOps(1);
819 
820    // now fetch more entries
821    do {
822      thisListing = dfs.listPaths(src, thisListing.getLastName());
823 
824      if (thisListing == null) { // the directory is deleted
825        throw new FileNotFoundException("File " + p + " does not exist.");
826      }
827 
828      partialListing = thisListing.getPartialListing();
829      for (HdfsFileStatus fileStatus : partialListing) {
830        listing.add(fileStatus.makeQualified(getUri(), p));
831      }
832      statistics.incrementLargeReadOps(1);
833    } while (thisListing.hasMore());
834 
835    return listing.toArray(new FileStatus[listing.size()]);
836  }

918  /**
919   * This class defines an iterator that returns
920   * the file status of each file/subdirectory of a directory
921   * 
922   * if needLocation, status contains block location if it is a file
923   * throws a RuntimeException with the error as its cause.
924   * 
925   * @param <T> the type of the file status
926   */
927  private class  DirListingIterator<T extends FileStatus>
928  implements RemoteIterator<T> {
929    private DirectoryListing thisListing;
930    private int i;
931    private Path p;
932    private String src;
933    private T curStat = null;
934    private PathFilter filter;
935    private boolean needLocation;
936
937    private DirListingIterator(Path p, PathFilter filter,
938        boolean needLocation) throws IOException {
939      this.p = p;
940      this.src = getPathName(p);
941      this.filter = filter;
942      this.needLocation = needLocation;
943      // fetch the first batch of entries in the directory
944      thisListing = dfs.listPaths(src, HdfsFileStatus.EMPTY_NAME,
945          needLocation);
946      statistics.incrementReadOps(1);
947      if (thisListing == null) { // the directory does not exist
948        throw new FileNotFoundException("File " + p + " does not exist.");
949      }
950      i = 0;
951    }
952
953    private DirListingIterator(Path p, boolean needLocation)
954        throws IOException {
955      this(p, null, needLocation);
956    }
957
958    @Override
959    @SuppressWarnings("unchecked")
960    public boolean hasNext() throws IOException {
961      while (curStat == null && hasNextNoFilter()) {
962        T next;
963        HdfsFileStatus fileStat = thisListing.getPartialListing()[i++];
964        if (needLocation) {
965          next = (T)((HdfsLocatedFileStatus)fileStat)
966              .makeQualifiedLocated(getUri(), p);
967        } else {
968          next = (T)fileStat.makeQualified(getUri(), p);
969        }
970          // apply filter if not null
971        if (filter == null || filter.accept(next.getPath())) {
972          curStat = next;
973        }
974      }
975      return curStat != null;
976    }
977      
978    /** Check if there is a next item before applying the given filter */
979    private boolean hasNextNoFilter() throws IOException {
980      if (thisListing == null) {
981        return false;
982      }
983      if (i >= thisListing.getPartialListing().length
984          && thisListing.hasMore()) { 
985        // current listing is exhausted & fetch a new listing
986        thisListing = dfs.listPaths(src, thisListing.getLastName(),
987            needLocation);
988        statistics.incrementReadOps(1);
989        if (thisListing == null) {
990          return false;
991        }
992        i = 0;
993      }
994      return (i < thisListing.getPartialListing().length);
995    }
996
997    @Override
998    public T next() throws IOException {
999      if (hasNext()) {
1000        T tmp = curStat;
1001        curStat = null;
1002        return tmp;
1003      } 
1004      throw new java.util.NoSuchElementException("No more entry in " + p);
1005    }
1006  }

1898        Path p = info.getPath().makeQualified(getUri(), getWorkingDirectory());
1899        return new CacheDirectiveEntry(
