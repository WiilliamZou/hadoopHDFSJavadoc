061 * FSImageLoader loads fsimage and provide methods to return JSON formatted
062 * file status of the namespace of the fsimage.

065  public static final Log LOG = LogFactory.getLog(FSImageHandler.class);

067  private final String[] stringTable;
068  // byte representation of inodes, sorted by id
069  private final byte[][] inodes;
070  private final Map<Long, long[]> dirmap;
071  private static final Comparator<byte[]> INODE_BYTES_COMPARATOR = new
072          Comparator<byte[]>() {
073    @Override
074    public int compare(byte[] o1, byte[] o2) {
075      try {
076        final FsImageProto.INodeSection.INode l = FsImageProto.INodeSection
077                .INode.parseFrom(o1);
078        final FsImageProto.INodeSection.INode r = FsImageProto.INodeSection
079                .INode.parseFrom(o2);
080        if (l.getId() < r.getId()) {
081          return -1;
082        } else if (l.getId() > r.getId()) {
083          return 1;
084        } else {
085          return 0;
086        }
087      } catch (InvalidProtocolBufferException e) {
088        throw new RuntimeException(e);
089      }
090    }
091  };

093  private FSImageLoader(String[] stringTable, byte[][] inodes,
094                        Map<Long, long[]> dirmap) {
095    this.stringTable = stringTable;
096    this.inodes = inodes;
097    this.dirmap = dirmap;
098  }

100  /**
101   * Load fsimage into the memory.
102   * @param inputFile the filepath of the fsimage to load.
103   * @return FSImageLoader
104   * @throws IOException if failed to load fsimage.
105   */
106  static FSImageLoader load(String inputFile) throws IOException {
107    Configuration conf = new Configuration();
108    RandomAccessFile file = new RandomAccessFile(inputFile, "r");
109    if (!FSImageUtil.checkFileFormat(file)) {
110      throw new IOException("Unrecognized FSImage");
111    }
112
113    FsImageProto.FileSummary summary = FSImageUtil.loadSummary(file);
114
115
116    try (FileInputStream fin = new FileInputStream(file.getFD())) {
117      // Map to record INodeReference to the referred id
118      ImmutableList<Long> refIdList = null;
119      String[] stringTable = null;
120      byte[][] inodes = null;
121      Map<Long, long[]> dirmap = null;
122
123      ArrayList<FsImageProto.FileSummary.Section> sections =
124          Lists.newArrayList(summary.getSectionsList());
125      Collections.sort(sections,
126          new Comparator<FsImageProto.FileSummary.Section>() {
127            @Override
128            public int compare(FsImageProto.FileSummary.Section s1,
129                               FsImageProto.FileSummary.Section s2) {
130              FSImageFormatProtobuf.SectionName n1 =
131                  FSImageFormatProtobuf.SectionName.fromString(s1.getName());
132              FSImageFormatProtobuf.SectionName n2 =
133                  FSImageFormatProtobuf.SectionName.fromString(s2.getName());
134              if (n1 == null) {
135                return n2 == null ? 0 : -1;
136              } else if (n2 == null) {
137                return -1;
138              } else {
139                return n1.ordinal() - n2.ordinal();
140              }
141            }
142          });
143
144      for (FsImageProto.FileSummary.Section s : sections) {
145        fin.getChannel().position(s.getOffset());
146        InputStream is = FSImageUtil.wrapInputStreamForCompression(conf,
147            summary.getCodec(), new BufferedInputStream(new LimitInputStream(
148            fin, s.getLength())));
149
150        LOG.debug("Loading section " + s.getName() + " length: " + s.getLength
151                ());
152        switch (FSImageFormatProtobuf.SectionName.fromString(s.getName())) {
153          case STRING_TABLE:
154            stringTable = loadStringTable(is);
155            break;
156          case INODE:
157            inodes = loadINodeSection(is);
158            break;
159          case INODE_REFERENCE:
160            refIdList = loadINodeReferenceSection(is);
161            break;
162          case INODE_DIR:
163            dirmap = loadINodeDirectorySection(is, refIdList);
164            break;
165          default:
166            break;
167        }
168      }
169      return new FSImageLoader(stringTable, inodes, dirmap);
170    }
171  }

173  private static Map<Long, long[]> loadINodeDirectorySection
174          (InputStream in, List<Long> refIdList)
175      throws IOException {
176    LOG.info("Loading inode directory section");
177    Map<Long, long[]> dirs = Maps.newHashMap();
178    long counter = 0;
179    while (true) {
180      FsImageProto.INodeDirectorySection.DirEntry e =
181          FsImageProto.INodeDirectorySection.DirEntry.parseDelimitedFrom(in);
182      // note that in is a LimitedInputStream
183      if (e == null) {
184        break;
185      }
186      ++counter;
187
188      long[] l = new long[e.getChildrenCount() + e.getRefChildrenCount()];
189      for (int i = 0; i < e.getChildrenCount(); ++i) {
190        l[i] = e.getChildren(i);
191      }
192      for (int i = e.getChildrenCount(); i < l.length; i++) {
193        int refId = e.getRefChildren(i - e.getChildrenCount());
194        l[i] = refIdList.get(refId);
195      }
196      dirs.put(e.getParent(), l);
197    }
198    LOG.info("Loaded " + counter + " directories");
199    return dirs;
200  }

202  private static ImmutableList<Long> loadINodeReferenceSection(InputStream in)
203      throws IOException {
204    LOG.info("Loading inode references");
205    ImmutableList.Builder<Long> builder = ImmutableList.builder();
206    long counter = 0;
207    while (true) {
208      FsImageProto.INodeReferenceSection.INodeReference e =
209          FsImageProto.INodeReferenceSection.INodeReference
210              .parseDelimitedFrom(in);
211      if (e == null) {
212        break;
213      }
214      ++counter;
215      builder.add(e.getReferredId());
216    }
217    LOG.info("Loaded " + counter + " inode references");
218    return builder.build();
219  }
220
221  private static byte[][] loadINodeSection(InputStream in)
222          throws IOException {
223    FsImageProto.INodeSection s = FsImageProto.INodeSection
224        .parseDelimitedFrom(in);
225    LOG.info("Loading " + s.getNumInodes() + " inodes.");
226    final byte[][] inodes = new byte[(int) s.getNumInodes()][];
227
228    for (int i = 0; i < s.getNumInodes(); ++i) {
229      int size = CodedInputStream.readRawVarint32(in.read(), in);
230      byte[] bytes = new byte[size];
231      IOUtils.readFully(in, bytes, 0, size);
232      inodes[i] = bytes;
233    }
234    LOG.debug("Sorting inodes");
235    Arrays.sort(inodes, INODE_BYTES_COMPARATOR);
236    LOG.debug("Finished sorting inodes");
237    return inodes;
238  }
239
240  static String[] loadStringTable(InputStream in) throws
241  IOException {
242    FsImageProto.StringTableSection s = FsImageProto.StringTableSection
243        .parseDelimitedFrom(in);
244    LOG.info("Loading " + s.getNumEntry() + " strings");
245    String[] stringTable = new String[s.getNumEntry() + 1];
246    for (int i = 0; i < s.getNumEntry(); ++i) {
247      FsImageProto.StringTableSection.Entry e = FsImageProto
248          .StringTableSection.Entry.parseDelimitedFrom(in);
249      stringTable[e.getId()] = e.getStr();
250    }
251    return stringTable;
252  }

254  /**
255   * Return the JSON formatted FileStatus of the specified file.
256   * @param path a path specifies a file
257   * @return JSON formatted FileStatus
258   * @throws IOException if failed to serialize fileStatus to JSON.
259   */
260  String getFileStatus(String path) throws IOException {
261    ObjectMapper mapper = new ObjectMapper();
262    FsImageProto.INodeSection.INode inode = fromINodeId(lookup(path));
263    return "{\"FileStatus\":\n"
264        + mapper.writeValueAsString(getFileStatus(inode, false)) + "\n}\n";
265  }

267  /**
268   * Return the JSON formatted list of the files in the specified directory.
269   * @param path a path specifies a directory to list
270   * @return JSON formatted file list in the directory
271   * @throws IOException if failed to serialize fileStatus to JSON.
272   */
273  String listStatus(String path) throws IOException {
274    StringBuilder sb = new StringBuilder();
275    ObjectMapper mapper = new ObjectMapper();
276    List<Map<String, Object>> fileStatusList = getFileStatusList(path);
277    sb.append("{\"FileStatuses\":{\"FileStatus\":[\n");
278    int i = 0;
279    for (Map<String, Object> fileStatusMap : fileStatusList) {
280      if (i++ != 0) {
281        sb.append(',');
282      }
283      sb.append(mapper.writeValueAsString(fileStatusMap));
284    }
285    sb.append("\n]}}\n");
286    return sb.toString();
287  }

289  private List<Map<String, Object>> getFileStatusList(String path)
290          throws IOException {
291    List<Map<String, Object>> list = new ArrayList<Map<String, Object>>();
292    long id = lookup(path);
293    FsImageProto.INodeSection.INode inode = fromINodeId(id);
294    if (inode.getType() == FsImageProto.INodeSection.INode.Type.DIRECTORY) {
295      if (!dirmap.containsKey(id)) {
296        // if the directory is empty, return empty list
297        return list;
298      }
299      long[] children = dirmap.get(id);
300      for (long cid : children) {
301        list.add(getFileStatus(fromINodeId(cid), true));
302      }
303    } else {
304      list.add(getFileStatus(inode, false));
305    }
306    return list;
307  }

309  /**
310   * Return the JSON formatted ACL status of the specified file.
311   * @param path a path specifies a file
312   * @return JSON formatted AclStatus
313   * @throws IOException if failed to serialize fileStatus to JSON.
314   */
315  String getAclStatus(String path) throws IOException {
316    PermissionStatus p = getPermissionStatus(path);
317    List<AclEntry> aclEntryList = getAclEntryList(path);
318    FsPermission permission = p.getPermission();
319    AclStatus.Builder builder = new AclStatus.Builder();
320    builder.owner(p.getUserName()).group(p.getGroupName())
321        .addEntries(aclEntryList).setPermission(permission)
322        .stickyBit(permission.getStickyBit());
323    AclStatus aclStatus = builder.build();
324    return JsonUtil.toJsonString(aclStatus);
325  }
326
327  private List<AclEntry> getAclEntryList(String path) throws IOException {
328    long id = lookup(path);
329    FsImageProto.INodeSection.INode inode = fromINodeId(id);
330    switch (inode.getType()) {
331      case FILE: {
332        FsImageProto.INodeSection.INodeFile f = inode.getFile();
333        return FSImageFormatPBINode.Loader.loadAclEntries(
334            f.getAcl(), stringTable);
335      }
336      case DIRECTORY: {
337        FsImageProto.INodeSection.INodeDirectory d = inode.getDirectory();
338        return FSImageFormatPBINode.Loader.loadAclEntries(
339            d.getAcl(), stringTable);
340      }
341      default: {
342        return new ArrayList<AclEntry>();
343      }
344    }
345  }

347  private PermissionStatus getPermissionStatus(String path) throws IOException {
348    long id = lookup(path);
349    FsImageProto.INodeSection.INode inode = fromINodeId(id);
350    switch (inode.getType()) {
351      case FILE: {
352        FsImageProto.INodeSection.INodeFile f = inode.getFile();
353        return FSImageFormatPBINode.Loader.loadPermission(
354            f.getPermission(), stringTable);
355      }
356      case DIRECTORY: {
357        FsImageProto.INodeSection.INodeDirectory d = inode.getDirectory();
358        return FSImageFormatPBINode.Loader.loadPermission(
359            d.getPermission(), stringTable);
360      }
361      case SYMLINK: {
362        FsImageProto.INodeSection.INodeSymlink s = inode.getSymlink();
363        return FSImageFormatPBINode.Loader.loadPermission(
364            s.getPermission(), stringTable);
365      }
366      default: {
367        return null;
368      }
369    }
370  }

372  /**
373   * Return the INodeId of the specified path.
374   */
375  private long lookup(String path) throws IOException {
376    Preconditions.checkArgument(path.startsWith("/"));
377    long id = INodeId.ROOT_INODE_ID;
378    for (int offset = 0, next; offset < path.length(); offset = next) {
379      next = path.indexOf('/', offset + 1);
380      if (next == -1) {
381        next = path.length();
382      }
383      if (offset + 1 > next) {
384        break;
385      }
386
387      final String component = path.substring(offset + 1, next);
388
389      if (component.isEmpty()) {
390        continue;
391      }
392
393      final long[] children = dirmap.get(id);
394      if (children == null) {
395        throw new FileNotFoundException(path);
396      }
397
398      boolean found = false;
399      for (long cid : children) {
400        FsImageProto.INodeSection.INode child = fromINodeId(cid);
401        if (component.equals(child.getName().toStringUtf8())) {
402          found = true;
403          id = child.getId();
404          break;
405        }
406      }
407      if (!found) {
408        throw new FileNotFoundException(path);
409      }
410    }
411    return id;
412  }

414  private Map<String, Object> getFileStatus
415      (FsImageProto.INodeSection.INode inode, boolean printSuffix){
416    Map<String, Object> map = Maps.newHashMap();
417    switch (inode.getType()) {
418      case FILE: {
419        FsImageProto.INodeSection.INodeFile f = inode.getFile();
420        PermissionStatus p = FSImageFormatPBINode.Loader.loadPermission(
421            f.getPermission(), stringTable);
422        map.put("accessTime", f.getAccessTime());
423        map.put("blockSize", f.getPreferredBlockSize());
424        map.put("group", p.getGroupName());
425        map.put("length", getFileSize(f));
426        map.put("modificationTime", f.getModificationTime());
427        map.put("owner", p.getUserName());
428        map.put("pathSuffix",
429            printSuffix ? inode.getName().toStringUtf8() : "");
430        map.put("permission", toString(p.getPermission()));
431        map.put("replication", f.getReplication());
432        map.put("type", inode.getType());
433        map.put("fileId", inode.getId());
434        map.put("childrenNum", 0);
435        return map;
436      }
437      case DIRECTORY: {
438        FsImageProto.INodeSection.INodeDirectory d = inode.getDirectory();
439        PermissionStatus p = FSImageFormatPBINode.Loader.loadPermission(
440            d.getPermission(), stringTable);
441        map.put("accessTime", 0);
442        map.put("blockSize", 0);
443        map.put("group", p.getGroupName());
444        map.put("length", 0);
445        map.put("modificationTime", d.getModificationTime());
446        map.put("owner", p.getUserName());
447        map.put("pathSuffix",
448            printSuffix ? inode.getName().toStringUtf8() : "");
449        map.put("permission", toString(p.getPermission()));
450        map.put("replication", 0);
451        map.put("type", inode.getType());
452        map.put("fileId", inode.getId());
453        map.put("childrenNum", dirmap.containsKey(inode.getId()) ?
454            dirmap.get(inode.getId()).length : 0);
455        return map;
456      }
457      case SYMLINK: {
458        FsImageProto.INodeSection.INodeSymlink d = inode.getSymlink();
459        PermissionStatus p = FSImageFormatPBINode.Loader.loadPermission(
460            d.getPermission(), stringTable);
461        map.put("accessTime", d.getAccessTime());
462        map.put("blockSize", 0);
463        map.put("group", p.getGroupName());
464        map.put("length", 0);
465        map.put("modificationTime", d.getModificationTime());
466        map.put("owner", p.getUserName());
467        map.put("pathSuffix",
468            printSuffix ? inode.getName().toStringUtf8() : "");
469        map.put("permission", toString(p.getPermission()));
470        map.put("replication", 0);
471        map.put("type", inode.getType());
472        map.put("symlink", d.getTarget().toStringUtf8());
473        map.put("fileId", inode.getId());
474        map.put("childrenNum", 0);
475        return map;
476      }
477      default:
478        return null;
479    }
480  }

482  static long getFileSize(FsImageProto.INodeSection.INodeFile f) {
483    long size = 0;
484    for (HdfsProtos.BlockProto p : f.getBlocksList()) {
485      size += p.getNumBytes();
486    }
487    return size;
488  }
489
490  private String toString(FsPermission permission) {
491    return String.format("%o", permission.toShort());
492  }
493
494  private FsImageProto.INodeSection.INode fromINodeId(final long id)
495          throws IOException {
496    int l = 0, r = inodes.length;
497    while (l < r) {
498      int mid = l + (r - l) / 2;
499      FsImageProto.INodeSection.INode n = FsImageProto.INodeSection.INode
500              .parseFrom(inodes[mid]);
501      long nid = n.getId();
502      if (id > nid) {
503        l = mid + 1;
504      } else if (id < nid) {
505        r = mid;
506      } else {
507        return n;
508      }
509    }
510    return null;
511  }
