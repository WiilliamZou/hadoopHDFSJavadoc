327    static final String DESCRIPTION = USAGE + ":\n"

341    private static void printMessage(RollingUpgradeInfo info,
342        PrintStream out) {
343      if (info != null && info.isStarted()) {
344        if (!info.createdRollbackImages() && !info.isFinalized()) {
345          out.println(
346              "Preparing for upgrade. Data is being saved for rollback."
347              + "\nRun \"dfsadmin -rollingUpgrade query\" to check the status"
348              + "\nfor proceeding with rolling upgrade");
349            out.println(info);
350        } else if (!info.isFinalized()) {
351          out.println("Proceed with rolling upgrade:");
352          out.println(info);
353        } else {
354          out.println("Rolling upgrade is finalized.");
355          out.println(info);
356        }
357      } else {
358        out.println("There is no rolling upgrade in progress or rolling " +
359            "upgrade has already been finalized.");
360      }
361    }
362
363    static int run(DistributedFileSystem dfs, String[] argv, int idx) throws IOException {
364      final RollingUpgradeAction action = RollingUpgradeAction.fromString(
365          argv.length >= 2? argv[1]: "");
366      if (action == null) {
367        throw new IllegalArgumentException("Failed to covert \"" + argv[1]
368            +"\" to " + RollingUpgradeAction.class.getSimpleName());
369      }
370
371      System.out.println(action + " rolling upgrade ...");
372
373      final RollingUpgradeInfo info = dfs.rollingUpgrade(action);
374      switch(action){
375      case QUERY:
376        break;
377      case PREPARE:
378        Preconditions.checkState(info.isStarted());
379        break;
380      case FINALIZE:
381        Preconditions.checkState(info == null || info.isFinalized());
382        break;
383      }
384      printMessage(info, System.out);
385      return 0;
386    }

449   * Gives a report on how the FileSystem is doing.
450   * @exception IOException if the filesystem does not exist.
451   */
452  public void report(String[] argv, int i) throws IOException {
453    DistributedFileSystem dfs = getDFS();
454    FsStatus ds = dfs.getStatus();
455    long capacity = ds.getCapacity();
456    long used = ds.getUsed();
457    long remaining = ds.getRemaining();
458    long presentCapacity = used + remaining;
459    boolean mode = dfs.setSafeMode(HdfsConstants.SafeModeAction.SAFEMODE_GET);
460    if (mode) {
461      System.out.println("Safe mode is ON");
462    }
463    System.out.println("Configured Capacity: " + capacity
464                       + " (" + StringUtils.byteDesc(capacity) + ")");
465    System.out.println("Present Capacity: " + presentCapacity
466        + " (" + StringUtils.byteDesc(presentCapacity) + ")");
467    System.out.println("DFS Remaining: " + remaining
468        + " (" + StringUtils.byteDesc(remaining) + ")");
469    System.out.println("DFS Used: " + used
470                       + " (" + StringUtils.byteDesc(used) + ")");
471    System.out.println("DFS Used%: "
472        + StringUtils.formatPercent(used/(double)presentCapacity, 2));
473    
474    /* These counts are not always upto date. They are updated after  
475     * iteration of an internal list. Should be updated in a few seconds to 
476     * minutes. Use "-metaSave" to list of all such blocks and accurate 
477     * counts.
478     */
479    System.out.println("Under replicated blocks: " + 
480                       dfs.getUnderReplicatedBlocksCount());
481    System.out.println("Blocks with corrupt replicas: " + 
482                       dfs.getCorruptBlocksCount());
483    System.out.println("Missing blocks: " + 
484                       dfs.getMissingBlocksCount());
485    System.out.println("Missing blocks (with replication factor 1): " +
486                      dfs.getMissingReplOneBlocksCount());
487
488    System.out.println();
489
490    System.out.println("-------------------------------------------------");
491    
492    // Parse arguments for filtering the node list
493    List<String> args = Arrays.asList(argv);
494    // Truncate already handled arguments before parsing report()-specific ones
495    args = new ArrayList<String>(args.subList(i, args.size()));
496    final boolean listLive = StringUtils.popOption("-live", args);
497    final boolean listDead = StringUtils.popOption("-dead", args);
498    final boolean listDecommissioning =
499        StringUtils.popOption("-decommissioning", args);
500
501    // If no filter flags are found, then list all DN types
502    boolean listAll = (!listLive && !listDead && !listDecommissioning);
503
504    if (listAll || listLive) {
505      DatanodeInfo[] live = dfs.getDataNodeStats(DatanodeReportType.LIVE);
506      if (live.length > 0 || listLive) {
507        System.out.println("Live datanodes (" + live.length + "):\n");
508      }
509      if (live.length > 0) {
510        for (DatanodeInfo dn : live) {
511          System.out.println(dn.getDatanodeReport());
512          System.out.println();
513        }
514      }
515    }
516
517    if (listAll || listDead) {
518      DatanodeInfo[] dead = dfs.getDataNodeStats(DatanodeReportType.DEAD);
519      if (dead.length > 0 || listDead) {
520        System.out.println("Dead datanodes (" + dead.length + "):\n");
521      }
522      if (dead.length > 0) {
523        for (DatanodeInfo dn : dead) {
524          System.out.println(dn.getDatanodeReport());
525          System.out.println();
526        }
527      }
528    }
529
530    if (listAll || listDecommissioning) {
531      DatanodeInfo[] decom =
532          dfs.getDataNodeStats(DatanodeReportType.DECOMMISSIONING);
533      if (decom.length > 0 || listDecommissioning) {
534        System.out.println("Decommissioning datanodes (" + decom.length
535            + "):\n");
536      }
537      if (decom.length > 0) {
538        for (DatanodeInfo dn : decom) {
539          System.out.println(dn.getDatanodeReport());
540          System.out.println();
541        }
542      }
543    }

546  /**
547   * Safe mode maintenance command.
548   * Usage: hdfs dfsadmin -safemode [enter | leave | get]
549   * @param argv List of of command line parameters.
550   * @param idx The index of the command that is being processed.
551   * @exception IOException if the filesystem does not exist.
552   */
553  public void setSafeMode(String[] argv, int idx) throws IOException {
554    if (idx != argv.length - 1) {
555      printUsage("-safemode");
556      return;
557    }
558    HdfsConstants.SafeModeAction action;
559    Boolean waitExitSafe = false;
560
561    if ("leave".equalsIgnoreCase(argv[idx])) {
562      action = HdfsConstants.SafeModeAction.SAFEMODE_LEAVE;
563    } else if ("enter".equalsIgnoreCase(argv[idx])) {
564      action = HdfsConstants.SafeModeAction.SAFEMODE_ENTER;
565    } else if ("get".equalsIgnoreCase(argv[idx])) {
566      action = HdfsConstants.SafeModeAction.SAFEMODE_GET;
567    } else if ("wait".equalsIgnoreCase(argv[idx])) {
568      action = HdfsConstants.SafeModeAction.SAFEMODE_GET;
569      waitExitSafe = true;
570    } else {
571      printUsage("-safemode");
572      return;
573    }
574
575    DistributedFileSystem dfs = getDFS();
576    Configuration dfsConf = dfs.getConf();
577    URI dfsUri = dfs.getUri();
578    boolean isHaEnabled = HAUtil.isLogicalUri(dfsConf, dfsUri);
579
580    if (isHaEnabled) {
581      String nsId = dfsUri.getHost();
582      List<ProxyAndInfo<ClientProtocol>> proxies =
583          HAUtil.getProxiesForAllNameNodesInNameservice(
584          dfsConf, nsId, ClientProtocol.class);
585      for (ProxyAndInfo<ClientProtocol> proxy : proxies) {
586        ClientProtocol haNn = proxy.getProxy();
587        boolean inSafeMode = haNn.setSafeMode(action, false);
588        if (waitExitSafe) {
589          inSafeMode = waitExitSafeMode(haNn, inSafeMode);
590        }
591        System.out.println("Safe mode is " + (inSafeMode ? "ON" : "OFF")
592            + " in " + proxy.getAddress());
593      }
594    } else {
595      boolean inSafeMode = dfs.setSafeMode(action);
596      if (waitExitSafe) {
597        inSafeMode = waitExitSafeMode(dfs, inSafeMode);
598      }
599      System.out.println("Safe mode is " + (inSafeMode ? "ON" : "OFF"));
600    }
601
602  }

604  private boolean waitExitSafeMode(DistributedFileSystem dfs, boolean inSafeMode)
605      throws IOException {
606    while (inSafeMode) {

1761      if (argv.length < 1) {
1762        printUsage(cmd);
1763        return exitCode;
1764      }
1765    }
1766    
1767    // initialize DFSAdmin
1768    try {
1769      init();
1770    } catch (RPC.VersionMismatch v) {
1771      System.err.println("Version Mismatch between client and server"
1772                         + "... command aborted.");
1773      return exitCode;
1774    } catch (IOException e) {
1775      System.err.println("Bad connection to DFS... command aborted.");
1776      return exitCode;
1777    }
1778
1779    Exception debugException = null;
1780    exitCode = 0;
1781    try {
1782      if ("-report".equals(cmd)) {
1783        report(argv, i);
1784      } else if ("-safemode".equals(cmd)) {
1785        setSafeMode(argv, i);
1786      } else if ("-allowSnapshot".equalsIgnoreCase(cmd)) {
1787        allowSnapshot(argv);
1788      } else if ("-disallowSnapshot".equalsIgnoreCase(cmd)) {
1789        disallowSnapshot(argv);
1790      } else if ("-saveNamespace".equals(cmd)) {
1791        exitCode = saveNamespace();
1792      } else if ("-rollEdits".equals(cmd)) {
1793        exitCode = rollEdits();
1794      } else if ("-restoreFailedStorage".equals(cmd)) {
1795        exitCode = restoreFailedStorage(argv[i]);
1796      } else if ("-refreshNodes".equals(cmd)) {
1797        exitCode = refreshNodes();
1798      } else if ("-finalizeUpgrade".equals(cmd)) {
1799        exitCode = finalizeUpgrade();
1800      } else if (RollingUpgradeCommand.matches(cmd)) {
1801        exitCode = RollingUpgradeCommand.run(getDFS(), argv, i);
1802      } else if ("-metasave".equals(cmd)) {
1803        exitCode = metaSave(argv, i);
1804      } else if (ClearQuotaCommand.matches(cmd)) {
1805        exitCode = new ClearQuotaCommand(argv, i, getDFS()).runAll();
1806      } else if (SetQuotaCommand.matches(cmd)) {
1807        exitCode = new SetQuotaCommand(argv, i, getDFS()).runAll();
1808      } else if (ClearSpaceQuotaCommand.matches(cmd)) {
1809        exitCode = new ClearSpaceQuotaCommand(argv, i, getDFS()).runAll();
1810      } else if (SetSpaceQuotaCommand.matches(cmd)) {
1811        exitCode = new SetSpaceQuotaCommand(argv, i, getDFS()).runAll();
1812      } else if ("-refreshServiceAcl".equals(cmd)) {
1813        exitCode = refreshServiceAcl();
1814      } else if ("-refreshUserToGroupsMappings".equals(cmd)) {
1815        exitCode = refreshUserToGroupsMappings();
1816      } else if ("-refreshSuperUserGroupsConfiguration".equals(cmd)) {
1817        exitCode = refreshSuperUserGroupsConfiguration();
1818      } else if ("-refreshCallQueue".equals(cmd)) {
1819        exitCode = refreshCallQueue();

1821        exitCode = genericRefresh(argv, i);
1822      } else if ("-printTopology".equals(cmd)) {
1823        exitCode = printTopology();
1824      } else if ("-refreshNamenodes".equals(cmd)) {
1825        exitCode = refreshNamenodes(argv, i);
1826      } else if ("-deleteBlockPool".equals(cmd)) {
1827        exitCode = deleteBlockPool(argv, i);

1831        exitCode = fetchImage(argv, i);
1832      } else if ("-shutdownDatanode".equals(cmd)) {
1833        exitCode = shutdownDatanode(argv, i);
1834      } else if ("-getDatanodeInfo".equals(cmd)) {
1835        exitCode = getDatanodeInfo(argv, i);
1836      } else if ("-reconfig".equals(cmd)) {
1837        exitCode = reconfig(argv, i);
1838      } else if ("-triggerBlockReport".equals(cmd)) {
1839        exitCode = triggerBlockReport(argv);
1840      } else if ("-help".equals(cmd)) {
1841        if (i < argv.length) {
1842          printHelp(argv[i]);
1843        } else {
1844          printHelp("");
1845        }
1846      } else {
1847        exitCode = -1;
1848        System.err.println(cmd.substring(1) + ": Unknown command");
1849        printUsage("");
1850      }
1851    } catch (IllegalArgumentException arge) {
1852      debugException = arge;
1853      exitCode = -1;
1854      System.err.println(cmd.substring(1) + ": " + arge.getLocalizedMessage());
1855      printUsage(cmd);
1856    } catch (RemoteException e) {
1857      //
1858      // This is a error returned by hadoop server. Print
1859      // out the first line of the error message, ignore the stack trace.
1860      exitCode = -1;
1861      debugException = e;
1862      try {
1863        String[] content;
1864        content = e.getLocalizedMessage().split("\n");
1865        System.err.println(cmd.substring(1) + ": "
1866                           + content[0]);
1867      } catch (Exception ex) {
1868        System.err.println(cmd.substring(1) + ": "
1869                           + ex.getLocalizedMessage());
1870        debugException = ex;
1871      }
1872    } catch (Exception e) {
1873      exitCode = -1;
1874      debugException = e;
1875      System.err.println(cmd.substring(1) + ": "
1876                         + e.getLocalizedMessage());
1877    }
1878    if (LOG.isDebugEnabled()) {
1879      LOG.debug("Exception encountered:", debugException);
1880    }
1881    return exitCode;
1882  }

1884  private ClientDatanodeProtocol getDataNodeProxy(String datanode)
1885      throws IOException {
1886    InetSocketAddress datanodeAddr = NetUtils.createSocketAddr(datanode);
1887    // Get the current configuration
1888    Configuration conf = getConf();
1889
1890    // For datanode proxy the server principal should be DN's one.
1891    conf.set(CommonConfigurationKeys.HADOOP_SECURITY_SERVICE_USER_NAME_KEY,
1892        conf.get(DFSConfigKeys.DFS_DATANODE_KERBEROS_PRINCIPAL_KEY, ""));
1893
1894    // Create the client
1895    ClientDatanodeProtocol dnProtocol =     
1896        DFSUtil.createClientDatanodeProtocolProxy(datanodeAddr, getUGI(), conf,
1897            NetUtils.getSocketFactory(conf, ClientDatanodeProtocol.class));

1899  }
