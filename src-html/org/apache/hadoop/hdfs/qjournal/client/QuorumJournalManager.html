152  static String parseJournalId(URI uri) {
153    String path = uri.getPath();
154    Preconditions.checkArgument(path != null && !path.isEmpty(),
155        "Bad URI '%s': must identify journal in path component",
156        uri);
157    String journalId = path.substring(1);
158    checkJournalId(journalId);
159    return journalId;
160  }

162  public static void checkJournalId(String jid) {
163    Preconditions.checkArgument(jid != null &&
164        !jid.isEmpty() &&
165        !jid.contains("/") &&
166        !jid.startsWith("."),
167        "bad journal id: " + jid);
168  }

171  /**
172   * Fence any previous writers, and obtain a unique epoch number
173   * for write-access to the journal nodes.
174   *
175   * @return the new, unique epoch number
176   */
177  Map<AsyncLogger, NewEpochResponseProto> createNewUniqueEpoch()
178      throws IOException {
179    Preconditions.checkState(!loggers.isEpochEstablished(),
180        "epoch already created");
181    
182    Map<AsyncLogger, GetJournalStateResponseProto> lastPromises =
183      loggers.waitForWriteQuorum(loggers.getJournalState(),
184          getJournalStateTimeoutMs, "getJournalState()");
185    
186    long maxPromised = Long.MIN_VALUE;
187    for (GetJournalStateResponseProto resp : lastPromises.values()) {
188      maxPromised = Math.max(maxPromised, resp.getLastPromisedEpoch());
189    }
190    assert maxPromised >= 0;
191    
192    long myEpoch = maxPromised + 1;
193    Map<AsyncLogger, NewEpochResponseProto> resps =
194        loggers.waitForWriteQuorum(loggers.newEpoch(nsInfo, myEpoch),
195            newEpochTimeoutMs, "newEpoch(" + myEpoch + ")");
196        
197    loggers.setEpoch(myEpoch);
198    return resps;
199  }

201  @Override
202  public void format(NamespaceInfo nsInfo) throws IOException {
203    QuorumCall<AsyncLogger,Void> call = loggers.format(nsInfo);
204    try {
205      call.waitFor(loggers.size(), loggers.size(), 0, FORMAT_TIMEOUT_MS,
206          "format");
207    } catch (InterruptedException e) {
208      throw new IOException("Interrupted waiting for format() response");
209    } catch (TimeoutException e) {
210      throw new IOException("Timed out waiting for format() response");
211    }
212    
213    if (call.countExceptions() > 0) {
214      call.rethrowException("Could not format one or more JournalNodes");
215    }
216  }

218  @Override
219  public boolean hasSomeData() throws IOException {
220    QuorumCall<AsyncLogger, Boolean> call =
221        loggers.isFormatted();
222
223    try {
224      call.waitFor(loggers.size(), 0, 0, HASDATA_TIMEOUT_MS, "hasSomeData");
225    } catch (InterruptedException e) {
226      throw new IOException("Interrupted while determining if JNs have data");
227    } catch (TimeoutException e) {
228      throw new IOException("Timed out waiting for response from loggers");
229    }
230    
231    if (call.countExceptions() > 0) {
232      call.rethrowException(
233          "Unable to check if JNs are ready for formatting");
234    }
235    
236    // If any of the loggers returned with a non-empty manifest, then
237    // we should prompt for format.
238    for (Boolean hasData : call.getResults().values()) {
239      if (hasData) {
240        return true;
241      }
242    }
243
244    // Otherwise, none were formatted, we can safely format.
245    return false;
246  }

248  /**
249   * Run recovery/synchronization for a specific segment.
250   * Postconditions:
251   * <ul>
252   * <li>This segment will be finalized on a majority
253   * of nodes.</li>
254   * <li>All nodes which contain the finalized segment will
255   * agree on the length.</li>
256   * </ul>
257   * 
258   * @param segmentTxId the starting txid of the segment
259   * @throws IOException
260   */
261  private void recoverUnclosedSegment(long segmentTxId) throws IOException {
262    Preconditions.checkArgument(segmentTxId > 0);
263    LOG.info("Beginning recovery of unclosed segment starting at txid " +
264        segmentTxId);
265    
266    // Step 1. Prepare recovery
267    QuorumCall<AsyncLogger,PrepareRecoveryResponseProto> prepare =
268        loggers.prepareRecovery(segmentTxId);
269    Map<AsyncLogger, PrepareRecoveryResponseProto> prepareResponses=
270        loggers.waitForWriteQuorum(prepare, prepareRecoveryTimeoutMs,
271            "prepareRecovery(" + segmentTxId + ")");
272    LOG.info("Recovery prepare phase complete. Responses:\n" +
273        QuorumCall.mapToString(prepareResponses));
274
275    // Determine the logger who either:
276    // a) Has already accepted a previous proposal that's higher than any
277    //    other
278    //
279    //  OR, if no such logger exists:
280    //
281    // b) Has the longest log starting at this transaction ID
282    
283    // TODO: we should collect any "ties" and pass the URL for all of them
284    // when syncing, so we can tolerate failure during recovery better.
285    Entry<AsyncLogger, PrepareRecoveryResponseProto> bestEntry = Collections.max(
286        prepareResponses.entrySet(), SegmentRecoveryComparator.INSTANCE); 
287    AsyncLogger bestLogger = bestEntry.getKey();
288    PrepareRecoveryResponseProto bestResponse = bestEntry.getValue();
289    
290    // Log the above decision, check invariants.
291    if (bestResponse.hasAcceptedInEpoch()) {
292      LOG.info("Using already-accepted recovery for segment " +
293          "starting at txid " + segmentTxId + ": " +
294          bestEntry);
295    } else if (bestResponse.hasSegmentState()) {
296      LOG.info("Using longest log: " + bestEntry);
297    } else {
298      // None of the responses to prepareRecovery() had a segment at the given
299      // txid. This can happen for example in the following situation:
300      // - 3 JNs: JN1, JN2, JN3
301      // - writer starts segment 101 on JN1, then crashes before
302      //   writing to JN2 and JN3
303      // - during newEpoch(), we saw the segment on JN1 and decide to
304      //   recover segment 101
305      // - before prepare(), JN1 crashes, and we only talk to JN2 and JN3,
306      //   neither of which has any entry for this log.
307      // In this case, it is allowed to do nothing for recovery, since the
308      // segment wasn't started on a quorum of nodes.
309
310      // Sanity check: we should only get here if none of the responses had
311      // a log. This should be a postcondition of the recovery comparator,
312      // but a bug in the comparator might cause us to get here.
313      for (PrepareRecoveryResponseProto resp : prepareResponses.values()) {
314        assert !resp.hasSegmentState() :
315          "One of the loggers had a response, but no best logger " +
316          "was found.";
317      }
318
319      LOG.info("None of the responders had a log to recover: " +
320          QuorumCall.mapToString(prepareResponses));
321      return;
322    }
323    
324    SegmentStateProto logToSync = bestResponse.getSegmentState();
325    assert segmentTxId == logToSync.getStartTxId();
326    
327    // Sanity check: none of the loggers should be aware of a higher
328    // txid than the txid we intend to truncate to
329    for (Map.Entry<AsyncLogger, PrepareRecoveryResponseProto> e :
330         prepareResponses.entrySet()) {
331      AsyncLogger logger = e.getKey();
332      PrepareRecoveryResponseProto resp = e.getValue();
333
334      if (resp.hasLastCommittedTxId() &&
335          resp.getLastCommittedTxId() > logToSync.getEndTxId()) {
336        throw new AssertionError("Decided to synchronize log to " + logToSync +
337            " but logger " + logger + " had seen txid " +
338            resp.getLastCommittedTxId() + " committed");
339      }
340    }
341    
342    URL syncFromUrl = bestLogger.buildURLToFetchLogs(segmentTxId);
343    
344    QuorumCall<AsyncLogger,Void> accept = loggers.acceptRecovery(logToSync, syncFromUrl);
345    loggers.waitForWriteQuorum(accept, acceptRecoveryTimeoutMs,
346        "acceptRecovery(" + TextFormat.shortDebugString(logToSync) + ")");
347
348    // If one of the loggers above missed the synchronization step above, but
349    // we send a finalize() here, that's OK. It validates the log before
350    // finalizing. Hence, even if it is not "in sync", it won't incorrectly
351    // finalize.
352    QuorumCall<AsyncLogger, Void> finalize =
353        loggers.finalizeLogSegment(logToSync.getStartTxId(), logToSync.getEndTxId()); 
354    loggers.waitForWriteQuorum(finalize, finalizeSegmentTimeoutMs,
355        String.format("finalizeLogSegment(%s-%s)",
356            logToSync.getStartTxId(),
357            logToSync.getEndTxId()));
358  }

360  static List<AsyncLogger> createLoggers(Configuration conf,
361      URI uri, NamespaceInfo nsInfo, AsyncLogger.Factory factory)
362          throws IOException {
363    List<AsyncLogger> ret = Lists.newArrayList();
364    List<InetSocketAddress> addrs = getLoggerAddresses(uri);
365    String jid = parseJournalId(uri);
366    for (InetSocketAddress addr : addrs) {
367      ret.add(factory.createLogger(conf, nsInfo, jid, addr));
368    }
369    return ret;
370  }
371 
372  private static List<InetSocketAddress> getLoggerAddresses(URI uri)
373      throws IOException {
374    String authority = uri.getAuthority();
375    Preconditions.checkArgument(authority != null && !authority.isEmpty(),
376        "URI has no authority: " + uri);
377    
378    String[] parts = StringUtils.split(authority, ';');
379    for (int i = 0; i < parts.length; i++) {
380      parts[i] = parts[i].trim();
381    }
382
383    if (parts.length % 2 == 0) {
384      LOG.warn("Quorum journal URI '" + uri + "' has an even number " +
385          "of Journal Nodes specified. This is not recommended!");
386    }
387    
388    List<InetSocketAddress> addrs = Lists.newArrayList();
389    for (String addr : parts) {
390      addrs.add(NetUtils.createSocketAddr(
391          addr, DFSConfigKeys.DFS_JOURNALNODE_RPC_PORT_DEFAULT));
392    }
393    return addrs;
394  }

398      throws IOException {
399    Preconditions.checkState(isActiveWriter,
400        "must recover segments before starting a new one");
401    QuorumCall<AsyncLogger, Void> q = loggers.startLogSegment(txId,
402        layoutVersion);
403    loggers.waitForWriteQuorum(q, startSegmentTimeoutMs,

406        outputBufferCapacity, writeTxnsTimeoutMs);

431  @Override
432  public void recoverUnfinalizedSegments() throws IOException {
433    Preconditions.checkState(!isActiveWriter, "already active writer");
434    
435    LOG.info("Starting recovery process for unclosed journal segments...");
436    Map<AsyncLogger, NewEpochResponseProto> resps = createNewUniqueEpoch();
437    LOG.info("Successfully started new epoch " + loggers.getEpoch());
438
439    if (LOG.isDebugEnabled()) {
440      LOG.debug("newEpoch(" + loggers.getEpoch() + ") responses:\n" +
441        QuorumCall.mapToString(resps));
442    }
443    
444    long mostRecentSegmentTxId = Long.MIN_VALUE;
445    for (NewEpochResponseProto r : resps.values()) {
446      if (r.hasLastSegmentTxId()) {
447        mostRecentSegmentTxId = Math.max(mostRecentSegmentTxId,
448            r.getLastSegmentTxId());
449      }
450    }
451    
452    // On a completely fresh system, none of the journals have any
453    // segments, so there's nothing to recover.
454    if (mostRecentSegmentTxId != Long.MIN_VALUE) {
455      recoverUnclosedSegment(mostRecentSegmentTxId);
456    }
457    isActiveWriter = true;
458  }

465  @Override
466  public void selectInputStreams(Collection<EditLogInputStream> streams,
467      long fromTxnId, boolean inProgressOk) throws IOException {
468
469    QuorumCall<AsyncLogger, RemoteEditLogManifest> q =
470        loggers.getEditLogManifest(fromTxnId, inProgressOk);
471    Map<AsyncLogger, RemoteEditLogManifest> resps =
472        loggers.waitForWriteQuorum(q, selectInputStreamsTimeoutMs,
473            "selectInputStreams");
474    
475    LOG.debug("selectInputStream manifests:\n" +
476        Joiner.on("\n").withKeyValueSeparator(": ").join(resps));
477    
478    final PriorityQueue<EditLogInputStream> allStreams = 
479        new PriorityQueue<EditLogInputStream>(64,
480            JournalSet.EDIT_LOG_INPUT_STREAM_COMPARATOR);
481    for (Map.Entry<AsyncLogger, RemoteEditLogManifest> e : resps.entrySet()) {
482      AsyncLogger logger = e.getKey();
483      RemoteEditLogManifest manifest = e.getValue();
484      
485      for (RemoteEditLog remoteLog : manifest.getLogs()) {
486        URL url = logger.buildURLToFetchLogs(remoteLog.getStartTxId());
487
488        EditLogInputStream elis = EditLogFileInputStream.fromUrl(
489            connectionFactory, url, remoteLog.getStartTxId(),
490            remoteLog.getEndTxId(), remoteLog.isInProgress());
491        allStreams.add(elis);
492      }
493    }
494    JournalSet.chainAndMakeRedundantStreams(streams, allStreams, fromTxnId);
495  }

558  @Override
559  public boolean canRollBack(StorageInfo storage, StorageInfo prevStorage,
560      int targetLayoutVersion) throws IOException {
561    QuorumCall<AsyncLogger, Boolean> call = loggers.canRollBack(storage,
562        prevStorage, targetLayoutVersion);
563    try {
564      call.waitFor(loggers.size(), loggers.size(), 0, CAN_ROLL_BACK_TIMEOUT_MS,
565          "lockSharedStorage");
566      
567      if (call.countExceptions() > 0) {
568        call.rethrowException("Could not check if roll back possible for"
569            + " one or more JournalNodes");
570      }
571      
572      // Either they all return the same thing or this call fails, so we can
573      // just return the first result.
574      try {
575        DFSUtil.assertAllResultsEqual(call.getResults().values());
576      } catch (AssertionError ae) {
577        throw new IOException("Results differed for canRollBack", ae);
578      }
579      for (Boolean result : call.getResults().values()) {
580        return result;
581      }
582    } catch (InterruptedException e) {
583      throw new IOException("Interrupted waiting for lockSharedStorage() " +
584          "response");
585    } catch (TimeoutException e) {
586      throw new IOException("Timed out waiting for lockSharedStorage() " +
587          "response");
588    }
589    
590    throw new AssertionError("Unreachable code.");
591  }

610  @Override
611  public long getJournalCTime() throws IOException {
612    QuorumCall<AsyncLogger, Long> call = loggers.getJournalCTime();
613    try {
614      call.waitFor(loggers.size(), loggers.size(), 0,
615          GET_JOURNAL_CTIME_TIMEOUT_MS, "getJournalCTime");
616      
617      if (call.countExceptions() > 0) {
618        call.rethrowException("Could not journal CTime for one "
619            + "more JournalNodes");
620      }
621      
622      // Either they all return the same thing or this call fails, so we can
623      // just return the first result.
624      try {
625        DFSUtil.assertAllResultsEqual(call.getResults().values());
626      } catch (AssertionError ae) {
627        throw new IOException("Results differed for getJournalCTime", ae);
628      }
629      for (Long result : call.getResults().values()) {
630        return result;
631      }
632    } catch (InterruptedException e) {
633      throw new IOException("Interrupted waiting for getJournalCTime() " +
634          "response");
635    } catch (TimeoutException e) {
636      throw new IOException("Timed out waiting for getJournalCTime() " +
637          "response");
638    }
639    
640    throw new AssertionError("Unreachable code.");
641  }

646    try {
647      call.waitFor(loggers.size(), loggers.size(), 0,
648          DISCARD_SEGMENTS_TIMEOUT_MS, "discardSegments");
649      if (call.countExceptions() > 0) {
650        call.rethrowException(

652      }
653    } catch (InterruptedException e) {
654      throw new IOException(
655          "Interrupted waiting for discardSegments() response");
656    } catch (TimeoutException e) {
657      throw new IOException(
658          "Timed out waiting for discardSegments() response");
659    }
660  }
