068/**
069 * Channel to a remote JournalNode using Hadoop IPC.
070 * All of the calls are run on a separate thread, and return
071 * {@link ListenableFuture} instances to wait for their result.
072 * This allows calls to be bound together using the {@link QuorumCall}
073 * class.
074 */
075@InterfaceAudience.Private
076public class IPCLoggerChannel implements AsyncLogger {
077

079  protected final InetSocketAddress addr;
080  private QJournalProtocol proxy;

085   */

088   * Executes tasks submitted to it in parallel with each other and with those
089   * submitted to singleThreadExecutor (generally used for read tasks that can
090   * be safely reordered and interleaved with writes).

104  /**
105   * The number of bytes of edits data still in the queue.
106   */
107  private int queuedEditsSizeBytes = 0;
108  
109  /**
110   * The highest txid that has been successfully logged on the remote JN.
111   */
112  private long highestAckedTxId = 0;
113
114  /**
115   * Nanotime of the last time we successfully journaled some edits
116   * to the remote node.
117   */
118  private long lastAckNanos = 0;
119
120  /**
121   * Nanotime of the last time that committedTxId was update. Used
122   * to calculate the lag in terms of time, rather than just a number

127  /**

139   * the writer sets this flag to true to avoid sending useless RPCs.
140   */

160  public IPCLoggerChannel(Configuration conf,
161      NamespaceInfo nsInfo,
162      String journalId,
163      InetSocketAddress addr) {
164    this.conf = conf;
165    this.nsInfo = nsInfo;
166    this.journalId = journalId;
167    this.addr = addr;
168    
169    this.queueSizeLimitBytes = 1024 * 1024 * conf.getInt(
170        DFSConfigKeys.DFS_QJOURNAL_QUEUE_SIZE_LIMIT_KEY,
171        DFSConfigKeys.DFS_QJOURNAL_QUEUE_SIZE_LIMIT_DEFAULT);
172    
173    singleThreadExecutor = MoreExecutors.listeningDecorator(
174        createSingleThreadExecutor());
175    parallelExecutor = MoreExecutors.listeningDecorator(
176        createParallelExecutor());
177    
178    metrics = IPCLoggerChannelMetrics.create(this);
179  }

188    Preconditions.checkArgument(txid >= committedTxId,
189        "Trying to move committed txid backwards in client " +
190         "old: %s new: %s", committedTxId, txid);

195  @Override
196  public void close() {
197    // No more tasks may be submitted after this point.
198    singleThreadExecutor.shutdown();
199    parallelExecutor.shutdown();
200    if (proxy != null) {
201      // TODO: this can hang for quite some time if the client
202      // is currently in the middle of a call to a downed JN.
203      // We should instead do this asynchronously, and just stop
204      // making any more calls after this point (eg clear the queue)
205      RPC.stopProxy(proxy);
206    }
207  }

209  protected QJournalProtocol getProxy() throws IOException {
210    if (proxy != null) return proxy;
211    proxy = createProxy();
212    return proxy;
213  }
214  
215  protected QJournalProtocol createProxy() throws IOException {
216    final Configuration confCopy = new Configuration(conf);
217    
218    // Need to set NODELAY or else batches larger than MTU can trigger 
219    // 40ms nagling delays.
220    confCopy.setBoolean(
221        CommonConfigurationKeysPublic.IPC_CLIENT_TCPNODELAY_KEY,
222        true);
223    
224    RPC.setProtocolEngine(confCopy,
225        QJournalProtocolPB.class, ProtobufRpcEngine.class);
226    return SecurityUtil.doAsLoginUser(
227        new PrivilegedExceptionAction<QJournalProtocol>() {
228          @Override
229          public QJournalProtocol run() throws IOException {
230            RPC.setProtocolEngine(confCopy,
231                QJournalProtocolPB.class, ProtobufRpcEngine.class);
232            QJournalProtocolPB pbproxy = RPC.getProxy(
233                QJournalProtocolPB.class,
234                RPC.getProtocolVersion(QJournalProtocolPB.class),
235                addr, confCopy);
236            return new QJournalProtocolTranslatorPB(pbproxy);
237          }
238        });
239  }

362  @Override
363  public ListenableFuture<Void> sendEdits(
364      final long segmentTxId, final long firstTxnId,
365      final int numTxns, final byte[] data) {
366    try {
367      reserveQueueSpace(data.length);
368    } catch (LoggerTooFarBehindException e) {
369      return Futures.immediateFailedFuture(e);
370    }
371    
372    // When this batch is acked, we use its submission time in order
373    // to calculate how far we are lagging.
374    final long submitNanos = System.nanoTime();
375    
376    ListenableFuture<Void> ret = null;
377    try {
378      ret = singleThreadExecutor.submit(new Callable<Void>() {
379        @Override
380        public Void call() throws IOException {
381          throwIfOutOfSync();
382
383          long rpcSendTimeNanos = System.nanoTime();
384          try {
385            getProxy().journal(createReqInfo(),
386                segmentTxId, firstTxnId, numTxns, data);
387          } catch (IOException e) {
388            QuorumJournalManager.LOG.warn(
389                "Remote journal " + IPCLoggerChannel.this + " failed to " +
390                "write txns " + firstTxnId + "-" + (firstTxnId + numTxns - 1) +
391                ". Will try to write to this JN again after the next " +
392                "log roll.", e); 
393            synchronized (IPCLoggerChannel.this) {
394              outOfSync = true;
395            }
396            throw e;
397          } finally {
398            long now = System.nanoTime();
399            long rpcTime = TimeUnit.MICROSECONDS.convert(
400                now - rpcSendTimeNanos, TimeUnit.NANOSECONDS);
401            long endToEndTime = TimeUnit.MICROSECONDS.convert(
402                now - submitNanos, TimeUnit.NANOSECONDS);
403            metrics.addWriteEndToEndLatency(endToEndTime);
404            metrics.addWriteRpcLatency(rpcTime);
405            if (rpcTime / 1000 > WARN_JOURNAL_MILLIS_THRESHOLD) {
406              QuorumJournalManager.LOG.warn(
407                  "Took " + (rpcTime / 1000) + "ms to send a batch of " +
408                  numTxns + " edits (" + data.length + " bytes) to " +
409                  "remote journal " + IPCLoggerChannel.this);
410            }
411          }
412          synchronized (IPCLoggerChannel.this) {
413            highestAckedTxId = firstTxnId + numTxns - 1;
414            lastAckNanos = submitNanos;
415          }
416          return null;
417        }
418      });
419    } finally {
420      if (ret == null) {
421        // it didn't successfully get submitted,
422        // so adjust the queue size back down.
423        unreserveQueueSpace(data.length);
424      } else {
425        // It was submitted to the queue, so adjust the length
426        // once the call completes, regardless of whether it
427        // succeeds or fails.
428        Futures.addCallback(ret, new FutureCallback<Void>() {
429          @Override
430          public void onFailure(Throwable t) {
431            unreserveQueueSpace(data.length);
432          }
433
434          @Override
435          public void onSuccess(Void t) {
436            unreserveQueueSpace(data.length);
437          }
438        });
439      }
440    }
441    return ret;
442  }

444  private void throwIfOutOfSync()
445      throws JournalOutOfSyncException, IOException {
446    if (isOutOfSync()) {
447      // Even if we're out of sync, it's useful to send an RPC
448      // to the remote node in order to update its lag metrics, etc.
449      heartbeatIfNecessary();
450      throw new JournalOutOfSyncException(
451          "Journal disabled until next roll");
452    }
453  }
454
455  /**
456   * When we've entered an out-of-sync state, it's still useful to periodically
457   * send an empty RPC to the server, such that it has the up to date
458   * committedTxId. This acts as a sanity check during recovery, and also allows
459   * that node's metrics to be up-to-date about its lag.
460   * 
461   * In the future, this method may also be used in order to check that the
462   * current node is still the current writer, even if no edits are being
463   * written.
464   */
465  private void heartbeatIfNecessary() throws IOException {
466    if (lastHeartbeatStopwatch.now(TimeUnit.MILLISECONDS)
467        > HEARTBEAT_INTERVAL_MILLIS || !lastHeartbeatStopwatch.isRunning()) {
468      try {
469        getProxy().heartbeat(createReqInfo());
470      } finally {
471        // Don't send heartbeats more often than the configured interval,
472        // even if they fail.
473        lastHeartbeatStopwatch.reset().start();
474      }
475    }
476  }

478  private synchronized void reserveQueueSpace(int size)
479      throws LoggerTooFarBehindException {
480    Preconditions.checkArgument(size >= 0);
481    if (queuedEditsSizeBytes + size > queueSizeLimitBytes &&
482        queuedEditsSizeBytes > 0) {
483      throw new LoggerTooFarBehindException();
484    }
485    queuedEditsSizeBytes += size;
486  }
487  
488  private synchronized void unreserveQueueSpace(int size) {
489    Preconditions.checkArgument(size >= 0);
490    queuedEditsSizeBytes -= size;
491  }

679  @Override
680  public synchronized void appendReport(StringBuilder sb) {
681    sb.append("Written txid ").append(highestAckedTxId);
682    long behind = getLagTxns();
683    if (behind > 0) {
684      if (lastAckNanos != 0) {
685        long lagMillis = getLagTimeMillis();
686        sb.append(" (" + behind + " txns/" + lagMillis + "ms behind)");
687      } else {
688        sb.append(" (never written");
689      }
690    }
691    if (outOfSync) {
692      sb.append(" (will try to re-sync on next segment)");
693    }
694  }
695  
696  public synchronized long getLagTxns() {
697    return Math.max(committedTxId - highestAckedTxId, 0);
698  }
699  
700  public synchronized long getLagTimeMillis() {
701    return TimeUnit.MILLISECONDS.convert(
702        Math.max(lastCommitNanos - lastAckNanos, 0),
703        TimeUnit.NANOSECONDS);
704  }

707    if (ret.hasFromURL()) {
708      URI uri = URI.create(ret.getFromURL());
709      httpServerURL = getHttpServerURI(uri.getScheme(), uri.getPort());
710    } else {
711      httpServerURL = getHttpServerURI("http", ret.getHttpPort());;
712    }

715  private void constructHttpServerURI(GetJournalStateResponseProto ret) {
716    if (ret.hasFromURL()) {
717      URI uri = URI.create(ret.getFromURL());
718      httpServerURL = getHttpServerURI(uri.getScheme(), uri.getPort());
719    } else {
720      httpServerURL = getHttpServerURI("http", ret.getHttpPort());;
721    }
722  }

724  /**
725   * Construct the http server based on the response.
726   *
727   * The fromURL field in the response specifies the endpoint of the http
728   * server. However, the address might not be accurate since the server can
729   * bind to multiple interfaces. Here the client plugs in the address specified
730   * in the configuration and generates the URI.
731   */
732  private URL getHttpServerURI(String scheme, int port) {
733    try {
734      return new URL(scheme, addr.getHostName(), port, "");
735    } catch (MalformedURLException e) {
736      // Unreachable
737      throw new RuntimeException(e);
738    }
739  }
740
741  private boolean hasHttpServerEndPoint() {
742   return httpServerURL != null;
743  }
