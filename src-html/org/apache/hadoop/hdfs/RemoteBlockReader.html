109  /* same interface as inputStream java.io.InputStream#read()
110   * used by DFSInputStream#read()
111   * This violates one rule when there is a checksum error:
112   * "Read should not modify user buffer before successful read"
113   * because it first reads the data to user buffer and then checks
114   * the checksum.
115   */
116  @Override
117  public synchronized int read(byte[] buf, int off, int len) 
118                               throws IOException {
119    
120    // This has to be set here, *before* the skip, since we can
121    // hit EOS during the skip, in the case that our entire read
122    // is smaller than the checksum chunk.
123    boolean eosBefore = eos;
124
125    //for the first read, skip the extra bytes at the front.
126    if (lastChunkLen < 0 && startOffset > firstChunkOffset && len > 0) {
127      // Skip these bytes. But don't call this.skip()!
128      int toSkip = (int)(startOffset - firstChunkOffset);
129      if ( skipBuf == null ) {
130        skipBuf = new byte[bytesPerChecksum];
131      }
132      if ( super.read(skipBuf, 0, toSkip) != toSkip ) {
133        // should never happen
134        throw new IOException("Could not skip required number of bytes");
135      }
136    }
137    
138    int nRead = super.read(buf, off, len);
139
140    // if eos was set in the previous read, send a status code to the DN
141    if (eos && !eosBefore && nRead >= 0) {
142      if (needChecksum()) {
143        sendReadResult(peer, Status.CHECKSUM_OK);
144      } else {
145        sendReadResult(peer, Status.SUCCESS);
146      }
147    }
148    return nRead;
149  }

151  @Override
152  public synchronized long skip(long n) throws IOException {
153    /* How can we make sure we don't throw a ChecksumException, at least
154     * in majority of the cases?. This one throws. */  
155    if ( skipBuf == null ) {
156      skipBuf = new byte[bytesPerChecksum]; 
157    }
158
159    long nSkipped = 0;
160    while ( nSkipped < n ) {
161      int toSkip = (int)Math.min(n-nSkipped, skipBuf.length);
162      int ret = read(skipBuf, 0, toSkip);
163      if ( ret <= 0 ) {
164        return nSkipped;
165      }
166      nSkipped += ret;
167    }
168    return nSkipped;
169  }

197  /**
198   * Makes sure that checksumBytes has enough capacity 
199   * and limit is set to the number of checksum bytes needed 
200   * to be read.
201   */
202  private void adjustChecksumBytes(int dataLen) {
203    int requiredSize = 
204      ((dataLen + bytesPerChecksum - 1)/bytesPerChecksum)*checksumSize;
205    if (checksumBytes == null || requiredSize > checksumBytes.capacity()) {
206      checksumBytes =  ByteBuffer.wrap(new byte[requiredSize]);
207    } else {
208      checksumBytes.clear();
209    }
210    checksumBytes.limit(requiredSize);
211  }

213  @Override
214  protected synchronized int readChunk(long pos, byte[] buf, int offset, 
215                                       int len, byte[] checksumBuf) 
216                                       throws IOException {
217    TraceScope scope =
218        Trace.startSpan("RemoteBlockReader#readChunk(" + blockId + ")",
219            Sampler.NEVER);
220    try {
221      return readChunkImpl(pos, buf, offset, len, checksumBuf);
222    } finally {
223      scope.close();
224    }
225  }
226
227  private synchronized int readChunkImpl(long pos, byte[] buf, int offset,
228                                     int len, byte[] checksumBuf)
229                                     throws IOException {
230    // Read one chunk.
231    if (eos) {
232      // Already hit EOF
233      return -1;
234    }
235    
236    // Read one DATA_CHUNK.
237    long chunkOffset = lastChunkOffset;
238    if ( lastChunkLen > 0 ) {
239      chunkOffset += lastChunkLen;
240    }
241    
242    // pos is relative to the start of the first chunk of the read.
243    // chunkOffset is relative to the start of the block.
244    // This makes sure that the read passed from FSInputChecker is the
245    // for the same chunk we expect to be reading from the DN.
246    if ( (pos + firstChunkOffset) != chunkOffset ) {
247      throw new IOException("Mismatch in pos : " + pos + " + " + 
248                            firstChunkOffset + " != " + chunkOffset);
249    }
250
251    // Read next packet if the previous packet has been read completely.
252    if (dataLeft <= 0) {
253      //Read packet headers.
254      PacketHeader header = new PacketHeader();
255      header.readFields(in);
256
257      if (LOG.isDebugEnabled()) {
258        LOG.debug("DFSClient readChunk got header " + header);
259      }
260
261      // Sanity check the lengths
262      if (!header.sanityCheck(lastSeqNo)) {
263           throw new IOException("BlockReader: error in packet header " +
264                                 header);
265      }
266
267      lastSeqNo = header.getSeqno();
268      dataLeft = header.getDataLen();
269      adjustChecksumBytes(header.getDataLen());
270      if (header.getDataLen() > 0) {
271        IOUtils.readFully(in, checksumBytes.array(), 0,
272                          checksumBytes.limit());
273      }
274    }
275
276    // Sanity checks
277    assert len >= bytesPerChecksum;
278    assert checksum != null;
279    assert checksumSize == 0 || (checksumBuf.length % checksumSize == 0);
280
281
282    int checksumsToRead, bytesToRead;
283
284    if (checksumSize > 0) {
285
286      // How many chunks left in our packet - this is a ceiling
287      // since we may have a partial chunk at the end of the file
288      int chunksLeft = (dataLeft - 1) / bytesPerChecksum + 1;
289
290      // How many chunks we can fit in databuffer
291      //  - note this is a floor since we always read full chunks
292      int chunksCanFit = Math.min(len / bytesPerChecksum,
293                                  checksumBuf.length / checksumSize);
294
295      // How many chunks should we read
296      checksumsToRead = Math.min(chunksLeft, chunksCanFit);
297      // How many bytes should we actually read
298      bytesToRead = Math.min(
299        checksumsToRead * bytesPerChecksum, // full chunks
300        dataLeft); // in case we have a partial
301    } else {
302      // no checksum
303      bytesToRead = Math.min(dataLeft, len);
304      checksumsToRead = 0;
305    }
306
307    if ( bytesToRead > 0 ) {
308      // Assert we have enough space
309      assert bytesToRead <= len;
310      assert checksumBytes.remaining() >= checksumSize * checksumsToRead;
311      assert checksumBuf.length >= checksumSize * checksumsToRead;
312      IOUtils.readFully(in, buf, offset, bytesToRead);
313      checksumBytes.get(checksumBuf, 0, checksumSize * checksumsToRead);
314    }
315
316    dataLeft -= bytesToRead;
317    assert dataLeft >= 0;
318
319    lastChunkOffset = chunkOffset;
320    lastChunkLen = bytesToRead;
321
322    // If there's no data left in the current packet after satisfying
323    // this read, and we have satisfied the client read, we expect
324    // an empty packet header from the DN to signify this.
325    // Note that pos + bytesToRead may in fact be greater since the
326    // DN finishes off the entire last chunk.
327    if (dataLeft == 0 &&
328        pos + bytesToRead >= bytesNeededToFinish) {
329
330      // Read header
331      PacketHeader hdr = new PacketHeader();
332      hdr.readFields(in);
333
334      if (!hdr.isLastPacketInBlock() ||
335          hdr.getDataLen() != 0) {
336        throw new IOException("Expected empty end-of-read packet! Header: " +
337                              hdr);
338      }
339
340      eos = true;
341    }
342
343    if ( bytesToRead == 0 ) {
344      return -1;
345    }
346
347    return bytesToRead;
348  }

350  private RemoteBlockReader(String file, String bpid, long blockId,
351      DataInputStream in, DataChecksum checksum, boolean verifyChecksum,
352      long startOffset, long firstChunkOffset, long bytesToRead, Peer peer,
353      DatanodeID datanodeID, PeerCache peerCache) {
354    // Path is used only for printing block and file information in debug
355    super(new Path("/" + Block.BLOCK_FILE_PREFIX + blockId +
356                    ":" + bpid + ":of:"+ file)/*too non path-like?*/,
357          1, verifyChecksum,
358          checksum.getChecksumSize() > 0? checksum : null, 
359          checksum.getBytesPerChecksum(),
360          checksum.getChecksumSize());
361
362    this.isLocal = DFSClient.isLocalAddress(NetUtils.
363        createSocketAddr(datanodeID.getXferAddr()));
364    
365    this.peer = peer;
366    this.datanodeID = datanodeID;
367    this.in = in;
368    this.checksum = checksum;
369    this.startOffset = Math.max( startOffset, 0 );
370    this.blockId = blockId;
371
372    // The total number of bytes that we need to transfer from the DN is
373    // the amount that the user wants (bytesToRead), plus the padding at
374    // the beginning in order to chunk-align. Note that the DN may elect
375    // to send more than this amount if the read starts/ends mid-chunk.
376    this.bytesNeededToFinish = bytesToRead + (startOffset - firstChunkOffset);
377
378    this.firstChunkOffset = firstChunkOffset;
379    lastChunkOffset = firstChunkOffset;
380    lastChunkLen = -1;
381
382    bytesPerChecksum = this.checksum.getBytesPerChecksum();
383    checksumSize = this.checksum.getChecksumSize();
384    this.peerCache = peerCache;
385  }

387  /**
388   * Create a new BlockReader specifically to satisfy a read.
389   * This method also sends the OP_READ_BLOCK request.
390   *
391   * @param file  File location
392   * @param block  The block object
393   * @param blockToken  The block token for security
394   * @param startOffset  The read offset, relative to block head
395   * @param len  The number of bytes to read
396   * @param bufferSize  The IO buffer size (not the client buffer size)
397   * @param verifyChecksum  Whether to verify checksum
398   * @param clientName  Client name
399   * @return New BlockReader instance, or null on error.
400   */
401  public static RemoteBlockReader newBlockReader(String file,
402                                     ExtendedBlock block, 
403                                     Token<BlockTokenIdentifier> blockToken,
404                                     long startOffset, long len,
405                                     int bufferSize, boolean verifyChecksum,
406                                     String clientName, Peer peer,
407                                     DatanodeID datanodeID,
408                                     PeerCache peerCache,
409                                     CachingStrategy cachingStrategy)
410                                       throws IOException {
411    // in and out will be closed when sock is closed (by the caller)
412    final DataOutputStream out =
413        new DataOutputStream(new BufferedOutputStream(peer.getOutputStream()));
414    new Sender(out).readBlock(block, blockToken, clientName, startOffset, len,
415        verifyChecksum, cachingStrategy);
416    
417    //
418    // Get bytes in block, set streams
419    //
420
421    DataInputStream in = new DataInputStream(
422        new BufferedInputStream(peer.getInputStream(), bufferSize));
423    
424    BlockOpResponseProto status = BlockOpResponseProto.parseFrom(
425        PBHelper.vintPrefixed(in));
426    RemoteBlockReader2.checkSuccess(status, peer, block, file);
427    ReadOpChecksumInfoProto checksumInfo =
428      status.getReadOpChecksumInfo();
429    DataChecksum checksum = DataTransferProtoUtil.fromProto(
430        checksumInfo.getChecksum());
431    //Warning when we get CHECKSUM_NULL?
432    
433    // Read the first chunk offset.
434    long firstChunkOffset = checksumInfo.getChunkOffset();
435    
436    if ( firstChunkOffset < 0 || firstChunkOffset > startOffset ||
437        firstChunkOffset <= (startOffset - checksum.getBytesPerChecksum())) {
438      throw new IOException("BlockReader: error in first chunk offset (" +
439                            firstChunkOffset + ") startOffset is " + 
440                            startOffset + " for file " + file);
441    }
442
443    return new RemoteBlockReader(file, block.getBlockPoolId(), block.getBlockId(),
444        in, checksum, verifyChecksum, startOffset, firstChunkOffset, len,
445        peer, datanodeID, peerCache);
446  }

448  @Override
449  public synchronized void close() throws IOException {
450    startOffset = -1;
451    checksum = null;
452    if (peerCache != null & sentStatusCode) {
453      peerCache.put(datanodeID, peer);
454    } else {
455      peer.close();
456    }
457
458    // in will be closed when its Socket is closed.
459  }

472  /**
473   * When the reader reaches end of the read, it sends a status response
474   * (e.g. CHECKSUM_OK) to the DN. Failure to do so could lead to the DN
475   * closing our connection (which we will re-open), but won't affect
476   * data correctness.
477   */
478  void sendReadResult(Peer peer, Status statusCode) {
479    assert !sentStatusCode : "already sent status code to " + peer;
480    try {
481      RemoteBlockReader2.writeReadResult(peer.getOutputStream(), statusCode);
482      sentStatusCode = true;
483    } catch (IOException e) {
484      // It's ok not to be able to send this. But something is probably wrong.
485      LOG.info("Could not send read status (" + statusCode + ") to datanode " +
486               peer.getRemoteAddressString() + ": " + e.getMessage());
487    }
488  }
