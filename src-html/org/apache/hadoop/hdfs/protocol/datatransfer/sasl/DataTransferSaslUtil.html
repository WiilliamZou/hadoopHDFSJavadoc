085  /**
086   * Checks that SASL negotiation has completed for the given participant, and
087   * the negotiated quality of protection is included in the given SASL
088   * properties and therefore acceptable.
089   *
090   * @param sasl participant to check
091   * @param saslProps properties of SASL negotiation
092   * @throws IOException for any error
093   */
094  public static void checkSaslComplete(SaslParticipant sasl,
095      Map<String, String> saslProps) throws IOException {
096    if (!sasl.isComplete()) {
097      throw new IOException("Failed to complete SASL handshake");
098    }
099    Set<String> requestedQop = ImmutableSet.copyOf(Arrays.asList(
100      saslProps.get(Sasl.QOP).split(",")));
101    String negotiatedQop = sasl.getNegotiatedQop();
102    LOG.debug("Verifying QOP, requested QOP = {}, negotiated QOP = {}",
103      requestedQop, negotiatedQop);
104    if (!requestedQop.contains(negotiatedQop)) {
105      throw new IOException(String.format("SASL handshake completed, but " +
106        "channel does not have acceptable quality of protection, " +
107        "requested = %s, negotiated = %s", requestedQop, negotiatedQop));
108    }
109  }

111  /**
112   * Check whether requested SASL Qop contains privacy.
113   * 
114   * @param saslProps properties of SASL negotiation
115   * @return boolean true if privacy exists
116   */
117  public static boolean requestedQopContainsPrivacy(
118      Map<String, String> saslProps) {
119    Set<String> requestedQop = ImmutableSet.copyOf(Arrays.asList(
120        saslProps.get(Sasl.QOP).split(",")));
121    return requestedQop.contains("auth-conf");
122  }

124  /**
125   * Creates SASL properties required for an encrypted SASL negotiation.
126   *
127   * @param encryptionAlgorithm to use for SASL negotation
128   * @return properties of encrypted SASL negotiation
129   */
130  public static Map<String, String> createSaslPropertiesForEncryption(
131      String encryptionAlgorithm) {
132    Map<String, String> saslProps = Maps.newHashMapWithExpectedSize(3);
133    saslProps.put(Sasl.QOP, QualityOfProtection.PRIVACY.getSaslQop());
134    saslProps.put(Sasl.SERVER_AUTH, "true");
135    saslProps.put("com.sun.security.sasl.digest.cipher", encryptionAlgorithm);
136    return saslProps;
137  }

140   * For an encrypted SASL negotiation, encodes an encryption key to a SASL
141   * password.
142   *
143   * @param encryptionKey to encode

151  /**
152   * Returns InetAddress from peer.  The getRemoteAddressString has the form
153   * [host][/ip-address]:port.  The host may be missing.  The IP address (and
154   * preceding '/') may be missing.  The port preceded by ':' is always present.
155   *
156   * @param peer
157   * @return InetAddress from peer
158   */
159  public static InetAddress getPeerAddress(Peer peer) {
160    String remoteAddr = peer.getRemoteAddressString().split(":")[0];
161    int slashIdx = remoteAddr.indexOf('/');
162    return InetAddresses.forString(slashIdx != -1 ?
163        remoteAddr.substring(slashIdx + 1, remoteAddr.length()) :
164        remoteAddr);
165  }

167  /**
168   * Creates a SaslPropertiesResolver from the given configuration.  This method
169   * works by cloning the configuration, translating configuration properties
170   * specific to DataTransferProtocol to what SaslPropertiesResolver expects,
171   * and then delegating to SaslPropertiesResolver for initialization.  This
172   * method returns null if SASL protection has not been configured for
173   * DataTransferProtocol.
174   *
175   * @param conf configuration to read
176   * @return SaslPropertiesResolver for DataTransferProtocol, or null if not
177   *   configured
178   */
179  public static SaslPropertiesResolver getSaslPropertiesResolver(
180      Configuration conf) {
181    String qops = conf.get(DFS_DATA_TRANSFER_PROTECTION_KEY);
182    if (qops == null || qops.isEmpty()) {
183      LOG.debug("DataTransferProtocol not using SaslPropertiesResolver, no " +
184        "QOP found in configuration for {}", DFS_DATA_TRANSFER_PROTECTION_KEY);
185      return null;
186    }
187    Configuration saslPropsResolverConf = new Configuration(conf);
188    saslPropsResolverConf.set(HADOOP_RPC_PROTECTION, qops);
189    Class<? extends SaslPropertiesResolver> resolverClass = conf.getClass(
190      HADOOP_SECURITY_SASL_PROPS_RESOLVER_CLASS,
191      SaslPropertiesResolver.class, SaslPropertiesResolver.class);
192    resolverClass = conf.getClass(DFS_DATA_TRANSFER_SASL_PROPS_RESOLVER_CLASS_KEY,
193      resolverClass, SaslPropertiesResolver.class);
194    saslPropsResolverConf.setClass(HADOOP_SECURITY_SASL_PROPS_RESOLVER_CLASS,
195      resolverClass, SaslPropertiesResolver.class);
196    SaslPropertiesResolver resolver = SaslPropertiesResolver.getInstance(
197      saslPropsResolverConf);
198    LOG.debug("DataTransferProtocol using SaslPropertiesResolver, configured " +
199      "QOP {} = {}, configured class {} = {}", DFS_DATA_TRANSFER_PROTECTION_KEY, qops, 
200      DFS_DATA_TRANSFER_SASL_PROPS_RESOLVER_CLASS_KEY, resolverClass);
201    return resolver;
202  }

204  /**
205   * Reads a SASL negotiation message.
206   *
207   * @param in stream to read
208   * @return bytes of SASL negotiation messsage
209   * @throws IOException for any error
210   */
211  public static byte[] readSaslMessage(InputStream in) throws IOException {
212    DataTransferEncryptorMessageProto proto =
213        DataTransferEncryptorMessageProto.parseFrom(vintPrefixed(in));
214    if (proto.getStatus() == DataTransferEncryptorStatus.ERROR_UNKNOWN_KEY) {
215      throw new InvalidEncryptionKeyException(proto.getMessage());
216    } else if (proto.getStatus() == DataTransferEncryptorStatus.ERROR) {
217      throw new IOException(proto.getMessage());
218    } else {
219      return proto.getPayload().toByteArray();
220    }
221  }

223  /**
224   * Reads a SASL negotiation message and negotiation cipher options. 
225   * 
226   * @param in stream to read
227   * @param cipherOptions list to store negotiation cipher options
228   * @return byte[] SASL negotiation message
229   * @throws IOException for any error
230   */
231  public static byte[] readSaslMessageAndNegotiationCipherOptions(
232      InputStream in, List<CipherOption> cipherOptions) throws IOException {
233    DataTransferEncryptorMessageProto proto =
234        DataTransferEncryptorMessageProto.parseFrom(vintPrefixed(in));
235    if (proto.getStatus() == DataTransferEncryptorStatus.ERROR_UNKNOWN_KEY) {
236      throw new InvalidEncryptionKeyException(proto.getMessage());
237    } else if (proto.getStatus() == DataTransferEncryptorStatus.ERROR) {
238      throw new IOException(proto.getMessage());
239    } else {
240      List<CipherOptionProto> optionProtos = proto.getCipherOptionList();
241      if (optionProtos != null) {
242        for (CipherOptionProto optionProto : optionProtos) {
243          cipherOptions.add(PBHelper.convert(optionProto));
244        }
245      }
246      return proto.getPayload().toByteArray();
247    }
248  }

250  /**
251   * Negotiate a cipher option which server supports.
252   * 
253   * @param conf the configuration
254   * @param options the cipher options which client supports
255   * @return CipherOption negotiated cipher option
256   */
257  public static CipherOption negotiateCipherOption(Configuration conf,
258      List<CipherOption> options) throws IOException {
259    // Negotiate cipher suites if configured.  Currently, the only supported
260    // cipher suite is AES/CTR/NoPadding, but the protocol allows multiple
261    // values for future expansion.
262    String cipherSuites = conf.get(DFS_ENCRYPT_DATA_TRANSFER_CIPHER_SUITES_KEY);
263    if (cipherSuites == null || cipherSuites.isEmpty()) {
264      return null;
265    }
266    if (!cipherSuites.equals(CipherSuite.AES_CTR_NOPADDING.getName())) {
267      throw new IOException(String.format("Invalid cipher suite, %s=%s",
268          DFS_ENCRYPT_DATA_TRANSFER_CIPHER_SUITES_KEY, cipherSuites));
269    }
270    if (options != null) {
271      for (CipherOption option : options) {
272        CipherSuite suite = option.getCipherSuite();
273        if (suite == CipherSuite.AES_CTR_NOPADDING) {
274          int keyLen = conf.getInt(
275              DFS_ENCRYPT_DATA_TRANSFER_CIPHER_KEY_BITLENGTH_KEY,
276              DFS_ENCRYPT_DATA_TRANSFER_CIPHER_KEY_BITLENGTH_DEFAULT) / 8;
277          CryptoCodec codec = CryptoCodec.getInstance(conf, suite);
278          byte[] inKey = new byte[keyLen];
279          byte[] inIv = new byte[suite.getAlgorithmBlockSize()];
280          byte[] outKey = new byte[keyLen];
281          byte[] outIv = new byte[suite.getAlgorithmBlockSize()];
282          codec.generateSecureRandom(inKey);
283          codec.generateSecureRandom(inIv);
284          codec.generateSecureRandom(outKey);
285          codec.generateSecureRandom(outIv);
286          return new CipherOption(suite, inKey, inIv, outKey, outIv);
287        }
288      }
289    }
290    return null;
291  }

293  /**
294   * Send SASL message and negotiated cipher option to client.
295   * 
296   * @param out stream to receive message
297   * @param payload to send
298   * @param option negotiated cipher option
299   * @throws IOException for any error
300   */
301  public static void sendSaslMessageAndNegotiatedCipherOption(
302      OutputStream out, byte[] payload, CipherOption option) 
303          throws IOException {
304    DataTransferEncryptorMessageProto.Builder builder =
305        DataTransferEncryptorMessageProto.newBuilder();
306    
307    builder.setStatus(DataTransferEncryptorStatus.SUCCESS);
308    if (payload != null) {
309      builder.setPayload(ByteString.copyFrom(payload));
310    }
311    if (option != null) {
312      builder.addCipherOption(PBHelper.convert(option));
313    }
314    
315    DataTransferEncryptorMessageProto proto = builder.build();
316    proto.writeDelimitedTo(out);
317    out.flush();
318  }

320  /**
321   * Create IOStreamPair of {@link org.apache.hadoop.crypto.CryptoInputStream}
322   * and {@link org.apache.hadoop.crypto.CryptoOutputStream}
323   * 
324   * @param conf the configuration
325   * @param cipherOption negotiated cipher option
326   * @param out underlying output stream
327   * @param in underlying input stream
328   * @param isServer is server side
329   * @return IOStreamPair the stream pair
330   * @throws IOException for any error
331   */
332  public static IOStreamPair createStreamPair(Configuration conf,
333      CipherOption cipherOption, OutputStream out, InputStream in, 
334      boolean isServer) throws IOException {
335    if (LOG.isDebugEnabled()) {
336      LOG.debug("Creating IOStreamPair of CryptoInputStream and " +
337          "CryptoOutputStream.");
338    }
339    CryptoCodec codec = CryptoCodec.getInstance(conf, 
340        cipherOption.getCipherSuite());
341    byte[] inKey = cipherOption.getInKey();
342    byte[] inIv = cipherOption.getInIv();
343    byte[] outKey = cipherOption.getOutKey();
344    byte[] outIv = cipherOption.getOutIv();
345    InputStream cIn = new CryptoInputStream(in, codec, 
346        isServer ? inKey : outKey, isServer ? inIv : outIv);
347    OutputStream cOut = new CryptoOutputStream(out, codec, 
348        isServer ? outKey : inKey, isServer ? outIv : inIv);
349    return new IOStreamPair(cIn, cOut);
350  }

403  /**
404   * Read SASL message and negotiated cipher option from server.
405   * 
406   * @param in stream to read
407   * @return SaslResponseWithNegotiatedCipherOption SASL message and 
408   * negotiated cipher option
409   * @throws IOException for any error
410   */
411  public static SaslResponseWithNegotiatedCipherOption
412      readSaslMessageAndNegotiatedCipherOption(InputStream in)
413          throws IOException {
414    DataTransferEncryptorMessageProto proto =
415        DataTransferEncryptorMessageProto.parseFrom(vintPrefixed(in));
416    if (proto.getStatus() == DataTransferEncryptorStatus.ERROR_UNKNOWN_KEY) {
417      throw new InvalidEncryptionKeyException(proto.getMessage());
418    } else if (proto.getStatus() == DataTransferEncryptorStatus.ERROR) {
419      throw new IOException(proto.getMessage());
420    } else {
421      byte[] response = proto.getPayload().toByteArray();
422      List<CipherOption> options = PBHelper.convertCipherOptionProtos(
423          proto.getCipherOptionList());
424      CipherOption option = null;
425      if (options != null && !options.isEmpty()) {
426        option = options.get(0);
427      }
428      return new SaslResponseWithNegotiatedCipherOption(response, option);
429    }
430  }

432  /**
433   * Encrypt the key and iv of the negotiated cipher option.
434   * 
435   * @param option negotiated cipher option
436   * @param sasl SASL participant representing server
437   * @return CipherOption negotiated cipher option which contains the 
438   * encrypted key and iv
439   * @throws IOException for any error
440   */
441  public static CipherOption wrap(CipherOption option, SaslParticipant sasl) 
442      throws IOException {
443    if (option != null) {
444      byte[] inKey = option.getInKey();
445      if (inKey != null) {
446        inKey = sasl.wrap(inKey, 0, inKey.length);
447      }
448      byte[] outKey = option.getOutKey();
449      if (outKey != null) {
450        outKey = sasl.wrap(outKey, 0, outKey.length);
451      }
452      return new CipherOption(option.getCipherSuite(), inKey, option.getInIv(),
453          outKey, option.getOutIv());
454    }
455    
456    return null;
457  }
458  
459  /**
460   * Decrypt the key and iv of the negotiated cipher option.
461   * 
462   * @param option negotiated cipher option
463   * @param sasl SASL participant representing client
464   * @return CipherOption negotiated cipher option which contains the 
465   * decrypted key and iv
466   * @throws IOException for any error
467   */
468  public static CipherOption unwrap(CipherOption option, SaslParticipant sasl)
469      throws IOException {
470    if (option != null) {
471      byte[] inKey = option.getInKey();
472      if (inKey != null) {
473        inKey = sasl.unwrap(inKey, 0, inKey.length);
474      }
475      byte[] outKey = option.getOutKey();
476      if (outKey != null) {
477        outKey = sasl.unwrap(outKey, 0, outKey.length);
478      }
479      return new CipherOption(option.getCipherSuite(), inKey, option.getInIv(),
480          outKey, option.getOutIv());
481    }
482    
483    return null;
484  }

507    }
508

510    proto.writeDelimitedTo(out);
