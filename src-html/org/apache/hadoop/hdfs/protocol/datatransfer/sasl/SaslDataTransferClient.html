064/**
065 * Negotiates SASL for DataTransferProtocol on behalf of a client.  There are
066 * two possible supported variants of SASL negotiation: either a general-purpose
067 * negotiation supporting any quality of protection, or a specialized
068 * negotiation that enforces privacy as the quality of protection using a
069 * cryptographically strong encryption key.
070 *
071 * This class is used in both the HDFS client and the DataNode.  The DataNode
072 * needs it, because it acts as a client to other DataNodes during write
073 * pipelines and block transfers.
074 */
075@InterfaceAudience.Private
076public class SaslDataTransferClient {
077
078  private static final Logger LOG = LoggerFactory.getLogger(
079    SaslDataTransferClient.class);
080
081  private final Configuration conf;
082  private final AtomicBoolean fallbackToSimpleAuth;
083  private final SaslPropertiesResolver saslPropsResolver;
084  private final TrustedChannelResolver trustedChannelResolver;
085
086  /**
087   * Creates a new SaslDataTransferClient.  This constructor is used in cases
088   * where it is not relevant to track if a secure client did a fallback to
089   * simple auth.  For intra-cluster connections between data nodes in the same
090   * cluster, we can assume that all run under the same security configuration.
091   *
092   * @param conf the configuration
093   * @param saslPropsResolver for determining properties of SASL negotiation
094   * @param trustedChannelResolver for identifying trusted connections that do
095   *   not require SASL negotiation
096   */
097  public SaslDataTransferClient(Configuration conf, 
098      SaslPropertiesResolver saslPropsResolver,
099      TrustedChannelResolver trustedChannelResolver) {
100    this(conf, saslPropsResolver, trustedChannelResolver, null);
101  }

108   * @param trustedChannelResolver for identifying trusted connections that do

123  /**
124   * Sends client SASL negotiation for a newly allocated socket if required.
125   *
126   * @param socket connection socket
127   * @param underlyingOut connection output stream
128   * @param underlyingIn connection input stream
129   * @param encryptionKeyFactory for creation of an encryption key
130   * @param accessToken connection block access token
131   * @param datanodeId ID of destination DataNode
132   * @return new pair of streams, wrapped after SASL negotiation
133   * @throws IOException for any error
134   */
135  public IOStreamPair newSocketSend(Socket socket, OutputStream underlyingOut,
136      InputStream underlyingIn, DataEncryptionKeyFactory encryptionKeyFactory,
137      Token<BlockTokenIdentifier> accessToken, DatanodeID datanodeId)
138      throws IOException {
139    // The encryption key factory only returns a key if encryption is enabled.
140    DataEncryptionKey encryptionKey = !trustedChannelResolver.isTrusted() ?
141      encryptionKeyFactory.newDataEncryptionKey() : null;
142    IOStreamPair ios = send(socket.getInetAddress(), underlyingOut,
143      underlyingIn, encryptionKey, accessToken, datanodeId);
144    return ios != null ? ios : new IOStreamPair(underlyingIn, underlyingOut);
145  }

147  /**
148   * Sends client SASL negotiation for a peer if required.
149   *
150   * @param peer connection peer
151   * @param encryptionKeyFactory for creation of an encryption key
152   * @param accessToken connection block access token
153   * @param datanodeId ID of destination DataNode
154   * @return new pair of streams, wrapped after SASL negotiation
155   * @throws IOException for any error
156   */
157  public Peer peerSend(Peer peer, DataEncryptionKeyFactory encryptionKeyFactory,
158      Token<BlockTokenIdentifier> accessToken, DatanodeID datanodeId)
159      throws IOException {
160    IOStreamPair ios = checkTrustAndSend(getPeerAddress(peer),
161      peer.getOutputStream(), peer.getInputStream(), encryptionKeyFactory,
162      accessToken, datanodeId);
163    // TODO: Consider renaming EncryptedPeer to SaslPeer.
164    return ios != null ? new EncryptedPeer(peer, ios) : peer;
165  }

188  /**
189   * Checks if an address is already trusted and then sends client SASL
190   * negotiation if required.
191   *
192   * @param addr connection address
193   * @param underlyingOut connection output stream
194   * @param underlyingIn connection input stream
195   * @param encryptionKeyFactory for creation of an encryption key
196   * @param accessToken connection block access token
197   * @param datanodeId ID of destination DataNode
198   * @return new pair of streams, wrapped after SASL negotiation
199   * @throws IOException for any error
200   */
201  private IOStreamPair checkTrustAndSend(InetAddress addr,
202      OutputStream underlyingOut, InputStream underlyingIn,
203      DataEncryptionKeyFactory encryptionKeyFactory,
204      Token<BlockTokenIdentifier> accessToken, DatanodeID datanodeId)
205      throws IOException {
206    if (!trustedChannelResolver.isTrusted() &&
207        !trustedChannelResolver.isTrusted(addr)) {
208      // The encryption key factory only returns a key if encryption is enabled.
209      DataEncryptionKey encryptionKey =
210        encryptionKeyFactory.newDataEncryptionKey();
211      return send(addr, underlyingOut, underlyingIn, encryptionKey, accessToken,
212        datanodeId);
213    } else {
214      LOG.debug(
215        "SASL client skipping handshake on trusted connection for addr = {}, "
216        + "datanodeId = {}", addr, datanodeId);
217      return null;
218    }
219  }
220
221  /**
222   * Sends client SASL negotiation if required.  Determines the correct type of
223   * SASL handshake based on configuration.
224   *
225   * @param addr connection address
226   * @param underlyingOut connection output stream
227   * @param underlyingIn connection input stream
228   * @param encryptionKey for an encrypted SASL handshake
229   * @param accessToken connection block access token
230   * @param datanodeId ID of destination DataNode
231   * @return new pair of streams, wrapped after SASL negotiation
232   * @throws IOException for any error
233   */
234  private IOStreamPair send(InetAddress addr, OutputStream underlyingOut,
235      InputStream underlyingIn, DataEncryptionKey encryptionKey,
236      Token<BlockTokenIdentifier> accessToken, DatanodeID datanodeId)
237      throws IOException {
238    if (encryptionKey != null) {
239      LOG.debug(
240        "SASL client doing encrypted handshake for addr = {}, datanodeId = {}",
241        addr, datanodeId);
242      return getEncryptedStreams(underlyingOut, underlyingIn,
243        encryptionKey);
244    } else if (!UserGroupInformation.isSecurityEnabled()) {
245      LOG.debug(
246        "SASL client skipping handshake in unsecured configuration for "
247        + "addr = {}, datanodeId = {}", addr, datanodeId);
248      return null;
249    } else if (SecurityUtil.isPrivilegedPort(datanodeId.getXferPort())) {
250      LOG.debug(
251        "SASL client skipping handshake in secured configuration with "
252        + "privileged port for addr = {}, datanodeId = {}", addr, datanodeId);
253      return null;
254    } else if (fallbackToSimpleAuth != null && fallbackToSimpleAuth.get()) {
255      LOG.debug(
256        "SASL client skipping handshake in secured configuration with "
257        + "unsecured cluster for addr = {}, datanodeId = {}", addr, datanodeId);
258      return null;
259    } else if (saslPropsResolver != null) {
260      LOG.debug(
261        "SASL client doing general handshake for addr = {}, datanodeId = {}",
262        addr, datanodeId);
263      return getSaslStreams(addr, underlyingOut, underlyingIn, accessToken,
264        datanodeId);
265    } else {
266      // It's a secured cluster using non-privileged ports, but no SASL.  The
267      // only way this can happen is if the DataNode has
268      // ignore.secure.ports.for.testing configured, so this is a rare edge case.
269      LOG.debug(
270        "SASL client skipping handshake in secured configuration with no SASL "
271        + "protection configured for addr = {}, datanodeId = {}",
272        addr, datanodeId);
273      return null;
274    }
275  }

277  /**
278   * Sends client SASL negotiation for specialized encrypted handshake.
279   *
280   * @param underlyingOut connection output stream
281   * @param underlyingIn connection input stream
282   * @param encryptionKey for an encrypted SASL handshake
283   * @return new pair of streams, wrapped after SASL negotiation
284   * @throws IOException for any error
285   */
286  private IOStreamPair getEncryptedStreams(OutputStream underlyingOut,
287      InputStream underlyingIn, DataEncryptionKey encryptionKey)
288      throws IOException {
289    Map<String, String> saslProps = createSaslPropertiesForEncryption(
290      encryptionKey.encryptionAlgorithm);
291
292    LOG.debug("Client using encryption algorithm {}",
293      encryptionKey.encryptionAlgorithm);
294
295    String userName = getUserNameFromEncryptionKey(encryptionKey);
296    char[] password = encryptionKeyToPassword(encryptionKey.encryptionKey);
297    CallbackHandler callbackHandler = new SaslClientCallbackHandler(userName,
298      password);
299    return doSaslHandshake(underlyingOut, underlyingIn, userName, saslProps,
300      callbackHandler);
301  }

303  /**
304   * The SASL username for an encrypted handshake consists of the keyId,
305   * blockPoolId, and nonce with the first two encoded as Strings, and the third
306   * encoded using Base64. The fields are each separated by a single space.
307   * 
308   * @param encryptionKey the encryption key to encode as a SASL username.
309   * @return encoded username containing keyId, blockPoolId, and nonce
310   */
311  private static String getUserNameFromEncryptionKey(
312      DataEncryptionKey encryptionKey) {
313    return encryptionKey.keyId + NAME_DELIMITER +
314        encryptionKey.blockPoolId + NAME_DELIMITER +
315        new String(Base64.encodeBase64(encryptionKey.nonce, false), Charsets.UTF_8);
316  }

318  /**
319   * Sets user name and password when asked by the client-side SASL object.
320   */
321  private static final class SaslClientCallbackHandler
322      implements CallbackHandler {
323
324    private final char[] password;
325    private final String userName;
326
327    /**
328     * Creates a new SaslClientCallbackHandler.
329     *
330     * @param userName SASL user name
331     * @Param password SASL password
332     */
333    public SaslClientCallbackHandler(String userName, char[] password) {
334      this.password = password;
335      this.userName = userName;
336    }
337
338    @Override
339    public void handle(Callback[] callbacks) throws IOException,
340        UnsupportedCallbackException {
341      NameCallback nc = null;
342      PasswordCallback pc = null;
343      RealmCallback rc = null;
344      for (Callback callback : callbacks) {
345        if (callback instanceof RealmChoiceCallback) {
346          continue;
347        } else if (callback instanceof NameCallback) {
348          nc = (NameCallback) callback;
349        } else if (callback instanceof PasswordCallback) {
350          pc = (PasswordCallback) callback;
351        } else if (callback instanceof RealmCallback) {
352          rc = (RealmCallback) callback;
353        } else {
354          throw new UnsupportedCallbackException(callback,
355              "Unrecognized SASL client callback");
356        }
357      }
358      if (nc != null) {
359        nc.setName(userName);
360      }
361      if (pc != null) {
362        pc.setPassword(password);
363      }
364      if (rc != null) {
365        rc.setText(rc.getDefaultText());
366      }
367    }
368  }

370  /**
371   * Sends client SASL negotiation for general-purpose handshake.
372   *
373   * @param addr connection address
374   * @param underlyingOut connection output stream
375   * @param underlyingIn connection input stream
376   * @param accessToken connection block access token
377   * @param datanodeId ID of destination DataNode
378   * @return new pair of streams, wrapped after SASL negotiation
379   * @throws IOException for any error
380   */
381  private IOStreamPair getSaslStreams(InetAddress addr,
382      OutputStream underlyingOut, InputStream underlyingIn,
383      Token<BlockTokenIdentifier> accessToken, DatanodeID datanodeId)
384      throws IOException {
385    Map<String, String> saslProps = saslPropsResolver.getClientProperties(addr);
386
387    String userName = buildUserName(accessToken);
388    char[] password = buildClientPassword(accessToken);
389    CallbackHandler callbackHandler = new SaslClientCallbackHandler(userName,
390      password);
391    return doSaslHandshake(underlyingOut, underlyingIn, userName, saslProps,
392      callbackHandler);
393  }
394
395  /**
396   * Builds the client's user name for the general-purpose handshake, consisting
397   * of the base64-encoded serialized block access token identifier.  Note that
398   * this includes only the token identifier, not the token itself, which would
399   * include the password.  The password is a shared secret, and we must not
400   * write it on the network during the SASL authentication exchange.
401   *
402   * @param blockToken for block access
403   * @return SASL user name
404   */
405  private static String buildUserName(Token<BlockTokenIdentifier> blockToken) {
406    return new String(Base64.encodeBase64(blockToken.getIdentifier(), false),
407      Charsets.UTF_8);
408  }
409
410  /**
411   * Calculates the password on the client side for the general-purpose
412   * handshake.  The password consists of the block access token's password.
413   *
414   * @param blockToken for block access
415   * @return SASL password
416   */    
417  private char[] buildClientPassword(Token<BlockTokenIdentifier> blockToken) {
418    return new String(Base64.encodeBase64(blockToken.getPassword(), false),
419      Charsets.UTF_8).toCharArray();
420  }

422  /**
423   * This method actually executes the client-side SASL handshake.
424   *
425   * @param underlyingOut connection output stream
426   * @param underlyingIn connection input stream
427   * @param userName SASL user name
428   * @param saslProps properties of SASL negotiation
429   * @param callbackHandler for responding to SASL callbacks
430   * @return new pair of streams, wrapped after SASL negotiation
431   * @throws IOException for any error
432   */
433  private IOStreamPair doSaslHandshake(OutputStream underlyingOut,
434      InputStream underlyingIn, String userName, Map<String, String> saslProps,
435      CallbackHandler callbackHandler) throws IOException {
436
437    DataOutputStream out = new DataOutputStream(underlyingOut);
438    DataInputStream in = new DataInputStream(underlyingIn);
439
440    SaslParticipant sasl= SaslParticipant.createClientSaslParticipant(userName,
441      saslProps, callbackHandler);
442
443    out.writeInt(SASL_TRANSFER_MAGIC_NUMBER);
444    out.flush();
445
446    try {
447      // Start of handshake - "initial response" in SASL terminology.
448      sendSaslMessage(out, new byte[0]);
449
450      // step 1
451      byte[] remoteResponse = readSaslMessage(in);
452      byte[] localResponse = sasl.evaluateChallengeOrResponse(remoteResponse);
453      List<CipherOption> cipherOptions = null;
454      if (requestedQopContainsPrivacy(saslProps)) {
455        // Negotiate cipher suites if configured.  Currently, the only supported
456        // cipher suite is AES/CTR/NoPadding, but the protocol allows multiple
457        // values for future expansion.
458        String cipherSuites = conf.get(
459            DFS_ENCRYPT_DATA_TRANSFER_CIPHER_SUITES_KEY);
460        if (cipherSuites != null && !cipherSuites.isEmpty()) {
461          if (!cipherSuites.equals(CipherSuite.AES_CTR_NOPADDING.getName())) {
462            throw new IOException(String.format("Invalid cipher suite, %s=%s",
463                DFS_ENCRYPT_DATA_TRANSFER_CIPHER_SUITES_KEY, cipherSuites));
464          }
465          CipherOption option = new CipherOption(CipherSuite.AES_CTR_NOPADDING);
466          cipherOptions = Lists.newArrayListWithCapacity(1);
467          cipherOptions.add(option);
468        }
469      }
470      sendSaslMessageAndNegotiationCipherOptions(out, localResponse, 
471          cipherOptions);
472
473      // step 2 (client-side only)
474      SaslResponseWithNegotiatedCipherOption response = 
475          readSaslMessageAndNegotiatedCipherOption(in);
476      localResponse = sasl.evaluateChallengeOrResponse(response.payload);
477      assert localResponse == null;
478
479      // SASL handshake is complete
480      checkSaslComplete(sasl, saslProps);
481
482      CipherOption cipherOption = null;
483      if (sasl.isNegotiatedQopPrivacy()) {
484        // Unwrap the negotiated cipher option
485        cipherOption = unwrap(response.cipherOption, sasl);
486      }
487
488      // If negotiated cipher option is not null, we will use it to create 
489      // stream pair.
490      return cipherOption != null ? createStreamPair(
491          conf, cipherOption, underlyingOut, underlyingIn, false) : 
492            sasl.createStreamPair(out, in);
493    } catch (IOException ioe) {
494      sendGenericSaslErrorMessage(out, ioe.getMessage());
495      throw ioe;
496    }
497  }
