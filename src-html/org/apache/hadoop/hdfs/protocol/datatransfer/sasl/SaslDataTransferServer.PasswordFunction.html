061/**
062 * Negotiates SASL for DataTransferProtocol on behalf of a server.  There are
063 * two possible supported variants of SASL negotiation: either a general-purpose
064 * negotiation supporting any quality of protection, or a specialized
065 * negotiation that enforces privacy as the quality of protection using a
066 * cryptographically strong encryption key.
067 *
068 * This class is used in the DataNode for handling inbound connections.
069 */
070@InterfaceAudience.Private
071public class SaslDataTransferServer {
072
073  private static final Logger LOG = LoggerFactory.getLogger(
074    SaslDataTransferServer.class);
075
076  private final BlockPoolTokenSecretManager blockPoolTokenSecretManager;
077  private final DNConf dnConf;
078
079  /**
080   * Creates a new SaslDataTransferServer.
081   *
082   * @param dnConf configuration of DataNode
083   * @param blockPoolTokenSecretManager used for checking block access tokens
084   *   and encryption keys
085   */
086  public SaslDataTransferServer(DNConf dnConf,
087      BlockPoolTokenSecretManager blockPoolTokenSecretManager) {
088    this.blockPoolTokenSecretManager = blockPoolTokenSecretManager;
089    this.dnConf = dnConf;
090  }
091
092  /**
093   * Receives SASL negotiation from a peer on behalf of a server.
094   *
095   * @param peer connection peer
096   * @param underlyingOut connection output stream
097   * @param underlyingIn connection input stream
098   * @param int xferPort data transfer port of DataNode accepting connection
099   * @param datanodeId ID of DataNode accepting connection
100   * @return new pair of streams, wrapped after SASL negotiation
101   * @throws IOException for any error
102   */
103  public IOStreamPair receive(Peer peer, OutputStream underlyingOut,
104      InputStream underlyingIn, int xferPort, DatanodeID datanodeId)
105      throws IOException {
106    if (dnConf.getEncryptDataTransfer()) {
107      LOG.debug(
108        "SASL server doing encrypted handshake for peer = {}, datanodeId = {}",
109        peer, datanodeId);
110      return getEncryptedStreams(peer, underlyingOut, underlyingIn);
111    } else if (!UserGroupInformation.isSecurityEnabled()) {
112      LOG.debug(
113        "SASL server skipping handshake in unsecured configuration for "
114        + "peer = {}, datanodeId = {}", peer, datanodeId);
115      return new IOStreamPair(underlyingIn, underlyingOut);
116    } else if (SecurityUtil.isPrivilegedPort(xferPort)) {
117      LOG.debug(
118        "SASL server skipping handshake in secured configuration for "
119        + "peer = {}, datanodeId = {}", peer, datanodeId);
120      return new IOStreamPair(underlyingIn, underlyingOut);
121    } else if (dnConf.getSaslPropsResolver() != null) {
122      LOG.debug(
123        "SASL server doing general handshake for peer = {}, datanodeId = {}",
124        peer, datanodeId);
125      return getSaslStreams(peer, underlyingOut, underlyingIn);
126    } else if (dnConf.getIgnoreSecurePortsForTesting()) {
127      // It's a secured cluster using non-privileged ports, but no SASL.  The
128      // only way this can happen is if the DataNode has
129      // ignore.secure.ports.for.testing configured, so this is a rare edge case.
130      LOG.debug(
131        "SASL server skipping handshake in secured configuration with no SASL "
132        + "protection configured for peer = {}, datanodeId = {}",
133        peer, datanodeId);
134      return new IOStreamPair(underlyingIn, underlyingOut);
135    } else {
136      // The error message here intentionally does not mention
137      // ignore.secure.ports.for.testing.  That's intended for dev use only.
138      // This code path is not expected to execute ever, because DataNode startup
139      // checks for invalid configuration and aborts.
140      throw new IOException(String.format("Cannot create a secured " +
141        "connection if DataNode listens on unprivileged port (%d) and no " +
142        "protection is defined in configuration property %s.",
143        datanodeId.getXferPort(), DFS_DATA_TRANSFER_PROTECTION_KEY));
144    }
145  }
146
147  /**
148   * Receives SASL negotiation for specialized encrypted handshake.
149   *
150   * @param peer connection peer
151   * @param underlyingOut connection output stream
152   * @param underlyingIn connection input stream
153   * @return new pair of streams, wrapped after SASL negotiation
154   * @throws IOException for any error
155   */
156  private IOStreamPair getEncryptedStreams(Peer peer,
157      OutputStream underlyingOut, InputStream underlyingIn) throws IOException {
158    if (peer.hasSecureChannel() ||
159        dnConf.getTrustedChannelResolver().isTrusted(getPeerAddress(peer))) {
160      return new IOStreamPair(underlyingIn, underlyingOut);
161    }
162
163    Map<String, String> saslProps = createSaslPropertiesForEncryption(
164      dnConf.getEncryptionAlgorithm());
165
166    if (LOG.isDebugEnabled()) {
167      LOG.debug("Server using encryption algorithm " +
168        dnConf.getEncryptionAlgorithm());
169    }
170
171    CallbackHandler callbackHandler = new SaslServerCallbackHandler(
172      new PasswordFunction() {
173        @Override
174        public char[] apply(String userName) throws IOException {
175          return encryptionKeyToPassword(getEncryptionKeyFromUserName(userName));
176        }
177      });
178    return doSaslHandshake(underlyingOut, underlyingIn, saslProps,
179        callbackHandler);
180  }
181
182  /**
183   * The SASL handshake for encrypted vs. general-purpose uses different logic
184   * for determining the password.  This interface is used to parameterize that
185   * logic.  It's similar to a Guava Function, but we need to let it throw
186   * exceptions.
187   */
188  private interface PasswordFunction {
189
190    /**
191     * Returns the SASL password for the given user name.
192     *
193     * @param userName SASL user name
194     * @return SASL password
195     * @throws IOException for any error
196     */
197    char[] apply(String userName) throws IOException;
198  }
199
200  /**
201   * Sets user name and password when asked by the server-side SASL object.
202   */
203  private static final class SaslServerCallbackHandler
204      implements CallbackHandler {
205
206    private final PasswordFunction passwordFunction;
207
208    /**
209     * Creates a new SaslServerCallbackHandler.
210     *
211     * @param passwordFunction for determing the user's password
212     */
213    public SaslServerCallbackHandler(PasswordFunction passwordFunction) {
214      this.passwordFunction = passwordFunction;
215    }
216
217    @Override
218    public void handle(Callback[] callbacks) throws IOException,
219        UnsupportedCallbackException {
220      NameCallback nc = null;
221      PasswordCallback pc = null;
222      AuthorizeCallback ac = null;
223      for (Callback callback : callbacks) {
224        if (callback instanceof AuthorizeCallback) {
225          ac = (AuthorizeCallback) callback;
226        } else if (callback instanceof PasswordCallback) {
227          pc = (PasswordCallback) callback;
228        } else if (callback instanceof NameCallback) {
229          nc = (NameCallback) callback;
230        } else if (callback instanceof RealmCallback) {
231          continue; // realm is ignored
232        } else {
233          throw new UnsupportedCallbackException(callback,
234              "Unrecognized SASL DIGEST-MD5 Callback: " + callback);
235        }
236      }
237
238      if (pc != null) {
239        pc.setPassword(passwordFunction.apply(nc.getDefaultName()));
240      }
241
242      if (ac != null) {
243        ac.setAuthorized(true);
244        ac.setAuthorizedID(ac.getAuthorizationID());
245      }
246    }
247  }
248
249  /**
250   * Given a secret manager and a username encoded for the encrypted handshake,
251   * determine the encryption key.
252   * 
253   * @param userName containing the keyId, blockPoolId, and nonce.
254   * @return secret encryption key.
255   * @throws IOException
256   */
257  private byte[] getEncryptionKeyFromUserName(String userName)
258      throws IOException {
259    String[] nameComponents = userName.split(NAME_DELIMITER);
260    if (nameComponents.length != 3) {
261      throw new IOException("Provided name '" + userName + "' has " +
262          nameComponents.length + " components instead of the expected 3.");
263    }
264    int keyId = Integer.parseInt(nameComponents[0]);
265    String blockPoolId = nameComponents[1];
266    byte[] nonce = Base64.decodeBase64(nameComponents[2]);
267    return blockPoolTokenSecretManager.retrieveDataEncryptionKey(keyId,
268        blockPoolId, nonce);
269  }

271  /**
272   * Receives SASL negotiation for general-purpose handshake.
273   *
274   * @param peer connection peer
275   * @param underlyingOut connection output stream
276   * @param underlyingIn connection input stream
277   * @return new pair of streams, wrapped after SASL negotiation
278   * @throws IOException for any error
279   */
280  private IOStreamPair getSaslStreams(Peer peer, OutputStream underlyingOut,
281      InputStream underlyingIn) throws IOException {
282    if (peer.hasSecureChannel() ||
283        dnConf.getTrustedChannelResolver().isTrusted(getPeerAddress(peer))) {
284      return new IOStreamPair(underlyingIn, underlyingOut);
285    }
286
287    SaslPropertiesResolver saslPropsResolver = dnConf.getSaslPropsResolver();
288    Map<String, String> saslProps = saslPropsResolver.getServerProperties(
289      getPeerAddress(peer));
290
291    CallbackHandler callbackHandler = new SaslServerCallbackHandler(
292      new PasswordFunction() {
293        @Override
294        public char[] apply(String userName) throws IOException {
295          return buildServerPassword(userName);
296        }
297    });
298    return doSaslHandshake(underlyingOut, underlyingIn, saslProps,
299        callbackHandler);
300  }
301
302  /**
303   * Calculates the expected correct password on the server side for the
304   * general-purpose handshake.  The password consists of the block access
305   * token's password (known to the DataNode via its secret manager).  This
306   * expects that the client has supplied a user name consisting of its
307   * serialized block access token identifier.
308   *
309   * @param userName SASL user name containing serialized block access token
310   *   identifier
311   * @return expected correct SASL password
312   * @throws IOException for any error
313   */    
314  private char[] buildServerPassword(String userName) throws IOException {
315    BlockTokenIdentifier identifier = deserializeIdentifier(userName);
316    byte[] tokenPassword = blockPoolTokenSecretManager.retrievePassword(
317      identifier);
318    return (new String(Base64.encodeBase64(tokenPassword, false),
319      Charsets.UTF_8)).toCharArray();
320  }

322  /**
323   * Deserializes a base64-encoded binary representation of a block access
324   * token.
325   *
326   * @param str String to deserialize
327   * @return BlockTokenIdentifier deserialized from str
328   * @throws IOException if there is any I/O error
329   */
330  private BlockTokenIdentifier deserializeIdentifier(String str)
331      throws IOException {
332    BlockTokenIdentifier identifier = new BlockTokenIdentifier();
333    identifier.readFields(new DataInputStream(new ByteArrayInputStream(
334      Base64.decodeBase64(str))));
335    return identifier;
336  }

338  /**
339   * This method actually executes the server-side SASL handshake.
340   *
341   * @param underlyingOut connection output stream
342   * @param underlyingIn connection input stream
343   * @param saslProps properties of SASL negotiation
344   * @param callbackHandler for responding to SASL callbacks
345   * @return new pair of streams, wrapped after SASL negotiation
346   * @throws IOException for any error
347   */
348  private IOStreamPair doSaslHandshake(OutputStream underlyingOut,
349      InputStream underlyingIn, Map<String, String> saslProps,
350      CallbackHandler callbackHandler) throws IOException {
351
352    DataInputStream in = new DataInputStream(underlyingIn);
353    DataOutputStream out = new DataOutputStream(underlyingOut);
354
355    SaslParticipant sasl = SaslParticipant.createServerSaslParticipant(saslProps,
356      callbackHandler);
357
358    int magicNumber = in.readInt();
359    if (magicNumber != SASL_TRANSFER_MAGIC_NUMBER) {
360      throw new InvalidMagicNumberException(magicNumber, 
361          dnConf.getEncryptDataTransfer());
362    }
363    try {
364      // step 1
365      byte[] remoteResponse = readSaslMessage(in);
366      byte[] localResponse = sasl.evaluateChallengeOrResponse(remoteResponse);
367      sendSaslMessage(out, localResponse);
368
369      // step 2 (server-side only)
370      List<CipherOption> cipherOptions = Lists.newArrayList();
371      remoteResponse = readSaslMessageAndNegotiationCipherOptions(
372          in, cipherOptions);
373      localResponse = sasl.evaluateChallengeOrResponse(remoteResponse);
374
375      // SASL handshake is complete
376      checkSaslComplete(sasl, saslProps);
377
378      CipherOption cipherOption = null;
379      if (sasl.isNegotiatedQopPrivacy()) {
380        // Negotiate a cipher option
381        cipherOption = negotiateCipherOption(dnConf.getConf(), cipherOptions);
382        if (cipherOption != null) {
383          if (LOG.isDebugEnabled()) {
384            LOG.debug("Server using cipher suite " + 
385                cipherOption.getCipherSuite().getName());
386          }
387        }
388      }
389
390      // If negotiated cipher option is not null, wrap it before sending.
391      sendSaslMessageAndNegotiatedCipherOption(out, localResponse, 
392          wrap(cipherOption, sasl));
393
394      // If negotiated cipher option is not null, we will use it to create 
395      // stream pair.
396      return cipherOption != null ? createStreamPair(
397          dnConf.getConf(), cipherOption, underlyingOut, underlyingIn, true) : 
398            sasl.createStreamPair(out, in);
399    } catch (IOException ioe) {
400      if (ioe instanceof SaslException &&
401          ioe.getCause() != null &&
402          ioe.getCause() instanceof InvalidEncryptionKeyException) {
403        // This could just be because the client is long-lived and hasn't gotten
404        // a new encryption key from the NN in a while. Upon receiving this
405        // error, the client will get a new encryption key from the NN and retry
406        // connecting to this DN.
407        sendInvalidKeySaslErrorMessage(out, ioe.getCause().getMessage());
408      } else {
409        sendGenericSaslErrorMessage(out, ioe.getMessage());
410      }
411      throw ioe;
412    }
413  }

415  /**
416   * Sends a SASL negotiation message indicating an invalid key error.
417   *
418   * @param out stream to receive message
419   * @param message to send
420   * @throws IOException for any error
421   */
422  private static void sendInvalidKeySaslErrorMessage(DataOutputStream out,
423      String message) throws IOException {
424    sendSaslMessage(out, DataTransferEncryptorStatus.ERROR_UNKNOWN_KEY, null,
425        message);
426  }
