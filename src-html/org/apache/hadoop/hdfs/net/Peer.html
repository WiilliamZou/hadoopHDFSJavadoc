018package org.apache.hadoop.hdfs.net;
019
020import java.io.Closeable;
021import java.io.IOException;
022import java.io.InputStream;
023import java.io.OutputStream;
024import java.nio.channels.ReadableByteChannel;
025import org.apache.hadoop.classification.InterfaceAudience;
026import org.apache.hadoop.net.unix.DomainSocket;
027
028/**
029 * Represents a connection to a peer.
030 */
031@InterfaceAudience.Private
032public interface Peer extends Closeable {
033  /**
034   * @return                The input stream channel associated with this
035   *                        peer, or null if it has none.
036   */
037  public ReadableByteChannel getInputStreamChannel();
038
039  /**
040   * Set the read timeout on this peer.
041   *
042   * @param timeoutMs       The timeout in milliseconds.
043   */
044  public void setReadTimeout(int timeoutMs) throws IOException;
045
046  /**
047   * @return                The receive buffer size.
048   */
049  public int getReceiveBufferSize() throws IOException;
050
051  /**
052   * @return                True if TCP_NODELAY is turned on.
053   */
054  public boolean getTcpNoDelay() throws IOException;
055
056  /**
057   * Set the write timeout on this peer.
058   *
059   * Note: this is not honored for BasicInetPeer.
060   * See {@link BasicSocketPeer#setWriteTimeout} for details.
061   * 
062   * @param timeoutMs       The timeout in milliseconds.
063   */
064  public void setWriteTimeout(int timeoutMs) throws IOException;
065
066  /**
067   * @return                true only if the peer is closed.
068   */
069  public boolean isClosed();
070  
071  /**
072   * Close the peer.
073   *
074   * It's safe to re-close a Peer that is already closed.
075   */
076  public void close() throws IOException;
077
078  /**
079   * @return               A string representing the remote end of our 
080   *                       connection to the peer.
081   */
082  public String getRemoteAddressString();
083
084  /**
085   * @return               A string representing the local end of our 
086   *                       connection to the peer.
087   */
088  public String getLocalAddressString();
089  
090  /**
091   * @return               An InputStream associated with the Peer.
092   *                       This InputStream will be valid until you close
093   *                       this peer with Peer#close.
094   */
095  public InputStream getInputStream() throws IOException;
096  
097  /**
098   * @return               An OutputStream associated with the Peer.
099   *                       This OutputStream will be valid until you close
100   *                       this peer with Peer#close.
101   */
102  public OutputStream getOutputStream() throws IOException;
103
104  /**
105   * @return               True if the peer resides on the same
106   *                       computer as we.
107   */
108  public boolean isLocal();
109
110  /**
111   * @return               The DomainSocket associated with the current
112   *                       peer, or null if there is none.
113   */
114  public DomainSocket getDomainSocket();
115  
116  /**
117   * Return true if the channel is secure.
118   *
119   * @return               True if our channel to this peer is not
120   *                       susceptible to man-in-the-middle attacks.
121   */
122  public boolean hasSecureChannel();
123}
